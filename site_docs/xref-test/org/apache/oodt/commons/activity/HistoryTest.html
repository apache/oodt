78  	 * Test a normal, well-behaved lifecylce of (incident, incident, stop) with no
79  	 * idle times.
80  	 */
81  	public void testLifecycle() {
82  		History.closeTime = 50;
83  		Incident one = new Incident();
84  		one.setActivityID("test");
85  		History history = new History(one, this);
86  		rest();
87  		Incident two = new Incident();
88  		two.setActivityID("test");
89  		history.addIncident(two);
90  		rest();
91  		ActivityStopped three = new ActivityStopped();
92  		three.setActivityID("test");
93  		history.addIncident(three);
94  		rest();
95  		assertNotNull("History never called store", id);
96  		assertEquals("test", id);
97  		assertNotNull(incidents);
98  		assertEquals(3, incidents.size());
99  		assertEquals(one, incidents.get(0));
100 		assertEquals(two, incidents.get(1));
101 		assertEquals(three, incidents.get(2));
102 	}
