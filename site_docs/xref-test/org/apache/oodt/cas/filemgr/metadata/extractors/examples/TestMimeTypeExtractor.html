41  public class TestMimeTypeExtractor extends TestCase implements CoreMetKeys {
42  
43    private Properties initialProperties = new Properties(System.getProperties());
44  
45    public void setUp() throws Exception {
46      Properties properties = new Properties(System.getProperties());
47      URL url = this.getClass().getResource("/mime-types.xml");
48      properties.setProperty("org.apache.oodt.cas.filemgr.mime.type.repository",
49          new File(url.getFile()).getAbsolutePath());
50      System.setProperties(properties);
51    }
52  
53    public void tearDown() throws Exception {
54      System.setProperties(initialProperties);
55    }
56  
57    /**
58     * @since OODT-58
59     */
60    public void testExtract() {
61      MimeTypeExtractor extractor = new MimeTypeExtractor();
62      Product p = Product.getDefaultFlatProduct("test", "urn:oodt:GenericFile");
63      Reference r = new Reference("file:///tmp/test.he5",
64          "file:///archive/test.he5/test.he5", 0L);
65      p.getProductReferences().add(r);
66      Metadata met = new Metadata();
67      try {
68        met = extractor.doExtract(p, met);
69      } catch (MetExtractionException e) {
70        fail(e.getMessage());
71      }
72  
73      assertNotNull(met);
74      assertNotNull(met.getAllMetadata(MIME_TYPE));
75      assertEquals(3, met.getAllMetadata(MIME_TYPE).size());
76      System.out.println(met.getAllMetadata(MIME_TYPE));
77      assertEquals("application/x-hdf", met.getAllMetadata(MIME_TYPE).get(0));
78      assertEquals("application", met.getAllMetadata(MIME_TYPE).get(1));
79      assertEquals("x-hdf", met.getAllMetadata(MIME_TYPE).get(2));
80    }
