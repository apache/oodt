197   private void deleteAllFiles(String startDir) {
198     File startDirFile = new File(startDir);
199     File[] delFiles = startDirFile.listFiles();
200 
201     if (delFiles != null && delFiles.length > 0) {
202       for (File delFile : delFiles) {
203         delFile.delete();
204       }
205     }
206 
207     startDirFile.delete();
208 
209   }

211   private void ingestTestFile() {
212     Metadata prodMet;
213     StdIngester ingester = new StdIngester(transferServiceFacClass);
214 
215     try {
216       URL ingestUrl = this.getClass().getResource("/ingest");
217       URL refUrl = this.getClass().getResource("/ingest/test.txt");
218       URL metUrl = this.getClass().getResource("/ingest/test.txt.met");
219 
220       prodMet = new SerializableMetadata(new FileInputStream(
221         new File(metUrl.getFile())));
222 
223       // now add the right file location
224       prodMet.addMetadata(CoreMetKeys.FILE_LOCATION, new File(
225         ingestUrl.getFile()).getCanonicalPath());
226       prodMet.addMetadata(CoreMetKeys.FILENAME, "test.txt");
227       prodMet.addMetadata(CoreMetKeys.PRODUCT_TYPE, "GenericFile");
228       ingester.ingest(new URL("http://localhost:" + FM_PORT), new File(
229         refUrl.getFile()), prodMet);
230     } catch (Exception e) {
231       fail(e.getMessage());
232     }

235   private void startXmlRpcFileManager() {
236 
237     Properties properties = new Properties(System.getProperties());
238 
239     // first make sure to load properties for the file manager
240     // and make sure to load logging properties as well
241 
242     // set the log levels
243     URL loggingPropertiesUrl = this.getClass().getResource(
244       "/test.logging.properties");
245     properties.setProperty("java.util.logging.config.file", new File(
246       loggingPropertiesUrl.getFile()).getAbsolutePath());
247 
248     // first load the example configuration
249     try {
250       URL filemgrPropertiesUrl = this.getClass().getResource(
251         "/filemgr.properties");
252       properties.load(
253         new FileInputStream(new File(filemgrPropertiesUrl.getFile())));
254     } catch (Exception e) {
255       fail(e.getMessage());
256     }
257 
258     // override the catalog to use: we'll use lucene
259     try {
260       URL ingestUrl = this.getClass().getResource("/ingest");
261       luceneCatLoc = new File(ingestUrl.getFile()).getCanonicalPath() + "/cat";
262     } catch (Exception e) {
263       fail(e.getMessage());
264     }
265 
266     properties.setProperty("filemgr.catalog.factory",
267         "org.apache.oodt.cas.filemgr.catalog.LuceneCatalogFactory");
268     properties.setProperty("org.apache.oodt.cas.filemgr.catalog.lucene.idxPath",
269         luceneCatLoc);
270 
271     // now override the repo mgr policy
272     URL fmpolicyUrl = this.getClass().getResource("/ingest/fmpolicy");
273     try {
274       properties.setProperty("org.apache.oodt.cas.filemgr.repositorymgr.dirs",
275         "file://" + new File(fmpolicyUrl.getFile()).getCanonicalPath());
276     } catch (Exception e) {
277       fail(e.getMessage());
278     }
279 
280     // now override the val layer ones
281     properties.setProperty("org.apache.oodt.cas.filemgr.validation.dirs",
282       "file://" + new File(fmpolicyUrl.getFile()).getAbsolutePath());
283 
284     // set up mime repo path
285     URL mimeTypesUrl = this.getClass().getResource("/mime-types.xml");
286     properties.setProperty("org.apache.oodt.cas.filemgr.mime.type.repository",
287       new File(mimeTypesUrl.getFile()).getAbsolutePath());
288 
289     // override expand product met
290     properties.setProperty("org.apache.oodt.cas.filemgr.metadata.expandProduct",
291       "true");
292 
293     System.setProperties(properties);
294 
295     try {
296       fm = new XmlRpcFileManager(FM_PORT);
297     } catch (Exception e) {
298       fail(e.getMessage());
299     }
300   }
