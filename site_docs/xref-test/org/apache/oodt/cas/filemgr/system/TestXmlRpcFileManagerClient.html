75      public void testGetReducedMetadata() {
76          List arrayListElems = new ArrayList();
77          List vectorElemList = new Vector();
78          List linkedListElemList = new LinkedList();
79  
80          arrayListElems.add(CoreMetKeys.FILENAME);
81          vectorElemList.add(CoreMetKeys.FILENAME);
82          linkedListElemList.add(CoreMetKeys.FILENAME);
83  
84          try {
85              XmlRpcFileManagerClient fmc = new XmlRpcFileManagerClient(new URL(
86                      "http://localhost:" + FM_PORT));
87              
88              Metadata reducedMet;
89              List pTypes = fmc.getProductTypes();
90              assertNotNull(pTypes);
91              assertTrue(pTypes.size() > 0);
92              ProductType genericFileType = fmc.getProductTypeByName("GenericFile");
93              assertNotNull(genericFileType);
94              List products = fmc.getProductsByProductType(genericFileType);
95              assertNotNull(products);
96              assertTrue(products.size() > 0);
97              Product product = (Product) products.get(0);
98              assertNotNull(product);
99  
100             reducedMet = fmc.getReducedMetadata(product, arrayListElems);
101             assertNotNull(reducedMet);
102             assertTrue(reducedMet.containsKey(CoreMetKeys.FILENAME));
103             assertEquals(reducedMet.getMap().keySet().size(), 1);
104 
105             reducedMet = fmc.getReducedMetadata(product, vectorElemList);
106             assertNotNull(reducedMet);
107             assertTrue(reducedMet.containsKey(CoreMetKeys.FILENAME));
108             assertEquals(reducedMet.getMap().keySet().size(), 1);
109 
110             reducedMet = fmc.getReducedMetadata(product, linkedListElemList);
111             assertNotNull(reducedMet);
112             assertTrue(reducedMet.containsKey(CoreMetKeys.FILENAME));
113             assertEquals(reducedMet.getMap().keySet().size(), 1);
114             
115         } catch (Exception e) {
116             fail(e.getMessage());
117         }finally {
118             fm.shutdown();
119         }

199     public void testComplexQuery() throws Exception {
200         URL ingestUrl = this.getClass().getResource("/ingest");
201         URL refUrl1 = this.getClass().getResource("/ingest/test-file-1.txt");
202         URL refUrl2 = this.getClass().getResource("/ingest/test-file-2.txt");
203 
204         StdIngester ingester = new StdIngester(transferServiceFacClass);
205 
206         //ingest first file
207         Metadata prodMet = new Metadata();
208         prodMet.addMetadata(CoreMetKeys.FILE_LOCATION, new File(
209             ingestUrl.getFile()).getCanonicalPath());
210         prodMet.addMetadata(CoreMetKeys.FILENAME, "test-file-1.txt");
211         prodMet.addMetadata(CoreMetKeys.PRODUCT_NAME, "TestFile1");
212         prodMet.addMetadata(CoreMetKeys.PRODUCT_TYPE, "GenericFile");
213         ingester.ingest(new URL("http://localhost:" + FM_PORT), new File(
214             refUrl1.getFile()), prodMet);           
215         
216         //ingest second file
217         prodMet.replaceMetadata(CoreMetKeys.FILENAME, "test-file-2.txt");
218         prodMet.replaceMetadata(CoreMetKeys.PRODUCT_NAME, "TestFile2");
219         ingester.ingest(new URL("http://localhost:" + FM_PORT), new File(
220             refUrl2.getFile()), prodMet);   
221         
222         //perform complex query
223         ComplexQuery complexQuery = new ComplexQuery();
224         List<String> reducedMetadata = new Vector<String>();
225         reducedMetadata.add(CoreMetKeys.FILENAME);
226         complexQuery.setReducedMetadata(reducedMetadata);
227         List<String> productTypeNames = new Vector<String>();
228         productTypeNames.add("GenericFile");
229         complexQuery.setReducedProductTypeNames(productTypeNames);
230         complexQuery.setSortByMetKey(CoreMetKeys.FILENAME);
231         complexQuery.setToStringResultFormat("$" + CoreMetKeys.FILENAME);
232         complexQuery.addCriterion(SqlParser.parseSqlWhereClause("Filename != 'test.txt'"));
233         XmlRpcFileManagerClient fmc = new XmlRpcFileManagerClient(new URL(
234                 "http://localhost:" + FM_PORT));
235         List<QueryResult> queryResults = fmc.complexQuery(complexQuery);
236         assertEquals("[test-file-1.txt, test-file-2.txt]", queryResults.toString());
237     }

306     private void startXmlRpcFileManager() {
307 
308         Properties properties = new Properties(System.getProperties());
309 
310         // first make sure to load properties for the file manager
311         // and make sure to load logging properties as well
312 
313         // set the log levels
314         URL loggingPropertiesUrl = this.getClass().getResource(
315             "/test.logging.properties");
316         properties.setProperty("java.util.logging.config.file", new File(
317             loggingPropertiesUrl.getFile()).getAbsolutePath());
318 
319         // first load the example configuration
320         try {
321             URL filemgrPropertiesUrl = this.getClass().getResource(
322                 "/filemgr.properties");
323             properties.load(
324                 new FileInputStream(new File(filemgrPropertiesUrl.getFile())));
325         } catch (Exception e) {
326             fail(e.getMessage());
327         }
328 
329         // override the catalog to use: we'll use lucene
330         try {
331             URL ingestUrl = this.getClass().getResource("/ingest");
332             luceneCatLoc = new File(ingestUrl.getFile()).getCanonicalPath()
333                 + "/cat";
334         } catch (Exception e) {
335             fail(e.getMessage());
336         }
337 
338         properties.setProperty("filemgr.catalog.factory",
339                 "org.apache.oodt.cas.filemgr.catalog.LuceneCatalogFactory");
340         properties.setProperty(
341                 "org.apache.oodt.cas.filemgr.catalog.lucene.idxPath",
342                 luceneCatLoc);
343 
344         // now override the repo mgr policy
345         URL fmpolicyUrl = this.getClass().getResource("/ingest/fmpolicy");
346         try {
347             properties.setProperty(
348                 "org.apache.oodt.cas.filemgr.repositorymgr.dirs",
349                 "file://" + new File(fmpolicyUrl.getFile()).getCanonicalPath());
350         } catch (Exception e) {
351             fail(e.getMessage());
352         }
353 
354         // now override the val layer ones
355         properties.setProperty("org.apache.oodt.cas.filemgr.validation.dirs",
356             "file://" + new File(fmpolicyUrl.getFile()).getAbsolutePath());
357 
358         // set up mime repo path
359         URL mimeTypesUrl = this.getClass().getResource("/mime-types.xml");
360         properties.setProperty(
361             "org.apache.oodt.cas.filemgr.mime.type.repository",
362             new File(mimeTypesUrl.getFile()).getAbsolutePath());
363 
364         System.setProperties(properties);
365 
366         try {
367             fm = new XmlRpcFileManager(FM_PORT);
368         } catch (Exception e) {
369             fail(e.getMessage());
370         }
371     }
