52    private static final Logger LOG = Logger
53        .getLogger(TestXmlRpcWorkflowManager.class.getName());

55    public void testGetWorkflowInstances() {
56  
57      List workflowInsts = null;
58  
59      int numInsts = -1;
60      while (numInsts != 2) {
61        try {
62          workflowInsts = wmgr.getWorkflowInstances();
63        } catch (Exception e) {
64          LOG.log(Level.SEVERE, e.getMessage());
65        }
66  
67        assertNotNull(workflowInsts);
68        numInsts = workflowInsts.size();
69      }
70  
71      assertEquals(2, workflowInsts.size());
72    }

83    private void startWorkflow() {
84      XmlRpcWorkflowManagerClient client = null;
85      try {
86        client = new XmlRpcWorkflowManagerClient(new URL("http://localhost:"
87            + WM_PORT));
88      } catch (Exception e) {
89        fail(e.getMessage());
90      }
91  
92      try {
93        client.sendEvent("long", new Metadata());
94      } catch (Exception e) {
95        LOG.log(Level.SEVERE, e.getMessage());
96        throw new RuntimeException(e);
97      }
98  
99    }

101   private void startXmlRpcWorkflowManager() {
102     System.setProperty("java.util.logging.config.file", new File(
103         "./src/main/resources/logging.properties").getAbsolutePath());
104 
105     try {
106       System.getProperties().load(
107           new FileInputStream("./src/main/resources/workflow.properties"));
108     } catch (Exception e) {
109       fail(e.getMessage());
110     }
111 
112     try {
113       luceneCatLoc = File.createTempFile("blah", "txt").getParentFile()
114           .getCanonicalPath();
115       luceneCatLoc = !luceneCatLoc.endsWith("/") ? luceneCatLoc + "/"
116           : luceneCatLoc;
117       luceneCatLoc += "repo";
118       LOG.log(Level.INFO, "Lucene instance repository: [" + luceneCatLoc + "]");
119     } catch (Exception e) {
120       fail(e.getMessage());
121     }
122 
123     if (new File(luceneCatLoc).exists()) {
124       // blow away lucene cat
125       LOG.log(Level.INFO, "Removing workflow instance repository: ["
126           + luceneCatLoc + "]");
127       try {
128         FileUtils.deleteDirectory(new File(luceneCatLoc));
129       } catch (IOException e) {
130         fail(e.getMessage());
131       }
132     }
133 
134     System
135         .setProperty("workflow.engine.instanceRep.factory",
136             "org.apache.oodt.cas.workflow.instrepo.LuceneWorkflowInstanceRepositoryFactory");
137     System
138         .setProperty("org.apache.oodt.cas.workflow.instanceRep.lucene.idxPath",
139             luceneCatLoc);
140 
141     try {
142       System.setProperty("org.apache.oodt.cas.workflow.repo.dirs", "file://"
143           + new File("./src/main/resources/examples").getCanonicalPath());
144       System.setProperty("org.apache.oodt.cas.workflow.lifecycle.filePath",
145           new File("./src/main/resources/examples/workflow-lifecycle.xml")
146               .getCanonicalPath());
147     } catch (Exception e) {
148       fail(e.getMessage());
149     }
150 
151     try {
152       wmgr = new XmlRpcWorkflowManager(WM_PORT);
153     } catch (Exception e) {
154       LOG.log(Level.SEVERE, e.getMessage());
155       fail(e.getMessage());
156     }
