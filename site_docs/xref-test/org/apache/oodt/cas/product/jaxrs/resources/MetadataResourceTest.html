39  /**
40   * Implements tests for methods in the {@link MetadataResource} class.
41   * @author rlaidlaw
42   * @version $Revision$
43   */
44  public class MetadataResourceTest
45  {
46    /**
47     * Tests that {@link MetadataResource metadata resources} are marshalled to
48     * the expected XML format.
49     * @throws IOException if the {@link Diff} constructor fails
50     * @throws JAXBException if the {@link JAXBContext} or {@link Marshaller} fail
51     * @throws SAXException if the {@link Diff} constructor fails
52     */
53    @Test
54    public void testXmlMarshalling() throws IOException, JAXBException,
55      SAXException
56    {
57      // Generate the expected output.
58      String expectedXml =
59        "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"yes\"?>"
60        + "<metadata>"
61        + "<keyval><key>1</key><val>one</val></keyval>"
62        + "<keyval><key>2</key><val>two</val></keyval>"
63        + "<keyval><key>3</key><val>three</val></keyval>"
64        + "<keyval><key>4</key><val>a</val><val>b</val><val>c</val></keyval>"
65        + "</metadata>";
66  
67      // Create a MetadataResource using a Metadata instance.
68      Hashtable metadataEntries = new Hashtable<String, Object>();
69      metadataEntries.put("1", "one");
70      metadataEntries.put("2", "two");
71      metadataEntries.put("3", "three");
72      List<String> list = new ArrayList<String>();
73      list.add("a");
74      list.add("b");
75      list.add("c");
76      metadataEntries.put("4", list);
77  
78      Metadata metadata = new Metadata();
79      metadata.addMetadata(metadataEntries);
80      MetadataResource resource = new MetadataResource(metadata);
81  
82  
83      // Set up a JAXB context and marshall the ReferenceResource to XML.
84      JAXBContext context = JAXBContext.newInstance(resource.getClass());
85      Marshaller marshaller = context.createMarshaller();
86      StringWriter writer = new StringWriter();
87      marshaller.marshal(resource, writer);
88  
89      // Compare the expected and actual outputs.
90      XMLUnit.setIgnoreWhitespace(true);
91      XMLUnit.setIgnoreComments(true);
92      XMLUnit.setIgnoreAttributeOrder(true);
93      Diff diff = new Diff(expectedXml, writer.toString());
94      diff.overrideElementQualifier(new RecursiveElementNameAndTextQualifier());
95      assertTrue("The output XML was different to the expected XML: "
96        + diff.toString(), diff.similar());
97    }
