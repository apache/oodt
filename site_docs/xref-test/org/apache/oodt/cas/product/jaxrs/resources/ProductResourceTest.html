44  /**
45   * Implements tests for methods in the {@link ProductResource} class.
46   * @author rlaidlaw
47   * @version $Revision$
48   */
49  public class ProductResourceTest
50  {
51    /**
52     * Tests that {@link ProductResource product resources} are marshalled to
53     * the expected XML format.
54     * @throws IOException if the {@link Diff} constructor fails
55     * @throws JAXBException if the {@link JAXBContext} or {@link Marshaller} fail
56     * @throws MimeTypeException if {@link MimeTypes#forName(String)} fails
57     * @throws SAXException if the {@link Diff} constructor fails
58     */
59    @Test
60    public void testXmlMarshalling() throws IOException, JAXBException,
61      MimeTypeException, SAXException
62    {
63      // Create a ProductResource using ProductType, Reference, Metadata and
64      // Product instances.
65      Hashtable metadataEntries = new Hashtable<String, Object>();
66      metadataEntries.put("CAS.Test", "test value");
67      Metadata metadata = new Metadata();
68      metadata.addMetadata(metadataEntries);
69  
70      Reference reference = new Reference("original", "dataStore", 1000,
71        new MimeTypes().forName("text/plain"));
72      List<Reference> references = new ArrayList<Reference>();
73      references.add(reference);
74  
75      ProductType productType = new ProductType("1", "GenericFile", "test type",
76        "repository", "versioner");
77  
78      Product product = new Product();
79      product.setProductId("123");
80      product.setProductName("test.txt");
81      product.setProductStructure(Product.STRUCTURE_FLAT);
82      product.setProductType(productType);
83  
84      ProductResource resource = new ProductResource(product, metadata,
85        references, new File("/tmp"));
86  
87  
88      // Generate the expected output.
89      String expectedXml =
90          "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"yes\"?>"
91          + "<product>"
92          + "<id>" + product.getProductId() + "</id>"
93          + "<name>" + product.getProductName() + "</name>"
94          + "<structure>" + product.getProductStructure() + "</structure>"
95          + "<type>" + productType.getName() + "</type>"
96          + "<metadata>"
97          + "<keyval>"
98          + "<key>" + metadata.getAllKeys().get(0) + "</key>"
99          + "<val>" + metadata.getAllValues().get(0) + "</val>"
100         + "</keyval>"
101         + "</metadata>"
102         + "<references>"
103         + "<reference>"
104         + "<productId>" + product.getProductId() + "</productId>"
105         + "<refIndex>0</refIndex>"
106         + "<dataStoreReference>"
107         +    reference.getDataStoreReference()
108         + "</dataStoreReference>"
109         + "<originalReference>"
110         +    reference.getOrigReference()
111         + "</originalReference>"
112         + "<mimeType>" + reference.getMimeType().getName() + "</mimeType>"
113         + "<fileSize>" + reference.getFileSize() + "</fileSize>"
114         + "</reference>"
115         + "</references>"
116         + "</product>";
117 
118 
119     // Set up a JAXB context and marshall the ProductResource to XML.
120     JAXBContext context = JAXBContext.newInstance(resource.getClass());
121     Marshaller marshaller = context.createMarshaller();
122     StringWriter writer = new StringWriter();
123     marshaller.marshal(resource, writer);
124 
125     // Compare the expected and actual outputs.
126     XMLUnit.setIgnoreWhitespace(true);
127     XMLUnit.setIgnoreComments(true);
128     XMLUnit.setIgnoreAttributeOrder(true);
129     Diff diff = new Diff(expectedXml, writer.toString());
130     assertTrue("The output XML was different to the expected XML: "
131       + diff.toString(), diff.identical());
132   }
