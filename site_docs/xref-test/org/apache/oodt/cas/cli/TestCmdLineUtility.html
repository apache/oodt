52  /**
53   * Test class for {@link CmdLineUtility}.
54   * 
55   * @author bfoster (Brian Foster)
56   */
57  public class TestCmdLineUtility extends TestCase {
58  
59     public void testCheck() {
60        CmdLineArgs args = getArgs();
61  
62        // Expect pass.
63        assertEquals(0, CmdLineUtility.check(args).size());
64  
65        // Expect fail.
66        args.getSupportedOptions().add(
67              TestUtils.createSimpleOption("ReqTestAction", true));
68        assertEquals(1, CmdLineUtility.check(args).size());
69     }
70  
71     public void testValidate() {
72        CmdLineArgs args = getArgs();
73  
74        // Expect pass.
75        assertEquals(1, CmdLineUtility.validate(args).size());
76        assertEquals(Result.Grade.PASS, CmdLineUtility.validate(args).get(0)
77              .getGrade());
78  
79        // Add validator which will cause fail.
80        AdvancedCmdLineOption option = (AdvancedCmdLineOption) getOptionByName("message", args.getSupportedOptions());
81        ArgRegExpCmdLineOptionValidator validator = new ArgRegExpCmdLineOptionValidator();
82        validator.setAllowedArgs(Lists.newArrayList("\\d{1,2}"));
83        option.addValidator(validator);
84  
85        // Expect fail.
86        assertFalse(determineFailedValidation(validate(args)).isEmpty());
87     }
88  
89     public void testHandle() {
90        CmdLineArgs args = getArgs();
91  
92        // Verify handling works.
93        PrintMessageAction action = (PrintMessageAction) args
94              .getSpecifiedAction();
95        assertNull(action.getMessage());
96        CmdLineUtility.handle(args);
97        assertEquals("Test Message", action.getMessage());
98     }
99  
100    private CmdLineArgs getArgs() {
101       // Setup Supported Actions.
102       String actionName = "TestAction";
103       PrintMessageAction action = new PrintMessageAction();
104       action.setName(actionName);
105       Set<CmdLineAction> actions = Sets.newHashSet((CmdLineAction) action);
106 
107       // Setup Supported Options.
108       Set<CmdLineOption> options = Sets.newHashSet();
109       options.add(new HelpCmdLineOption());
110       options.add(new PrintSupportedActionsCmdLineOption());
111       options.add(createActionOption("action"));
112       AdvancedCmdLineOption option = createAdvancedOption("message",
113             createApplyToActionHandler(actionName, "setMessage"));
114       ArgRegExpCmdLineOptionValidator validator = new ArgRegExpCmdLineOptionValidator();
115       validator.setAllowedArgs(Lists.newArrayList(".*"));
116       option.addValidator(validator);
117       options.add(option);
118 
119       // Setup Specified Options.
120       Set<CmdLineOptionInstance> specifiedOptions = Sets.newHashSet();
121       specifiedOptions.add(createOptionInstance(option, "Test Message"));
122       specifiedOptions.add(createOptionInstance(createActionOption("action"),
123             "TestAction"));
124 
125       // Setup CmdLineArgs.
126       return new CmdLineArgs(actions, options, specifiedOptions);
127    }
