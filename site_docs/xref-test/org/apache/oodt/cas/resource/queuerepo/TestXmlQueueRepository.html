41  public class TestXmlQueueRepository extends TestCase {

43  	private QueueManager queueManager;
44  	
45  	protected void setUp() {
46  		System.setProperty("org.apache.oodt.cas.resource.nodetoqueues.dirs",
47  				"file:"
48  						+ new File("./src/main/resources/examples")
49  								.getAbsolutePath());
50  		this.queueManager = new XmlQueueRepositoryFactory().createQueueRepository().loadQueues();
51  	}
52  	
53  	public void testMapping() throws QueueManagerException {
54  	  assertTrue(this.queueManager.getQueues().containsAll(Arrays.asList("quick", "high", "long")));
55  		assertEquals(this.queueManager.getNodes("quick"), Collections.singletonList("localhost"));
56  		assertEquals(this.queueManager.getNodes("high"), Collections.singletonList("localhost"));
57  		assertEquals(this.queueManager.getNodes("long"), Collections.singletonList("localhost"));
58  		assertTrue(this.queueManager.getQueues("localhost").containsAll(Arrays.asList("quick", "high", "long")));
59  		
60  		this.queueManager.addQueue("test-queue-1");
61  		this.queueManager.addNodeToQueue("test-node-1", "test-queue-1");
62  		
63  		assertEquals(this.queueManager.getQueues("test-node-1"), Collections.singletonList("test-queue-1"));
64  
65  		this.queueManager.addNodeToQueue("test-node-1","quick");
66  		assertEquals(this.queueManager.getQueues("test-node-1"), Arrays.asList("quick", "test-queue-1"));
67  		
68  		this.queueManager.removeQueue("quick");
69  		assertEquals(this.queueManager.getQueues("test-node-1"), Collections.singletonList("test-queue-1"));
70  		assertEquals(this.queueManager.getQueues("localhost"), Arrays.asList("high", "long"));
71  	}
