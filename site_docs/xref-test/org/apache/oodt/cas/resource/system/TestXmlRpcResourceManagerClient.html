58    private static void generateTestConfiguration() throws IOException {
59      Properties config = new Properties();
60  
61      String propertiesFile = "." + File.separator + "src" + File.separator +
62                              "test" + File.separator + "resources" + File.separator + "test.resource.properties";
63      System.getProperties().load(new FileInputStream(new File(propertiesFile)));
64  
65      // stage policy
66      File tmpPolicyDir = null;
67      try {
68        tmpPolicyDir = File.createTempFile("test", "ignore").getParentFile();
69      } catch (Exception e) {
70        fail(e.getMessage());
71      }
72      for (File policyFile : new File("./src/test/resources/policy")
73          .listFiles(new FileFilter() {
74  
75            @Override
76            public boolean accept(File pathname) {
77              return pathname.isFile() && pathname.getName().endsWith(".xml");
78            }
79          })) {
80        try {
81          FileUtils.copyFileToDirectory(policyFile, tmpPolicyDir);
82        } catch (Exception e) {
83          fail(e.getMessage());
84        }
85      }
86  
87      config.setProperty("org.apache.oodt.cas.resource.nodes.dirs", tmpPolicyDir
88          .toURI().toString());
89      config.setProperty("org.apache.oodt.cas.resource.nodetoqueues.dirs",
90          tmpPolicyDir.toURI().toString());
91  
92      System.getProperties().putAll(config);
93  
94    }
