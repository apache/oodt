55      protected void setUp() {
56          try {
57              //Map monitor to nodes list
58              map = new ConcurrentHashMap<MockMonitor,List<ResourceNode>>();
59              List<ResourceNode> nodes1 = getNodesList("mock-1");
60              List<ResourceNode> nodes2 = getNodesList("mock-2");
61              //Backend Manager setup
62              BackendManager back = new StandardBackendManager();
63              back.addSet("queue-1",(mock1 = addMonitor(0,map,nodes1)), null, null);
64              back.addSet("queue-2",(mock2 = addMonitor(5,map,nodes2)), null, null);
65              //Make sure the queue manager is setup
66              qm = new QueueManager();
67              qm.addQueue("queue-1");
68              qm.addQueue("queue-2");
69              qm.addQueue("queue-3");
70              for (ResourceNode rn : nodes1)
71                  qm.addNodeToQueue(rn.getNodeId(), "queue-1");
72              for (ResourceNode rn : nodes2)
73                  qm.addNodeToQueue(rn.getNodeId(), "queue-2");
74              //Add an extra node to test "unknown queue"
75              qm.addNodeToQueue((superfluous = new ResourceNode("superfluous-1",new URL("http://superfluous-1"),-2)).getNodeId(), "queue-3");
76              monitor = new QueueMuxMonitor(back, qm);
77          } catch (QueueManagerException e) {
78              TestCase.fail("Unanticipated queue manager exception caught: "+e.getMessage());
79          } catch (MalformedURLException e) {
80              TestCase.fail("Unanticipated URL exception caught: "+e.getMessage());
81          }
82      }

187     private MockMonitor addMonitor(int load,Map<MockMonitor, List<ResourceNode>> map, List<ResourceNode> list) {
188         MockMonitor mon = new MockMonitor(load, list, list.get(0), list.get(1), list.get(2));
