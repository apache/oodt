16  package org.apache.oodt.commons;
17  
18  import java.io.File;
19  
20  import junit.framework.TestCase;
21  
22  /**
23   * Abstract base class for test cases.
24   *
25   * @author <a href="jason@zenplex.com">Jason van Zyl</a>
26   */
27  public abstract class AbstractTestCase
28      extends TestCase 
29  {
30      /** 
31       * Basedir for all file I/O. Important when running tests from
32       * the reactor.
33       */
34      public String basedir = System.getProperty("basedir");
35      
36      /**
37       * Constructor.
38       */
39      public AbstractTestCase(String testName)
40      {
41          super(testName);
42      }
43      
44      /**
45       * Get test input file.
46       *
47       * @param path Path to test input file.
48       */
49      public String getTestFile(String path)
50      {
51          return new File(basedir,path).getAbsolutePath();
52      }
53  }
