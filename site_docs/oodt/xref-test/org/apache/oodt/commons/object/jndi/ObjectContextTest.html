46  	public void setUp() throws Exception {
47  		super.setUp();
48  
49  		aliasFile = File.createTempFile("test", ".properties");
50  		aliasFile.deleteOnExit();
51  		Properties aliases = new Properties();
52  		aliases.setProperty("urn:alias:x", "urn:a:x");
53  		FileOutputStream out = new FileOutputStream(aliasFile);
54  		aliases.save(out, "Temporary properties");
55  		out.close();
56  
57  		a1 = new TContext("urn:a");
58  		a2 = new TContext("urn:a");
59  		b = new TContext("urn:b");
60  
61  		oldValue = System.getProperty("org.apache.oodt.commons.object.jndi.aliases");
62  		System.setProperty("org.apache.oodt.commons.object.jndi.aliases", aliasFile.toString());
63  
64  		List contexts = new ArrayList();
65  		contexts.add(a1);
66  		contexts.add(a2);
67  		contexts.add(b);
68  		context = new ObjectContext(contexts);
69  	}
70  
71  	public void tearDown() throws Exception {
72  		aliasFile.delete();
73  		if (oldValue == null)
74  			System.getProperties().remove("org.apache.oodt.commons.object.jndi.aliases");
75  		else
76  			System.setProperty("org.apache.oodt.commons.object.jndi.aliases", oldValue);
77  		super.tearDown();
78  	}

80  	/**
81  	 * Test various aspects of the object context.
82  	 *
83  	 * @throws NamingException if an error occurs.
84  	 */
85  	public void testObjectContext() throws NamingException {
86  		// Test retrieval of nonexistent bindings
87  		try {
88  			context.lookup("urn:a:x");
89  			fail("Found nonexistent object");
90  		} catch (NamingException ignored) {}
91  
92  		// Test binding names that don't match any delegate's namespace prefix.
93  		try {
94  			context.bind("urn:c:x", this);
95  			fail("Bound nonconforming name");
96  		} catch (NamingException ignored) {}
97  
98  		// Test binding and retrieval
99  		context.bind("urn:a:x", this);					       // Bind something
100 		assertSame(this, context.lookup("urn:a:x"));			       // Look it up
101 		assertSame(this, context.lookup("urn:alias:x"));		       // Try the alias
102 		assertTrue(a1.bindings.values().contains(this));		       // It should be in both a1...
103 		assertTrue(a2.bindings.values().contains(this));		       // ...and a2
104 		assertTrue(!b.bindings.values().contains(this));		       // But not b.
105 
106 		context.bind("urn:b:x", getClass());				       // Now bind something for b
107 		assertSame(getClass(), context.lookup("urn:b:x"));		       // Look it up
108 		assertTrue(!a1.bindings.values().contains(getClass()));		       // It should not be in a1...
109 		assertTrue(!a2.bindings.values().contains(getClass()));		       // ...nor a2
110 		assertTrue(b.bindings.values().contains(getClass()));		       // But should be in b.
111 
112 		// Test binding a bound name
113 		try {
114 			context.bind("urn:a:x", "");
115 			fail("Able to bind an already-bound name");
116 		} catch (NamingException ignored) {}
117 
118 		// Test rebinding a bound name
119 		context.rebind("urn:a:x", context);				       // Bind to a different object
120 		assertSame(context, context.lookup("urn:a:x"));			       // Look it up
121 		assertTrue(!a1.bindings.values().contains(this));		       // The old object should be gone from a1...
122 		assertTrue(!a2.bindings.values().contains(this));		       // ...and from a2
123 		assertTrue(!b.bindings.values().contains(context));		       // And the new object isn't in b
124 
125 		// Test renaming
126 		context.rename("urn:a:x", "urn:a:y");				       // Change x to y
127 		try {
128 			context.lookup("urn:a:x");				       // Look it up
129 			fail("Found object under old name");
130 		} catch (NamingException ignored) {}
131 		assertSame(context, context.lookup("urn:a:y"));			       // Just the name has changed
132 		assertTrue(a1.bindings.keySet().contains("urn:a:y"));		       // The new name is in a1
133 		assertTrue(!a1.bindings.keySet().contains("urn:a:x"));		       // But not the old
134 		assertTrue(a2.bindings.keySet().contains("urn:a:y"));		       // The new name is in a2
135 		assertTrue(!a2.bindings.keySet().contains("urn:a:x"));		       // But not the old
136 		assertTrue(!b.bindings.values().contains(context));		       // It was never in b
137 
138 		// Test listing
139 		int count = 0;
140 		boolean sawA = false;
141 		boolean sawB = false;
142 		NamingEnumeration e = context.list("");
143 		while (e.hasMore()) {
144 			NameClassPair p = (NameClassPair) e.next();
145 			if ("urn:a:y".equals(p.getName()) && context.getClass().getName().equals(p.getClassName()))
146 				sawA = true;
147 			else if ("urn:b:x".equals(p.getName()) && "java.lang.Class".equals(p.getClassName()))
148 				sawB = true;
149 			else
150 				fail("Unexpected binding \"" + p.getName() + "\" to " + p.getClassName());
151 			++count;
152 		}
153 		assertEquals(3, count);
154 		assertTrue(sawA);
155 		assertTrue(sawB);
156 
157 		// Test listing of bindings
158 		count = 0;
159 		sawA = false;
160 		sawB = false;
161 		e = context.listBindings("");
162 		while (e.hasMore()) {
163 			Binding b = (Binding) e.next();
164 			if ("urn:a:y".equals(b.getName()) && context == b.getObject())
165 				sawA = true;
166 			else if ("urn:b:x".equals(b.getName()) && getClass() == b.getObject())
167 				sawB = true;
168 			else
169 				fail("Unexpected binding \"" + b.getName() + "\" to " + b.getObject());
170 			++count;
171 		}
172 		assertEquals(3, count);
173 		assertTrue(sawA);
174 		assertTrue(sawB);
175 			
176 		// Test unbinding
177 		context.unbind("urn:a:y");					       // Unbind it
178 		try {
179 			context.lookup("urn:a:y");				       // Look it up
180 			fail("Found unbound object");
181 		} catch (NamingException ignored) {}
182 		assertTrue(a1.bindings.isEmpty());				       // It's not in a1...
183 		assertTrue(a2.bindings.isEmpty());				       // ...nor in a2
184 	}
