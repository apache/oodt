58     public void testValidateErrors() throws CmdLineActionException {
59        ActionMessagePrinter printer = new ActionMessagePrinter();
60        MockDeleteProductByIdCliAction cliAction = new MockDeleteProductByIdCliAction();
61        try {
62           cliAction.execute(printer);
63           fail("Expected throw CmdLineActionException");
64        } catch (CmdLineActionException ignore) {
65        }
66        cliAction.setProductId(PRODUCT_ID);
67        cliAction.execute(printer); // Should not throw exception.
68        cliAction = new NullProductDeleteProductByIdCliAction();
69        cliAction.setProductId(PRODUCT_ID);

71           cliAction.execute(printer);

91     public void testDataFlow() throws CmdLineActionException {
92        ActionMessagePrinter printer = new ActionMessagePrinter();
93        MockDeleteProductByIdCliAction cliAction = new MockDeleteProductByIdCliAction();
94        cliAction.setProductId(PRODUCT_ID);
95        cliAction.execute(printer);
96        assertEquals(2, printer.getPrintedMessages().size());
97        assertEquals("Successfully deleted product '" + PRODUCT_NAME + "'",
98              printer.getPrintedMessages().get(0));
99        assertEquals("\n", printer.getPrintedMessages().get(1));
100       assertEquals(PRODUCT_ID, clientSetProduct.getProductId());
101       assertEquals(PRODUCT_NAME, clientSetProduct.getProductName());
102       assertEquals(2, clientSetProduct.getProductReferences().size());
103       assertEquals(REF_1, clientSetProduct.getProductReferences().get(0)
104             .getDataStoreReference());
105       assertEquals(REF_2, clientSetProduct.getProductReferences().get(1)
106             .getDataStoreReference());
107    }
