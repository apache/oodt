65      public void testIngest() {
66          Metadata prodMet;
67  
68          try {
69              URL ingestUrl = this.getClass().getResource("/ingest");
70              URL refUrl = this.getClass().getResource("/ingest/test.txt");
71              URL metUrl = this.getClass().getResource("/ingest/test.txt.met");
72  
73              prodMet = new SerializableMetadata(new FileInputStream(
74                  new File(metUrl.getFile())));
75  
76              // now add the right file location
77              prodMet.addMetadata(CoreMetKeys.FILE_LOCATION, new File(
78                  ingestUrl.getFile()).getCanonicalPath());
79              ingester.ingest(new URL("http://localhost:" + FM_PORT), new File(
80                  refUrl.getFile()), prodMet);
81          } catch (Exception e) {
82              fail(e.getMessage());
83          }
84          
85          // now make sure that the file is ingested
86          try {
87              XmlRpcFileManagerClient fmClient = new XmlRpcFileManagerClient(new URL("http://localhost:"+FM_PORT));
88              Product p = fmClient.getProductByName("test.txt");
89              assertNotNull(p);
90              assertEquals(Product.STATUS_RECEIVED, p.getTransferStatus());
91              assertTrue(fmClient.hasProduct("test.txt"));
92          } catch (Exception e){
93              fail(e.getMessage());
94          }
95      }

125     private void deleteAllFiles(String startDir) {
126         File startDirFile = new File(startDir);
127         File[] delFiles = startDirFile.listFiles();
128 
129         if (delFiles != null && delFiles.length > 0) {
130             for (File delFile : delFiles) {
131                 delFile.delete();
132             }
133         }
134 
135         startDirFile.delete();

139     private void startXmlRpcFileManager() {
140 
141         Properties properties = new Properties(System.getProperties());
142 
143         // first make sure to load properties for the file manager
144         // and make sure to load logging properties as well
145 
146         // set the log levels
147         URL loggingPropertiesUrl = this.getClass().getResource(
148             "/test.logging.properties");
149         properties.setProperty("java.util.logging.config.file", new File(
150             loggingPropertiesUrl.getFile()).getAbsolutePath());
151 
152         // first load the example configuration
153         try {
154           URL filemgrPropertiesUrl = this.getClass().getResource(
155               "/filemgr.properties");
156           properties.load(
157               new FileInputStream(new File(filemgrPropertiesUrl.getFile())));
158         } catch (Exception e) {
159             fail(e.getMessage());
160         }
161 
162         // override the catalog to use: we'll use lucene
163         try {
164             URL ingestUrl = this.getClass().getResource("/ingest");
165             luceneCatLoc = new File(ingestUrl.getFile()).getCanonicalPath()
166                 + "/cat";
167         } catch (Exception e) {
168             fail(e.getMessage());
169         }
170 
171         properties.setProperty("filemgr.catalog.factory",
172                 "org.apache.oodt.cas.filemgr.catalog.LuceneCatalogFactory");
173         properties.setProperty(
174                 "org.apache.oodt.cas.filemgr.catalog.lucene.idxPath",
175                 luceneCatLoc);
176 
177         // now override the repo mgr policy
178         try {
179             URL fmpolicyUrl = this.getClass().getResource("/ingest/fmpolicy");
180             properties.setProperty(
181                     "org.apache.oodt.cas.filemgr.repositorymgr.dirs",
182                     "file://"
183                         + new File(fmpolicyUrl.getFile()).getCanonicalPath());
184         } catch (Exception e) {
185             fail(e.getMessage());
186         }
187 
188         // now override the val layer ones
189         URL examplesCoreUrl = this.getClass().getResource("/examples/core");
190         properties.setProperty("org.apache.oodt.cas.filemgr.validation.dirs",
191                 "file://"
192                     + new File(examplesCoreUrl.getFile()).getAbsolutePath());
193 
194         // set up mime repo path
195         URL mimeTypesUrl = this.getClass().getResource("/mime-types.xml");
196         properties.setProperty(
197                 "org.apache.oodt.cas.filemgr.mime.type.repository", new File(
198                      mimeTypesUrl.getFile()).getAbsolutePath());
199 
200         System.setProperties(properties);
201 
202         try {
203             fm = new XmlRpcFileManager(FM_PORT);
204         } catch (Exception e) {
205             fail(e.getMessage());
206         }
207     }
