44  public class TestThreadPoolWorkflowEngine extends TestCase {
45  
46      public TestThreadPoolWorkflowEngine() {
47          // suppress WARNING level and below because we don't want
48          // the warning message where we test if start date time is AFTER
49          // end date time below
50          LogManager.getLogManager().getLogger("").setLevel(Level.SEVERE);
51      }
52  
53      public void testCurrentTaskWallClockTime() {
54          // at first, there is no start date time
55          WorkflowInstance inst = new WorkflowInstance();
56          WorkflowTask task = new WorkflowTask();
57          task.setTaskId("urn:oodt:testTask");
58          ParentChildWorkflow workflow = new ParentChildWorkflow(new Graph());
59          workflow.getTasks().add(task);
60          inst.setParentChildWorkflow(workflow);
61          inst.setCurrentTaskId("urn:oodt:testTask");
62          assertEquals(0.0, ThreadPoolWorkflowEngine
63              .getCurrentTaskWallClockMinutes(inst));
64  
65          // now set start date time, and assert that wall clock minutes > 0
66          inst.setCurrentTaskStartDateTimeIsoStr(DateConvert
67                  .isoFormat(new Date()));
68          System.out.println(ThreadPoolWorkflowEngine.getCurrentTaskWallClockMinutes(inst));
69          assertTrue(ThreadPoolWorkflowEngine
70                  .getCurrentTaskWallClockMinutes(inst) > 0.0);
71  
72          // set end date time to "" and make sure wall clock mins still greater
73          // than 0
74          inst.setCurrentTaskEndDateTimeIsoStr("");
75          assertTrue(ThreadPoolWorkflowEngine
76                  .getCurrentTaskWallClockMinutes(inst) > 0.0);
77  
78          // set the end date time, compute it, and make sure it stays the same
79          String endDateTimeIsoStr = DateConvert.isoFormat(new Date());
80          inst.setCurrentTaskEndDateTimeIsoStr(endDateTimeIsoStr);
81          double wallClockMins = ThreadPoolWorkflowEngine
82                  .getCurrentTaskWallClockMinutes(inst);
83          assertEquals(wallClockMins, ThreadPoolWorkflowEngine
84              .getCurrentTaskWallClockMinutes(inst));
85          assertEquals(wallClockMins, ThreadPoolWorkflowEngine
86              .getCurrentTaskWallClockMinutes(inst));
87  
88          // set the start date time after the end date time
89          // make sure that the wall cock time is 0.0
90          inst.setCurrentTaskStartDateTimeIsoStr(DateConvert
91                  .isoFormat(new Date()));
92          assertEquals(0.0, ThreadPoolWorkflowEngine
93              .getCurrentTaskWallClockMinutes(inst));
94  
95      }
