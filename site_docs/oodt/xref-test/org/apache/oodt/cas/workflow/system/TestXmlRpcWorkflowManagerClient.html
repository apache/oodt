485   @Test
486   public void testGetWorkflowInstanceMetadata() {
487 
488     try {
489       repo.addWorkflowInstance(testWrkInst);
490     } catch (InstanceRepositoryException e) {
491       fail(e.getMessage());
492     }
493     String testWrkInstId = testWrkInst.getId();
494     assertNotNull(testWrkInstId);
495 
496     // get workflow instance from instance id
497     WorkflowInstance WInst = null;
498     try {
499       WInst = repo.getWorkflowInstanceById(testWrkInstId);
500     } catch (InstanceRepositoryException e) {
501       fail(e.getMessage());
502     }
503 
504     assertNotNull(WInst);
505 
506     // get Metadata for the workflow instance
507     Metadata met;
508     met = WInst.getSharedContext();
509     assertNotNull(met);
510 
511     assertNotNull(met.getMap());
512     assertEquals(2, met.getMap().keySet().size());
513     assertNotNull(met.getAllMetadata("key1"));
514     assertEquals(3, met.getAllMetadata("key1").size());
515     assertNotNull(met.getAllMetadata("key2"));
516     assertEquals(2, met.getAllMetadata("key2").size());
517 
518     // check key-values for key1
519     boolean checkVal1 = false, checkVal2 = false, checkVal3 = false;
520 
521     for (String val : met.getAllMetadata("key1")) {
522       if (val.equals("val1")) {
523         checkVal1 = true;
524       } else if (val.equals("val2")) {
525         checkVal2 = true;
526       } else if (val.equals("val3")) {
527         checkVal3 = true;
528       }
529     }
530 
531     assert (checkVal1 && checkVal2 && checkVal3);
532 
533     // check key-values for key2
534     boolean checkVal4 = false, checkVal5 = false;
535 
536     for (String val : met.getAllMetadata("key2")) {
537       if (val.equals("val4")) {
538         checkVal4 = true;
539       } else if (val.equals("val5")) {
540         checkVal5 = true;
541       }
542     }
543 
544     assertTrue(checkVal4 && checkVal5);
545   }

548   private static void startXmlRpcWorkflowManager() {
549     System.setProperty("java.util.logging.config.file", new File(
550         "./src/main/resources/logging.properties").getAbsolutePath());
551 
552     try {
553       System.getProperties().load(
554           new FileInputStream("./src/main/resources/workflow.properties"));
555     } catch (Exception e) {
556       fail(e.getMessage());
557     }
558     try {
559       luceneCatLoc = Files.createTempDirectory("repo").toString();
560       LOG.log(Level.INFO, "Lucene instance repository: [" + luceneCatLoc + "]");
561     } catch (Exception e) {
562       fail(e.getMessage());
563     }
564 
565 
566     if (new File(luceneCatLoc).exists()) {
567       // blow away lucene cat
568       LOG.log(Level.INFO, "Removing workflow instance repository: ["
569                           + luceneCatLoc + "]");
570       try {
571         FileUtils.deleteDirectory(new File(luceneCatLoc));
572       } catch (IOException e) {
573         fail(e.getMessage());
574       }
575     }
576 
577     System
578         .setProperty("workflow.engine.instanceRep.factory",
579             "org.apache.oodt.cas.workflow.instrepo.LuceneWorkflowInstanceRepositoryFactory");
580     System
581         .setProperty("org.apache.oodt.cas.workflow.instanceRep.lucene.idxPath",
582             luceneCatLoc);
583 
584     try {
585       System.setProperty("org.apache.oodt.cas.workflow.repo.dirs", "file://"
586                                                                    + new File("./src/main/resources/examples")
587                                                                        .getCanonicalPath());
588       System.setProperty("org.apache.oodt.cas.workflow.lifecycle.filePath",
589           new File("./src/main/resources/examples/workflow-lifecycle.xml")
590               .getCanonicalPath());
591     } catch (Exception e) {
592       fail(e.getMessage());
593     }
594 
595     try {
596       wmgr = new XmlRpcWorkflowManager(WM_PORT);
597       Thread.sleep(MILLIS);
598     } catch (Exception e) {
599       LOG.log(Level.SEVERE, e.getMessage());
600       fail(e.getMessage());
601     }
602 
603   }

606   private static void stopXmlRpcWorkflowManager() {
607     System.setProperty("java.util.logging.config.file", new File(
608         "./src/main/resources/logging.properties").getAbsolutePath());
609 
610     try {
611       System.getProperties().load(
612           new FileInputStream("./src/main/resources/workflow.properties"));
613     } catch (Exception e) {
614       fail(e.getMessage());
615     }
616     System
617         .setProperty("workflow.engine.instanceRep.factory",
618             "org.apache.oodt.cas.workflow.instrepo.LuceneWorkflowInstanceRepositoryFactory");
619     System
620         .setProperty("org.apache.oodt.cas.workflow.instanceRep.lucene.idxPath",
621             luceneCatLoc);
622 
623     try {
624       System.setProperty("org.apache.oodt.cas.workflow.repo.dirs", "file://"
625                                                                    + new File("./src/main/resources/examples")
626                                                                        .getCanonicalPath());
627       System.setProperty("org.apache.oodt.cas.workflow.lifecycle.filePath",
628           new File("./src/main/resources/examples/workflow-lifecycle.xml")
629               .getCanonicalPath());
630     } catch (Exception e) {
631       fail(e.getMessage());
632     }
633 
634     try {
635       wmgr.shutdown();
636     } catch (Exception e) {
637       LOG.log(Level.SEVERE, e.getMessage());
638       fail(e.getMessage());
639     }
640 
641     /**
642      * Sleep before removing to prevent file not found issues.
643      */
644 
645     try {
646       Thread.sleep(MILLIS);
647     } catch (InterruptedException e) {
648       e.printStackTrace();
649     }
650 
651     if (new File(luceneCatLoc).exists()) {
652       // blow away lucene cat
653       LOG.log(Level.INFO, "Removing workflow instance repository: ["
654                           + luceneCatLoc + "]");
655       try {
656         FileUtils.deleteDirectory(new File(luceneCatLoc));
657       } catch (IOException e) {
658         fail(e.getMessage());
659       }
660     }
