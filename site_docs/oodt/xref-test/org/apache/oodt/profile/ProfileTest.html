31   * Unit test the Profile class.

35  public class ProfileTest extends TestCase {
36  	/** Construct the test case for the Profile class. */
37  	public ProfileTest(String name) {
38  		super(name);
39  	}
40  
41  	protected void setUp() throws Exception {
42  		super.setUp();
43  		oldProfNS = System.getProperty("jpl.rdf.ns");
44  		System.setProperty("jpl.rdf.ns", "http://enterprise.jpl.nasa.gov/rdfs/prof.rdf#");
45  
46  		StringBuilder buffer = new StringBuilder("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
47  		buffer.append("<!DOCTYPE profile PUBLIC \"").append(Profile.PROFILES_DTD_FPI).append("\" \"")
48  			.append(Profile.PROFILES_DTD_URL).append("\">\n");
49  		BufferedReader reader = new BufferedReader(new InputStreamReader(getClass().getResourceAsStream("test.xml")));
50  		String line;
51  		while ((line = reader.readLine()) != null) {
52  			buffer.append(line);
53  			buffer.append('\n');
54  		}
55  		reader.close();
56  		Document doc = XML.parse(buffer.toString());
57  		profile1 = new Profile(buffer.toString());
58  		profile2 = new Profile(doc.getDocumentElement());
59  	}
60  
61  	protected void tearDown() throws Exception {
62  		if (oldProfNS != null)
63  			System.setProperty("jpl.rdf.ns", oldProfNS);
64  		else
65  			System.getProperties().remove("jpl.rdf.ns");
66  	}
67  
68  	public void testQueries() {
69  		// We test both profile1 (built from a string) and profile2 (from an XML
70  		// document node); they should yield the same results because they
71  		// represent the same document.
72  
73  		// Test the getResourceAttribute method.
74  		assertEquals("PDS_PROFILE_SERVER", profile1.getResourceAttributes().getIdentifier());
75  		assertEquals("PDS_PROFILE_SERVER", profile2.getResourceAttributes().getIdentifier());
76  		assertEquals("text/html", profile1.getResourceAttributes().getFormats().get(0));
77  		assertEquals("text/html", profile2.getResourceAttributes().getFormats().get(0));
78  
79  		// Test the getProfileID method.
80  		assertEquals("OODT_PDS_PROFILE_SERVER", profile1.getProfileAttributes().getID());
81  		assertEquals("OODT_PDS_PROFILE_SERVER", profile2.getProfileAttributes().getID());
82  
83  		// Test the getProfileAttribute method
84  		assertEquals("profile", profile1.getProfileAttributes().getType());
85  		assertEquals("profile", profile2.getProfileAttributes().getType());
86  		assertEquals("NULL", profile1.getProfileAttributes().getSecurityType());
87  		assertEquals("NULL", profile2.getProfileAttributes().getSecurityType());
88  
89  		// Test the getProfileElementItem method.
90  		Map elements1 = profile1.getProfileElements();
91  		Map elements2 = profile2.getProfileElements();
92  		assertEquals(3, elements1.size());
93  		assertEquals(3, elements2.size());
94  		assertTrue(elements1.containsKey("TEST"));
95  		assertTrue(elements2.containsKey("TEST"));
96  		assertTrue(!elements1.containsKey("does-not-exist"));
97  		assertTrue(!elements2.containsKey("does-not-exist"));
98  		ProfileElement element1 = (ProfileElement) elements1.get("TEST2");
99  		ProfileElement element2 = (ProfileElement) elements2.get("TEST2");
100 		assertEquals("Testing", element1.getType());
101 		assertEquals("Testing", element2.getType());
102 
103 		// Test the toString and getProfileString methods ... NB: this test should
104 		// actually check value, not just see if they're equal.
105 		assertEquals(profile1.toString(), profile2.toString());
106 
107 		// Test some miscellaneous query methods
108 		assertEquals("Planetary Data System (PDS) - Profile Server V1.0", profile1.getResourceAttributes().getTitle());
109 		assertEquals("iiop://oodt.jpl.nasa.gov:10000/JPL.PDS.PROFILE", profile1.getResourceAttributes().getResLocations()
110 			.get(0));
111 		assertEquals("system.profileServer", profile1.getResourceAttributes().getResClass());
112 		assertEquals("Planetary Data System (PDS) - Profile Server V1.0", profile2.getResourceAttributes().getTitle());
113 		assertEquals("iiop://oodt.jpl.nasa.gov:10000/JPL.PDS.PROFILE", profile2.getResourceAttributes().getResLocations()
114 			.get(0));
115 		assertEquals("system.profileServer", profile2.getResourceAttributes().getResClass());
116 	}
