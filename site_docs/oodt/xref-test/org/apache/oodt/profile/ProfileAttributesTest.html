112 		Document doc = XML.createDocument();
113 		Node root = TEST_PROFILE_ATTRIBUTES.toXML(doc);
114 		assertEquals("profAttributes", root.getNodeName());
115 		NodeList children = root.getChildNodes();
116 		for (int i = 0; i < children.getLength(); ++i) {
117 			Node child = children.item(i);
118 			String name = child.getNodeName();
119 			if ("profId".equals(name)) {
120 				assertEquals("id", XML.text(child));
121 			} else if ("profVersion".equals(name)) {
122 				assertEquals("version", XML.text(child));
123 			} else if ("profType".equals(name)) {
124 				assertEquals("type", XML.text(child));
125 			} else if ("profStatusId".equals(name)) {
126 				assertEquals("statusID", XML.text(child));
127 			} else if ("profSecurityType".equals(name)) {
128 				assertEquals("securityType", XML.text(child));
129 			} else if ("profParentId".equals(name)) {
130 				assertEquals("parent", XML.text(child));
131 			} else if ("profChildId".equals(name)) {
132 			  // ignore, list serialization tested in XMLTest
133 			} else if ("profRegAuthority".equals(name)) {
134 				assertEquals("regAuthority", XML.text(child));
135 			} else if ("profRevisionNote".equals(name)) {
136 			  // ignore, list serialization tested in XMLTest
137 			} else fail("Unknown node \"" + name + "\" in XML result");
138 		}
139 		ProfileAttributes p = new ProfileAttributes(root);
140 		assertEquals(TEST_PROFILE_ATTRIBUTES, p);
141 	}
