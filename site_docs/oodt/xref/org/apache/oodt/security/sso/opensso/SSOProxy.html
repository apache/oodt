86    public String authenticate(String username, String password) {
87      HttpClient httpClient = new HttpClient();
88      PostMethod post = new PostMethod(AUTH_ENDPOINT);
89      String response;
90      String ssoToken = null;
91  
92      NameValuePair[] data = { new NameValuePair("username", username),
93          new NameValuePair("password", password),
94          new NameValuePair("uri", "realm/lmmp") };
95  
96      post.setRequestBody(data);
97  
98      try {
99        httpClient.executeMethod(post);
100       if (post.getStatusCode() != HttpStatus.SC_OK) {
101         throw new HttpException(post.getStatusLine().toString());
102       }
103       response = post.getResponseBodyAsString().trim();
104       ssoToken = response.substring(9);
105     } catch (Exception e) {
106       LOG.log(Level.SEVERE, e.getMessage());
107     } finally {
108       post.releaseConnection();
109     }
110 
111     return ssoToken;
112   }

114   public IdentityDetails readIdentity(String username, String token)
115       throws IOException, SingleSignOnException {
116     HttpClient httpClient = new HttpClient();
117     PostMethod post = new PostMethod(IDENT_READ_ENDPOINT);
118     LOG.log(Level.INFO, "Obtaining identity: username: [" + username
119         + "]: token: [" + token + "]: REST url: [" + IDENT_READ_ENDPOINT
120         + "]");
121     NameValuePair[] data = { new NameValuePair("name", username),
122         new NameValuePair("admin", token) };
123 
124     post.setRequestBody(data);
125 
126     httpClient.executeMethod(post);
127     if (post.getStatusCode() != HttpStatus.SC_OK) {
128       throw new SingleSignOnException(post.getStatusLine().toString());
129     }
130 
131     return parseIdentityDetails(post.getResponseBodyAsString().trim());
132 
133   }

177   private IdentityDetails parseIdentityDetails(String serviceResponse) {
178     ByteArrayInputStream is = new ByteArrayInputStream(serviceResponse
179         .getBytes());
180     BufferedReader br = new BufferedReader(new InputStreamReader(is));
181     IdentityDetails details = new IdentityDetails();
182     String line = null, lastAttrKeyRead = null;
183 
184     try {
185       while ((line = br.readLine()) != null) {
186         if (line.equals(IDENTITY_DETAILS_ATTR_SKIP_LINE)) {
187           continue;
188         }
189         String key, val;
190         if (line.startsWith(IDENTITY_DETAILS_REALM)) {
191           // can't parse it the same way
192           key = line.substring(0, IDENTITY_DETAILS_REALM.length());
193           val = line.substring(IDENTITY_DETAILS_REALM.length() + 1);
194         } else {
195           String[] lineToks = line.split("=");
196           key = lineToks[0];
197           val = lineToks[1];
198         }
199 
200         if (key.equals(IDENTITY_DETAILS_NAME)) {
201           details.setName(val);
202         } else if (key.equals(IDENTITY_DETAILS_TYPE)) {
203           details.setType(val);
204         } else if (key.equals(IDENTITY_DETAILS_REALM)) {
205           details.setRealm(val);
206         } else if (key.equals(IDENTITY_DETAILS_GROUP)) {
207           details.getGroups().add(val);
208         } else if (key.equals(IDENTITY_DETAILS_ATTR_NAME)) {
209           lastAttrKeyRead = val;
210         } else if (key.equals(IDENTITY_DETAILS_ATTR_VALUE)) {
211           details.getAttributes().addMetadata(lastAttrKeyRead, val);
212         }
213       }
214     } catch (IOException e) {
215       LOG.log(Level.SEVERE, e.getMessage());
216       LOG.log(Level.WARNING, "Error reading service response line: [" + line
217           + "]: Message: " + e.getMessage());
218     } finally {
219       try {
220         is.close();
221       } catch (Exception ignore) {
222       }
223 
224       try {
225         br.close();
226       } catch (Exception ignore) {
227       }
228 
229     }
230 
231     return details;
232   }

234   private UserDetails parseUserDetails(String serviceResponse) {
235     ByteArrayInputStream is = new ByteArrayInputStream(serviceResponse
236         .getBytes());
237     BufferedReader br = new BufferedReader(new InputStreamReader(is));
238     UserDetails details = new UserDetails();
239     String line = null, lastAttrKeyRead = null;
240 
241     try {
242       while ((line = br.readLine()) != null) {
243         String key, val;
244         if (line.startsWith(USER_DETAILS_ROLE)) {
245           // can't parse by splitting, parse by using substring
246           key = line.substring(0, USER_DETAILS_ROLE.length());
247           val = line.substring(USER_DETAILS_ROLE.length() + 1);
248         } else {
249           String[] lineToks = line.split("=");
250           key = lineToks[0];
251           val = lineToks[1];
252         }
253 
254         if (key.equals(USER_DETAILS_TOKEN)) {
255           details.setToken(val);
256         } else if (key.equals(USER_DETAILS_ROLE)) {
257           details.getRoles().add(val);
258         } else if (key.equals(USER_DETAILS_ATTR_NAME)) {
259           lastAttrKeyRead = val;
260         } else if (key.equals(USER_DETAILS_ATTR_VALUE)) {
261           details.getAttributes().addMetadata(lastAttrKeyRead, val);
262         }
263       }
264     } catch (IOException e) {
265       LOG.log(Level.SEVERE, e.getMessage());
266       LOG.log(Level.WARNING, "Error reading service response line: [" + line
267           + "]: Message: " + e.getMessage());
268     } finally {
269       try {
270         is.close();
271       } catch (Exception ignore) {
272       }
273 
274       try {
275         br.close();
276       } catch (Exception ignore) {
277       }
278 
279     }
280 
281     return details;
282   }

284   public static void main(String[] args) throws IOException, SingleSignOnException {
285     String usage = "SSOProxy <cmd> [args]\n\n" + "Where cmd is one of:\n"
286         + "authenticate <user> <pass>\n" + "identity <user> <token>\n"
287         + "attributes <token>\nlogout <token>\n";
288 
289     if (args.length < 2 || args.length > 3) {
290       System.err.println(usage);
291       System.exit(1);
292     }
293 
294     String cmd = args[0];
295     SSOProxy sso = new SSOProxy();
296     if (cmd.equals(AUTH_COMMAND)) {
297       System.out.println(sso.authenticate(args[1], args[2]));
298     } else if (cmd.equals(IDENTITY_COMMAND)) {
299       System.out.println(sso.readIdentity(args[1], args[2]));
300     } else if (cmd.equals(ATTRIBUTES_COMMAND)) {
301       System.out.println(sso.getUserAttributes(args[1]));
302     } else if (cmd.equals(LOGOUT_COMMAND)) {
303       sso.logout(args[1]);
304     }
305 
306   }
