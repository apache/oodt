26  /**
27   * Resource attributes that can be searched.
28   *
29   * @author Kelly
30   * @version $Revision: 1.1.1.1 $
31   */
32  public class SearchableResourceAttributes extends ResourceAttributes {
33  	public SearchableResourceAttributes(SearchableProfile profile, Element node) {
34  		super(profile, node);
35  	}

37  	/**
38  	 * Produce a search result.
39  	 *
40  	 * @param value Desired value.
41  	 * @param operator What operator to use for comparison.
42  	 * @return a <code>Result</code> value.
43  	 */
44  	public Result result(String name, String value, String operator) {
45  		if ("Identifier".equals(name)) {
46  		  return computeResult(identifier, value, operator);
47  		} else if ("Title".equals(name)) {
48  		  return computeResult(title, value, operator);
49  		} else if ("Format".equals(name)) {
50  		  return computeResult(formats, value, operator);
51  		} else if ("Description".equals(name)) {
52  		  return computeResult(description, value, operator);
53  		} else if ("Creator".equals(name)) {
54  		  return computeResult(creators, value, operator);
55  		} else if ("Subject".equals(name)) {
56  		  return computeResult(subjects, value, operator);
57  		} else if ("Publisher".equals(name)) {
58  		  return computeResult(publishers, value, operator);
59  		} else if ("Contributor".equals(name)) {
60  		  return computeResult(contributors, value, operator);
61  		} else if ("Date".equals(name)) {
62  		  return computeResult(dates, value, operator);
63  		} else if ("Type".equals(name)) {
64  		  return computeResult(types, value, operator);
65  		} else if ("Source".equals(name)) {
66  		  return computeResult(sources, value, operator);
67  		} else if ("Language".equals(name)) {
68  		  return computeResult(languages, value, operator);
69  		} else if ("Relation".equals(name)) {
70  		  return computeResult(relations, value, operator);
71  		} else if ("Coverage".equals(name)) {
72  		  return computeResult(coverages, value, operator);
73  		} else if ("Rights".equals(name)) {
74  		  return computeResult(rights, value, operator);
75  		} else if ("resContext".equals(name)) {
76  		  return computeResult(contexts, value, operator);
77  		} else if ("resClass".equals(name)) {
78  		  return computeResult(clazz, value, operator);
79  		} else if ("resLocation".equals(name)) {
80  		  return computeResult(locations, value, operator);
81  		} else {
82  		  throw new IllegalArgumentException("Unknown attribute \"" + name + "\"");
83  		}
84  	}
85  
86  	private Result computeResult(String a, String b, String op) {
87  		int c = a.compareTo(b);
88  		boolean t;
89  		if ("EQ".equals(op) || "LIKE".equals(op)) {
90  		  t = c == 0;
91  		} else if ("GE".equals(op)) {
92  		  t = c >= 0;
93  		} else if ("GT".equals(op)) {
94  		  t = c > 0;
95  		} else if ("LE".equals(op)) {
96  		  t = c <= 0;
97  		} else if ("LT".equals(op)) {
98  		  t = c < 0;
99  		} else if ("NE".equals(op) || "NOTLIKE".equals(op)) {
100 		  t = c != 0;
101 		} else {
102 		  throw new IllegalArgumentException("Unknown relational operator \"" + op + "\"");
103 		}
104 		if (t) {
105 		  return new MatchingResult(new HashSet(profile.getProfileElements().values()));
106 		} else {
107 		  return FalseResult.INSTANCE;
108 		}
109 	}
110 
111 	private Result computeResult(List a, String b, String op) {		
112 		if (a == null || a.isEmpty()) {
113 		  return FalseResult.INSTANCE;
114 		}
115 
116 		Result f = FalseResult.INSTANCE;
117 		Result t = new MatchingResult(new HashSet(profile.getProfileElements().values()));
118 		Result rc = f;
119 		if ("EQ".equals(op) || "LIKE".equals(op)) {
120 		  if (a.contains(b)) {
121 			rc = t;
122 		  } else if ("NE".equals(op) || "NOTLIKE".equals(op)) {
123 			if (!a.contains(b)) {
124 			  rc = t;
125 			} else if ("LT".equals(op) || "GT".equals(op) || "LE".equals(op) || "GE".equals(op)) {
126 			  for (Object anA : a) {
127 				String value = (String) anA;
128 				rc = computeResult(value, b, op);
129 				if (rc != f) {
130 				  break;
131 				}
132 			  }
133 			} else {
134 			  throw new IllegalArgumentException("Unknown relational operator \"" + op + "\"");
135 			}
136 		  }
137 		}
138 		return rc;
139 	}
