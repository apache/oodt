63  	/**
64  	 * Create a list of profiles by deserializing data from the given XML element.
65  	 *
66  	 * @param root Either a &lt;profiles&gt; or a &lt;profile&gt; element.
67  	 * @param factory Factory for creation of profile-related objects.
68  	 * @return A list of profiles.
69  	 */
70  	public static List<Profile> createProfiles(Element root, ObjectFactory factory) {
71  		List<Profile> profiles = new ArrayList<Profile>();
72  		if ("profile".equals(root.getNodeName()))
73  			// The root is a <profile>, so add the single profile to the list.
74  		{
75  		  profiles.add(factory.createProfile((Element) root));
76  		} else if ("profiles".equals(root.getNodeName())) {
77  			// The root is a <profiles>, so add each <profile> to the list.
78  			NodeList children = root.getChildNodes();
79  			for (int i = 0; i < children.getLength(); ++i) {
80  				Node node = children.item(i);
81  				if ("profile".equals(node.getNodeName())) {
82  				  profiles.add(factory.createProfile((Element) node));
83  				}
84  			}
85  		} else {
86  		  throw new IllegalArgumentException("Expected a <profiles> or <profile> top level element but got "
87  											 + root.getNodeName());
88  		}
89  		return profiles;
90  	}

145 	/**
146 	 * Create a profile from an XML document.
147 	 *
148 	 * @param root The &lt;profile&gt; element.
149 	 */
150 	public Profile(Node root, ObjectFactory factory) {
151 		if (!root.getNodeName().equals("profile")) {
152 		  throw new IllegalArgumentException("Construct a Profile from a <profile> element, not a <"
153 											 + root.getNodeName() + ">");
154 		}
155 		NodeList children = root.getChildNodes();
156 		for (int i = 0; i < children.getLength(); ++i) {
157 			Node node = children.item(i);
158 			if ("profAttributes".equals(node.getNodeName())) {
159 			  profAttr = factory.createProfileAttributes((Element) node);
160 			} else if ("resAttributes".equals(node.getNodeName())) {
161 			  resAttr = factory.createResourceAttributes(this, (Element) node);
162 			} else if ("profElement".equals(node.getNodeName())) {
163 				ProfileElement element = ProfileElement.createProfileElement((Element) node, this, factory);
164 				elements.put(element.getName(), element);
165 			}
166 		}
167 	}

309 	/**
310 	 * Serialize this profile as an XML node.
311 	 *
312 	 * @param doc The document that will own this node.
313 	 * @param withElements Include the profile elements?
314 	 * @return The XML element &lt;profile&gt; representing these attributes.
315 	 * @throws DOMException If an error occurs creating the XML nodes.
316 	 */
317 	private Node toXML(Document doc, boolean withElements) throws DOMException {
318 		Element profile = doc.createElement("profile");
319 		profile.appendChild(profAttr.toXML(doc));
320 		profile.appendChild(resAttr.toXML(doc));
321 		if (withElements) {
322 		  for (ProfileElement profileElement : elements.values()) {
323 			profile.appendChild((profileElement).toXML(doc));
324 		  }
325 		}
326 		return profile;
327 	}

370 	/**
371 	 * Try to parse an XML profile in a file in its XML vocabulary.  If successful,
372 	 * you get the profile as RDF document (in XML format) to the standard output
373 	 * after it's been digested by the profile class.  If not, then you get an
374 	 * exception.
375 	 *
376 	 * @param argv Command-line arguments, of which there should be one, the name of XML file containing the profile to parse
377 	 * @throws Throwable if an error occurs.
378 	 */
379 	public static void main(String[] argv) throws Throwable {
380 		if (argv.length != 1) {
381 			System.err.println("Usage: <profile.xml>");
382 			System.exit(1);
383 		}
384 		StringBuilder b = new StringBuilder();
385 		BufferedReader reader = new BufferedReader(new FileReader(argv[0]));
386 		char[] buf = new char[INT];
387 		int num;
388 		while ((num = reader.read(buf)) != -1) {
389 		  b.append(buf, 0, num);
390 		}
391 		reader.close();
392 		Profile p = new Profile(b.toString());
393 
394 		Model model = ModelFactory.createDefaultModel();
395 		p.addToModel(model);
396 		OutputStreamWriter writer = new OutputStreamWriter(System.out);
397 		model.write(writer);
398 		writer.close();
399 		System.exit(0);
400 	}
