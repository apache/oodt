66  	/**
67  	 * Create resource attributes from an XML document.
68  	 *
69  	 * @param profile Owning profile.
70  	 * @param root The &lt;resAttributes&gt; element.
71  	 */
72  	public ResourceAttributes(Profile profile, Node root) {
73  		this.profile = profile;
74  		initializeLists();
75  		NodeList children = root.getChildNodes();
76  		for (int i = 0; i < children.getLength(); ++i) {
77  			Node node = children.item(i);
78  			if ("Identifier".equals(node.getNodeName())) {
79  			  identifier = XML.unwrappedText(node);
80  			} else if ("Title".equals(node.getNodeName())) {
81  			  title = XML.unwrappedText(node);
82  			} else if ("Format".equals(node.getNodeName())) {
83  			  formats.add(XML.unwrappedText(node));
84  			} else if ("Description".equals(node.getNodeName())) {
85  			  description = XML.unwrappedText(node);
86  			} else if ("Creator".equals(node.getNodeName())) {
87  			  creators.add(XML.unwrappedText(node));
88  			} else if ("Subject".equals(node.getNodeName())) {
89  			  subjects.add(XML.unwrappedText(node));
90  			} else if ("Publisher".equals(node.getNodeName())) {
91  			  publishers.add(XML.unwrappedText(node));
92  			} else if ("Contributor".equals(node.getNodeName())) {
93  			  contributors.add(XML.unwrappedText(node));
94  			} else if ("Date".equals(node.getNodeName())) {
95  			  dates.add(XML.unwrappedText(node));
96  			} else if ("Type".equals(node.getNodeName())) {
97  			  types.add(XML.unwrappedText(node));
98  			} else if ("Source".equals(node.getNodeName())) {
99  			  sources.add(XML.unwrappedText(node));
100 			} else if ("Language".equals(node.getNodeName())) {
101 			  languages.add(XML.unwrappedText(node));
102 			} else if ("Relation".equals(node.getNodeName())) {
103 			  relations.add(XML.unwrappedText(node));
104 			} else if ("Coverage".equals(node.getNodeName())) {
105 			  coverages.add(XML.unwrappedText(node));
106 			} else if ("Rights".equals(node.getNodeName())) {
107 			  rights.add(XML.unwrappedText(node));
108 			} else if ("resContext".equals(node.getNodeName())) {
109 			  contexts.add(XML.unwrappedText(node));
110 			} else if ("resAggregation".equals(node.getNodeName())) {
111 			  aggregation = XML.unwrappedText(node);
112 			} else if ("resClass".equals(node.getNodeName())) {
113 			  clazz = XML.unwrappedText(node);
114 			} else if ("resLocation".equals(node.getNodeName())) {
115 			  locations.add(XML.unwrappedText(node));
116 			}
117 		}
118 	}

235 	public URI getURI() {
236 		String identification;
237 		if (identifier == null || identifier.length() == 0) {
238 			if (locations.isEmpty()) {
239 			  identification = null;
240 			} else {
241 			  identification = (String) locations.get(0);
242 			}
243 		} else {
244 		  identification = identifier;
245 		}
246 
247 		return identification == null? UNKNOWN_URI : URI.create(identification);
248 	}

549 	/**
550 	 * Serialize this attributes as an XML node.
551 	 *
552 	 * @param doc The document that will own this node.
553 	 * @return The XML element &lt;resAttributes&gt; representing these attributes.
554 	 * @throws DOMException If an error occurs creating the XML nodes.
555 	 */
556 	public Node toXML(Document doc) throws DOMException {
557 		Element root = doc.createElement("resAttributes");
558 		XML.add(root, "Identifier", identifier);
559 		XML.addNonNull(root, "Title", title);
560 		XML.add(root, "Format", formats);
561 		XML.addNonNull(root, "Description", description);
562 		XML.add(root, "Creator", creators);
563 		XML.add(root, "Subject", subjects);
564 		XML.add(root, "Publisher", publishers);
565 		XML.add(root, "Contributor", contributors);
566 		XML.add(root, "Date", dates);
567 		XML.add(root, "Type", types);
568 		XML.add(root, "Source", sources);
569 		XML.add(root, "Language", languages);
570 		XML.add(root, "Relation", relations);
571 		XML.add(root, "Coverage", coverages);
572 		XML.add(root, "Rights", rights);
573 		List<String> contexts = new ArrayList<String>(this.contexts);
574 		if (contexts.isEmpty()) {
575 		  contexts.add("UNKNOWN");
576 		}
577 		XML.add(root, "resContext", contexts);
578 		XML.addNonNull(root, "resAggregation", aggregation);
579 		if(clazz==null) {
580 		  clazz = "UNKNOWN";
581 		}
582 		XML.addNonNull(root, "resClass", clazz);
583 		XML.add(root, "resLocation", locations);
584 
585 		return root;
586 	}
