48  	 * Create a profile element from the given XML node.
49  	 *
50  	 * @param root The &lt;profElement&gt; node.
51  	 * @param profile To what profile the element belongs.
52  	 * @return A profile element.
53  	 */
54  	public static ProfileElement createProfileElement(Element root, Profile profile, ObjectFactory factory) {
55  		String name = null;
56  		String id = null;
57  		String desc = null;
58  		String type = null;
59  		String unit = null;
60  		List<String> synonyms = new ArrayList<String>();
61  		boolean obligation = false;
62  		int maxOccurrence = 0;
63  		String comments = null;
64  		boolean ranged = false;
65  		NodeList children = root.getChildNodes();
66  		String min = "0.0", max = "0.0";
67  		boolean gotMin = false;
68  		boolean gotMax = false;
69  		List<String> values = new ArrayList<String>();
70  		for (int i = 0; i < children.getLength(); ++i) {
71  			Node node = children.item(i);
72  			if ("elemId".equals(node.getNodeName())) {
73  			  id = XML.unwrappedText(node);
74  			} else if ("elemName".equals(node.getNodeName())) {
75  			  name = XML.unwrappedText(node);
76  			} else if ("elemDesc".equals(node.getNodeName())) {
77  			  desc = XML.unwrappedText(node);
78  			} else if ("elemType".equals(node.getNodeName())) {
79  			  type = XML.unwrappedText(node);
80  			} else if ("elemUnit".equals(node.getNodeName())) {
81  			  unit = XML.unwrappedText(node);
82  			} else if ("elemEnumFlag".equals(node.getNodeName())) {
83  			  ranged = "F".equals(XML.unwrappedText(node));
84  			} else if ("elemSynonym".equals(node.getNodeName())) {
85  			  synonyms.add(XML.unwrappedText(node));
86  			} else if ("elemObligation".equals(node.getNodeName())) {
87  				String value = XML.unwrappedText(node);
88  				obligation = "Required".equals(value) || "T".equals(value);
89  			} else if ("elemMaxOccurrence".equals(node.getNodeName())) {
90  			  try {
91  				maxOccurrence = Integer.parseInt(XML.unwrappedText(node));
92  			  } catch (NumberFormatException ignore) {
93  			  }
94  			} else if ("elemComment".equals(node.getNodeName())) {
95  			  comments = XML.unwrappedText(node);
96  			} else if ("elemValue".equals(node.getNodeName())) {
97  				values.add(text(node));
98  			} else if ("elemMinValue".equals(node.getNodeName())) {
99  			  try {
100 				min = XML.unwrappedText(node);
101 				gotMin = true;
102 			  } catch (NumberFormatException ignore) {
103 			  }
104 			} else if ("elemMaxValue".equals(node.getNodeName())) {
105 			  try {
106 				max = XML.unwrappedText(node);
107 				gotMax = true;
108 			  } catch (NumberFormatException ignore) {
109 			  }
110 			}
111 		}
112 		if (ranged) {
113 			if (gotMin && gotMax) {
114 				return factory.createRangedProfileElement(profile, name, id, desc, type, unit, synonyms, obligation,
115 					maxOccurrence, comments, min, max);
116 			}
117 			return factory.createUnspecifiedProfileElement(profile, name, id, desc, type, unit, synonyms, obligation,
118 				maxOccurrence, comments);
119 		} else {
120 		  return factory.createEnumeratedProfileElement(profile, name, id, desc, type, unit, synonyms, obligation,
121 			  maxOccurrence, comments, values);
122 		}
123 	}

575 	static void text0(StringBuffer b, Node node) {
576 		if (node.getNodeType() == Node.TEXT_NODE || node.getNodeType() == Node.CDATA_SECTION_NODE) {
577 			b.append(node.getNodeValue());
578 			return;
579 		}
580 		NodeList children = node.getChildNodes();
581 		for (int i = 0; i < children.getLength(); ++i) {
582 		  text0(b, children.item(i));
583 		}
584 	}
