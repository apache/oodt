109 	private static DocumentBuilder getStandardDocumentBuilder() {
110 		try {
111 			DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
112 			factory.setCoalescing(false);
113 			factory.setExpandEntityReferences(false);
114 			factory.setIgnoringComments(false);
115 			factory.setIgnoringElementContentWhitespace(true);
116 			factory.setNamespaceAware(true);
117 			factory.setValidating(true);
118 			return factory.newDocumentBuilder();
119 		} catch (ParserConfigurationException ex) {
120 			throw new IllegalStateException("Unexpected ParserConfigurationException: " + ex.getMessage());
121 		}
122 	}

143 	 * ignorable whitespace not included, and has a default error handler that prints
144 	 * error messages and warnings to the standard error stream.

164 		return new DOMParser(builder);

227 	/** Serialize an XML DOM document into a writer.
228 	 *
229 	 * This method takes a DOM document and pretty-prints (or pretty-serializes, in
230 	 * XML parlance) it into a writer.
231 	 *
232 	 * @param doc The document.
233 	 * @param writer Where to write it.
234 	 * @param omitXMLDeclaration True if we should omit the XML declaration, false to keep the XML declaration.
235 	 * @throws IOException If an I/O error occurs.
236 	 */
237 	public static void serialize(Document doc, Writer writer, boolean omitXMLDeclaration) {
238 		try {
239 			TransformerFactory factory = TransformerFactory.newInstance();
240 			Transformer transformer = factory.newTransformer();
241 			
242 			//update 05/01/2005
243 			//author: Chris Mattmann
244 			//set properties to pretty print
245 			//and also to include the DOCTYPE when serializing the XML
246 			
247 			//only set this if it's not null else we get a nice NullPointerException
248 			if(doc.getDoctype().getPublicId() != null){
249 				transformer.setOutputProperty(OutputKeys.DOCTYPE_PUBLIC, doc.getDoctype().getPublicId());			    
250 			}
251 
252 			//only set this if it's not null else we get a nice NullPointerException
253 			if(doc.getDoctype().getSystemId() != null){
254 				transformer.setOutputProperty(OutputKeys.DOCTYPE_SYSTEM, doc.getDoctype().getSystemId());			    
255 			}
256 
257 			transformer.setOutputProperty(OutputKeys.INDENT,"yes");
258 			
259 			DOMSource source = new DOMSource(doc);
260 			StreamResult result = new StreamResult(writer);
261 			transformer.transform(source, result);
262 		} catch (TransformerConfigurationException ex) {
263 			throw new IllegalStateException("Unexpected TransformerConfigurationException: " + ex.getMessage());
264 		} catch (TransformerException ex) {
265 			throw new IllegalStateException("Unexpected TransformerException: " + ex.getMessage());
266 		}
267 	}

389 	 *
390 	 * This method modifies your DOM tree so that
391 	 * <pre>&lt;node&gt;
392 	 *   ...
393 	 * &lt;/node&gt;</pre>
394 	 * becomes
395 	 * <pre>&lt;node&gt;
396 	 *   ...
397 	 *   &lt;name&gt;text&gt;/name&gt;
398 	 * &lt;/node&gt;</pre>
399 	 *
400 	 * Adding a null <var>name</var> does nothing.  Adding null <var>text</var>
401 	 * results in an empty <var>name</var> tag.
402 	 *
403 	 * @param node Node to which to add a child element.
404 	 * @param name Name of the child element to add to <var>node</var>.
405 	 * @param text What text the text-node child the child element named <var>name</var> should have.
406 	 * @throws DOMException If a DOM error occurs.
407 	 */
408 	public static void add(Node node, String name, String text) throws DOMException {
409 		if (name == null) {
410 		  return;
411 		}
412 		if (node == null) {
413 		  throw new IllegalArgumentException("Can't add to a null node");
414 		}
415 		Document doc = node.getOwnerDocument();
416 		Element element = doc.createElement(name);
417 		if (text != null) {
418 		  element.appendChild(doc.createTextNode(text));
419 		}
420 		node.appendChild(element);
421 	}

449 	/** Get unwrapped text from the given DOM node
450 	 *
451 	 * This method unwraps any wrapped text.  For example, if the document contains
452 	 * <pre>&lt;node&gt;Hello, world.  This is
453 	 *    my first document.
454          * &lt;/node&gt;</pre>
455 	 * then the node's unwrapped text is
456 	 * <pre>Hello, world.  This is my first document.</pre>
457 	 * while the {@link #text} method would return the wrapped value
458 	 * <pre>Hello, world.  This is
459 	 *    my first document.</pre>
460 	 *
461 	 * <p>In other words, it collects the text nodes under the given node and replaces
462 	 * strings of newlines and spaces with a single space.  Unwrapping a null node
463 	 * returns a null string.
464 	 *
465 	 * @param node The node.
466 	 * @return The text in its children, unwrapped.
467 	 */
468 	public static String unwrappedText(Node node) {
469 		if (node == null) {
470 		  return null;
471 		}
472 		StringBuffer buffer = new StringBuffer();
473 		StringBuilder wrapped = new StringBuilder(text1(node, buffer));
474 		boolean newline = false;
475 		for (int i = 0; i < wrapped.length(); ++i) {
476 			if (!newline) {
477 				if (wrapped.charAt(i) == '\n') {
478 					newline = true;
479 					wrapped.setCharAt(i, ' ');
480 				}
481 			} else {
482 				if (Character.isWhitespace(wrapped.charAt(i))) {
483 					wrapped.deleteCharAt(i);
484 					--i;
485 				} else {
486 				  newline = false;
487 				}
488 			}
489 		}
490 		return wrapped.toString().trim();
491 	}

541 	/** Identifies the last printable character in the Unicode range that is supported
542 	 * by the encoding used with this serializer.  For 8-bit encodings this will be either
543 	 * 0x7E or 0xFF.  For 16-bit encodings this will be 0xFFFF. Characters that are not
544 	 * printable will be escaped using character references.
545 	 *
546 	 * <p>Taken from Xerces 1.0.3.  Apache license applies; see source code for
547 	 * license.
548 	 *
549 	 */
550 	private static int LAST_PRINTABLE = 0x7E;
551 
552 	/** Escapes a string so it may be printed as text content or attribute value. Non
553 	 * printable characters are escaped using character references.  Where the format
554 	 * specifies a deault entity reference, that reference is used
555 	 * (e.g. <code>&amp;lt;</code>).
556 	 *
557 	 * <p>Taken from Xerces 1.0.3.  Apache license applies; see source code for
558 	 * license.
559 	 *
560 	 * @param source The string to escape
561 	 * @return The escaped string
562 	 */
563 	public static String escape(String source) {
564 		StringBuffer    result;
565 		int             i;
566 		char            ch;
567 		String          charRef;
568         
569 		result = new StringBuffer(source.length());
570 		for (i = 0; i < source.length(); ++i) {
571 			ch = source.charAt(i);
572 			// If the character is not printable, print as character
573 			// reference.  Non printables are below ASCII space but not tab or
574 			// line terminator, ASCII delete, or above a certain Unicode
575 			// threshold.
576 			if ((ch < ' ' && ch != '\t' && ch != '\n' && ch != '\r') || ch > LAST_PRINTABLE || ch == 0xF7) {
577 			  result.append("&#").append(Integer.toString(ch)).append(';');
578 			} else {
579 				// If there is a suitable entity reference for this
580 				// character, print it. The list of available entity
581 				// references is almost but not identical between XML and
582 				// HTML.
583 				charRef = getEntityRef(ch);
584 				if (charRef == null) {
585 				  result.append(ch);
586 				} else {
587 				  result.append('&').append(charRef).append(';');
588 				}
589 			}
590 		}
591 		return result.toString();
592 	}

595 	 *
596 	 * @param list List to add to.
597 	 * @param node Node to search.
598 	 */
599 	private static void findCommentNodes(List list, Node node) {
600 		if (node.getNodeType() == Node.COMMENT_NODE) {
601 		  list.add(node);
602 		} else {
603 			NodeList children = node.getChildNodes();
604 			for (int i = 0; i < children.getLength(); ++i) {
605 				findCommentNodes(list, children.item(i));
606 			}
607 		}

610 	/** Get the entity reference for the given character.
611 	 *
612 	 * Taken from Xerces 1.0.3; see source code for license.
613 	 */
614 	private static String getEntityRef(char ch) {
615 		// Encode special XML characters into the equivalent character references.
