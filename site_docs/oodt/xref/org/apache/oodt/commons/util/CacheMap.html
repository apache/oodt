18  import java.util.Collection;
19  import java.util.LinkedList;

24  // Definitions for intended functions:
25  //
26  // DEFN size(x) = The size of an object x, usually by the size(), length(), or getLength()
27  //                methods defined for x.
28  // NFED
29  //
30  // DEFN keys(x) = Set of keys from map x.
31  // NFED
32  //
33  // DEFN dup(x) = Duplicate of object x.
34  // NFED
35  //
36  // DEFN presence(x, y) = True if (x, z) or (z, x) exists in map y for any z.
37  // NFED
38  //
39  // DEFN advance(x, y) = sequence y such that object x in y is moved to the front of the
40  //                      sequence, or y if x is not an element of x.
41  // NFED
42  //
43  // DEFN value(v, y) = value v of (k, v) mapping in y.
44  // NFED
45  //
46  // A cache-map Z is a triple (c, C, M) such that c is the capacity, C is an ordered
47  // sequence of n keys = {k_0, k_1, ..., k_(n-1)}, and M is a set of mappings (k, v) such
48  // that k is the for value v, and all k's are unique.  For all k_i in C, there exists
49  // (k_i, v_i) in M.  size(Z) = size(C) = size(M) <= c.  The most recently used v is at the
50  // head of C, and the least recently used is at the tail.  When a new (k_new, v_new) is
51  // added to Z, C = {k_new, k_0, ..., k_(n-1)} and if size(C) >= c, the previous k_(n-1) is
52  // removed from C and M; and M = M U (k_new, v_new).
53  
54  
55  /** A cache map is a map with a built-in cache.
56   *
57   * It operates in all ways like a regular {@link java.util.Map}, except that you can store
58   * only a limited number of entries.  After adding an entry that exceeds the size of the
59   * cache, the map ejects the least recently used entry.
60   *
61   * @author Kelly
62   */
63  public class CacheMap implements Map {
64  	/** Create a cache map with a default capacity of 5 entries.
65  	 */
66  	public CacheMap() {
67  		// FXN: [ c, C, M := 5, {}, {} ]
68  		this(DEFAULT_CAPACITY);
69  	}
70  
71  	/** Create a cache map with the given capacity.
72  	 *
73  	 * @param capacity How big the cache is.
74  	 */
75  	public CacheMap(int capacity) {
76  		// FXN: [ c, C, M := capacity, {}, {} ]
77  
78  		if (capacity < 0) {
79  		  throw new IllegalArgumentException("Can't have a negative size " + capacity + " cache map");
80  		}
81  		this.capacity = capacity;
82  	}
83  
84  	/** Get the cache map's capacity.
85  	 *
86  	 * @return Its capacity.
87  	 */
88  	public int getCapacity() {
89  		return capacity;
90  	}
91  
92  	/** Create a cache map from the given map, having the capacity of the given map.
93  	 *
94  	 * @param map The map to copy.
95  	 */
96  	public CacheMap(Map map) {
97  		// FXN: [ c, C, M := size(map), keys(map), dup(map) ]
98  
99  		this.capacity = map.size();
100 		putAll(map);
101 	}
102 
103 	public int size() {
104 		// FXN: [ return value := size(M) ]
105 		return map.size();
106 	}
107 
108 	public boolean isEmpty() {
109 		// FXN: [ return value := size(M) = 0? ]
110 		return map.size() == 0;
111 	}
112 
113 	/** Returns true if there's a mapping for the specified key in the cache map.
114 	 *
115 	 * This method does not otherwise affect the cache.
116 	 *
117 	 * @param key Key to check.
118 	 * @return True if the cache map contains a mapping for the <var>key</var>.
119 	 */
120 	public boolean containsKey(Object key) {
121 		// FXN: [ return value := presence(key, M) ]
122 		return (map.containsKey(key));
123 	}
124 
125 	/** Returns true if there's a mapping for the specified value in the cache map.
126 	 *
127 	 * This method does not otherwise affect the cache.
128 	 *
129 	 * @param value Value to check.
130 	 * @return True if the cache map contains a mapping for the <var>value</var>.
131 	 */
132 	public boolean containsValue(Object value) {
133 		// FXN: [ return value := presence(value, M) ]
134 		return map.containsValue(value);
135 	}
136 
137 	public Object get(Object key) {
138 		// FXN: [ key in M -> C, return value := advance(key, C), value(key, M)
139 		//      | true     -> C, return value := C, null ]
140 
141 		advance(key);
142 		return map.get(key);
143 	}
144 
145 	public Object put(Object key, Object value) {
146 		// FXN: [ key in M -> C, M, return value := advance(key, C), M U (key, value), value(key, M)
147 		//      | size(C) < c -> C, M, return value := key || C, M U (key, value), null
148 		//      | true -> C, M, return value := key || {k_i | k elem C, 0 <= i <= c - 2},
149 		//                                      (M U (key, value)) - (k_(c-1), v), null ]
150 
151 		Object old = map.put(key, value);
152 		if (old != null) {
153 			advance(key);
154 			return old;
155 		}
156 
157 		cache.addFirst(key);
158 		if (cache.size() > capacity) {
159 		  map.remove(cache.removeLast());
160 		}
161 		return null;
162 	}
163 	
164 	public Object remove(Object key) {
165 		// FXN: [ key in M -> C, M, return value := C - key, M - (key, v), v
166 		//      | true -> return value := null ]
167 
168 		if (!map.containsKey(key)) {
169 		  return null;
170 		}
171 		cache.remove(key);
172 		return map.remove(key);
173 	}
174 
175 	public void putAll(Map t) {
176 		// FXN: [ C, M := (keys(t) || C)[0..(c-1)], { (k_i, v_i) | k_i elem of (keys(t) || C)[0..(c-1)]} ]
177 	  for (Object o : t.entrySet()) {
178 		Entry entry = (Entry) o;
179 		put(entry.getKey(), entry.getValue());
180 	  }
181 	}
182 
183 	public void clear() {
184 		cache.clear();
185 		map.clear();
186 	}
187 
188 	public Set keySet() {
189 		throw new UnsupportedOperationException("Not implemented for CacheMap");
190 	}
191 	public Collection values() {
192 		throw new UnsupportedOperationException("Not implemented for CacheMap");
193 	}
194 	public Set entrySet() {
195 		throw new UnsupportedOperationException("Not implemented for CacheMap");
196 	}
197 
198 	public boolean equals(Object rhs) {
199 		if (rhs == this) {
200 		  return true;
201 		}
202 		if (rhs == null || !(rhs instanceof CacheMap)) {
203 		  return false;
204 		}
205 		CacheMap obj = (CacheMap) rhs;
206 		return obj.cache.equals(cache);
207 	}
208 
209 	public int hashCode() {
210 		return cache.hashCode();
211 	}
212 
213 	/** Advance the given key to the front of the cache.
214 	 *
215 	 * If the key isn't in the cache, leave the cache alone.
216 	 *
217 	 * @param key The key to advance.
218 	 */
219 	private void advance(Object key) {
220 		// FXN: [ C = advance(key, C) ]
221 
222 		boolean present = cache.remove(key);
223 		if (!present) {
224 		  return;
225 		}
226 		cache.addFirst(key);
227 	}
228 
229 	/** What the default capacity of a cache map is. */
230 	private static final int DEFAULT_CAPACITY = 5;
231 
232 	/** The cache (C). */
233 	private LinkedList cache = new LinkedList();
234 
235 	/** The map (M). */
236 	private Map map = new ConcurrentHashMap();
237 
238 	/** The capacity of this cache map (c). */
239 	private int capacity;
240 }
