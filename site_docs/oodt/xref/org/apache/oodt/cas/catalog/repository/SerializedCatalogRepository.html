47  	private static Logger LOG = Logger.getLogger(SerializedCatalogRepository.class.getName());
48  	protected String storageDir;

50  	public SerializedCatalogRepository(String storageDir) throws InstantiationException {
51  		try {
52  			this.storageDir = storageDir;
53  			new File(this.storageDir + "/catalogs").mkdirs();
54  			new File(this.storageDir + "/classloaders").mkdirs();
55  		}catch(Exception e) {
56  			LOG.log(Level.SEVERE, e.getMessage());
57  		  	LOG.log(Level.SEVERE, e.getMessage());
58  			throw new InstantiationException("Failed to instantiate SerializedCatalogRepository : " + e.getMessage());
59  		}
60  	}

63  	 * (non-Javadoc)
64  	 * @see org.apache.oodt.cas.catalog.repository.MemoryBasedCatalogRepository#deleteSerializedCatalog(java.lang.String)
65  	 */
66  	public void deleteSerializedCatalog(String catalogUrn)
67  			throws CatalogRepositoryException {
68  		LOG.log(Level.INFO, "Deleting Catalog: '" + catalogUrn + "' . . . ");
69  		boolean catalogFileDelete = this.getCatalogFile(catalogUrn).delete();
70  		if (!catalogFileDelete) {
71  		  throw new CatalogRepositoryException(
72  			  "Failed to deserialize catalog '" + catalogUrn + "', delete files returned false");
73  		} else {
74  		  LOG.log(Level.INFO, "Successfully deleting Catalog: '" + catalogUrn + "'");
75  		}
76  	}

79  	 * (non-Javadoc)
80  	 * @see org.apache.oodt.cas.catalog.repository.MemoryBasedCatalogRepository#deserializeCatalog(java.lang.String)
81  	 */
82  	public Catalog deserializeCatalog(String catalogUrn)
83  			throws CatalogRepositoryException {
84  		LOG.log(Level.INFO, "Deserializing Catalog: " + catalogUrn);
85  		FileInputStream catalogIn = null;
86  		try {
87  			return new Serializer().deserializeObject(Catalog.class, catalogIn = new FileInputStream(this.getCatalogFile(catalogUrn)));
88  		}catch (Exception e) {
89  			throw new CatalogRepositoryException("Failed to Deserialized Catalogs from '" + this.storageDir + "' : " + e.getMessage(), e);
90  		}finally {
91  			try {
92  				catalogIn.close();
93  			}catch (Exception ignored) {}
94  		}
95  	}

97  	/*
98  	 * (non-Javadoc)
99  	 * @see org.apache.oodt.cas.catalog.repository.MemoryBasedCatalogRepository#deserializeAllCatalogs()
100 	 */
101 	public Set<Catalog> deserializeAllCatalogs()
102 			throws CatalogRepositoryException {
103 		HashSet<Catalog> catalogs = new HashSet<Catalog>();
104 		for (String catalogFile : new File(this.storageDir + "/catalogs").list()) {
105 			Catalog catalog = this.deserializeCatalog(catalogFile.split("\\.ser")[0]);
106 			catalogs.add(catalog);
107 		}
108 		return catalogs;
109 	}

121 	 * @see org.apache.oodt.cas.catalog.repository.MemoryBasedCatalogRepository#serializeCatalog(org.apache.oodt.cas.catalog.Catalog)
122 	 */
123 	public void serializeCatalog(Catalog catalog)
124 			throws CatalogRepositoryException {
125 		LOG.log(Level.INFO, "Serializing Catalog: " + catalog.getId());
126 		FileOutputStream catalogOut = null;
127 		try {
128 			//serialize Catalog
129 			new Serializer().serializeObject(catalog, (catalogOut = new FileOutputStream(this.getCatalogFileWorker(catalog.getId()))));
130 			if (this.getCatalogFile(catalog.getId()).exists()) {
131 			  FileUtils.copyFile(this.getCatalogFile(catalog.getId()), this.getCatalogFileBkup(catalog.getId()), true);
132 			}
133 			FileUtils.copyFile(this.getCatalogFileWorker(catalog.getId()), this.getCatalogFile(catalog.getId()), true);
134 			this.getCatalogFileWorker(catalog.getId()).delete();
135 			this.getCatalogFileBkup(catalog.getId()).delete();
136 		}catch (Exception e) {
137 			throw new CatalogRepositoryException("Failed to Serialized Catalogs to '" + this.storageDir + "' : " + e.getMessage(), e);
138 		}finally {
139 			try {
140 				catalogOut.close();
141 			}catch (Exception ignored) {}
142 		}	
143 	}

145 	public void serializePluginURLs(List<PluginURL> urls) 
146 			throws CatalogRepositoryException {
147 		FileOutputStream urlsOut = null;
148 		try {
149 			//serialize URLs
150 			new Serializer().serializeObject(urls, (urlsOut = new FileOutputStream(this.getClassLoaderUrlsFileWorker())));
151 			if (this.getClassLoaderUrlsFile().exists()) {
152 			  FileUtils.copyFile(this.getClassLoaderUrlsFile(), this.getClassLoaderUrlsFileBkup(), true);
153 			}
154 			FileUtils.copyFile(this.getClassLoaderUrlsFileWorker(), this.getClassLoaderUrlsFile(), true);
155 			this.getClassLoaderUrlsFileWorker().delete();
156 			this.getClassLoaderUrlsFileBkup().delete();
157 		}catch (Exception e) {
158 			throw new CatalogRepositoryException("Failed to Serialized ClassLoader URLs to '" + this.storageDir + "' : " + e.getMessage(), e);
159 		}finally {
160 			try {
161 				urlsOut.close();
162 			}catch (Exception ignored) {}
163 		}	
164 	}

166 	public List<PluginURL> deserializePluginURLs() throws CatalogRepositoryException {
167 		FileInputStream urlsIn = null;
168 		try {
169 			if (this.getClassLoaderUrlsFile().exists()) {
170 			  return new Serializer()
171 				  .deserializeObject(List.class, (urlsIn = new FileInputStream(this.getClassLoaderUrlsFile())));
172 			} else {
173 			  return Collections.emptyList();
174 			}
175 		}catch (Exception e) {
176 			throw new CatalogRepositoryException("Failed to Deserialized All ClassLoader URLs from '" + this.storageDir + "' : " + e.getMessage(), e);
177 		}finally {
178 			try {
179 				urlsIn.close();
180 			}catch (Exception ignored) {}
181 		}
182 	}
