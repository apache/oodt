199 		Connection conn = null;
200 		Statement stmt = null;
201 		TransactionId<?> catalogTransactionId = null;
202 		try {
203 			catalogTransactionId = this.getTransactionIdFactory().createNewTransactionId();
204 			conn = this.dataSource.getConnection();
205 			stmt = conn.createStatement();
206 			Calendar calendar = DateUtils.getCurrentLocalTime();
207 			stmt.execute("INSERT INTO transactions VALUES ('" + catalogTransactionId + "','" + DateUtils.toString(calendar) + "')");
208 			for (TermBucket termBucket : termBuckets) {
209 				for (Term term : termBucket.getTerms()) {
210 					for (String value : term.getValues()) {
211 						try {
212 							stmt.execute("INSERT INTO transaction_terms VALUES ('" + catalogTransactionId + "','" + termBucket.getName() + "','" + term.getName() + "','" + (this.useUTF8 ? URLEncoder.encode(value, "UTF8") : value) + "')");
213 						}catch (Exception e) {
214 							LOG.log(Level.WARNING, "Failed to ingest term: '" + catalogTransactionId + "','" + termBucket.getName() + "','" + term.getName() + "','" + value + "'");
215 						}
216 					}
217 				}
218 			}
219 			conn.commit();
220 			return new IngestReceipt(catalogTransactionId, calendar.getTime());
221 		}catch (Exception e) {
222 			throw new IngestServiceException("Failed to ingest metadata for transaction id '" + catalogTransactionId + "' : " + e.getMessage(), e);
223 		}finally {
224 			try {
225 				conn.close();

319 	public List<TermBucket> getBuckets(TransactionId<?> transactionId)
320 			throws QueryServiceException {
321 		Connection conn = null;
322 		Statement stmt = null;
323 		ResultSet rs = null;
324 		try {
325 			ConcurrentHashMap<String, TermBucket> termBuckets = new ConcurrentHashMap<String, TermBucket>();
326 			conn = this.dataSource.getConnection();
327 			stmt = conn.createStatement();
328 			rs = stmt.executeQuery("SELECT bucket_name,term_name,term_value FROM transaction_terms WHERE transaction_id = '" + transactionId + "'");
329 			while (rs.next()) {
330                 String bucketName = rs.getString("bucket_name");
331                 String termName = rs.getString("term_name");
332                 String termValue = rs.getString("term_value");
333                 TermBucket bucket = termBuckets.get(bucketName);
334                 if (bucket == null) {
335 				  bucket = new TermBucket(bucketName);
336 				}
337                 Term term = new Term(termName, Collections.singletonList((this.useUTF8 ? URLDecoder.decode(termValue, "UTF8") : termValue)));
338                 bucket.addTerm(term);
339                 termBuckets.put(bucketName, bucket);
340 			}
341 			return new Vector<TermBucket>(termBuckets.values());
342 		}catch (Exception e) {
343 			throw new QueryServiceException("Failed to get term buckets for transaction id '" + transactionId + "' : " + e.getMessage(), e);
344 		}finally {
345 			try {
346 				conn.close();
347 			}catch(Exception ignored) {}
348 			try {
349 				stmt.close();
350 			}catch(Exception ignored) {}
351 			try {
352 				rs.close();
353 			}catch(Exception ignored) {}
354 		}
355 	}

484     private String getSqlQuery(QueryExpression queryExpression) throws QueryServiceException, UnsupportedEncodingException {
485         StringBuilder sqlQuery = new StringBuilder();
486 		StringBuilder bucketNameFilter = new StringBuilder("");
487 		if (queryExpression.getBucketNames() != null) {
488 			if (queryExpression.getBucketNames().size() == 1) {
489 			  bucketNameFilter.append("bucket_name = '").append(queryExpression.getBucketNames().iterator().next())
490 							  .append("' ").append("AND ");
491 			} else if (queryExpression.getBucketNames().size() > 1) {
492 			  bucketNameFilter.append("(bucket_name = '")
493 							  .append(StringUtils.join(queryExpression.getBucketNames().iterator(),
494 								  "' OR bucket_name = '")).append("') AND ");
495 			}
496 		}
497         if (queryExpression instanceof QueryLogicalGroup) {
498         	QueryLogicalGroup qlg = (QueryLogicalGroup) queryExpression;
499             sqlQuery.append("(").append(this.getSqlQuery(qlg.getExpressions().get(0)));
500             String op = qlg.getOperator() == QueryLogicalGroup.Operator.AND ? "INTERSECT" : "UNION";
501             for (int i = 1; i < qlg.getExpressions().size(); i++) {
502 			  sqlQuery.append(") ").append(op).append(" (").append(this.getSqlQuery(qlg.getExpressions().get(i)));
503 			}
504             sqlQuery.append(")");
505         }else if (queryExpression instanceof ComparisonQueryExpression){
506         	ComparisonQueryExpression cqe = (ComparisonQueryExpression) queryExpression;
507         	String operator;
508             if (cqe.getOperator().equals(ComparisonQueryExpression.Operator.EQUAL_TO)) {
509             	operator = "=";
510             } else if (cqe.getOperator().equals(ComparisonQueryExpression.Operator.GREATER_THAN)) {
511             	operator = ">";
512             } else if (cqe.getOperator().equals(ComparisonQueryExpression.Operator.GREATER_THAN_EQUAL_TO)) {
513             	operator = ">=";
514             } else if (cqe.getOperator().equals(ComparisonQueryExpression.Operator.LESS_THAN)) {
515             	operator = "<";
516             } else if (cqe.getOperator().equals(ComparisonQueryExpression.Operator.LESS_THAN_EQUAL_TO)) {
517             	operator = "<=";
518             } else {
519                 throw new QueryServiceException("Invalid ComparisonQueryExpression Operator '" + cqe.getOperator() + "'");
520             }
521             
522             sqlQuery.append("SELECT DISTINCT transaction_id FROM transaction_terms WHERE ")
523 					.append(bucketNameFilter.toString()).append(" ").append("term_name = '")
524 					.append(cqe.getTerm().getName()).append("' AND (");
525         	for (int i = 0; i < cqe.getTerm().getValues().size(); i++) {
526         		String value = cqe.getTerm().getValues().get(i);
527                 sqlQuery.append("term_value ").append(operator).append(" '")
528 						.append(this.useUTF8 ? URLEncoder.encode(value, "UTF-8") : value).append("'");
529 	            if ((i + 1) < cqe.getTerm().getValues().size()) {
530 				  sqlQuery.append(" OR ");
531 				}
532         	}
533         	sqlQuery.append(")");
534         }else if (queryExpression instanceof NotQueryExpression) {
535         	NotQueryExpression nqe = (NotQueryExpression) queryExpression;
536             sqlQuery.append("SELECT DISTINCT transaction_id FROM transaction_terms WHERE ")
537 					.append(bucketNameFilter.toString()).append(" NOT (")
538 					.append(this.getSqlQuery(nqe.getQueryExpression())).append(")");
539         }else if (queryExpression instanceof StdQueryExpression) {
540             sqlQuery.append("SELECT DISTINCT transaction_id FROM transaction_terms ")
541 					.append(bucketNameFilter.toString());
542         }else {
543             throw new QueryServiceException("Invalid QueryExpression '" + queryExpression.getClass().getCanonicalName() + "'");
544         }
545         return sqlQuery.toString();
546     }
