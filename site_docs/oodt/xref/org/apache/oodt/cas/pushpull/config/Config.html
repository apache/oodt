101     public void loadConfigFile(File configFile) throws ConfigException {
102 
103         // load properties from configuration file
104         try {
105             System.getProperties().load(new FileInputStream(configFile));
106         } catch (Exception e) {
107             throw new ConfigException("Failed to load properties file : "
108                     + e.getMessage());
109         }
110 
111         // parse properties and xml file specified in config file
112         try {
113             loadProperties();
114         } catch (Exception e) {
115             throw new ConfigException(
116                     "Failed to get properties from properties file : "
117                             + e.getMessage());
118         }
119     }

186     void loadExternalConfigFiles() throws ConfigException {
187         String[] externalConfigs = PropertiesUtils
188                 .getProperties(EXTERNAL_PROPERTIES_FILES);
189         for (String externalConfig : externalConfigs) {
190             try {
191                 System.getProperties().load(
192                         new FileInputStream(new File(externalConfig)));
193             } catch (Exception e) {
194                 throw new ConfigException("Failed to load default config file "
195                         + externalConfig + " : " + e.getMessage());
196             }
197         }
198     }

200     void loadProtocolTypes() throws ConfigException {
201         LOG.log(Level.INFO,
202                 "Associating protocol types with ProtocolFactories . . .");
203         String[] protocolFactoryInfoFiles = PropertiesUtils
204                 .getProperties(PROTOCOL_FACTORY_INFO_FILES);
205         for (String file : protocolFactoryInfoFiles) {
206             try {
207                 pi.loadProtocolFactoryInfoFromFile(new File(file));
208             } catch (Exception e) {

230     void loadIngester() throws ConfigException {
231         try {
232             String fmUrlStr = PropertiesUtils.getProperties(INGESTER_FM_URL,
233                     new String[] { NO_FM_SPECIFIED })[0];
234 
235             if (fmUrlStr.equals(NO_FM_SPECIFIED)) {
236                 LOG
237                         .log(Level.INFO,
238                                 "No file manager url specified: no ingester will be used");
239             } else {
240                 String ingesterClass = PropertiesUtils
241                         .getProperties(INGESTER_CLASS)[0];
242                 String dataTransferClass = PropertiesUtils
243                         .getProperties(INGESTER_DATA_TRANSFER)[0];
244                 String cacheFactoryClass = System.getProperty(CACHE_FACTORY_CLASS);
245 
246                 LOG.log(Level.INFO, "Configuring and building ingester: ["
247                         + ingesterClass + "]: data transfer: ["
248                         + dataTransferClass + "]: to ingest to file manager: ["
249                         + fmUrlStr + "]");
250 
251                 if (cacheFactoryClass != null) {
252                     LOG.log(Level.INFO, "Configuring Ingester cache: ["
253                             + cacheFactoryClass + "]");
254                 }
255 
256                 this.ingester = PushPullObjectFactory.createIngester(
257                         ingesterClass, cacheFactoryClass);
258                 
259                 this.fmUrl = safeGetUrlFromString(fmUrlStr);
260 
261             }
262 
263         } catch (Exception e) {
264             LOG.log(Level.SEVERE, e.getMessage());
265             throw new ConfigException("Failed to load Ingester : "
266                     + e.getMessage());
267         }
268     }

294     void loadMiscVariables() {
295         onlyDefinedTypes = (PropertiesUtils.getProperties(
296                 ALLOW_ONLY_DEFINED_TYPES, new String[] { "false" })[0]
297                 .toLowerCase().equals("true"));
298         useTracker = (PropertiesUtils.getProperties(USE_TRACKER,
299                 new String[] { "false" })[0].toLowerCase().equals("true"));
300         this.recommendedThreadCount = Integer.parseInt(PropertiesUtils
301                 .getProperties(FILE_RET_SYSTEM_REC_THREAD_COUNT,
302                         new String[] { "8" })[0]);
303         this.maxFailedDownloads = Integer.parseInt(PropertiesUtils
304                 .getProperties(FILE_RET_SYSTEM_MAX_ALLOWED_FAIL_DOWNLOADS,
305                         new String[] { "10" })[0]);
306         metFileExtension = PropertiesUtils.getProperties(MET_FILE_EXT,
307                 new String[] { "info.tmp" })[0];
308         this.writeMetFile = Boolean.getBoolean(WRITE_MET_FILE);
309         String timeoutString = PropertiesUtils.getProperties(
310                 PROTOCOL_TIMEOUT_MS, new String[] { "600000" })[0];
311         if (timeoutString == null) {
312             timeoutString = "0";
313         }
314         pi.setDownloadTimeout(Long.parseLong(timeoutString));
315         pi.setPageSize(Integer.parseInt(PropertiesUtils.getProperties(
316                 PROTOCOL_PAGE_SIZE, new String[] { "8" })[0]));
317         this.baseStagingArea = new File(PropertiesUtils.getProperties(
318                 DATA_FILE_BASE_STAGING_AREA,
319                 new String[] { this.baseStagingArea.getPath() })[0]);
320     }

392     private static URL safeGetUrlFromString(String urlStr) {
393         URL url = null;
394 
395         try {
396             url = new URL(urlStr);
397         } catch (MalformedURLException e) {
398             LOG.log(Level.WARNING, "Unable to generate url from url string: ["
399                     + urlStr + "]: Message: " + e.getMessage());
400         }
401 
402         return url;
403     }
