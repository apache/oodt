59      public VirtualFile(String path, boolean isDir) {
60          this(isDir);
61  
62          if (path != null) {
63              StringTokenizer st = new StringTokenizer(path, "/");
64  
65              if (st.countTokens() > 1) {
66                  VirtualFile vf = path.startsWith("/") ? VirtualFile
67                          .createRootDir()
68                          : new VirtualFile(st.nextToken(), true);
69                  String curRegExp = st.nextToken();
70  
71                  while (st.hasMoreTokens()) {
72                      VirtualFile temp = new VirtualFile(curRegExp, true);
73                      vf.addChild(temp);
74                      vf = temp;
75                      curRegExp = st.nextToken();
76                  }
77                  this.regExp = curRegExp;
78                  vf.addChild(this);
79              } else if (st.countTokens() > 0) {
80                  this.regExp = st.nextToken();
81                  if (path.startsWith("/")) {
82                      VirtualFile.createRootDir().addChild(this);
83                  }
84              } else {
85                  this.copy(VirtualFile.createRootDir());
86              }
87          } else {
88              this.copy(VirtualFile.createRootDir());
89          }
90      }

92      public VirtualFile(VirtualFile root, String path, boolean isDir) {
93          this(isDir);
94  
95          if (path != null) {
96              StringTokenizer st = new StringTokenizer(path, "/");
97  
98              if (st.countTokens() > 0) {
99                  VirtualFile vf = root;
100                 String curRegExp = st.nextToken();
101 
102                 while (st.hasMoreTokens()) {
103                     VirtualFile temp = new VirtualFile(curRegExp, true);
104                     vf.addChild(temp);
105                     vf = temp;
106                     curRegExp = st.nextToken();
107                 }
108                 this.regExp = curRegExp;
109                 vf.addChild(this);
110             } else {
111                 this.copy(root);
112             }
113         } else {
114             this.copy(root);
115         }
116     }

132     public static VirtualFile mergeTwoFiles(VirtualFile vf1, VirtualFile vf2) {
133         if (vf1.isDir) {

143     public void addChild(VirtualFile vf) {
144         if (this.isDir) {
145             VirtualFile existingChildWithSameName = this.getChild(vf.regExp,
146                     vf.isDir);
147             if (existingChildWithSameName == null) {
148                 children.add(vf);
149                 if (vf.isDir()) {
150                     allowNewDirs = false;
151                 } else {
152                     allowNewFiles = false;
153                 }
154                 vf.parent = this;
155             } else {
156                 vf.copy(existingChildWithSameName);
157             }
158         }
159     }

165     public VirtualFile getChild(String regExp, boolean isDirectory) {
166         for (VirtualFile vf : children) {
167             if ((regExp.equals(vf.regExp) || Pattern.matches(vf.regExp, regExp))
168                     && vf.isDir == isDirectory) {
169                 return vf;
170             }
171         }
172         return null;
173     }

179     public VirtualFile getChildRecursive(String path, boolean isDirectory) {
180         StringTokenizer st = new StringTokenizer(path, "/");
181         VirtualFile vf = this;
182         while (st.hasMoreTokens()) {
183             String curRegExp = st.nextToken();
184             if (st.hasMoreTokens()) {
185                 if ((vf = vf.getChild(curRegExp, true)) == null) {
186                     return null;
187                 }
188             } else {
189                 return vf.getChild(curRegExp, isDirectory);
190             }
191         }
192         return null;
193     }

199     public String getAbsolutePath() {
200         if (regExp == null) {
201             return null;
202         }
203         StringBuilder path = new StringBuilder(this.regExp);
204         VirtualFile parent = this.parent;
205         while (parent != null) {
206             path.insert(0, (!parent.regExp.equals("/") ? parent.regExp : "") + "/");
207             parent = parent.parent;
208         }
209         return path.toString();
210     }

290     private StringBuffer printVirtualFiles(LinkedList<VirtualFile> list,
291             String spacer) {
292         StringBuffer output = new StringBuffer("");
293         for (VirtualFile vf : list) {
294             output.append(spacer);
295             output.append(vf.getRegExp()).append(vf.isDir && !vf.regExp.equals("/") ? "/" : "").append("\t : ")
296                   .append(vf.allowNewDirs).append("/").append(vf.noDirs).append("\t\t : ").append(vf.allowNewFiles)
297                   .append("/").append(vf.noFiles).append("\n");
298             output.append(printVirtualFiles(vf.getChildren(), " " + spacer));
299         }
300         return output;
301     }
