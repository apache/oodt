22  import java.io.File;
23  import java.util.concurrent.ConcurrentHashMap;
24  import java.util.LinkedList;
25  import java.util.Map.Entry;
26  
27  //OODT imports
28  import org.apache.oodt.cas.protocol.ProtocolFile;

38   */
39  public class DataFileToPropFileLinker implements DownloadListener {
40  
41      private ConcurrentHashMap<String, File> protocolFilePathAndPropFileMap;
42  
43      private ConcurrentHashMap<File, String> propFileToErrorsMap;
44  
45      private LinkedList<ProtocolFile> downloadingDataFiles;
46  
47      private LinkedList<ProtocolFile> failedDataFiles;
48  
49      private LinkedList<ProtocolFile> successDataFiles;
50  
51      public DataFileToPropFileLinker() {
52          this.protocolFilePathAndPropFileMap = new ConcurrentHashMap<String, File>();
53          this.propFileToErrorsMap = new ConcurrentHashMap<File, String>();
54          downloadingDataFiles = new LinkedList<ProtocolFile>();
55          failedDataFiles = new LinkedList<ProtocolFile>();
56          successDataFiles = new LinkedList<ProtocolFile>();
57      }
58  
59      public synchronized void addPropFileToDataFileLink(File propFile,
60              ProtocolFile pFile) {
61          this.addPropFileToDataFileLink(propFile, pFile.getPath());
62      }
63  
64      public synchronized void addPropFileToDataFileLink(File propFile,
65              String remoteDataFilePath) {
66          this.protocolFilePathAndPropFileMap.put(remoteDataFilePath, propFile);
67      }
68  
69      public synchronized void markAsFailed(File propFile, String errorMsg) {
70          String errors = this.propFileToErrorsMap.get(propFile);
71          if (errors == null) {
72            this.propFileToErrorsMap.put(propFile, errorMsg);
73          } else {
74            this.propFileToErrorsMap.put(propFile, errors + "," + errorMsg);
75          }
76      }
77  
78      public synchronized void markAsFailed(ProtocolFile pFile, String errorMsg) {
79          this.markAsFailed(pFile.getPath(), errorMsg);
80      }
81  
82      public synchronized void markAsFailed(String pFilePath, String errorMsg) {
83          File propFile = this.protocolFilePathAndPropFileMap.get(pFilePath);
84          if (propFile != null) {
85              String errors = this.propFileToErrorsMap.get(propFile);
86              if (errors == null) {
87                this.propFileToErrorsMap.put(propFile, errorMsg);
88              } else {
89                this.propFileToErrorsMap.put(propFile, errors + "," + errorMsg);
90              }
91          }
92      }
93  
94      public synchronized String getErrorsAndEraseLinks(File propFile) {
95          this.eraseLinks(propFile);
96          return this.getErrors(propFile);
97      }
98  
99      public synchronized void eraseLinks(File propFile) {
100         LinkedList<String> keysToRemove = new LinkedList<String>();
101         for (Entry<String, File> entry : this.protocolFilePathAndPropFileMap
102                 .entrySet()) {
103           if (entry.getValue().equals(propFile)) {
104             keysToRemove.add(entry.getKey());
105           }
106         }
107         for (String key : keysToRemove) {
108           this.protocolFilePathAndPropFileMap.remove(key);
109         }
110     }
111 
112     public synchronized String getErrors(File propFile) {
113         return this.propFileToErrorsMap.remove(propFile);
114     }
115 
116     public synchronized String getStatusOf(File propFile) {
117         return "Status for " + propFile.getAbsolutePath() + ":" + " Errors: \n"
118                 + "   " + this.propFileToErrorsMap.get(propFile) + "\n"
119                 + " Data files specified which are currently downloading:\n"
120                 + "   " + this.getDownloadingFilesLinkedToPropFile(propFile)
121                 + "\n"
122                 + " Data files specified which successfully downloaded:\n"
123                 + "   "
124                 + this.getSuccessfullyDownloadedFilesLinkedToPropFile(propFile)
125                 + "\n" + " Data files specified which failed to download:\n"
126                 + "   "
127                 + this.getFailedToDownloadFilesLinkedToPropFile(propFile)
128                 + "\n";
129     }
130 
131     public synchronized LinkedList<ProtocolFile> getDownloadingFilesLinkedToPropFile(
132             File propFile) {
133         return this.getFilesLinkedToPropFileInList(propFile,
134                 this.downloadingDataFiles);
135     }
136 
137     public synchronized LinkedList<ProtocolFile> getSuccessfullyDownloadedFilesLinkedToPropFile(
138             File propFile) {
139         return this.getFilesLinkedToPropFileInList(propFile,
140                 this.successDataFiles);
141     }
142 
143     public synchronized LinkedList<ProtocolFile> getFailedToDownloadFilesLinkedToPropFile(
144             File propFile) {
145         return this.getFilesLinkedToPropFileInList(propFile,
146                 this.failedDataFiles);
147     }
148 
149     private LinkedList<ProtocolFile> getFilesLinkedToPropFileInList(
150             File propFile, LinkedList<ProtocolFile> list) {
151         LinkedList<ProtocolFile> returnList = new LinkedList<ProtocolFile>();
152         for (ProtocolFile pFile : list) {
153             if (this.protocolFilePathAndPropFileMap.get(pFile.getPath()) != null) {
154               returnList.add(pFile);
155             }
156         }
157         return returnList;
158     }
159 
160     public synchronized void downloadFailed(ProtocolFile file, String errorMsg) {
161         this.markAsFailed(file, "Failed to download '" + file
162                 + "' -- Logged Error msg: " + errorMsg);
163         this.failedDataFiles.add(file);
164         this.downloadingDataFiles.remove(file);
165     }
166 
167     public synchronized void downloadFinished(ProtocolFile file) {
168         this.successDataFiles.add(file);
169         this.downloadingDataFiles.remove(file);
170     }
171 
172     public synchronized void downloadStarted(ProtocolFile file) {
173         this.downloadingDataFiles.add(file);
174     }
175 
176     public synchronized void clear() {
177         this.propFileToErrorsMap.clear();
178         this.protocolFilePathAndPropFileMap.clear();
179         this.downloadingDataFiles.clear();
180         this.failedDataFiles.clear();
181         this.successDataFiles.clear();
182     }
