47  /**
48   * This class is responsible for creating the appropriate Protocol for the given
49   * RemoteSites. The boolean argument 'allowReuse' allows for one unique protocol
50   * for each URL. That is, if allowReuse is set to true, then if no Protocol has
51   * been created for the current site, the Protocol created will be saved and
52   * then returned for any later called with allowReuse equals true. This is to
53   * allow for the same Protocol object to be used by several classes. The
54   * Protocol class has been synchronized so this is thread-safe. If you set
55   * 'allowReuse' to false then a new Protocol object will be created and
56   * returned.<br>
57   * <br>
58   *
59   * @author bfoster
60   */

91    /**
92     * Returns the appropriate protocol for the given Path
93     *
94     * @param pFile
95     *          Used to determine the appropriate Protocol to be returned and the
96     *          path to navigate on if navigateToPathLoc is set to true.
97     * @param allowReuse
98     *          Set to true if you would like ProtocolHandler to take care of the
99     *          protocol returned (i.e. reuseable protocols may be returned by
100    *          this method again, if it is the appropriate protocol type for a
101    *          given Path. Also ProtocolHandler will take care of disconnecting
102    *          the reuseable protocols)
103    * @param navigateToPathLoc
104    *          If true, will navigate the to the end of the Path location
105    *          specified
106    * @return Protocol for the given Path
107    * @throws RemoteConnectionException
108    *           If there is an error creating the protocol
109    */
110   public Protocol getAppropriateProtocol(RemoteSiteFile pFile,
111       boolean allowReuse, boolean navigateToPathLoc)
112       throws RemoteConnectionException {
113     try {
114       Protocol protocol = getAppropriateProtocol(pFile, allowReuse);
115       if (protocol != null && navigateToPathLoc) {
116         if (pFile.isDir()) {
117           this.cd(protocol, pFile);
118         } else if (pFile.getParent() != null) {
119           this.cd(protocol, new RemoteSiteFile(pFile.getParent(), pFile.getSite()));
120         }
121       }
122       return protocol;
123     } catch (Exception e) {
124       throw new RemoteConnectionException(
125           "Failed to get appropriate protocol for " + pFile + " : "
126               + e.getMessage(), e);
127     }
128   }

135   public Protocol getAppropriateProtocolBySite(RemoteSite remoteSite,
136       boolean allowReuse) throws ProtocolException {
137     Protocol protocol = null;
138     try {
139       if ((allowReuse && ((protocol = reuseProtocols.get(remoteSite.getURL().toURI())) == null))
140           || !allowReuse) {
141         ProtocolFactory protocolFactory = null;
142         try {
143           protocolFactory = this.urlAndProtocolFactory
144               .get(remoteSite.getURL().toURI());
145         } catch (URISyntaxException e) {
146           LOG.log(Level.SEVERE, "could not convert url to uri: Message: " + e.getMessage());
147         }
148         if (protocolFactory == null) {
149           LinkedList<Class<ProtocolFactory>> protocolClasses = pi
150               .getProtocolClassesForProtocolType(remoteSite.getURL()
151                                                            .getProtocol());
152           for (Class<ProtocolFactory> clazz : protocolClasses) {
153             try {
154               if ((protocol = (protocolFactory = clazz.newInstance())
155                   .newInstance()) != null) {
156                 if (!connect(protocol, remoteSite, true)) {
157                   LOG.log(
158                       Level.WARNING,
159                       "ProtocolFactory "
160                           + protocolFactory.getClass().getCanonicalName()
161                           + " is not compatible with server at "
162                           + remoteSite.getURL());
163                   protocol = null;
164                 } else {
165                   this.urlAndProtocolFactory.put(remoteSite.getURL().toURI(),
166                       protocolFactory);
167                   break;
168                 }
169               }
170             } catch (Exception e) {
171               LOG.log(Level.WARNING, "Failed to instanciate protocol " + clazz
172                   + " for " + remoteSite.getURL());
173             }
174           }
175           if (protocol == null) {
176             throw new ProtocolException("Failed to get appropriate protocol for "
177                                         + remoteSite);
178           }
179         } else {
180           connect(protocol = protocolFactory.newInstance(), remoteSite, false);
181         }
182         if (allowReuse) {
183           try {
184             this.reuseProtocols.put(remoteSite.getURL().toURI(), protocol);
185           } catch (URISyntaxException e) {
186             LOG.log(Level.SEVERE, "Couildn't covert URL to URI Mesage: " + e.getMessage());
187           }
188         }
189       }
190     } catch (URISyntaxException e) {
191       LOG.log(Level.SEVERE, "could not convert url to uri: Message: "+e.getMessage());
192     }
193     return protocol;
194   }

207   public synchronized List<RemoteSiteFile> nextPage(RemoteSite site, Protocol protocol,
208       ProtocolFileFilter filter) throws RemoteConnectionException,
209       ProtocolException {
210 
211     PagingInfo pgInfo = this.getPagingInfo(this.pwd(site, protocol));
212     try {
213       System.out.println("PageSize: " + pi.getPageSize() + " PageLoc: "
214           + pgInfo.getPageLoc());
215       List<RemoteSiteFile> fileList = this.ls(site, protocol);
216       System.out.println("FileList size: " + fileList.size());
217 
218       if (this.getDynamicFileList(site, protocol) == null
219           && !this.passesDynamicDetection(pgInfo, fileList)) {
220         LOG.log(
221             Level.SEVERE,
222             "Remote directory '"
223                 + this.pwd(site, protocol)
224                 + "' file list size has changed -- setting directory as dynamic and resetting page location");
225         this.putDynamicFileList(site, protocol, fileList);
226         pgInfo.updatePageInfo(0, fileList);
227       }
228 
229       List<RemoteSiteFile> page = new LinkedList<RemoteSiteFile>();
230       int curLoc = pgInfo.getPageLoc();
231       for (; page.size() < pi.getPageSize() && curLoc < fileList.size(); curLoc++) {
232         if (filter == null || filter.accept(fileList.get(curLoc))) {
233           page.add(fileList.get(curLoc));
234         }
235       }
236       pgInfo.updatePageInfo(curLoc, fileList);
237 
238       return page;
239     } catch (Exception e) {
240       LOG.log(Level.SEVERE, e.getMessage());
241       throw new RemoteConnectionException(
242           "Failed getting next page for protocol " + protocol + "-- pgStart = "
243               + pgInfo.getPageLoc() + " pgSize = " + pi.getPageSize() + " : "
244               + e.getMessage());
245     }
246

249   private boolean passesDynamicDetection(PagingInfo pgInfo,
250       List<RemoteSiteFile> newLS) {
251     return !(pgInfo.getSizeOfLastLS() != -1
252              && (pgInfo.getSizeOfLastLS() != newLS.size() || (newLS.size() != 0
253                                                               && pgInfo.getPageLoc() < newLS.size() && (newLS.get(pgInfo
254         .getPageLoc()) == null || !newLS.get(pgInfo.getPageLoc()).equals(
255         pgInfo.getRemoteSiteFileAtPageLoc())))));
256   }

258   public void download(Protocol protocol, RemoteSiteFile fromFile, File toFile,
259       boolean delete) throws RemoteConnectionException {
260 
261     // rename file for download
262     File downloadFile = new File(
263         String.format("%s/Downloading_%s", toFile.getParent(), toFile.getName()));
264     toFile.renameTo(downloadFile);
265 
266     LOG.log(Level.INFO, "Starting to download " + fromFile);
267     try {
268       // try to download the file
269       protocol.get(fromFile, downloadFile);
270 
271       // rename file back to original name
272       if (!downloadFile.renameTo(toFile)) {
273         throw new IOException(
274             String.format("Failed to rename file %s to %s", downloadFile, toFile));
275       }
276 
277       // delete file is specified
278       if (delete) {
279         if (!this.delete(protocol, fromFile)) {
280           LOG.log(Level.WARNING, "Failed to delete file '" + fromFile
281                                  + "' from server '" + fromFile.getSite() + "'");
282         } else {
283           LOG.log(Level.INFO, "Successfully deleted file '" + fromFile
284                               + "' from server '" + fromFile.getSite() + "'");
285         }
286       }
287 
288       LOG.log(Level.INFO, "Finished downloading " + fromFile + " to " + toFile);
289 
290     } catch (Exception e) {
291       downloadFile.delete();
292       throw new RemoteConnectionException("Failed to download file " + fromFile
293           + " : " + e.getMessage(), e);
294     }
295   }

297   /**
298    * Connects the given Protocol to the given URL
299    *
300    * @param protocol
301    *          The Protocol that will be connected
302    * @param remoteSite
303    *          The server to which the Protocol will connect
304    * @param test
305    */
306   public boolean connect(Protocol protocol, RemoteSite remoteSite, boolean test) {
307     for (int tries = 0; tries < 3; tries++) {
308 
309       // wait for 5 secs before next retry
310       if (tries > 0) {
311         LOG.log(Level.INFO, "Will retry connecting to " + remoteSite
312             + " in 5 seconds");
313         synchronized (this) {
314           try {
315             System.out.print("Waiting");
316             for (int i = 0; i < 5; i++) {
317               System.out.print(" .");
318               wait(1000);
319             }
320             System.out.println();
321           } catch (Exception ignored) {
322           }
323         }
324       }
325 
326       try {
327         // make sure protocol is disconnected
328         try {
329           protocol.close();
330         } catch (Exception ignored) {
331         }
332 
333         // try connecting Protocol
334         protocol.connect(
335             remoteSite.getURL().getHost(),
336             new BasicAuthentication(remoteSite.getUsername(), remoteSite
337                 .getPassword()));
338 
339         // check connection
340         if (protocol.connected()
341             && (!test || isOkProtocol(protocol, remoteSite))) {
342           LOG.log(Level.INFO,
343               "Successfully connected to " + remoteSite.getURL()
344                   + " with protocol '" + protocol.getClass().getCanonicalName()
345                   + "' and username '" + remoteSite.getUsername() + "'");
346           return true;
347         } else {
348           return false;
349         }
350 
351       } catch (Exception e) {
352         LOG.log(Level.WARNING, "Error occurred while connecting to "
353             + remoteSite + " : " + e.getMessage(), e);
354       }
355 
356     }
357     return false;
358   }

360   private boolean isOkProtocol(Protocol protocol, RemoteSite remoteSite) {
361     try {
362       LOG.log(Level.INFO, "Testing protocol "
363           + protocol.getClass().getCanonicalName()
364           + " . . . this may take a few minutes . . .");
365       // test ls, cd, and pwd
366       this.cdToHOME(protocol);
367       RemoteSiteFile home = this.pwd(remoteSite, protocol);
368       this.ls(remoteSite, protocol);
369       if (remoteSite.getCdTestDir() != null) {
370         this.cd(protocol, new RemoteSiteFile(home, remoteSite.getCdTestDir(),
371             true, remoteSite));
372       } else {
373         this.cdToROOT(protocol);
374       }
375       this.cdToHOME(protocol);
376       if (home == null || !home.equals(this.pwd(remoteSite, protocol))) {
377         throw new ProtocolException("Home directory not the same after cd");
378       }
379     } catch (Exception e) {
380       LOG.log(Level.SEVERE, "Protocol "
381           + protocol.getClass().getCanonicalName()
382           + " failed compatibility test : " + e.getMessage(), e);
383       return false;
384     }
385     return true;
386   }

410   public synchronized boolean delete(Protocol protocol, RemoteSiteFile file) {
411     try {
412       PagingInfo pgInfo = this.getPagingInfo(file.getRemoteParent());
413       List<RemoteSiteFile> fileList = this.ls(protocol, file.getRemoteParent());
414       int indexOfFile = fileList.indexOf(file);
415       if (indexOfFile != -1) {
416         protocol.delete(file);
417         fileList.remove(indexOfFile);
418         System.out.println("IndexOfFile: " + indexOfFile + " PageIndex: "
419             + pgInfo.getPageLoc());
420         if (indexOfFile < pgInfo.getPageLoc()
421             || indexOfFile == fileList.size() - 1) {
422           pgInfo.updatePageInfo(pgInfo.getPageLoc() - 1, fileList);
423         } else {
424           pgInfo.updatePageInfo(pgInfo.getPageLoc(), fileList);
425         }
426         return true;
427       } else {
428         return false;
429       }
430     } catch (Exception e) {
431       LOG.log(Level.SEVERE, "Failed to delete file", e);
432       return false;
433     }
434   }

529    * the current Protocol
530    *
531    * @throws RemoteConnectionException
532    */
533   public void close() throws RemoteConnectionException {
534     Set<Entry<URI, Protocol>> entries = reuseProtocols.entrySet();
535     for (Entry<URI, Protocol> entry : entries) {
536       disconnect(entry.getValue());
537     }
538     this.reuseProtocols.clear();
539     this.urlAndProtocolFactory.clear();
540     this.pageInfos.clear();
541     this.pathAndFileListMap.clear();
542   }

558   private List<RemoteSiteFile> toRemoteSiteFiles(List<ProtocolFile> files, RemoteSite site) {
559     List<RemoteSiteFile> newFiles = new Vector<RemoteSiteFile>();
560     if (files != null) {

568   class PagingInfo {
569 
570     private int pageLoc;
571 
572     private int sizeOfLastLS;
573 
574     private RemoteSiteFile pFileAtPageLoc;
575 
576     PagingInfo() {
577       this.pageLoc = 0;
578       this.sizeOfLastLS = -1;
579       this.pFileAtPageLoc = null;
580     }
581 
582     synchronized void updatePageInfo(int newPageLoc, List<RemoteSiteFile> ls) {
583       this.sizeOfLastLS = ls.size();
584       this.pageLoc = newPageLoc < 0 ? 0 : newPageLoc;
585       this.pFileAtPageLoc = (this.sizeOfLastLS > 0 && newPageLoc < ls.size()) ? ls
586           .get(newPageLoc) : null;
587     }
588 
589     synchronized int getPageLoc() {
590       return this.pageLoc;
591     }
592 
593     synchronized int getSizeOfLastLS() {
594       return this.sizeOfLastLS;
595     }
596 
597     synchronized RemoteSiteFile getRemoteSiteFileAtPageLoc() {
598       return this.pFileAtPageLoc;
599     }
