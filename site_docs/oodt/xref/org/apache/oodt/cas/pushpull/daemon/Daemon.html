176     /**
177      * Loads and executes the Crawler specified by the properties file. It will
178      * crawl the URLs specified in the properties file in the sequence
179      * given--one at a time.
180      *
181      * @throws DirStructException
182      */
183     public void startDaemon() {
184         new Thread(new Runnable() {
185 
186             public void run() {
187 
188                 // check if Daemon should sleep first
189                 long timeTilNextRun;
190                 if ((timeTilNextRun = Daemon.this.calculateTimeTilNextRun()) != 0
191                     && !(Daemon.this.beforeToday(daemonInfo
192                     .getFirstRunDateTime()) && daemonInfo
193                              .runOnReboot())) {
194                     sleep(timeTilNextRun);
195                 }
196 
197                 for (keepRunning = true; keepRunning;) {
198                     long startTime = System.currentTimeMillis();
199 
200                     // get permission to run
201                     Daemon.this.notifyDaemonListenerOfStart();
202                     if (!keepRunning) {
203                         Daemon.this.notifyDaemonListenerOfFinish();
204                         System.out.println("BREAKING OUT");
205                         break;
206                     }
207 
208                     // run
209                     Daemon.this.isRunning = true;
210 
211                     try {
212                         rs.retrieveFiles(daemonInfo.getPropFilesInfo(),
213                             daemonInfo.getDataFilesInfo());
214                     } catch (Exception e) {
215                         LOG.log(Level.SEVERE, e.getMessage());
216                     } finally {
217                         numberOfCrawls++;
218                     }
219 
220                     Daemon.this.isRunning = false;
221 
222                     // calculate performance and sleep
223                     Daemon.this.notifyDaemonListenerOfFinish();
224                     Daemon.this.calculateAndStoreElapsedTime(startTime);
225                     if (Daemon.this.keepRunning
226                         && daemonInfo.getTimeIntervalInMilliseconds() >= 0) {
227                         sleep(Daemon.this.calculateTimeTilNextRun());
228                     } else {
229                         break;
230                     }
231                 }
232                 LOG.log(Level.INFO, "Daemon with ID = "
233                                     + Daemon.this.getDaemonID() + " on RMI registry port "
234                                     + Daemon.this.rmiRegPort + " is shutting down");
235                 Daemon.this.unregister();
236             }
237         }).start();
238     }

259     private long calculateTimeTilNextRun() {
260         GregorianCalendar now = new GregorianCalendar();
261         GregorianCalendar gcStartDateTime = new GregorianCalendar();
262         gcStartDateTime.setTime(daemonInfo.getFirstRunDateTime());
263         long diff = now.getTimeInMillis() - gcStartDateTime.getTimeInMillis();
264         if (Math.abs(diff) <= daemonInfo.getEpsilonInMilliseconds()) {
265             return 0;
266         } else if (diff < 0) {
267             return gcStartDateTime.getTimeInMillis() - now.getTimeInMillis();
268         } else if (daemonInfo.getTimeIntervalInMilliseconds() == 0) {
269             return 0;
270         } else {
271             int numOfPeriods = (int) (diff / daemonInfo
272                 .getTimeIntervalInMilliseconds());
273             long nextRunTime = gcStartDateTime.getTimeInMillis()
274                                + ((numOfPeriods + 1) * daemonInfo
275                 .getTimeIntervalInMilliseconds());
276             return nextRunTime - now.getTimeInMillis();
277         }
278     }

296     private void sleep(long length) {
297         if (length > 0) {
298             LOG.log(Level.INFO, "Daemon with ID = " + this.getDaemonID()
299                                 + " on RMI registry port " + this.rmiRegPort
300                                 + " is going to sleep until "
301                                 + new Date(System.currentTimeMillis() + length));
302             synchronized (this) {
303                 try {
304                     wait(length);
305                 } catch (InterruptedException ignored) {
306                 }
307             }
308         }
309     }

311     private long calculateAndStoreElapsedTime(long startTime) {
312         long elapsedTime = System.currentTimeMillis() - startTime;
313         daemonTotalRuntime += elapsedTime;
314         return elapsedTime;
315     }

317     public synchronized void pauseDaemon() {
318         try {
319             LOG.log(Level.INFO, "Daemon with ID = " + this.getDaemonID()
320                                 + " on RMI registry port " + this.rmiRegPort
321                                 + " has been stopped");
322             this.wait(0);
323         } catch (Exception ignored) {
324         }
325         LOG.log(Level.INFO, "Daemon with ID = " + this.getDaemonID()
326                             + " on RMI registry port " + this.rmiRegPort + " has resumed");
327     }

533      *
534      * @param args
535      *            Not Used
536      * @throws IOException
537      * @throws SecurityException
538      */
539     public static void main(String[] args) {
540         try {
541 
542             int rmiPort = -1;
543             boolean waitForCrawlNotification = false;
544 
545             for (int i = 0; i < args.length; ++i) {
546                 if (args[i].equals("--rmiPort")) {
547                     rmiPort = Integer.parseInt(args[++i]);
548                 } else if (args[i].equals("--waitForNotification")) {
549                     waitForCrawlNotification = true;
550                 }
551             }
552 
553             LocateRegistry.createRegistry(rmiPort);
554 
555             try {
556                 // registry CrawlDaemon as MBean so it can be used with jconsole
557                 MBeanServer mbs = ManagementFactory.getPlatformMBeanServer();
558                 ObjectName name = new ObjectName(
559                     "org.apache.oodt.cas.pushpull.daemon:type=Daemon");
560             } catch (Exception e) {
561                 LOG.log(Level.SEVERE,
562                     "Failed to register CrawlDaemon as a MBean Object : "
563                     + e.getMessage());
564             }
565 
566         } catch (Exception e) {
567             LOG.log(Level.SEVERE, "Failed to create CrawlDaemon : "
568                                   + e.getMessage());
569         } finally {
570             // terminate the CrawlDaemon
571             LOG.log(Level.INFO, "Terminating CrawlDaemon");
572         }
