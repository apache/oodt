56  public class ListRetriever implements RetrievalMethod {
57  
58      /* our log stream */
59      private static final Logger LOG = Logger.getLogger(ListRetriever.class
60              .getName());
61  
62      public void processPropFile(FileRetrievalSystem frs, Parser propFileParser,
63              File propFile, DataFilesInfo dfi, DataFileToPropFileLinker linker)
64          throws FileNotFoundException, ParserException, RetrievalMethodException {
65          RemoteSite remoteSite;
66  
67          // parse property file
68          Metadata fileMetadata = new Metadata();
69          VirtualFileStructure vfs = propFileParser.parse(new FileInputStream(
70                  propFile), fileMetadata);
71          DownloadInfo di = dfi.getDownloadInfo();
72          if (!di.isAllowAliasOverride()
73                  || (remoteSite = vfs.getRemoteSite()) == null) {
74            remoteSite = di.getRemoteSite();
75          }
76          LinkedList<String> fileList = FileRestrictions.toStringList(vfs
77                  .getRootVirtualFile());
78  
79          // download data files specified in property file
80          for (String file : fileList) {
81              try {
82                  linker.addPropFileToDataFileLink(propFile, file);
83                  if (!frs.addToDownloadQueue(remoteSite, file, di
84                          .getRenamingConv(), di.getStagingArea(), dfi
85                          .getQueryMetadataElementName(), di.deleteFromServer(), fileMetadata)) {
86                    linker.eraseLinks(propFile);
87                  }
88              } catch (ToManyFailedDownloadsException e) {
89                  throw new RetrievalMethodException(
90                          "Connection appears to be down. . .unusual number of download failures. . .stopping : "
91                                  + e.getMessage());
92              } catch (CatalogException e) {
93                  throw new RetrievalMethodException(
94                          "Failed to communicate with database : "
95                                  + e.getMessage());
96              } catch (AlreadyInDatabaseException e) {
97                  LOG.log(Level.WARNING, "Skipping file : " + e.getMessage());
98              } catch (UndefinedTypeException e) {
99                  LOG.log(Level.WARNING, "Skipping file : " + e.getMessage());
100             } catch (Exception e) {
101                 LOG.log(Level.SEVERE, e.getMessage());
102                 linker.markAsFailed(propFile, "Failed to download " + file
103                         + " from " + remoteSite + " : " + e.getMessage());
104                 throw new RetrievalMethodException("Uknown error accured while downloading "
105                         + file + " from " + remoteSite + " -- bailing out : "
106                         + e.getMessage());
107             }
108         }
109     }
