66    /**
67     * Constructs a new {@link JobInput} implementation from the given
68     * <code>className</code>.
69     *
70     * @param className
71     *          The implementation class for the {@link JobInput}
72     * @return A new {@link JobInput} implementation.
73     */
74    public static JobInput getJobInputFromClassName(String className) {
75      try {
76        Class inputClass = Class.forName(className);
77        return (JobInput) inputClass.newInstance();
78      } catch (ClassNotFoundException e) {
79        LOG.log(Level.SEVERE, e.getMessage());
80        LOG.log(Level.WARNING,
81            "ClassNotFoundException when loading job input class " + className
82                + " Message: " + e.getMessage());
83      } catch (InstantiationException e) {
84        LOG.log(Level.SEVERE, e.getMessage());
85        LOG.log(Level.WARNING,
86            "InstantiationException when loading job input class " + className
87                + " Message: " + e.getMessage());
88      } catch (IllegalAccessException e) {
89        LOG.log(Level.SEVERE, e.getMessage());
90        LOG.log(Level.WARNING,
91            "IllegalAccessException when loading job input class " + className
92                + " Message: " + e.getMessage());
93      }
94  
95      return null;
96    }
