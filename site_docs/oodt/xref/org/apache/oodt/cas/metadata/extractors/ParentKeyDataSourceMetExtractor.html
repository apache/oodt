18  package org.apache.oodt.cas.metadata.extractors;
19  
20  import java.io.File;
21  import java.util.List;
22  
23  import org.apache.oodt.cas.metadata.Metadata;
24  import org.apache.oodt.cas.metadata.exceptions.MetExtractionException;
25  
26  import com.google.common.annotations.VisibleForTesting;
27  import com.google.common.base.Splitter;
28  import com.google.common.collect.Lists;
29  
30  /**
31   * Supports data source key look via parent key if files key is no good.
32   * 
33   * Expected file name:
34   * <parent_key>-<primary_key>.<post_fix>
35   * 
36   * @author bfoster@apache.com (Brian Foster)
37   */
38  public class ParentKeyDataSourceMetExtractor extends DataSourceMetExtractor {
39  
40    private String key;
41    
42    @Override
43    protected Metadata extrMetadata(File file) throws MetExtractionException {
44      try {
45        key = getPrimaryKey(file);
46        return super.extrMetadata(file);
47      } catch (MetExtractionException e) {
48        key = getParentKey(file);
49        if (key != null) {
50          return super.extrMetadata(file);
51        } else {
52          throw e;
53        }
54      }
55    }
56  
57    @VisibleForTesting
58    protected String getKey(File file) {
59      return key;
60    }
61  
62    private String getPrimaryKey(File file) {
63      String key = getKeyAtIndex(file, Index.PRIMARY);
64      return key == null ? super.getKey(file) : key;
65    }
66  
67    private String getParentKey(File file) {
68      return getKeyAtIndex(file, Index.PARENT);
69    }
70  
71    private String getKeyAtIndex(File file, Index index) {
72      String key = super.getKey(file);
73      List<String> splitKey = Lists.newArrayList(Splitter.on("_").split(key));
74      if (splitKey.size() == 2) {
75        return splitKey.get(index.getNumeric());
76      } else {
77        return null;
78      }
79    }
80  
81    private enum Index {
82      PRIMARY(1),
83      PARENT(0);
84      
85      private int index;
86  
87      Index(int index) {
88        this.index = index;
89      }
90  
91      public int getNumeric() {
92        return index;
93      }
94    }
95  }
