67      public static void PostQuery(
68              org.apache.oodt.cas.filemgr.structs.Query casQuery) {
69          Vector products = new Vector();
70          try {
71              products = (Vector) client.getProductTypes();
72          } catch (RepositoryManagerException e) {

83      public static void PostQuery(String product,
84              org.apache.oodt.cas.filemgr.structs.Query casQuery) {
85          Vector results = new Vector();
86          ProductType productType = null;
87  
88          try {
89              productType = client.getProductTypeById(product);
90          } catch (RepositoryManagerException e) {
91              System.out.println("Could not access Product Type information");
92              System.exit(-1);
93          }
94  
95          try {
96              results = (Vector) client.query(casQuery, productType);
97          } catch (CatalogException ignore) {
98              System.out.println("Error querying the File Manager");
99              LOG.log(Level.SEVERE, ignore.getMessage());
100             System.exit(-1);
101         }
102 
103         if (results.isEmpty()) {
104             System.out.println("No Products Found Matching This Criteria.");
105         } else {
106             System.out.println("Products Matching Query");
107             for (Object result : results) {
108                 System.out.print(((Product) result).getProductName()
109                                  + "\t");
110                 System.out.println(((Product) result).getProductId());
111             }
112         }
113     }

163     public static void listElements(String prodID) {
164         Vector elements = new Vector();

172                     + prodID);
173         } catch (ValidationLayerException e) {
174             // TODO Auto-generated catch block
175             LOG.log(Level.SEVERE, e.getMessage());
176         }
177 
178         for (Object element : elements) {
179             Element e = (Element) element;
180             System.out.print(e.getElementId() + "\t");
181             System.out.println(e.getElementName());

224     public static Query ParseQuery(String query) {
225         // note that "__FREE__" is a control work for free text searching
226         QueryParser parser = new QueryParser(freeTextBlock, new CASAnalyzer());
227         Query luceneQ = null;
228         try {
229             luceneQ = (Query) parser.parse(query);
230         } catch (ParseException e) {
231             System.out.println("Error parsing query text.");
232             System.exit(-1);
233         }
234         return luceneQ;
235     }

237     public static void GenerateCASQuery(
238             org.apache.oodt.cas.filemgr.structs.Query casQuery,
239             Query luceneQuery) {
240         if (luceneQuery instanceof TermQuery) {
241             Term t = ((TermQuery) luceneQuery).getTerm();
242             if (!t.field().equals(freeTextBlock)) {
243                 casQuery
244                         .addCriterion(new TermQueryCriteria(t.field(), t.text()));
245             }
246         } else if (luceneQuery instanceof PhraseQuery) {
247             Term[] t = ((PhraseQuery) luceneQuery).getTerms();
248             if (!t[0].field().equals(freeTextBlock)) {
249                 for (Term aT : t) {
250                     casQuery.addCriterion(new TermQueryCriteria(aT.field(),
251                         aT.text()));
252                 }
253             }
254         } else if (luceneQuery instanceof TermRangeQuery) {
255             BytesRef startT = ((TermRangeQuery) luceneQuery).getLowerTerm();
256             BytesRef endT = ((TermRangeQuery) luceneQuery).getUpperTerm();
257 
258             //TODO CHECK THIS RANGE!
259             casQuery.addCriterion(new RangeQueryCriteria(((TermRangeQuery) luceneQuery).getField(), startT.utf8ToString(), endT.utf8ToString()));
260         } else if (luceneQuery instanceof BooleanQuery) {
261             List<BooleanClause> clauses = ((BooleanQuery) luceneQuery).clauses();
262             for (BooleanClause clause : clauses) {
263                 GenerateCASQuery(casQuery, (clause).getQuery());
264             }
265         } else {
266             System.out.println("Error Parsing Query");
267             System.exit(-1);
268         }
269     }

271     public static void CommandParser(String command) {
272         StringTokenizer tok = new StringTokenizer(command, " ");
273         int tokCount = tok.countTokens();
274 
275         if (tokCount > 0) {
276             String com = tok.nextToken();
277             if (com.equalsIgnoreCase("get")) {
278                 if (tokCount > 1) {
279                     String subcom = tok.nextToken();
280                     if (subcom.equalsIgnoreCase("products")) {
281                         ListProducts();
282                     } else if (subcom.equalsIgnoreCase("elements")) {
283                         if (tokCount == 4
284                                 && tok.nextToken().equalsIgnoreCase("for")) {
285                             String prodElements = tok.nextToken();
286                             listElements(prodElements);
287                         } else {
288                             if (tokCount == 2) {
289                                 listElements();
290                             } else {
291                                 System.out.println("Error parsing command");
292                             }
293                         }
294                     }
295                 } else {
296                     System.out.println("Error parsing command");
297                 }
298             } else if (com.equalsIgnoreCase("add")) {
299                 if (tokCount == 3 && tok.nextToken().equalsIgnoreCase("filter")) {
300                     setFilter(tok.nextToken());
301                 } else {
302                     System.out.println("Error parsing command");
303                 }
304             } else if (com.equalsIgnoreCase("remove")) {
305                 if (tokCount == 2 && tok.nextToken().equalsIgnoreCase("filter")) {
306                     removeFilter();
307                 } else {
308                     System.out.println("Error parsing command");
309                 }
310             } else if (com.equalsIgnoreCase("help")) {
311                 printHelp();
312             } else if (com.equalsIgnoreCase("exit")
313                     || com.equalsIgnoreCase("quit")) {
314                 System.out.println("Exiting...");
315                 System.exit(0);
316             } else if (com.equalsIgnoreCase("query")) {
317                 StringBuilder query = new StringBuilder();
318                 while (tok.hasMoreTokens()) {
319                     query.append(tok.nextToken()).append(" ");
320                 }
321                 System.out.println("querying for: " + query);
322                 Query parsedQuery = ParseQuery(query.toString());
323                 org.apache.oodt.cas.filemgr.structs.Query casQuery = new org.apache.oodt.cas.filemgr.structs.Query();
324 
325                 GenerateCASQuery(casQuery, parsedQuery);
326                 PostQuery(productFilter, casQuery);
327             }
328         }

331     public static void main(String[] args) {
332 
333         String fileManagerUrl = null;
334         String welcomeMessage = "CatalogSearch v0.1\n";
335         welcomeMessage += "Copyright 2006. California Institute of Technology.\n";
336         String usage = "CatalogSearch --url <url to File Manager service>\n";
337         BufferedReader in = new BufferedReader(new InputStreamReader(System.in));
338 
339         // determine url
340         for (int i = 0; i < args.length; i++) {
341             if (args[i].equals("--url")) {
342                 fileManagerUrl = args[++i];
343             }
344         }
345 
346         if (fileManagerUrl == null) {
347             System.err.println(usage);
348             System.exit(1);
349         }
350 
351         // connect with Filemgr Client
352         boolean clientConnect = true;
353         try {
354             client = new XmlRpcFileManagerClient(new URL(fileManagerUrl));
355         } catch (Exception e) {
356             System.out
357                     .println("Exception when communicating with file manager, errors to follow: message: "
358                             + e.getMessage());
359             clientConnect = false;
360         }
361 
362         if (clientConnect) {
363 
364             System.out.println(welcomeMessage);
365 
366             String command;
367 
368             for (;;) {
369                 System.out.print("CatalogSearch>");
370 
371                 try {
372                     command = in.readLine();
373                     CommandParser(command);
374 
375                 } catch (IOException e) {
376                     System.out.println("\nError Reading Query\n");
377                     System.exit(-1);
378                 }
379             }
380         }
381     }
