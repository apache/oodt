111   /**
112    * <p> Creates a new XmlRpcFileManager with the given metadata store factory, and the given data store factory, on the
113    * given port. </p>
114    *
115    * @param port The web server port to run the XML Rpc server on, defaults to 1999.
116    */
117   public XmlRpcFileManager(int port) throws IOException {
118     webServerPort = port;
119 
120     // start up the web server
121     webServer = new WebServer(webServerPort);
122     webServer.addHandler("filemgr", this);
123     webServer.start();
124 
125     this.loadConfiguration();
126     LOG.log(Level.INFO, "File Manager started by "
127                         + System.getProperty("user.name", "unknown"));
128 
129   }

245   public Map<String, Object> pagedQueryCore(
246       Map<String, Object> queryHash,
247       Map<String, Object> productTypeHash,
248       int pageNum) throws CatalogException {
249     ProductType type = XmlRpcStructFactory
250         .getProductTypeFromXmlRpc(productTypeHash);
251     Query query = XmlRpcStructFactory.getQueryFromXmlRpc(queryHash);
252 
253     ProductPage prodPage;
254 
255     try {
256       prodPage = catalog.pagedQuery(this.getCatalogQuery(query, type), type, pageNum);
257 
258       if (prodPage == null) {
259         prodPage = ProductPage.blankPage();
260       } else {
261         // it is possible here that the underlying catalog did not
262         // set the ProductType
263         // to obey the contract of the File Manager, we need to make
264         // sure its set here
265         setProductType(prodPage.getPageProducts());
266       }
267     } catch (Exception e) {
268       LOG.log(Level.SEVERE, e.getMessage());
269       LOG.log(Level.WARNING,
270           "Catalog exception performing paged query for product type: ["
271           + type.getProductTypeId() + "] query: [" + query
272           + "]: Message: " + e.getMessage());
273       throw new CatalogException(e.getMessage(), e);
274     }
275 
276     return XmlRpcStructFactory.getXmlRpcProductPage(prodPage);
277   }

674   public List<Map<String, Object>> complexQueryCore(
675       Map<String, Object> complexQueryHash) throws CatalogException {
676     try {
677       ComplexQuery complexQuery = XmlRpcStructFactory
678           .getComplexQueryFromXmlRpc(complexQueryHash);
679 
680       // get ProductTypes
681       List<ProductType> productTypes;
682       if (complexQuery.getReducedProductTypeNames() == null) {
683         productTypes = this.repositoryManager.getProductTypes();
684       } else {
685         productTypes = new Vector<ProductType>();
686         for (String productTypeName : complexQuery
687             .getReducedProductTypeNames()) {
688           productTypes.add(this.repositoryManager
689               .getProductTypeByName(productTypeName));
690         }
691       }
692 
693       // get Metadata
694       List<QueryResult> queryResults = new LinkedList<QueryResult>();
695       for (ProductType productType : productTypes) {
696         List<String> productIds = catalog.query(this.getCatalogQuery(
697             complexQuery, productType), productType);
698         for (String productId : productIds) {
699           Product product = catalog.getProductById(productId);
700           product.setProductType(productType);
701           QueryResult qr = new QueryResult(product, this
702               .getReducedMetadata(product, complexQuery
703                   .getReducedMetadata()));
704           qr.setToStringFormat(complexQuery
705               .getToStringResultFormat());
706           queryResults.add(qr);
707         }
708       }
709 
710       LOG.log(Level.INFO, "Query returned " + queryResults.size()
711                           + " results");
712 
713       // filter query results
714       if (complexQuery.getQueryFilter() != null) {
715         queryResults = applyFilterToResults(queryResults, complexQuery
716             .getQueryFilter());
717         LOG.log(Level.INFO, "Filter returned " + queryResults.size()
718                             + " results");
719       }
720 
721       // sort query results
722       if (complexQuery.getSortByMetKey() != null) {
723         queryResults = sortQueryResultList(queryResults, complexQuery
724             .getSortByMetKey());
725       }
726 
727       return XmlRpcStructFactory.getXmlRpcQueryResults(queryResults);
728     } catch (Exception e) {
729       LOG.log(Level.SEVERE, e.getMessage());
730       throw new CatalogException("Failed to perform complex query : "
731                                  + e.getMessage(), e);
732     }
733   }

835   public String ingestProductCore(Map<String, Object> productHash,
836                                   Map<String, String> metadata, boolean clientTransfer)
837       throws
838       CatalogException {
839 
840     Product p = XmlRpcStructFactory.getProductFromXmlRpc(productHash);
841 
842     try {
843       // first, create the product
844       p.setTransferStatus(Product.STATUS_TRANSFER);
845       catalogProduct(p);
846 
847       // now add the metadata
848       Metadata m = new Metadata();
849       m.addMetadata((Map) metadata);
850       Metadata expandedMetdata = addMetadata(p, m);
851 
852       // version the product
853       if (!clientTransfer || (Boolean.getBoolean("org.apache.oodt.cas.filemgr.serverside.versioning"))) {
854         Versioner versioner;
855         try {
856           versioner = GenericFileManagerObjectFactory
857               .getVersionerFromClassName(p.getProductType().getVersioner());
858           if (versioner != null) {
859             versioner.createDataStoreReferences(p, expandedMetdata);
860           }
861         } catch (Exception e) {
862           LOG.log(Level.SEVERE,
863               "ingestProduct: VersioningException when versioning Product: "
864               + p.getProductName() + " with Versioner "
865               + p.getProductType().getVersioner() + ": Message: "
866               + e.getMessage());
867           throw new VersioningException(e);
868         }
869 
870         // add the newly versioned references to the data store
871         addProductReferences(p);
872       }
873 
874       if (!clientTransfer) {
875         LOG.log(Level.FINEST,
876             "File Manager: ingest: no client transfer enabled, "
877             + "server transfering product: [" + p.getProductName() + "]");
878 
879         // now transfer the product
880         try {
881           dataTransfer.transferProduct(p);
882           // now update the product's transfer status in the data store
883           p.setTransferStatus(Product.STATUS_RECEIVED);
884 
885           try {
886             catalog.setProductTransferStatus(p);
887           } catch (CatalogException e) {
888             LOG.log(Level.SEVERE, "ingestProduct: CatalogException "
889                                   + "when updating product transfer status for Product: "
890                                   + p.getProductName() + " Message: " + e.getMessage());
891             throw e;
892           }
893         } catch (Exception e) {
894           LOG.log(Level.SEVERE,
895               "ingestProduct: DataTransferException when transfering Product: "
896               + p.getProductName() + ": Message: " + e.getMessage());
897           throw new DataTransferException(e);
898         }
899       }
900 
901       // that's it!
902       return p.getProductId();
903     } catch (Exception e) {
904       LOG.log(Level.SEVERE, e.getMessage());
905       throw new CatalogException("Error ingesting product [" + p + "] : "
906                                  + e.getMessage(), e);
907     }
908 
909   }

911   public byte[] retrieveFile(String filePath, int offset, int numBytes)
912       throws DataTransferException {
913     FileInputStream is = null;
914     try {
915       byte[] fileData = new byte[numBytes];
916       (is = new FileInputStream(filePath)).skip(offset);
917       int bytesRead = is.read(fileData);
918       if (bytesRead != -1) {
919         byte[] fileDataTruncated = new byte[bytesRead];
920         System.arraycopy(fileData, 0, fileDataTruncated, 0, bytesRead);
921         return fileDataTruncated;
922       } else {
923         return new byte[0];
924       }
925     } catch (Exception e) {
926       LOG.log(Level.SEVERE, "Failed to read '" + numBytes
927                             + "' bytes from file '" + filePath + "' at index '" + offset
928                             + "' : " + e.getMessage(), e);
929       throw new DataTransferException("Failed to read '" + numBytes
930                                       + "' bytes from file '" + filePath + "' at index '" + offset
931                                       + "' : " + e.getMessage(), e);
932     } finally {
933       try {
934         if (is != null) {
935           is.close();
936         }
937       } catch (Exception ignored) {
938       }
939     }
940   }

942   public boolean transferFile(String filePath, byte[] fileData, int offset,
943                               int numBytes) {
944     File outFile = new File(filePath);
945     boolean success = true;
946 
947     FileOutputStream fOut = null;
948 
949     if (outFile.exists()) {
950       try {
951         fOut = new FileOutputStream(outFile, true);
952       } catch (FileNotFoundException e) {
953         LOG.log(Level.SEVERE, e.getMessage());
954         LOG.log(Level.SEVERE,
955             "FileNotFoundException when trying to use RandomAccess file on "
956             + filePath + ": Message: " + e.getMessage());
957         success = false;
958       }
959       finally {
960         try {
961           if (fOut != null) {
962             fOut.close();
963           }
964         } catch (IOException e) {
965           LOG.log(Level.SEVERE, "Could not close file stream", e.getMessage());
966         }
967       }
968     } else {
969       // create the output directory
970       String outFileDirPath = outFile.getAbsolutePath().substring(0,
971           outFile.getAbsolutePath().lastIndexOf("/"));
972       LOG.log(Level.INFO, "Outfile directory: " + outFileDirPath);
973       File outFileDir = new File(outFileDirPath);
974       outFileDir.mkdirs();
975 
976       try {
977         fOut = new FileOutputStream(outFile, false);
978       } catch (FileNotFoundException e) {
979         LOG.log(Level.SEVERE, e.getMessage());
980         LOG.log(Level.SEVERE,
981             "FileNotFoundException when trying to use RandomAccess file on "
982             + filePath + ": Message: " + e.getMessage());
983         success = false;
984       }
985     }
986 
987     if (success) {
988       try {
989         fOut.write(fileData, (int) offset, (int) numBytes);
990       } catch (IOException e) {
991         LOG.log(Level.SEVERE, e.getMessage());
992         LOG.log(Level.SEVERE, "IOException when trying to write file "
993                               + filePath + ": Message: " + e.getMessage());
994         success = false;
995       } finally {
996         try {
997           fOut.close();
998         } catch (Exception ignore) {
999         }
1000 
1001       }
1002     }
1003 
1004     return success;
1005   }

1013   public boolean moveProductCore(Map<String, Object> productHash, String newPath)
1014       throws DataTransferException {
1015 
1016     Product p = XmlRpcStructFactory.getProductFromXmlRpc(productHash);
1017 
1018     // first thing we care about is if the product is flat or heirarchical
1019     if (p.getProductStructure().equals(Product.STRUCTURE_FLAT)) {
1020       // we just need to get its first reference
1021       if (p.getProductReferences() == null || (p.getProductReferences().size() != 1)) {
1022         throw new DataTransferException(
1023             "Flat products must have a single reference: cannot move");
1024       }
1025 
1026       // okay, it's fine to move it
1027       // first, we need to update the data store ref
1028       Reference r = p.getProductReferences().get(0);
1029       if (r.getDataStoreReference().equals(
1030           new File(newPath).toURI().toString())) {
1031         throw new DataTransferException("cannot move product: ["
1032                                         + p.getProductName() + "] to same location: ["
1033                                         + r.getDataStoreReference() + "]");
1034       }
1035 
1036       // create a copy of the current data store path: we'll need it to
1037       // do the data transfer
1038       Reference copyRef = new Reference(r);
1039 
1040       // update the copyRef to have the data store ref as the orig ref
1041       // the the newLoc as the new ref
1042       copyRef.setOrigReference(r.getDataStoreReference());
1043       copyRef.setDataStoreReference(new File(newPath).toURI().toString());
1044 
1045       p.getProductReferences().clear();
1046       p.getProductReferences().add(copyRef);
1047 
1048       // now transfer it
1049       try {
1050         this.dataTransfer.transferProduct(p);
1051       } catch (IOException e) {
1052         throw new DataTransferException(e);
1053       }
1054 
1055       // now delete the original copy
1056       try {
1057         if (!new File(new URI(copyRef.getOrigReference())).delete()) {
1058           LOG.log(Level.WARNING, "Deletion of original file: ["
1059                                  + r.getDataStoreReference()
1060                                  + "] on product move returned false");
1061         }
1062       } catch (URISyntaxException e) {
1063         throw new DataTransferException(
1064             "URI Syntax exception trying to remove original product ref: Message: "
1065             + e.getMessage(), e);
1066       }
1067 
1068       // now save the updated reference
1069       try {
1070         this.catalog.modifyProduct(p);
1071         return true;
1072       } catch (CatalogException e) {
1073         throw new DataTransferException(e.getMessage(), e);
1074       }
1075     } else {
1076       throw new UnsupportedOperationException(
1077           "Moving of heirarhical and stream products not supported yet");
1078     }
1079   }

1273   private Metadata runExtractors(Product product, Metadata metadata) throws CatalogException {
1274     // make sure that the product type definition is present
1275     if (product.getProductType() == null) {
1276       LOG.log(Level.SEVERE, "Failed to run extractor for: " + product.getProductId() + ":" + product
1277           .getProductName() + " product type cannot be null.");
1278       throw new CatalogException("Failed to run extractor for: " + product.getProductId() + ":" + product
1279           .getProductName() + " product type cannot be null.");
1280     }
1281     try {
1282       product.setProductType(repositoryManager.getProductTypeById(product
1283           .getProductType().getProductTypeId()));
1284     } catch (RepositoryManagerException e) {
1285       LOG.log(Level.SEVERE, "Failed to load ProductType " + product
1286           .getProductType().getProductTypeId(), e);
1287       return null;
1288     }
1289 
1290     Metadata met = new Metadata();
1291     met.addMetadata(metadata.getHashTable());
1292 
1293     if (product.getProductType().getExtractors() != null) {
1294       for (ExtractorSpec spec : product.getProductType().getExtractors()) {
1295         FilemgrMetExtractor extractor = GenericFileManagerObjectFactory
1296             .getExtractorFromClassName(spec.getClassName());
1297         if (extractor != null) {
1298           extractor.configure(spec.getConfiguration());
1299         }
1300         LOG.log(Level.INFO, "Running Met Extractor: ["
1301                             + (extractor != null ? extractor.getClass().getName() : null)
1302                             + "] for product type: ["
1303                             + product.getProductType().getName() + "]");
1304         try {
1305           met = extractor != null ? extractor.extractMetadata(product, met) : null;
1306         } catch (MetExtractionException e) {
1307           LOG.log(Level.SEVERE,
1308               "Exception extractor metadata from product: ["
1309               + product.getProductName()
1310               + "]: using extractor: ["
1311               + extractor.getClass().getName()
1312               + "]: Message: " + e.getMessage(), e);
1313         }
1314       }
1315     }
1316 
1317     return met;
1318   }
