47      try {
48        filemgrUrl = new URL(cas);

115   public boolean issueQuery(org.apache.oodt.cas.filemgr.structs.Query query,
116       String productType) {
117     results = new Results();
118     try {
119       ProductType type = client.getProductTypeByName(productType);
120       Vector<Product> products = (Vector<Product>) client.query(query, type);
121       int maxVal = INT;
122       if (products.size() < maxVal) {
123         maxVal = products.size();
124       }
125       for (int i = 0; i < maxVal; i++) {
126         Metadata m = client.getMetadata(products.get(i));
127         results.addProduct(m);
128       }
129     } catch (Exception e) {
130       return false;
131     }
132 
133     return true;
134   }

137 
138     results = new Results();
139     if (queryText.trim().equals("*")) {
140       ProductType type;
141       try {
142         type = client.getProductTypeByName(productType);
143         Vector<Product> products = (Vector<Product>) client
144             .getProductsByProductType(type);
145         int maxVal = INT;

147           maxVal = products.size();
148         }
149         for (int i = 0; i < maxVal; i++) {
150           Metadata m = client.getMetadata(products.get(i));
151           results.addProduct(m);
152         }
153       } catch (Exception e) {
154         return false;
155       }
156 
157     } else {
158 
159       QueryBuilder qb = new QueryBuilder(this);
160       org.apache.oodt.cas.filemgr.structs.Query casQ = qb.ParseQuery(queryText);
161       ProductType type;
162       try {
163         type = client.getProductTypeByName(productType);
164         Vector<Product> products = (Vector<Product>) client.query(casQ, type);
165         int maxVal = INT;
166         if (products.size() < maxVal) {
167           maxVal = products.size();
168         }
169         for (int i = 0; i < maxVal; i++) {
170           Metadata m = client.getMetadata(products.get(i));
171           results.addProduct(m);
172         }
173       } catch (Exception e) {
174         return false;
175       }

177     }

179     return true;
180   }
