106   /**
107    * Constructs a new WorkflowTask.
108    * 
109    * @param taskId
110    *          The identifier for this task.
111    * @param taskName
112    *          The name for this task.
113    * @param taskConfig
114    *          The associated {@link WorkflowTaskConfiguration}.
115    * @param requiredMetFields
116    *          A {@link List} of String met field names that this task requires.
117    * @param preConditions
118    *          The {@link List} of pre-{@link WorkflowCondition}s.
119    * @param postConditions
120    *          The {@link List} of post-{@link WorkflowCondition}s.
121    * @param taskInstanceClassName
122    *          The implementing class name of this WorkflowTask.
123    * @param startDate
124    *          The time that this task started executing.
125    * @param endDate
126    *          The time that this task stopped executing.
127    * @param order
128    *          The order in which this task should be run.
129    */
130   public WorkflowTask(String taskId, String taskName,
131       WorkflowTaskConfiguration taskConfig,
132       List<WorkflowCondition> preConditions,
133       List<WorkflowCondition> postConditions, List<String> requiredMetFields,
134       String taskInstanceClassName, Date startDate, Date endDate, int order) {
135     this.taskId = taskId;
136     this.taskName = taskName;
137     this.taskConfig = taskConfig;
138     this.requiredMetFields = requiredMetFields;
139     this.preConditions = preConditions;
140     this.postConditions = postConditions;
141     this.taskInstanceClassName = taskInstanceClassName;
142     this.startDate = startDate;
143     this.endDate = endDate;
144     this.order = order;
