69    public PrioritizedQueueBasedWorkflowEngine(WorkflowInstanceRepository repo,
70        PrioritySorter prioritizer, WorkflowLifecycleManager lifecycle,
71        EngineRunner runner, WorkflowRepository modelRepo, long querierWaitSeconds) {
72      this.repo = repo;
73      this.prioritizer = prioritizer != null ? new HighestFIFOPrioritySorter(1,
74          50, 1) : prioritizer;
75      this.lifecycle = lifecycle;
76      this.modelRepo = modelRepo;
77      this.processorQueue = new WorkflowProcessorQueue(repo, lifecycle, modelRepo);
78      this.runner = runner;
79      this.runner.setInstanceRepository(repo);
80  
81      // Task QUEUER thread
82      TaskQuerier querier = new TaskQuerier(processorQueue, this.prioritizer,
83          this.repo, querierWaitSeconds);
84      queuerThread = new Thread(querier);
85      queuerThread.start();
86  
87      // Task Runner thread
88      runnerThread = new Thread(new TaskRunner(querier, runner));
89      runnerThread.start();
90  
91    }

104   @Override
105   public WorkflowInstance startWorkflow(Workflow workflow, Metadata metadata)
106       throws EngineException {
107     // TODO Auto-generated method stub
108 
109     // looks like the work to do here is
110     // create a new WorkflowInstance
111     // create a new WorkflowProcessor around it
112     // set it in Queued status
113     // commit it to workflow instance repo and it will get picked up
114 
115     WorkflowInstance inst = new WorkflowInstance();
116     inst.setParentChildWorkflow(workflow instanceof ParentChildWorkflow ? (ParentChildWorkflow) workflow
117         : new ParentChildWorkflow(workflow));
118     inst.setStartDate(Calendar.getInstance().getTime());
119     inst.setCurrentTaskId(workflow.getTasks().get(0).getTaskId());
120     inst.setSharedContext(metadata);
121     inst.setPriority(Priority.getDefault()); // FIXME: this should be sensed or
122                                              // passed in
123     WorkflowLifecycle cycle = getLifecycleForWorkflow(workflow);
124     WorkflowState state = cycle.createState("Null", "initial",
125         "Workflow created by Engine.");
126     inst.setState(state);
127     persist(inst);
128     return inst;
129   }

245   private synchronized void persist(WorkflowInstance inst) throws EngineException {
246     try {
247       if (inst.getId() == null || (inst.getId().equals(""))) {
248         // we have to persist it by adding it
249         // rather than updating it
250         repo.addWorkflowInstance(inst);
251       } else {
252         // persist by update
253         repo.updateWorkflowInstance(inst);
254       }
255     } catch (InstanceRepositoryException e) {
256       LOG.log(Level.SEVERE, e.getMessage());
257       throw new EngineException(e.getMessage());
258     }
259   }

263         .getLifecycleForWorkflow(workflow) : lifecycle.getDefaultLifecycle();
264   }
