103   private JTable createTable(final ViewState state) {
104     JTable table;
105     final ModelGraph selected = state.getSelected();
106     if (selected != null) {
107       final Vector<Vector<String>> rows = new Vector<Vector<String>>();
108       ConcurrentHashMap<String, String> keyToGroupMap = new ConcurrentHashMap<String, String>();
109       Metadata staticMet = selected.getModel().getStaticMetadata();
110       Metadata inheritedMet = selected.getInheritedStaticMetadata(state);
111       Metadata completeMet = new Metadata();
112       if (staticMet != null) {
113         completeMet.replaceMetadata(staticMet.getSubMetadata(state
114             .getCurrentMetGroup()));
115       }
116       if (selected.getModel().getExtendsConfig() != null) {
117         for (String configGroup : selected.getModel().getExtendsConfig()) {
118           Metadata extendsMetadata = state.getGlobalConfigGroups()
119               .get(configGroup).getMetadata()
120               .getSubMetadata(state.getCurrentMetGroup());
121           for (String key : extendsMetadata.getAllKeys()) {
122             if (!completeMet.containsKey(key)) {
123               keyToGroupMap.put(key, configGroup);
124               completeMet.replaceMetadata(key,
125                   extendsMetadata.getAllMetadata(key));
126             }
127           }
128         }
129       }
130       if (inheritedMet != null) {
131         Metadata inheritedMetadata = inheritedMet.getSubMetadata(state
132             .getCurrentMetGroup());
133         for (String key : inheritedMetadata.getAllKeys()) {
134           if (!completeMet.containsKey(key)) {
135             keyToGroupMap.put(key, "__inherited__");
136             completeMet.replaceMetadata(key,
137                 inheritedMetadata.getAllMetadata(key));
138           }
139         }
140       }
141       List<String> keys = completeMet.getAllKeys();
142       Collections.sort(keys);
143       for (String key : keys) {
144         if (key.endsWith("/envReplace")) {
145           continue;
146         }
147         String values = StringUtils.join(completeMet.getAllMetadata(key), ",");
148         Vector<String> row = new Vector<String>();
149         row.add(keyToGroupMap.get(key));
150         row.add(key);
151         row.add(values);
152         row.add(Boolean.toString(Boolean.parseBoolean(completeMet
153             .getMetadata(key + "/envReplace"))));
154         rows.add(row);
155       }
156       table = new JTable();
157       table.setModel(new AbstractTableModel() {
158         public String getColumnName(int col) {
159           switch (col) {
160           case 0:
161             return "group";
162           case 1:
163             return "key";
164           case 2:
165             return "values";
166           case 3:
167             return "envReplace";
168           default:
169             return null;
170           }
171         }
172 
173         public int getRowCount() {
174           return rows.size() + 1;
175         }
176 
177         public int getColumnCount() {
178           return 4;
179         }
180 
181         public Object getValueAt(int row, int col) {
182           if (row >= rows.size()) {
183             return null;
184           }
185           String value = rows.get(row).get(col);
186           if (value == null && col == 3) {
187             return "false";
188           }
189           if (value == null && col == 0) {
190             return "__local__";
191           }
192           return value;
193         }
194 
195         public boolean isCellEditable(int row, int col) {
196           if (row >= rows.size()) {
197             return selected.getModel().getStaticMetadata()
198                            .containsGroup(state.getCurrentMetGroup());
199           }
200           if (col == 0) {
201             return false;
202           }
203           String key = rows.get(row).get(1);
204           return key == null
205                  || (selected.getModel().getStaticMetadata() != null && selected
206               .getModel().getStaticMetadata()
207               .containsKey(getKey(key, state)));
208         }
209 
210         public void setValueAt(Object value, int row, int col) {
211           if (row >= rows.size()) {
212             Vector<String> newRow = new Vector<String>(Arrays
213                 .asList(new String[] { null, null, null, null }));
214             newRow.add(col, (String) value);
215             rows.add(newRow);
216           } else {
217             Vector<String> rowValues = rows.get(row);
218             rowValues.add(col, (String) value);
219             rowValues.remove(col + 1);
220           }
221           this.fireTableCellUpdated(row, col);
222         }
223 
224       });
225       MyTableListener tableListener = new MyTableListener(state);
226       table.getModel().addTableModelListener(tableListener);
227       table.getSelectionModel().addListSelectionListener(tableListener);
228     } else {
229       table = new JTable(new Vector<Vector<String>>(), new Vector<String>(
230           Arrays.asList(new String[] { "key", "values", "envReplace" })));
231     }
232 
233     table.setSelectionBackground(Color.cyan);
234     table.setAutoResizeMode(JTable.AUTO_RESIZE_OFF);
235     TableCellRenderer cellRenderer = new TableCellRenderer() {
236 
237       public Component getTableCellRendererComponent(JTable table,
238           Object value, boolean isSelected, boolean hasFocus, int row,
239           int column) {
240         JLabel field = new JLabel((String) value);
241         if (column == 0) {
242           field.setForeground(Color.gray);
243         } else {
244           if (isSelected) {
245             field.setBorder(new EtchedBorder(1));
246           }
247           if (table.isCellEditable(row, 1)) {
248             field.setForeground(Color.black);
249           } else {
250             field.setForeground(Color.gray);
251           }
252         }
253         return field;
254       }
255 
256     };
257     TableColumn groupCol = table.getColumnModel().getColumn(0);
258     groupCol.setPreferredWidth(75);
259     groupCol.setCellRenderer(cellRenderer);
260     TableColumn keyCol = table.getColumnModel().getColumn(1);
261     keyCol.setPreferredWidth(200);
262     keyCol.setCellRenderer(cellRenderer);
263     TableColumn valuesCol = table.getColumnModel().getColumn(2);
264     valuesCol.setPreferredWidth(300);
265     valuesCol.setCellRenderer(cellRenderer);
266     TableColumn envReplaceCol = table.getColumnModel().getColumn(3);
267     envReplaceCol.setPreferredWidth(75);
268     envReplaceCol.setCellRenderer(cellRenderer);
269 
270     table.addMouseListener(new MouseListener() {
271       public void mouseClicked(MouseEvent e) {
272         if (e.getButton() == MouseEvent.BUTTON3
273             && DefaultPropView.this.table.getSelectedRow() != -1) {
274           int row = DefaultPropView.this.table.getSelectedRow();
275           String key = getKey(
276               (String) DefaultPropView.this.table.getValueAt(row, 1), state);
277           Metadata staticMet = state.getSelected().getModel()
278               .getStaticMetadata();
279           override.setVisible(staticMet == null || !staticMet.containsKey(key));
280           delete.setVisible(staticMet != null && staticMet.containsKey(key));
281           tableMenu.show(DefaultPropView.this.table, e.getX(), e.getY());
282         }
283       }
284 
285       public void mouseEntered(MouseEvent e) {
286       }
287 
288       public void mouseExited(MouseEvent e) {
289       }
290 
291       public void mousePressed(MouseEvent e) {
292       }
293 
294       public void mouseReleased(MouseEvent e) {
295       }
296     });
297 
298     return table;
299   }

301   @Override
302   public void refreshView(final ViewState state) {
303     this.removeAll();
304 
305     tableMenu = new JPopupMenu("TableMenu");
306     this.add(tableMenu);
307     override = new JMenuItem(OVERRIDE);
308     override.addActionListener(new ActionListener() {
309       public void actionPerformed(ActionEvent e) {
310         int row = DefaultPropView.this.table.getSelectedRow();
311         String key = getKey(
312             (String) DefaultPropView.this.table.getValueAt(row, 1), state);
313         Metadata staticMet = state.getSelected().getModel().getStaticMetadata();
314         if (staticMet == null) {
315           staticMet = new Metadata();
316         }
317         if (e.getActionCommand().equals(OVERRIDE)) {
318           if (!staticMet.containsKey(key)) {
319             staticMet.addMetadata(key,
320                 (String) DefaultPropView.this.table.getValueAt(row, 2));
321             String envReplace = (String) DefaultPropView.this.table.getValueAt(
322                 row, 3);
323             if (Boolean.valueOf(envReplace)) {
324               staticMet.addMetadata(key + "/envReplace", envReplace);
325             }
326             state.getSelected().getModel().setStaticMetadata(staticMet);
327             DefaultPropView.this.notifyListeners();
328           }
329         }
330       }
331     });
332     delete = new JMenuItem(DELETE);
333     delete.addActionListener(new ActionListener() {
334 
335       public void actionPerformed(ActionEvent e) {
336         int row = DefaultPropView.this.table.getSelectedRow();
337         String key = getKey(
338             (String) DefaultPropView.this.table.getValueAt(row, 1), state);
339         Metadata staticMet = state.getSelected().getModel().getStaticMetadata();
340         if (staticMet == null) {
341           staticMet = new Metadata();
342         }
343         staticMet.removeMetadata(key);
344         staticMet.removeMetadata(key + "/envReplace");
345         state.getSelected().getModel().setStaticMetadata(staticMet);
346         DefaultPropView.this.notifyListeners();
347       }
348 
349     });
350     tableMenu.add(override);
351     tableMenu.add(delete);
352 
353     if (state.getSelected() != null) {
354       JPanel masterPanel = new JPanel();
355       masterPanel.setLayout(new BoxLayout(masterPanel, BoxLayout.Y_AXIS));
356       masterPanel.add(this.getModelIdPanel(state.getSelected(), state));
357       masterPanel.add(this.getModelNamePanel(state.getSelected(), state));
358       if (!state.getSelected().getModel().isParentType()) {
359         masterPanel.add(this.getInstanceClassPanel(state.getSelected(), state));
360       }
361       masterPanel.add(this.getExecutionTypePanel(state.getSelected(), state));
362       masterPanel.add(this.getPriorityPanel(state));
363       masterPanel.add(this.getExecusedIds(state.getSelected()));
364       if (state.getSelected().getModel().getExecutionType().equals("condition")) {
365         masterPanel.add(this.getTimeout(state.getSelected(), state));
366         masterPanel.add(this.getOptional(state.getSelected(), state));
367       }
368       JScrollPane scrollPane = new JScrollPane(table = this.createTable(state),
369           JScrollPane.VERTICAL_SCROLLBAR_ALWAYS,
370           JScrollPane.HORIZONTAL_SCROLLBAR_ALWAYS);
371       scrollPane.getHorizontalScrollBar().setUnitIncrement(10);
372       scrollPane.getVerticalScrollBar().setUnitIncrement(10);
373       JPanel panel = new JPanel();
374       panel.setLayout(new BorderLayout());
375       panel.setBorder(new EtchedBorder());
376       final JLabel metLabel = new JLabel("Static Metadata");
377       metLabel.setAlignmentX(Component.CENTER_ALIGNMENT);
378       final JLabel extendsLabel = new JLabel("<extends>");
379       extendsLabel.setFont(new Font("Serif", Font.PLAIN, 10));
380       extendsLabel.setAlignmentX(Component.CENTER_ALIGNMENT);
381       extendsLabel.addMouseListener(new MouseListener() {
382

504         private JList createJList(DefaultListModel model,
505             final List<String> list) {
506           for (String value : list) {
507             model.addElement(value);
508           }
509           JList jList = new JList(model);
510           jList.setSelectionMode(ListSelectionModel.SINGLE_INTERVAL_SELECTION);
511           jList.setLayoutOrientation(JList.VERTICAL);
512           return jList;
513         }
514       });
515       JLabel metGroupLabel = new JLabel("(Sub-Group: "
516           + (state.getCurrentMetGroup() != null ? state.getCurrentMetGroup()
517               : "<base>") + ")");
518       metGroupLabel.setAlignmentX(Component.CENTER_ALIGNMENT);
519       JPanel labelPanel = new JPanel();
520       labelPanel.setLayout(new BoxLayout(labelPanel, BoxLayout.Y_AXIS));
521       JPanel top = new JPanel();
522       top.setLayout(new BoxLayout(top, BoxLayout.Y_AXIS));
523       top.add(extendsLabel);
524       top.add(metLabel);
525       labelPanel.add(top);
526       labelPanel.add(metGroupLabel);
527       panel.add(labelPanel, BorderLayout.NORTH);
528       panel.add(scrollPane, BorderLayout.CENTER);
529       masterPanel.add(panel);
530       this.add(masterPanel);
531     } else {
532       this.add(new JPanel());
533     }
534     this.revalidate();
535   }

537   private JPanel getTimeout(final ModelGraph graph, final ViewState state) {
538     JPanel panel = new JPanel();
539     panel.setLayout(new BorderLayout());
540     panel.setBorder(new EtchedBorder());
541     panel.add(new JLabel("Timeout:"), BorderLayout.NORTH);
542     JTextField field = new JTextField(String.valueOf(graph.getModel()
543         .getTimeout()), 50);
544     field.addActionListener(new ActionListener() {
545 
546       public void actionPerformed(ActionEvent e) {
547         if (!graph.getModel().getModelId().equals(e.getActionCommand())) {
548           graph.getModel().setTimeout(Long.valueOf(
549               e.getActionCommand() != null && 
550               !e.getActionCommand().equals("") ? 
551                   e.getActionCommand():"-1"));
552           DefaultPropView.this.notifyListeners();
553           DefaultPropView.this.refreshView(state);
554         }
555       }
556 
557     });
558     panel.add(field, BorderLayout.CENTER);
559     return panel;
560   }

629   private JPanel getPriorityPanel(final ViewState state) {
630     JPanel panel = new JPanel();
631     panel.setBorder(new EtchedBorder());
632     panel.setLayout(new BorderLayout());
633     panel.add(new JLabel("Priority:  "), BorderLayout.WEST);
634     final JLabel priorityLabel = new JLabel(String.valueOf(DEFAULT_PRIORITY));
635     panel.add(priorityLabel, BorderLayout.CENTER);
636     JSlider slider = new JSlider(0, 100, (int) 5 * 10);
637     slider.setMajorTickSpacing(10);
638     slider.setMinorTickSpacing(1);
639     slider.setPaintLabels(true);
640     slider.setPaintTicks(true);
641     slider.setSnapToTicks(false);
642     Format f = new DecimalFormat("0.0");
643     Hashtable<Integer, JComponent> labels = new Hashtable<Integer, JComponent>();
644     for (int i = 0; i <= 10; i += 2) {
645       JLabel label = new JLabel(f.format(i));
646       label.setFont(label.getFont().deriveFont(Font.PLAIN));
647       labels.put(i * 10, label);
648     }
649     slider.setLabelTable(labels);
650     slider.addChangeListener(new ChangeListener() {
651 
652       public void stateChanged(ChangeEvent e) {
653         double value = ((JSlider) e.getSource()).getValue() / 10.0;
654         priorityLabel.setText(value + "");
655         priorityLabel.revalidate();
656         if (!((JSlider) e.getSource()).getValueIsAdjusting()) {
657           // FIXME: deal with priorities
658           DefaultPropView.this.notifyListeners();
659         }
660       }
661 
662     });
663 
664     panel.add(slider, BorderLayout.SOUTH);
665     return panel;
