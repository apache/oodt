21   * <p>
22   * A Jsp utility that processes a {@link PolicyAwareWorkflowTask} that is given
23   * in the form of a HttpServletRequestWrapper#getParameterMap().
24   * </p>.
25   */
26  public class TaskRequestFactory implements TaskRequestProcessorMetKeys {
27  
28      private TaskRequestFactory() throws InstantiationException {
29          throw new InstantiationException("Don't construct factories!");
30      }

32      /**
33       * Processes a given {@link PolicyAwareWorkflowTask} and unmarshalls it out
34       * of the given {@link Map}.
35       * 
36       * @param paramMap
37       *            A parameter {@link Map}, generated by a call to
38       *            {@link HttpServletRequestWrapper#getParameterMap()}.
39       * @return An unmarshalled, {@link PolicyAwareWorkflowTask}.
40       */
41      public static PolicyAwareWorkflowTask getTask(Map paramMap) {
42          PolicyAwareWorkflowTask task = new PolicyAwareWorkflowTask();
43          task.setTaskName(getFirst((String[]) paramMap.get(TASK_NAME)));
44          task.setTaskId(getFirst((String[]) paramMap.get(UPDATE_TASK_ID)));
45          task.setPolicyDirPath(getFirst((String[]) paramMap
46                  .get(TASK_POLICY_DIR_PATH)));
47          task.setTaskInstanceClassName(getFirst((String[]) paramMap
48                  .get(TASK_CLASS_NAME)));
49          int reqMetFieldCount = Integer.parseInt(getFirst((String[]) paramMap
50                  .get(REQ_MET_FIELDS_COUNT)));
51          if (reqMetFieldCount > 0) {
52              List metFields = new Vector();
53  
54              for (int i = 0; i < reqMetFieldCount; i++) {
55                  metFields.add(getFirst((String[]) paramMap
56                          .get(REQ_MET_FIELD_TAGBASE + (i))));
57              }
58  
59              task.setRequiredMetFields(metFields);
60          } else
61              task.setRequiredMetFields(Collections.EMPTY_LIST);
62  
63          String[] selectedCondIds = (String[]) paramMap.get(TASK_CONDITION_IDS);
64  
65          if (selectedCondIds != null && selectedCondIds.length > 0) {
66              List conditions = new Vector();
67              for (int i = 0; i < selectedCondIds.length; i++) {
68                  WorkflowCondition cond = new WorkflowCondition();
69                  cond.setConditionId(selectedCondIds[i]);
70                  conditions.add(cond);
71              }
72  
73              task.setConditions(conditions);
74          } else
75              task.setConditions(Collections.EMPTY_LIST);
76  
77          int numConfigProperties = Integer.parseInt(getFirst((String[]) paramMap
78                  .get(TASK_CONFIG_FIELD_COUNT)));
79          if (numConfigProperties > 0) {
80              EnvSavingConfiguration config = new EnvSavingConfiguration();
81              for (int i = 0; i < numConfigProperties; i++) {
82                  String propName = getFirst((String[]) paramMap
83                          .get(TASK_CONFIG_PROPNAME_BASE + (i)));
84                  String propVal = getFirst((String[]) paramMap
85                          .get(TASK_CONFIG_PROPVAL_BASE + (i)));
86                  config.addConfigProperty(propName, propVal);
87              }
88  
89              task.setTaskConfig(config);
90          }
91  
92          return task;
93      }
94  
95      /**
96       * Merges the <code>newTask</code> {@link PolicyAwareWorkflowTask} into
97       * the <code>orig</code> {@link PolicyAwareWorkflowTask}, taking care not
98       * to overwrite the {@link EnvSavingConfiguration}, and only its
99       * {@link EnvSavingConfiguration#getProperties()}.
100      * 
101      * @param newTask
102      *            The new {@link PolicyAwareWorkflowTask}, whose values will
103      *            overwrite those of <code>orig</code>.
104      * @param orig
105      *            The destination for the new values provided by
106      *            <code>newTask</code>.
107      */
108     public static void merge(PolicyAwareWorkflowTask newTask,
109             PolicyAwareWorkflowTask orig) {
110         orig.setTaskId(newTask.getTaskId());
111         orig.setTaskName(newTask.getTaskName());
112         orig.setTaskInstanceClassName(newTask.getTaskInstanceClassName());
113         orig.setPolicyDirPath(newTask.getPolicyDirPath());
114         orig.setConditions(newTask.getConditions());
115         orig.setRequiredMetFields(newTask.getRequiredMetFields());
116         // only copy through properties, don't reset config, because
117         // we need to save the orig envReplace info
118         orig.getTaskConfig().getProperties().clear();
119         orig.getTaskConfig().getProperties().putAll(
120                 newTask.getTaskConfig().getProperties());
121     }
122 
123     private static String getFirst(String[] strArray) {
124         if (strArray != null && strArray.length >= 1) {
125             return strArray[0];
126         } else
127             return null;
128     }
