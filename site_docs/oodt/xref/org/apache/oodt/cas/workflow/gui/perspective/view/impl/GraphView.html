497   private void hideDummyNodes(Map nested) {
498     for (Object cell : nested.keySet()) {
499       if (cell instanceof DefaultEdge) {
500         // do nothing
501       } else if (cell instanceof DefaultGraphCell) {
502         if (GuiUtils.isDummyNode((ModelNode) ((DefaultGraphCell) cell)
503             .getUserObject())) {
504           ((Map<Object, Object>) nested.get(cell)).put("opaque", false);
505           ((Map<Object, Object>) nested.get(cell)).put("backgroundColor",
506               Color.white);
507         }
508       }
509     }
510   }

512   private Map shiftMap(Map nested, ViewState state) {
513     Map shiftedNested = new Hashtable(nested);
514     for (Object obj : shiftedNested.entrySet()) {
515       Entry entry = (Entry) obj;
516       if (entry.getKey() instanceof DefaultEdge) {
517         ArrayList<Point2D.Double> points = (ArrayList<Point2D.Double>) ((Map<Object, Object>) entry
518             .getValue()).get("points");
519         ArrayList<Point2D.Double> newPoints = new ArrayList<Point2D.Double>();
520         Point shift = this.getShift(state, (DefaultGraphCell) entry.getKey(),
521             nested);
522         for (Point2D.Double point : points) {
523           newPoints
524               .add(new Point2D.Double(point.x + shift.x, point.y + shift.y));
525         }
526         ((Map<Object, Object>) entry.getValue()).put("points", newPoints);
527       } else if (entry.getKey() instanceof DefaultGraphCell) {
528         DefaultGraphCell cell = (DefaultGraphCell) entry.getKey();
529         Rectangle2D bounds = (Rectangle2D) ((Map<Object, Object>) entry
530             .getValue()).get("bounds");
531         Point shift = this.getShift(state, cell, nested);
532         AttributeMap.SerializableRectangle2D newBounds = new AttributeMap.SerializableRectangle2D(
533             bounds.getX() + shift.x, bounds.getY() + shift.y,
534             bounds.getWidth(), bounds.getHeight());
535         Map<Object, Object> newMap = new Hashtable<Object, Object>(
536             (Map<Object, Object>) entry.getValue());
537         newMap.put("bounds", newBounds);
538         shiftedNested.put(cell, newMap);
539       }
540     }
541     return shiftedNested;
542   }

544   private void ensureNoOverlap(Map nested, ViewState state) {
545     boolean changed;
546     do {
547       changed = false;
548       for (int i = 0; i < state.getGraphs().size(); i++) {
549         ModelGraph currentGraph = state.getGraphs().get(i);
550         if (this.ensureNoInternalOverlap(currentGraph, nested)) {
551           changed = true;
552         }
553         DefaultGraphCell currentCell = this.m_jgAdapter
554             .getVertexCell(currentGraph.getModel());
555         Rectangle2D currentBounds = (Rectangle2D) ((Map) nested
556             .get(currentCell)).get(GraphConstants.BOUNDS);
557         Point currentShift = this.getShift(state, currentCell, nested);
558         Rectangle currentShiftBounds = new Rectangle(
559             (int) (currentBounds.getX() + currentShift.getX()),
560             (int) (currentBounds.getY() + currentShift.getY()),
561             (int) currentBounds.getWidth(), (int) currentBounds.getHeight());
562         for (int j = 0; j < state.getGraphs().size(); j++) {
563           if (i == j) {
564             continue;
565           }
566           ModelGraph graph = state.getGraphs().get(j);
567           DefaultGraphCell cell = this.m_jgAdapter.getVertexCell(graph
568               .getModel());
569           Rectangle2D bounds = (Rectangle2D) ((Map) nested.get(cell))
570               .get(GraphConstants.BOUNDS);
571           Point shift = this.getShift(state, cell, nested);
572           Rectangle shiftBounds = new Rectangle(
573               (int) (bounds.getX() + shift.getX()),
574               (int) (bounds.getY() + shift.getY()), (int) bounds.getWidth(),
575               (int) bounds.getHeight());
576           if (currentShiftBounds.intersects(shiftBounds)) {
577             changed = true;
578             if (currentShiftBounds.getY() < shiftBounds.getY()) {
579               Rectangle intersection = currentShiftBounds
580                   .intersection(shiftBounds);
581               if (currentShiftBounds.getY() + currentShiftBounds.getHeight() > shiftBounds
582                   .getY() + shiftBounds.getHeight()) {
583                 this.setShift(state, graph,
584                     new Point((int) (currentShiftBounds.getX()
585                         + currentShiftBounds.getWidth() + 20.0),
586                         (int) shiftBounds.getY()));
587               } else {
588                 this.setShift(
589                     state,
590                     graph,
591                     new Point((int) shiftBounds.getX(), (int) (shiftBounds
592                         .getY() + intersection.getHeight() + 20.0)));
593               }
594             } else {
595               Rectangle intersection = shiftBounds
596                   .intersection(currentShiftBounds);
597               if (shiftBounds.getY() + shiftBounds.getHeight() > currentShiftBounds
598                   .getY() + currentShiftBounds.getHeight()) {
599                 this.setShift(
600                     state,

627   private boolean ensureNoInternalOverlap(final ModelGraph graph,
628       final Map nested) {
629     boolean changed = false;
630     if (graph.getChildren().size() > 1) {
631       List<ModelGraph> sortedChildren = new Vector<ModelGraph>(
632           graph.getChildren());
633       Collections.sort(sortedChildren, new Comparator<ModelGraph>() {
634 
635         public int compare(ModelGraph o1, ModelGraph o2) {
636           DefaultGraphCell child1Cell = GraphView.this.m_jgAdapter
637               .getVertexCell(o1.getModel());
638           DefaultGraphCell child2Cell = GraphView.this.m_jgAdapter
639               .getVertexCell(o2.getModel());
640           Rectangle2D child1Bounds = (Rectangle2D) ((Map) nested
641               .get(child1Cell)).get(GraphConstants.BOUNDS);
642           Rectangle2D child2Bounds = (Rectangle2D) ((Map) nested
643               .get(child2Cell)).get(GraphConstants.BOUNDS);
644           if (graph.getModel().getExecutionType().equals("parallel")) {
645             return Double.compare(child1Bounds.getMaxY(),
646                 child2Bounds.getMaxY());
647           } else {
648             return Double.compare(child1Bounds.getX(), child2Bounds.getX());
649           }
650         }
651 
652       });
653       changed = ensureNoInternalOverlap(sortedChildren.get(0), nested);
654       Rectangle2D graphRectangle = (Rectangle2D) ((Map) nested
655           .get(this.m_jgAdapter.getVertexCell(sortedChildren.get(0).getModel())))
656           .get(GraphConstants.BOUNDS);
657       for (int i = 1; i < sortedChildren.size(); i++) {
658         ModelGraph child2 = sortedChildren.get(i);
659         if (ensureNoInternalOverlap(child2, nested)) {
660           changed = true;
661         }
662         DefaultGraphCell child2Cell = this.m_jgAdapter.getVertexCell(child2
663             .getModel());
664         for (int j = i - 1; j >= 0; j--) {
665           ModelGraph child1 = sortedChildren.get(j);
666           DefaultGraphCell child1Cell = this.m_jgAdapter.getVertexCell(child1
667               .getModel());
668           Rectangle2D child1Bounds = (Rectangle2D) ((Map) nested
669               .get(child1Cell)).get(GraphConstants.BOUNDS);
670           Rectangle2D child2Bounds = (Rectangle2D) ((Map) nested
671               .get(child2Cell)).get(GraphConstants.BOUNDS);
672           if (child1Bounds.intersects(child2Bounds)) {
673             changed = true;
674             if (graph.getModel().getExecutionType().equals("parallel")) {
675               ((Map) nested.get(child2Cell)).put(GraphConstants.BOUNDS,
676                   new AttributeMap.SerializableRectangle2D(child2Bounds.getX(),
677                       child1Bounds.getMaxY() + 20.0, child2Bounds.getWidth(),
678                       child2Bounds.getHeight()));
679               this.shift(child2.getChildren(), nested, 0,
680                   child1Bounds.getMaxY() + 20.0 - child2Bounds.getY());
681             } else {
682               ((Map) nested.get(child2Cell)).put(
683                   GraphConstants.BOUNDS,
684                   new AttributeMap.SerializableRectangle2D(child1Bounds
685                       .getMaxX() + 20.0, child2Bounds.getY(), child2Bounds
686                       .getWidth(), child2Bounds.getHeight()));
687               this.shift(child2.getChildren(), nested, child1Bounds.getMaxX()
688                   + 20.0 - child2Bounds.getX(), 0);
689             }
690           }
691         }
692         graphRectangle = graphRectangle.createUnion((Rectangle2D) ((Map) nested
693             .get(child2Cell)).get(GraphConstants.BOUNDS));
694       }
695       ((Map) nested.get(this.m_jgAdapter.getVertexCell(graph.getModel()))).put(
696           GraphConstants.BOUNDS, new AttributeMap.SerializableRectangle2D(
697               graphRectangle.getX() - 5, graphRectangle.getY() - 20,
698               graphRectangle.getWidth() + 10, graphRectangle.getHeight() + 25));
699     }
700     return changed;
701   }

724   private DefaultGraphCell addGroups(ModelGraph modelGraph, Map nested,
725       ViewState state) {
726     if (modelGraph.getModel().isParentType()) {
727       double top_x = Double.MAX_VALUE, top_y = Double.MAX_VALUE, bottom_x = 0.0, bottom_y = 0.0;
728       this.directedGraph.addVertex(modelGraph.getModel());
729       DefaultGraphCell modelCell = this.m_jgAdapter.getVertexCell(modelGraph
730           .getModel());
731 
732 
733       ConcurrentHashMap<Object, Object> map = new ConcurrentHashMap<Object, Object>();
734       for (int i = 0; i < modelGraph.getChildren().size(); i++) {
735         ModelGraph child = modelGraph.getChildren().get(i);
736         DefaultGraphCell curCell = addGroups(child, nested, state);
737         Rectangle2D bounds = (Rectangle2D) ((Map<Object, Object>) nested
738             .get(curCell)).get("bounds");
739         if (bounds.getX() < top_x) {
740           top_x = bounds.getX();
741         }
742         if (bounds.getY() < top_y) {
743           top_y = bounds.getY();
744         }
745         if (bounds.getMaxX() > bottom_x) {
746           bottom_x = bounds.getMaxX();
747         }
748         if (bounds.getMaxY() > bottom_y) {
749           bottom_y = bounds.getMaxY();
750         }
751       }
752 
753       map.put(GraphConstants.BOUNDS, new AttributeMap.SerializableRectangle2D(
754           top_x - 5, top_y - 20, bottom_x - top_x + 10, bottom_y - top_y + 25));
755       map.put(GraphConstants.FOREGROUND, Color.black);
756       if (modelGraph.getModel().isRef()) {
757         map.put(GraphConstants.BACKGROUND, Color.lightGray);
758       } else {
759         map.put(GraphConstants.BACKGROUND, Color.white);
760       }
761       if (modelGraph.isExcused()) {
762         map.put(GraphConstants.GRADIENTCOLOR, Color.gray);
763       }
764       map.put(GraphConstants.HORIZONTAL_ALIGNMENT, SwingConstants.LEFT);
765       map.put(GraphConstants.VERTICAL_ALIGNMENT, SwingConstants.TOP);
766       map.put(GraphConstants.BORDER, new LineBorder(modelGraph.getModel()
767           .getColor(), 1));
768       jgraph.getGraphLayoutCache().toBack(new Object[] { modelCell });
769       nested.put(modelCell, map);
770       return modelCell;
771     }
772     DefaultGraphCell cell = this.m_jgAdapter.getVertexCell(modelGraph
773         .getModel());
774     ((Map<Object, Object>) nested.get(cell)).put(GraphConstants.FOREGROUND,
775         Color.black);
776     if (modelGraph.isExcused()) {
777       ((Map<Object, Object>) nested.get(cell)).put(
778           GraphConstants.GRADIENTCOLOR, Color.gray);
779     } else {
780       ((Map<Object, Object>) nested.get(cell)).put(
781           GraphConstants.GRADIENTCOLOR, Color.white);
782     }
783     if (!((ModelNode) ((DefaultGraphCell) cell).getUserObject()).isRef()) {
784       ((Map<Object, Object>) nested.get(cell)).put(GraphConstants.BACKGROUND,
785           modelGraph.getModel().getColor());
786     } else {
787       ((Map<Object, Object>) nested.get(cell)).put(GraphConstants.BACKGROUND,
788           Color.lightGray);
789     }
790     return cell;
791   }
