62   * <p>
63   * A {@link WorkflowRepository} that loads events, {@link Workflow}s,
64   * {@link WorkflowTask}s and {@link WorkflowCondition}s from specialized xml
65   * files. The WorkflowRepository is given an initial set of seed directory uris,
66   * where it looks for the following files:
67   * 
68   * <ul>
69   * <li>conditions.xml - defines workflow pre conditions</li>
70   * <li>tasks.xml - defines workflow tasks</li>
71   * <li>*.workflow.xml - individual workflow xml files specifying a single
72   * workflow</li>
73   * <li>events.xml - maps available workflows to event names</li>
74   * </ul>
75   * 
76   * All of the WorkflowTasks, WorkflowConditions and Workflows themselves are
77   * cached in memory by their ids (which are typically URNs).
78   * </p>
79   */
80  public class XMLWorkflowRepository implements WorkflowRepository {

273     @Override
274     public String addTask(WorkflowTask task) throws RepositoryException {
275       // check its conditions
276       if(task.getPreConditions() != null && task.getPreConditions().size() > 0){
277         for(WorkflowCondition cond: task.getPreConditions()){
278           if(!conditionMap.containsKey(cond.getConditionId())){
279             throw new RepositoryException("Reference in new task: ["+task.getTaskName()+"] to undefined pre condition ith id: ["+cond.getConditionId()+"]");            
280           }          
281         }
282         
283         for(WorkflowCondition cond: task.getPostConditions()){
284           if(!conditionMap.containsKey(cond.getConditionId())){
285             throw new RepositoryException("Reference in new task: ["+task.getTaskName()+"] to undefined post condition ith id: ["+cond.getConditionId()+"]");            
286           }              
287         }
288       }
289       
290         String taskId = task.getTaskId() != null ? 
291              task.getTaskId():UUID.randomUUID().toString();
292         taskMap.put(taskId, task);
293         return taskId;
294     }

298      * @see org.apache.oodt.cas.workflow.repository.WorkflowRepository#addWorkflow(org.apache.oodt.cas.workflow.structs.Workflow)
299      */
300     @Override
301     public String addWorkflow(Workflow workflow) throws RepositoryException {
302        // first check to see that its tasks are all present
303       if(workflow.getTasks() == null || (workflow.getTasks().size() == 0)){
304         throw new RepositoryException("Attempt to define a new worklfow: ["+workflow.getName()+"] with no tasks.");
305       }
306       
307       for(WorkflowTask task: (List<WorkflowTask>)workflow.getTasks()){
308         if(!taskMap.containsKey(task.getTaskId())){
309           throw new RepositoryException("Reference in new workflow: ["+workflow.getName()+"] to undefined task with id: ["+task.getTaskId()+"]");
310         }
311         
312         // check its conditions
313         if(task.getConditions() != null && task.getConditions().size() > 0){
314           for(WorkflowCondition cond: (List<WorkflowCondition>)task.getConditions()){
315             if(!conditionMap.containsKey(cond.getConditionId())){
316               throw new RepositoryException("Reference in new workflow: ["+workflow.getName()+"] to undefined condition ith id: ["+cond.getConditionId()+"]");
317             }
318           }
319         }
320       }
321       
322         String workflowId = workflow.getId();
323 		if (workflowId == null || (workflowId.equals(""))) {
324 			// generate its ID
325 			workflowId = UUID.randomUUID().toString();
326 			workflow.setId(workflowId);
327 		}
328 		workflowMap.put(workflowId, workflow);
329 		eventMap.put(workflowId, Collections.singletonList(workflow));
330 		return workflowId;
331 		
332     }

433     private void loadTasks(List dirUris) {
434         if (dirUris != null && dirUris.size() > 0) {
435           for (Object dirUri1 : dirUris) {
436             String dirUri = (String) dirUri1;
437 
438             try {
439               File workflowDir = new File(new URI(dirUri));
440               if (workflowDir.isDirectory()) {
441                 String workflowDirStr = workflowDir.getAbsolutePath();
442 
443                 if (!workflowDirStr.endsWith("/")) {
444                   workflowDirStr += "/";
445                 }
446 
447                 Document taskRoot = getDocumentRoot(workflowDirStr
448                                                     + "tasks.xml");
449 
450                 Element taskElement = taskRoot.getDocumentElement();
451 
452                 NodeList taskElemList = taskElement
453                     .getElementsByTagName("task");
454 
455                 if (taskElemList != null
456                     && taskElemList.getLength() > 0) {
457                   for (int j = 0; j < taskElemList.getLength(); j++) {
458                     Element taskElem = (Element) taskElemList
459                         .item(j);
460                     WorkflowTask task = XmlStructFactory
461                         .getWorkflowTask(taskElem, conditionMap);
462                     if (task != null) {
463                       taskMap.put(task.getTaskId(), task);
464                     }
465                   }
466 
467                 }
468               }
469             } catch (URISyntaxException e) {
470               LOG
471                   .log(
472                       Level.WARNING,
473                       "DirUri: "
474                       + dirUri
475                       + " is not a directory: skipping task loading for it.");
476             }
477 
478           }
479         }
480     }

534     private void loadWorkflows(List dirUris) {
535         if (dirUris != null && dirUris.size() > 0) {
536           for (Object dirUri1 : dirUris) {
537             String dirUri = (String) dirUri1;
538 
539             try {
540               File workflowDir = new File(new URI(dirUri));
541               if (workflowDir.isDirectory()) {
542 
543 
544                 // get all the workflow xml files
545                 File[] workflowFiles = workflowDir
546                     .listFiles(workflowXmlFilter);
547 
548                 if(workflowFiles!=null) {
549                   for (File workflowFile : workflowFiles) {
550                     String workflowXmlFile = workflowFile
551                         .getAbsolutePath();
552                     Document workflowRoot = getDocumentRoot(workflowXmlFile);
553 
554                     String workflowId = workflowRoot
555                         .getDocumentElement().getAttribute("id");
556                     if (workflowMap.get(workflowId) == null) {
557                       Workflow w = XmlStructFactory.getWorkflow(
558                           workflowRoot.getDocumentElement(),
559                           taskMap, conditionMap);
560 
561                       if (w.getConditions() != null && w.getConditions().size() > 0) {
562                         // add a virtual first task, with the conditions
563                         w.getTasks().add(0, getGlobalWorkflowConditionsTask(w.getName(), w.getId(), w.getConditions()));
564                       }
565                       workflowMap.put(workflowId, w);
566                     } else {
567                       LOG
568                           .log(
569                               Level.FINE,
570                               "Ignoring workflow file: "
571                               + workflowXmlFile
572                               + " when loading workflows, workflow id: "
573                               + workflowId
574                               + " already loaded");
575                     }
576 
577                   }
578                 }
579               }
580             } catch (URISyntaxException e) {
581               LOG
582                   .log(
583                       Level.WARNING,
584                       "DirUri: "
585                       + dirUri
586                       + " is not a directory: skipping workflow loading for it.");
587             }
588 
589           }
590         }

663     private Document getDocumentRoot(String xmlFile) {
664         // open up the XML file
665         DocumentBuilderFactory factory;
666         DocumentBuilder parser;
667         Document document;
668         InputSource inputSource;
669 
670         InputStream xmlInputStream;
671 
672         try {
673             xmlInputStream = new File(xmlFile).toURL().openStream();
674         } catch (IOException e) {
675             LOG.log(Level.WARNING,
676                     "IOException when getting input stream from [" + xmlFile
677                             + "]: returning null document root");
678             return null;
679         }
680 
681         inputSource = new InputSource(xmlInputStream);
682 
683         try {
684             factory = DocumentBuilderFactory.newInstance();
685             parser = factory.newDocumentBuilder();
686             document = parser.parse(inputSource);
687         } catch (Exception e) {
688             LOG.warning("Unable to parse xml file [" + xmlFile + "]."
689                     + "Reason is [" + e + "]");
690             return null;
691         }
692 
693         return document;
694     }

700       task.setTaskId(workflowId+"-global-conditions-eval");
701       task.setTaskName(workflowName+"-global-conditions-eval");
702       task.setTaskInstanceClassName(NoOpTask.class.getName());
703       taskMap.put(task.getTaskId(), task);
704       return task;
