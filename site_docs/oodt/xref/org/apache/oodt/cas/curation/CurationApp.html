50    @Override
51    protected void init() {
52      super.init();
53      Set<String> benchResources = Workbench.getImageFiles();
54      String localPath = HomePage.class.getPackage().getName();
55      Set<String> localResources = Workbench.getImageFiles(localPath);
56      benchResources = filterBenchResources(benchResources, localResources,
57          localPath);
58      doImageMounts(benchResources, Workbench.class);
59      doImageMounts(localResources, HomePage.class);
60  
61      MixedParamUrlCodingStrategy loginPageMount = new MixedParamUrlCodingStrategy(
62          "auth", LoginPage.class, new String[] { "action" });
63      mount(loginPageMount);
64    }

110   private Set<String> filterBenchResources(Set<String> bench,
111       Set<String> local, String localPrefix) {
112     if (local == null || (local.size() == 0)) {
113       return bench;
114     }
115     if (bench == null || (bench.size() == 0)) {
116       return bench;
117     }
118     Set<String> filtered = new HashSet<String>();
119     for (String bResource : bench) {
120       String localName = new File(bResource).getName();
121       String compare = localPrefix + localName;
122       if (!local.contains(compare)) {
123         filtered.add(bResource);
124       } else {
125         LOG.log(Level.INFO, "Filtered conflicting bench resource: ["
126             + bResource + "]");
127       }
128 
129     }
130     return filtered;
131   }

133   private void doImageMounts(Set<String> resources, Class<?> clazz) {
134     if (resources != null) {
135       for (String resource : resources) {
136         String resName = new File(resource).getName();
137         String resPath = "/images/" + resName;
138         LOG.log(Level.INFO, "Mounting: [" + resPath + "] origName: [" + resName
139             + "]: resource: [" + resource + "]");
140         mountSharedResource(resPath,
141             new ResourceReference(clazz, resName).getSharedResourceKey());
142       }
143     }
144   }
