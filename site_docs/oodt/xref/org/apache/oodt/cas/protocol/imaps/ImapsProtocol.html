79    public synchronized void cd(ProtocolFile file) throws ProtocolException {
80      try {
81        String remotePath = file.getPath();
82        if (remotePath.startsWith("/")) {
83          remotePath = remotePath.substring(1);
84        }
85        if (remotePath.trim().equals("")) {
86          homeFolder = currentFolder = store.getDefaultFolder();
87        } else {
88          homeFolder = currentFolder = store.getFolder(remotePath);
89        }
90      } catch (Exception e) {
91        throw new ProtocolException("Failed to change directory to '" 
92        		+ file + "' : " + e.getMessage(), e);
93      }
94    }

112   public synchronized void connect(String host, Authentication auth)
113       throws ProtocolException {
114     try {
115       if (store == null) {
116         store = (session = Session.getInstance(System.getProperties()))
117             .getStore("imaps");
118         store.connect(host, port, auth.getUser(), auth.getPass());
119         currentFolder = store.getDefaultFolder();
120       }
121       this.incrementConnections();
122     } catch (Exception e) {
123       throw new ProtocolException("Failed to connected to IMAPS server " + host
124           + " with username " + auth.getUser() + " : " + e.getMessage(), e);
125     }
126   }

128   public synchronized void close() throws ProtocolException {
129     decrementConnections();
130     if (connectCalls <= 0) {
131       try {
132         store.close();
133       } catch (Exception e) {

135       } finally {
136         store = null;
137       }
138     }

151   public synchronized void get(ProtocolFile fromFile, File toFile)
152       throws ProtocolException {
153     try {
154       openFolder(currentFolder);
155       Message[] messages = currentFolder.getMessages();
156       for (Message message : messages) {
157         if (this.getMessageName(message).equals(fromFile.getName())) {
158           writeMessageToLocalFile(message, toFile);
159           break;
160         }
161       }
162     } catch (Exception e) {
163       throw new ProtocolException("Failed to download " + fromFile + " to "
164           + toFile + " : " + e.getMessage(), e);
165     } finally {
166       try {
167         closeFolder(currentFolder);
168       } catch (Exception ignored) {
169       }
170     }
171   }

177   private void writeMessageToLocalFile(Message message, File toLocalFile)
178       throws MessagingException, IOException, SAXException,
179       TikaException {
180     PrintStream ps = new PrintStream(new FileOutputStream(toLocalFile));
181 
182     ps.print("From:");
183     Address[] senders = message.getFrom();
184     for (Address address : senders) {
185       ps.print(" " + address.toString());
186     }
187 
188     ps.print("\nTo:");
189 		Set<Address> recipients = new LinkedHashSet<Address>(Arrays.asList(message
190 				.getAllRecipients()));
191     for (Address address : recipients) {
192       ps.print(" " + address.toString());
193     }
194 
195     ps.println("\nSubject: " + message.getSubject());
196 
197     ps.println("----- ~ Message ~ -----");
198     String content = this.getContentFromHTML(message);
199     if (content.equals("")) {
200       content = this.getContentFromPlainText(message);
201     }
202     ps.println(content);
203 
204     ps.close();
205   }

280   private String getMessageName(Message msg) throws MessagingException {
281     Enumeration headers = msg.getAllHeaders();
282     while (headers.hasMoreElements()) {
283       Header header = (Header) headers.nextElement();
284       if (header.getName().toLowerCase().equals("message-id")) {
285         String stringHeader = header.getValue();
286         stringHeader = stringHeader.replace("<", "");
287         return stringHeader.substring(0, stringHeader.indexOf("@"));
288       }
289     }
290     return null;
291   }

293   private String getContentFromPlainText(Part p) throws MessagingException,
294       IOException {
295     StringBuilder content = new StringBuilder("");
296     if (p.isMimeType("text/plain")) {
297       content.append((String) p.getContent());
298     } else if (p.isMimeType("multipart/*")) {
299       Multipart mp = (Multipart) p.getContent();
300       int count = mp.getCount();
301       for (int i = 0; i < count; i++) {
302         content.append(getContentFromPlainText(mp.getBodyPart(i)));
303       }
304     } else {
305       Object obj = p.getContent();
306       if (obj instanceof Part) {
307         content.append(getContentFromPlainText((Part) p.getContent()));
308       }
309     }
310     return content.toString().replaceAll(" \\r\\n", "").replaceAll(" \\n", "");
311   }

313   private String getContentFromHTML(Part p) throws MessagingException,
314       IOException, SAXException, TikaException {
315     StringBuilder content = new StringBuilder("");
316     if (p.isMimeType("multipart/*")) {
317       Multipart mp = (Multipart) p.getContent();
318       int count = mp.getCount();
319       for (int i = 0; i < count; i++) {
320         content.append(getContentFromHTML(mp.getBodyPart(i)));
321       }
322     } else if (p.isMimeType("text/html")) {
323       HtmlParser parser = new HtmlParser();
324       Metadata met = new Metadata();
325       TextContentHandler handler = new TextContentHandler(
326           new BodyContentHandler());
327       parser.parse(new ByteArrayInputStream(((String) p.getContent())
328           .getBytes()), handler, met);
329       content.append(handler.toString());
330     } else {
331       Object obj = p.getContent();
332       if (obj instanceof Part) {
333         content.append(getContentFromHTML((Part) p.getContent()));
334       }
335     }
336     return content.toString();
337   }

339   private synchronized void openFolder(Folder folder) throws ProtocolException {
340     if (!folder.isOpen()) {
341       try {
342         folder.open(Folder.READ_WRITE);
343       } catch (Exception e) {
344         try {
345           folder.open(Folder.READ_ONLY);
346         } catch (Exception e2) {
347           throw new ProtocolException("Failed to open folder : "
348               + e.getMessage() + " : " + e2.getMessage());
349         }
350       }
351     }
352     openCalls++;
353   }
354 
355   private synchronized void closeFolder(Folder folder) {
356     if (openCalls > 0) {
357       openCalls--;
358     }
359 
360     if (openCalls <= 0) {
361       try {
362         folder.close(true);
363       } catch (Exception ignored) {
364       }
365     }
366   }

368   public synchronized void delete(ProtocolFile file) throws ProtocolException {
369     try {
370       openFolder(currentFolder);
371       Message[] messages = currentFolder.getMessages();
372       for (Message message : messages) {
373         if (this.getMessageName(message).equals(file.getName())) {
374           message.setFlag(Flags.Flag.DELETED, true);
375           break;
376         }
377       }
378     } catch (Exception e) {
379     	throw new ProtocolException("Failed to delete file '" + file + "' : " + e.getMessage(), e);
380     } finally {
381       closeFolder(currentFolder);
382     }
383   }
