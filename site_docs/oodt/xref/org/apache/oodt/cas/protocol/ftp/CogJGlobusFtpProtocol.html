56    public void cd(ProtocolFile file) throws ProtocolException {
57        try {
58            ftp.changeDir(file.getPath());
59        } catch (Exception e) {
60            throw new ProtocolException("Failed to cd to " + file + " : "
61                    + e.getMessage());
62        }
63    }
64    
65    public void cdRoot() throws ProtocolException {
66    	cd(new ProtocolFile(ProtocolFile.SEPARATOR, true));
67    }
68    
69    public void cdHome() throws ProtocolException {
70    	cd(new ProtocolFile(homeDir, true));
71    }
72  
73    public void connect(String host, Authentication auth) throws ProtocolException {
74        try {
75            ftp = new FTPClient(host, port);
76        } catch (Exception e) {
77            throw new ProtocolException("Failed to connect to: " + host + " : "
78                    + e.getMessage(), e);
79        }
80        isConnected = true;
81  
82        try {
83            ftp.authorize(auth.getUser(), auth.getPass());
84            ftp.setActive(ftp.setLocalPassive());
85            homeDir = ftp.getCurrentDir();
86        } catch (Exception e) {
87            throw new ProtocolException("Failed to login to: " + host + " : "
88                    + e.getMessage(), e);
89        }
90    }
91  
92    public void close() throws ProtocolException {
93        try {
94            ftp.close();
95            isConnected = false;
96        } catch (Exception e) {
97            throw new ProtocolException("Error disconnecting from "
98                    + ftp.getHost() + " : " + e.getMessage());
99        }
100   }

102   public void get(ProtocolFile fromFile, File toFile)
103           throws ProtocolException {
104       try {
105           ftp.setActive(ftp.setLocalPassive());
106           ftp.get(fromFile.getPath(), toFile);
107       } catch (Exception e) {
108           throw new ProtocolException("Failed to download: " + fromFile.getName()
109                   + " : " + e.getMessage());
110       }
111   }

153   public List<ProtocolFile> ls() throws ProtocolException {
154       try {
155           setActive();
156           @SuppressWarnings("unchecked")
157           Vector<FileInfo> fileList = (Vector<FileInfo>) ftpList("*", null);
158           Vector<ProtocolFile> returnList = new Vector<ProtocolFile>();
159           for (FileInfo file : fileList) {
160               returnList.add(new ProtocolFile(this.pwd(), file.getName(), file.isDirectory()));
161           }
162           return returnList;
163       } catch (Exception e) {
164           throw new ProtocolException("Failed to get list of files : "
165                   + e.getMessage());
166       }
167   }

169 	public List<ProtocolFile> ls(ProtocolFileFilter filter) throws ProtocolException {
170     try {
171       ftp.setActive(ftp.setLocalPassive());
172       @SuppressWarnings("unchecked")
173       Vector<FileInfo> fileList = (Vector<FileInfo>) ftp.list("*", null);
174       Vector<ProtocolFile> returnList = new Vector<ProtocolFile>();
175       for (FileInfo file : fileList) {
176       	ProtocolFile pFile = new ProtocolFile(this.pwd(), file.getName(), file.isDirectory());
177       	if (filter.accept(pFile)) {
178       		returnList.add(pFile);
179       	}
180       }
181       return returnList;
182 	  } catch (Exception e) {
183 	      throw new ProtocolException("Failed to get list of files : "
184 	              + e.getMessage());
185 	  }
186   }
