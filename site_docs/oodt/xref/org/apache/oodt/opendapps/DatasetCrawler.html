41  /**
42   * Crawls a catalog and returns all the datasets and their references.
43   * 
44   */
45  public class DatasetCrawler implements CatalogCrawler.Listener {
46  
47    private static Logger LOG = Logger.getLogger(DatasetCrawler.class.getName());
48  
49    private List<String> urls = new Vector<String>();
50  
51    private Map<String, Metadata> datasetMet;
52  
53    private String datasetURL = null;
54    
55    private OpendapConfig conf = null;
56  
57    public DatasetCrawler(String datasetURL, OpendapConfig conf) {
58      this.datasetURL = datasetURL.endsWith("/") ? datasetURL : datasetURL + "/";
59      this.datasetMet = new ConcurrentHashMap<String, Metadata>();
60      this.conf = conf;
61    }
62  
63    /*
64     * (non-Javadoc)
65     * 
66     * @see
67     * thredds.catalog.crawl.CatalogCrawler.Listener#getCatalogRef(thredds.catalog
68     * .InvCatalogRef, java.lang.Object)
69     */
70    public boolean getCatalogRef(InvCatalogRef dd, Object context) {
71      return true;
72    }
73  
74    /*
75     * (non-Javadoc)
76     * 
77     * @see
78     * thredds.catalog.crawl.CatalogCrawler.Listener#getDataset(thredds.catalog
79     * .InvDataset, java.lang.Object)
80     */
81    public void getDataset(InvDataset dd, Object context) {
82    	String url = this.datasetURL + dd.getCatalogUrl().split("#")[1];
83  
84      LOG.log(Level.FINE, url + " is the computed access URL for this dataset");
85      // look for an OpenDAP access URL, only extract metadata if it is found
86      List<InvAccess> datasets = dd.getAccess();
87      if (dd.getAccess() != null && dd.getAccess().size() > 0) {
88        for (InvAccess single : datasets) {
89          InvService service = single.getService();
90          // note: select the OpenDAP access URL based on THREDDS service type
91          if (service.getServiceType() == ServiceType.OPENDAP) {
92            LOG.log(Level.FINE, "Found OpenDAP access URL: " + single.getUrlPath());
93            String opendapurl = this.datasetURL + single.getUrlPath();
94            // extract metadata from THREDDS catalog
95            MetadataExtractor extractor = new ThreddsMetadataExtractor(dd);
96            Metadata met = new Metadata();
97            extractor.extract(met, conf);
98            // index metadata by opendap access URL
99            this.datasetMet.put(opendapurl, met);
100           this.urls.add(opendapurl);
101           break;
102         }
103       }
104     }
105   }
106 
107   /**
108    * Gets the set of String {@link java.net.URL}s crawled.
109    * 
110    * @return A {@link List} of {@link String} representations of {@link java.net.URL}s.
111    */
112   public List<String> getURLs() {
113     return this.urls;
114   }
115 
116   /**
117    * Returns the exracted THREDDS {@link InvDataset} metadata. The dataset
118    * metadata is mapped to the unique THREDDS dataset URL.
119    * 
120    * @return the exracted THREDDS {@link InvDataset} metadata.
121    */
122   public Map<String, Metadata> getDatasetMet() {
123     return this.datasetMet;
124   }
