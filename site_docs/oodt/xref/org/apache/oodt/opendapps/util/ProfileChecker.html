17  package org.apache.oodt.opendapps.util;
18  
19  // JDK imports
20  import java.util.List;
21  
22  // OODT imports
23  import org.apache.oodt.profile.Profile;
24  import org.apache.oodt.profile.ProfileElement;

26  // Spring imports
27  import org.springframework.util.StringUtils;

29  /**
30   * Utility class that checks an OODT Profile versus a list of required/optional elements,
31   * and provides a validation summary for quick inspection by the publisher.
32   * 
33   * @author Luca Cinquini
34   *
35   */
36  public class ProfileChecker {
37  	
38  	// list of mandatory profile elements
39  	private final static String[] mandatoryProfileElements = new String[] {  };
40  	
41  	// list of optional profile elements
42  	private final static String[] optionalProfileElements = new String[] { "mission_name", "sensor", "institute",
43  		                                                                     "variable", "cf_standard_name", "variable_long_name",
44  		                                                                     "spatial_coverage",
45  		                                                                     "north_degrees", "east_degrees", "south_degrees", "west_degrees",
46  		                                                                     "datetime_start", "datetime_stop" };
47  		
48  	/**
49  	 * Main method to check an OODT profile.
50  	 * 
51  	 * @param profile : the OODT profile that needs validation.
52  	 * @param sb : buffer to write the output to.
53  	 * @return : true if the profile is valid, false otherwise.
54  	 */
55  	public static boolean check(final Profile profile, final StringBuilder sb) {
56  		
57  		// profile passes by default
58  		boolean ok = true;
59  		sb.append("\nChecking profile=").append(profile.getProfileAttributes().getID());
60  		
61  		ok = checkResourceAttribute("Identifier", profile.getResourceAttributes().getIdentifier(), true, sb);
62  		
63  		ok = ok && checkResourceAttribute("Title", profile.getResourceAttributes().getTitle(), true, sb);
64  		
65  		ok = ok && checkResourceAttribute("Description", profile.getResourceAttributes().getDescription(), false, sb);
66  		
67  		ok = ok && checkResourceAttribute("Location of type "+ProfileUtils.MIME_TYPE_OPENDAP_HTML, 
68  				             selectResourceLocationByMimeType(profile.getResourceAttributes().getResLocations(), ProfileUtils.MIME_TYPE_OPENDAP_HTML), 
69  				             true, sb);
70  		
71  		ok = ok && checkResourceAttribute("Location of type "+ProfileUtils.MIME_TYPE_THREDDS, 
72          selectResourceLocationByMimeType(profile.getResourceAttributes().getResLocations(), ProfileUtils.MIME_TYPE_THREDDS), 
73          true, sb);
74  		
75  		ok = ok && checkResourceAttribute("Location of type "+ProfileUtils.MIME_TYPE_HTML, 
76          selectResourceLocationByMimeType(profile.getResourceAttributes().getResLocations(), ProfileUtils.MIME_TYPE_HTML), 
77          true, sb);
78  		
79  		ok = ok && checkResourceAttribute("Location of type "+ProfileUtils.MIME_TYPE_GIS, 
80          selectResourceLocationByMimeType(profile.getResourceAttributes().getResLocations(), ProfileUtils.MIME_TYPE_GIS), 
81          true, sb);
82  		
83  		for (String name : mandatoryProfileElements) {
84  			ok = ok && checkProfileElement(profile, name, true, sb);
85  		}
86  		for (String name : optionalProfileElements) {
87  			ok = ok && checkProfileElement(profile, name, false, sb);
88  		}
89  		
90  		return ok;
91  	}
92  	
93  	/**
94  	 * Method to check that a profile Resource Attribute has a valid value.
95  	 * 
96  	 * @param name
97  	 * @param value
98  	 * @param mandatory
99  	 * @param sb
100 	 * @return
101 	 */
102 	private static boolean checkResourceAttribute(String name, String value, boolean mandatory, StringBuilder sb) {
103 		sb.append("\n\tResource Attribute '").append(name).append("' = ");
104 		if (!StringUtils.hasText(value) || value.equalsIgnoreCase("null")) {
105 				if (mandatory) {
106 				  return false; // bad value
107 				}
108 		} else {
109 			sb.append(value);
110 		}
111 		return true;
112 	}
113 	
114 	/**
115 	 * Method to check that the profile contains at least one valid value for of a specific element name.
116 	 * 
117 	 * @param profile
118 	 * @param name
119 	 * @param mandatory
120 	 * @param sb
121 	 * @return
122 	 */
123 	private static boolean checkProfileElement(Profile profile, String name, boolean mandatory, StringBuilder sb) {
124 		
125 		sb.append("\n\tProfile Element '").append(name).append("' = ");
126 	  // profile element is valid by default
127 		boolean ok = true; 
128 	
129 		if (profile.getProfileElements().containsKey(name)) {
130 				ProfileElement profElement = (ProfileElement)profile.getProfileElements().get(name);
131 			
132   			// profile element found
133   			List<String> values = profElement.getValues();
134   			if (values.size()>0) {
135   				boolean first = true;
136   				for (String value : values) {
137   					if (!StringUtils.hasText(value) || value.equalsIgnoreCase("null")) {
138   						if (mandatory) {
139 						  ok = false; // invalid value for this profile element
140 						}
141   					} else {
142   						if (!first) {
143 						  sb.append(", ");
144 						}
145   						sb.append(value);
146   						first = false;
147   					}
148   				}
149   			} else {
150   				if (mandatory) {
151 				  ok = false; // no values found for this profile element
152 				}
153   			}
154 		
155 		} else {
156 			if (mandatory) {
157 			  ok = false; // no profile element found
158 			}
159 		}
160 		
161 		return ok;
162 		
163 	}
164 	
165 	/**
166 	 * Method to select the resource location of a specific mime type, if found.
167 	 * 
168 	 * @param resLocations
169 	 * @param mimeType
170 	 * @return
171 	 */
172 	private static String selectResourceLocationByMimeType(List<String> resLocations, String mimeType) {
173 		
174 		for (String resLocation : resLocations) {
175 			String[] parts = resLocation.split("\\|"); // regular expression of ProfileUtils.CHAR
176 			if (parts[1].equals(mimeType)) {
177 			  return parts[0];
178 			}
179 		}
180 		
181 		// resource location not found
182 		return null;
183 		
184 	}
