75      public XMLPSProductHandler() throws InstantiationException {
76          String MappingFilePath = System
77                  .getProperty("org.apache.oodt.xmlps.xml.mapFilePath");
78  
79          if (MappingFilePath == null) {
80              throw new InstantiationException(
81                      "Need to specify path to xml mapping file!");
82          }
83  
84          try {
85              mapping = MappingReader.getMapping(MappingFilePath);
86          } catch (Exception e) {
87              throw new InstantiationException(
88                      "Unable to parse mapping xml file: ["
89                              + MappingFilePath + "]: reason: "
90                              + e.getMessage());
91          }
92  
93          /* load the db properties file */
94          /*
95           * if one exists: otherwise, don't bother and just print out the SQL to
96           * the console.
97           */
98          String dbPropFilePath = System
99                  .getProperty("org.apache.oodt.xmlps.xml.dbPropFilePath");
100         if (dbPropFilePath != null) {
101             try {
102                 System.getProperties()
103                         .load(new FileInputStream(dbPropFilePath));
104             } catch (FileNotFoundException e) {
105                 // TODO Auto-generated catch block
106                 LOG.log(Level.SEVERE, e.getMessage());
107             } catch (IOException e) {
108                 LOG.log(Level.SEVERE, e.getMessage());
109                 throw new InstantiationException(e.getMessage());
110             }
111 
112             executor = new DBMSExecutor();
113         }
114     }

118      *
119      * @see org.apache.oodt.product.QueryHandler#query(org.apache.oodt.xmlquery.XMLQuery)
120      */
121     public XMLQuery query(XMLQuery query) throws ProductException {
122         List<QueryElement> whereSet = query.getWhereElementSet();
123         List<QueryElement> selectSet = query.getSelectElementSet();
124         try {
125             translateToDomain(selectSet, true);
126             translateToDomain(whereSet, false);
127         } catch (Exception e) {
128             LOG.severe(e.getMessage());
129             throw new ProductException(e.getMessage());
130         }
131 
132         queryAndPackageResults(query);
133 
134         return query;
135     }

151     protected List<QueryElement> getElemNamesFromQueryElemSet(
152             List<QueryElement> origSet) {
153         if (origSet == null || (origSet.size() == 0)) {
154             return Collections.emptyList();
155         }
156 
157         List<QueryElement> newSet = new Vector<QueryElement>();
158 
159         for (QueryElement elem : origSet) {
160             if (elem.getRole().equals(XMLQueryHelper.ROLE_ELEMNAME)
161                 && !mapping.constantField(elem.getValue())) {
162                 newSet.add(elem);
163 
164             }
165 
166         }
167 
168         return newSet;
169 
170     }

190     protected void queryAndPackageResults(XMLQuery query) {
191         Stack<QueryElement> queryStack = HandlerQueryParser
192                 .createQueryStack(query.getWhereElementSet());
193         Expression parsedQuery = HandlerQueryParser.parse(queryStack,
194                 this.mapping);
195         List<QueryElement> selectNames = getElemNamesFromQueryElemSet(query
196                 .getSelectElementSet());
197 
198         String querySelectNames = toSQLSelectColumns(selectNames);
199 
200         StringBuilder sqlBuf = new StringBuilder("SELECT ");
201         sqlBuf.append(querySelectNames);
202         sqlBuf.append(" FROM ");
203         sqlBuf.append(mapping.getDefaultTable());
204         sqlBuf.append(" ");
205 
206         if (mapping.getNumTables() > 0) {
207             List<QueryElement> whereNames = getElemNamesFromQueryElemSet(query.getWhereElementSet());
208             Set<DatabaseTable> requiredTables = getRequiredTables(whereNames, selectNames);
209             for (DatabaseTable tbl : requiredTables) {
210                 sqlBuf.append("INNER JOIN ");
211                 sqlBuf.append(tbl.getName());
212                 sqlBuf.append(" ON ");
213                 sqlBuf.append(tbl.getName());
214                 sqlBuf.append(".");
215                 sqlBuf.append(tbl.getJoinFieldName());
216                 sqlBuf.append(" = ");
217                 sqlBuf.append(tbl.getDefaultTableJoin());
218                 sqlBuf.append(".");
219                 sqlBuf.append(tbl.getDefaultTableJoinFieldName());
220                 sqlBuf.append(" ");
221             }
222         }
223 
224         if(parsedQuery != null){
225             sqlBuf.append(" WHERE ");
226             sqlBuf.append(parsedQuery.evaluate());
227         }
228 
229         LOG.log(Level.INFO, sqlBuf.toString());
230 
231         if (executor != null) {
232             try {
233                 CDEResult res = executor.executeLocalQuery(sqlBuf.toString());
234                 res.setOrderedFields(query.getSelectElementSet());
235                 res.setMapping(mapping);
236                 res.setConstValues(getConstValuesForQuery(query));
237                 query.getResults().add(res);
238             } catch (SQLException e) {
239                 LOG.log(Level.SEVERE, e.getMessage());
240                 LOG.log(Level.WARNING, "Error executing sql: ["
241                         + sqlBuf.toString() + "]: Message: " + e.getMessage());
242             }
243         }
244 
245     }

247     private List<CDEValue> getConstValuesForQuery(XMLQuery query) {
248         List<QueryElement> select = query.getSelectElementSet();
249         List<QueryElement> constNames = getConstElemNamesFromQueryElemSet(select);
250         List<CDEValue> constValues = new ArrayList<CDEValue>();
251         if (constNames != null) {
252             for (QueryElement qe : constNames) {
253                 MappingField fld = mapping.getFieldByLocalName(qe.getValue());
254                 if (fld != null) {
255                     constValues.add(new CDEValue(fld.getName(), fld.getConstantValue()));
256                 }
257             }
258         }
259         return constValues;
260     }

262     private String toSQLSelectColumns(List<QueryElement> elems) {
263         if (elems == null || (elems.size() == 0)) {
264             return null;
265         }
266 
267         StringBuilder buf = new StringBuilder();
268         for (QueryElement qe : elems) {
269             MappingField fld = this.mapping.getFieldByLocalName(qe.getValue());
270             if (fld != null) {
271                 buf.append(fld.getLocalName());
272                 buf.append(" as ");
273                 buf.append(fld.getName());
274                 buf.append(",");
275             }
276         }
277 
278         buf.deleteCharAt(buf.length() - 1);
279 
280         return buf.toString();
281     }

283     protected void translateToDomain(List<QueryElement> elemSet,
284             boolean selectSet) throws XmlpsException {
285         // go through each query element: use the mapping fields
286         // to translate the names
287 
288         for (Iterator<QueryElement> i = elemSet.iterator(); i.hasNext();) {
289             QueryElement elem = i.next();
290             if (elem.getRole().equals(XMLQueryHelper.ROLE_ELEMNAME)) {
291                 // do the translation
292                 String elemValue = elem.getValue();
293                 MappingField fld = this.mapping.getFieldByName(elemValue);
294                 // make sure fld is not null
295                 if (fld == null) {
296                     continue;
297                 }
298 
299                 // make sure scope is null, or if it's not null, then it's
300                 // FieldScope.QUERY
301 
302                 if (fld.getScope() != null
303                         && fld.getScope().equals(FieldScope.RETURN)) {
304                     // skip
305                     continue;
306                 }
307 
308                 // check to see if it has a dbname attr, if not, then the name
309                 // stays
310                 // the same
311                 String newFldName = fld.getLocalName();
312 
313                 elem.setValue(newFldName);
314 
315                 // now translate the domain vocab if there are translate funcs
316                 // present and this isn't the select set
317 
318                 if (!selectSet && fld.getFuncs() != null
319                         && fld.getFuncs().size() > 0) {
320                     // the next query element should be
321                     // XMLQueryHelper.ROLE_LITERAL
322                     if (!i.hasNext()) {
323                         break;
324                     }
325                     QueryElement litElem = i.next();
326                     if (!litElem.getRole().equals(XMLQueryHelper.ROLE_LITERAL)) {
327                         throw new XmlpsException("next query element not "
328                                 + XMLQueryHelper.ROLE_LITERAL + "! role is "
329                                 + litElem.getRole() + " instead!");
330                     }
331 
332                     for (MappingFunc func : fld.getFuncs()) {
333                         CDEValue origVal = new CDEValue(fld.getName(),
334                             litElem.getValue());
335                         CDEValue newVal = func.inverseTranslate(origVal);
336                         litElem.setValue(newVal.getVal());
337                     }
338 
339                 }
340 
341             }
342         }
343 
344     }

346     protected Set<DatabaseTable> getRequiredTables(
347             List<QueryElement> whereElemNames, List<QueryElement> selectElemNames) {
348         Set<DatabaseTable> tables = new HashSet<DatabaseTable>();
349         // add tables from where element set
350         if (whereElemNames != null) {
351             for (QueryElement qe : whereElemNames) {
352                 MappingField fld = mapping.getFieldByLocalName(qe.getValue());
353                 if (fld != null) {
354                     DatabaseTable t = mapping.getTableByName(fld.getTableName());
355                     if (t != null && !tables.contains(t) && !t.getName().equals(mapping.getDefaultTable())) {
356                         tables.add(t);
357                     }
358                 }
359             }
360         }
361         // add tables from select element set
362         if (selectElemNames != null) {
363             for (QueryElement qe : selectElemNames) {
364                 MappingField fld = mapping.getFieldByLocalName(qe.getValue());
365                 if (fld != null) {
366                     DatabaseTable t = mapping.getTableByName(fld.getTableName());
367                     if (t != null && !tables.contains(t) && !t.getName().equals(mapping.getDefaultTable())) {
368                         tables.add(t);
369                     }
370                 }
371             }
372         }
373         // the tables found may be joined on columns from tables we haven't found
374         // yet
375         // add additional required join tables
376         Set<DatabaseTable> moreTables = new HashSet<DatabaseTable>(tables);
377         for (DatabaseTable t : tables) {
378             DatabaseTable join = mapping.getTableByName(t.getDefaultTableJoin());
379             // recursively add all join tables until we get to either
380             // (a) the mapping default table (join == null)
381             // (b) or a table already found (moreTables.contains(join))
382             while (join != null && !moreTables.contains(join) && !join.getName().equals(mapping.getDefaultTable())) {
383                 moreTables.add(join);
384                 join = mapping.getTableByName(join.getDefaultTableJoin());
385             }
386         }
387         return moreTables;
388     }
