194   @Override
195   public String toString() {
196     return "[name=" + this.name + ",dbname=" + this.dbName + ",constant_value=" + this.constantValue + ",table_name="
197            + this.tableName + ",scope=" + (this.scope.equals(FieldScope.RETURN) ? "return" : "query") + ",type=" + (
198                this.type.equals(FieldType.CONSTANT) ? "constant" : "dynamic") + ",funcs=" + printClassNames(this.funcs)
199            + ",string=" + String.valueOf(this.string) + "]";
200   }
201 
202   private String printClassNames(List<MappingFunc> funcs) {
203     StringBuilder buf = new StringBuilder();
204 
205     if (funcs == null || (funcs.size() == 0)) {
206       return "";
207     } else {
208       for (MappingFunc func : funcs) {
209         buf.append(func.getClass().getName());
210         buf.append(",");
211       }
212 
213       buf.deleteCharAt(buf.length() - 1);
214 
215       return buf.toString();
216     }
217   }

234   /**
235    * If dbname exists and is not empty, it is used as the field name.
236    * If the table exists and is not empty,
237    * return tableName.fieldName, otherwise return fieldName.
238    * @return the column name understood by the local db
239    */
240   public String getLocalName() {
241     String dbColName = getName();
242     if (getDbName() != null && !getDbName().isEmpty()) {
243       dbColName = getDbName();
244     }
245     if (getTableName() == null || getTableName().isEmpty()) {
246       return dbColName;
247     }
248     return getTableName() + "." + dbColName;
249   }
