54    public static Mapping getMapping(InputStream is) throws XmlpsException {
55      Document mappingDoc = XMLUtils.getDocumentRoot(is);
56      Mapping map = new Mapping();
57  
58      Element rootElem = mappingDoc.getDocumentElement();
59  
60      map.setId(rootElem.getAttribute("id"));
61      map.setName(rootElem.getAttribute("name"));
62      readTables(rootElem, map);
63      readFields(rootElem, map);
64  
65      return map;
66  
67    }
68  
69    public static Mapping getMapping(URL mappingUrl) throws IOException, XmlpsException {
70      return getMapping(mappingUrl.openStream());
71    }
72  
73    public static Mapping getMapping(String filePath) throws FileNotFoundException, XmlpsException {
74      return getMapping(new FileInputStream(filePath));

76    }

78    private static void readTables(Element rootElem, Mapping map)
79        throws XmlpsException {
80      Element tblsElem = XMLUtils.getFirstElement(TABLES_OUTER_TAG, rootElem);
81      if (tblsElem == null) {
82        throw new XmlpsException("Unable to parse mapping XML file: [" + map.getName()
83            + "]: reason: no defined tables tag element!");
84  
85      }
86  
87      String defaultTbl = tblsElem.getAttribute("default");
88      // make sure that the default attribute is set
89      if (defaultTbl == null || (defaultTbl.equals(""))) {
90        throw new XmlpsException("Unable to parse mapping XML file: [" + map.getName()
91            + "]: reason: there needs to be a default table defined "
92            + "by the \"default\" attribute!");
93      }
94  
95      map.setDefaultTable(defaultTbl);
96  
97      // get a list of all the tables, and process them one by one
98      NodeList tableNodes = tblsElem.getElementsByTagName(TABLE_TAG);
99      if (tableNodes != null && tableNodes.getLength() > 0) {
100       for (int i = 0; i < tableNodes.getLength(); i++) {
101         Element tableElem = (Element) tableNodes.item(i);
102         DatabaseTable tbl = readTable(tableElem);
103         if (tbl.getDefaultTableJoin() == null || tbl.getDefaultTableJoin().isEmpty()) {
104           tbl.setDefaultTableJoin(map.getDefaultTable());
105         }
106         map.addTable(tbl.getName(), tbl);
107       }
108     }
109   }
110 
111   private static DatabaseTable readTable(Element tableElem) {
112     DatabaseTable tbl = new DatabaseTable();
113     tbl.setJoinFieldName(tableElem.getAttribute(TABLE_ATTR_JOIN_FLD));
114     tbl.setName(tableElem.getAttribute(TABLE_ATTR_NAME));
115     tbl.setDefaultTableJoinFieldName(tableElem
116         .getAttribute(TABLE_ATTR_BASE_TBL_JOIN_FLD));
117     tbl.setDefaultTableJoin(tableElem
118         .getAttribute(TABLE_ATTR_BASE_TBL_JOIN_TABLE));
119     return tbl;
120   }

122   private static void readFields(Element rootElem, Mapping map) {
123     NodeList fldNodes = rootElem.getElementsByTagName(FIELD_TAG);
124 
125     if (fldNodes != null && fldNodes.getLength() > 0) {
126       for (int i = 0; i < fldNodes.getLength(); i++) {
127         MappingField fld = readField((Element) fldNodes.item(i));
128         if (fld.getTableName() == null || fld.getTableName().isEmpty()) {
129           fld.setTableName(map.getDefaultTable());
130         }
131         map.addField(fld.getName(), fld);
132       }
133     }
134 
135   }
136 
137   private static MappingField readField(Element fldElem) {
138     MappingField field = new MappingField();
139     field.setTableName(fldElem.getAttribute(FIELD_ATTR_TABLE));
140     field.setName(fldElem.getAttribute(FIELD_ATTR_NAME));
141     field
142         .setType(fldElem.getAttribute(FIELD_ATTR_TYPE).equals("dynamic") ? FieldType.DYNAMIC
143             : FieldType.CONSTANT);
144     field.setString(Boolean.valueOf(fldElem.getAttribute(FIELD_ATTR_STRING)));
145     field.setConstantValue(fldElem.getAttribute(FIELD_ATTR_VALUE));
146     field.setDbName(fldElem.getAttribute(FIELD_ATTR_DBNAME));
147     if (fldElem.getAttribute(FIELD_ATTR_SCOPE) != null
148         && !fldElem.getAttribute(FIELD_ATTR_SCOPE).equals("")) {
149 
150       field
151           .setScope(fldElem.getAttribute(FIELD_ATTR_SCOPE).equals("query") ? FieldScope.QUERY
152               : FieldScope.RETURN);
153     }
154 
155     field.setFuncs(getTranslateFuncs(fldElem));
156 
157     return field;
158 
159   }
160 
161   private static List<MappingFunc> getTranslateFuncs(Element fldElem) {
162     Element translateElem = XMLUtils.getFirstElement(FIELD_TRANSLATE_TAG,
163         fldElem);
164 
165     List<MappingFunc> funcs = new Vector<MappingFunc>();
166 
167     if (translateElem != null) {
168       // check for func tags
169       NodeList funcNodes = translateElem.getElementsByTagName(FUNC_TAG);
170       if (funcNodes != null && funcNodes.getLength() > 0) {
171         for (int i = 0; i < funcNodes.getLength(); i++) {
172           Element funcElem = (Element) funcNodes.item(i);
173           funcs.add(getFunc(funcElem));
174         }
175       }
176     }
177 
178     return funcs;
179   }
180 
181   private static MappingFunc getFunc(Element funcElem) {
182     String funcClass = funcElem.getAttribute(FUNC_ATTR_CLASS);
183     MappingFunc func = GenericCDEObjectFactory
184         .getMappingFuncFromClassName(funcClass);
185 
186     if (func != null) {
187       func.configure(getPropsFromElementAttrs(funcElem.getAttributes()));
188     }
189 
190     return func;
191   }
192 
193   private static Properties getPropsFromElementAttrs(NamedNodeMap map) {
194     Properties props = new Properties();
195 
196     if (map != null) {
197       for (int i = 0; i < map.getLength(); i++) {
198         Node mapNode = map.item(i);
199         props.setProperty(mapNode.getNodeName(), mapNode.getNodeValue());
200       }
201     }
202 
203     return props;
204   }
