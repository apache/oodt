46  /**
47   *
48   * <p>
49   * Executes Profile Queries against an underlying JDBC database, backed by
50   * Apache commons-pool and commons-dbcp.
51   * </p>
52   * .
53   */
54  public class DBMSExecutor {
55  
56    private final DataSource dataSource;
57  
58    private static final Logger LOG = Logger.getLogger(DBMSExecutor.class
59        .getName());
60  
61    public DBMSExecutor() {
62      String jdbcUrl = System.getProperty("xmlps.datasource.jdbc.url");
63      String user = System.getProperty("xmlps.datasource.jdbc.user");
64      String pass = System.getProperty("xmlps.datasource.jdbc.pass");
65      String driver = System.getProperty("xmlps.datasource.jdbc.driver");
66      try {
67        Class.forName(driver);
68      } catch (ClassNotFoundException ignore) {
69      }
70  
71      this.dataSource = DatabaseConnectionBuilder.buildDataSource(user, pass,
72          driver, jdbcUrl);
73  
74    }
75  
76    public List<Profile> executeLocalQuery(Mapping map, String sql,
77        String resLocationSpec) throws SQLException {
78      Connection conn = null;
79      Statement statement = null;
80  
81      List<Profile> profiles = null;
82  
83      try {
84        conn = dataSource.getConnection();
85        statement = conn.createStatement();
86        ResultSet rs = statement.executeQuery(sql);
87  
88        profiles = new Vector<Profile>();
89  
90        while (rs.next()) {
91          Profile prof = toProfile(rs, map, resLocationSpec);
92          profiles.add(prof);
93        }
94  
95      } catch (SQLException e) {
96        LOG.log(Level.SEVERE, e.getMessage());
97        throw e;
98      } finally {
99        if (statement != null) {
100         try {
101           statement.close();
102         } catch (Exception ignore) {
103         }
104 
105       }
106 
107       if (conn != null) {
108         try {
109           conn.close();
110         } catch (Exception ignore) {
111         }
112 
113       }
114     }
115 
116     return profiles;
117 
118   }

120   @SuppressWarnings("unchecked")
121   private Profile toProfile(ResultSet rs, Mapping map, String resLocationSpec) {
122     Profile profile = new Profile();
123     ResourceAttributes resAttr = profile.getResourceAttributes();
124     ProfileAttributes profAttr = profile.getProfileAttributes();
125     resAttr.setResClass("system.profile");
126     profAttr.setStatusID("active");
127     profAttr.setType("profile");
128 
129     Metadata met = new Metadata();
130 
131     for (String fldName : map.getFieldNames()) {
132       MappingField fld = map.getFieldByName(fldName);
133       ProfileElement elem = new EnumeratedProfileElement(profile);
134       elem.setName(fld.getName());
135 
136       try {
137         if (fld.getType().equals(FieldType.CONSTANT)) {
138           elem.getValues().add(fld.getConstantValue());
139         } else {
140           String elemDbVal = rs.getString(fld.getDbName());
141           for (MappingFunc func : fld.getFuncs()) {
142             CDEValue origVal = new CDEValue(fld.getName(), elemDbVal);
143             CDEValue newVal = func.translate(origVal);
144             elemDbVal = newVal.getVal();
145           }
146 
147           elem.getValues().add(elemDbVal);
148         }
149       } catch (SQLException e) {
150         LOG.log(Level.SEVERE, e.getMessage());
151         LOG.log(Level.WARNING, "Unable to obtain field: ["
152                                + fld.getLocalName() + "] from result set: message: "
153                                + e.getMessage());
154       }
155 
156       met.addMetadata(elem.getName(), (String) elem.getValues().get(0));
157 
158       profile.getProfileElements().put(fld.getName(), elem);
159     }
160 
161     if (resLocationSpec != null) {
162       resAttr.getResLocations().add(
163           PathUtils.replaceEnvVariables(resLocationSpec, met));
164     }
165 
166     return profile;
167   }
