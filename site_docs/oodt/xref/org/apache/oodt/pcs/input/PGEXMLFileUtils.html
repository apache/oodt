56    public static Map getMatrixsAsMap(Element group)
57        throws PGEConfigFileException {
58      // get the nodelist for the matrixs
59      NodeList matrixs = group.getElementsByTagName("matrix");
60  
61      // if it's null, return null
62      if (matrixs.getLength()==0) {
63        return Collections.emptyMap();
64      }
65  
66      Map matrixMap = new ConcurrentHashMap(matrixs.getLength());
67  
68      // for each matrix in the list, create a PGEMatrix with the name
69      // attribute and the appropriate value
70  
71      for (int i = 0; i < matrixs.getLength(); i++) {
72        Element matrix = (Element) matrixs.item(i);
73  
74        // get the name of the matrix
75        String matrixName = matrix.getAttribute("name");
76  
77        // get the number of cols
78        NodeList rowNodeList = matrix.getElementsByTagName("tr");
79  
80        // there has to be at least one 1
81        if (rowNodeList == null || (rowNodeList.getLength() <= 0)) {
82          throw new PGEConfigFileException(
83              "there must be at least one row in a matrix!");
84        }
85  
86        PGEMatrix pgeMatrix = new PGEMatrix();
87        pgeMatrix.setName(matrixName);
88  
89        for (int j = 0; j < rowNodeList.getLength(); j++) {
90          Element rowElem = (Element) rowNodeList.item(j);
91  
92          // get the number of cols
93          NodeList colNodeList = rowElem.getElementsByTagName("td");
94  
95          // there must be at least one colum in each row
96          if (colNodeList == null || (colNodeList.getLength() <= 0)) {
97            throw new PGEConfigFileException(
98                "there must be at least one column a matrix row!");
99          }
100 
101         if (pgeMatrix.getNumCols() == 0) {
102           // then set it
103           pgeMatrix.setNumCols(colNodeList.getLength());
104         }
105 
106         List colList = new Vector(colNodeList.getLength());
107 
108         for (int k = 0; k < colNodeList.getLength(); k++) {
109           Element colElement = (Element) colNodeList.item(k);
110           colList.add(DOMUtil.getSimpleElementText(colElement));
111         }
112 
113         pgeMatrix.getRows().add(colList);
114       }
115 
116       matrixMap.put(pgeMatrix.getName(), pgeMatrix);
117     }
118     return matrixMap;
119   }

243       throws PGEConfigFileException {
244     // get the nodelist for scalars
245     NodeList vectors = group.getElementsByTagName("vector");
246 
247     // if it's null, return null
248     if (vectors.getLength()==0) {
249       return Collections.emptyMap();
250     }
251 
252     Map vectorMap = new ConcurrentHashMap(vectors.getLength());
253 
254     // for each vector in the list, create a PGEVector with the name
255     // attribute, and appropriate value
256     for (int i = 0; i < vectors.getLength(); i++) {
257       Element vector = (Element) vectors.item(i);
258 
259       // name of the vector is in the name attribute
260       String vectorName = vector.getAttribute("name");
261 
262       PGEVector vec = new PGEVector();
263       vec.setName(vectorName);
264 
265       // get the nodelist of elements
266       NodeList vecElems = vector.getElementsByTagName("element");
267 
268       if (vecElems == null || (vecElems.getLength() <= 0)) {
269         throw new PGEConfigFileException(
270             "There must be at least one element in a PGEVector!");
271       }
272 
273       List vecElemList = new Vector(vecElems.getLength());
274 
275       for (int j = 0; j < vecElems.getLength(); j++) {
276         Element vecElem = (Element) vecElems.item(j);
277         vecElemList.add(DOMUtil.getSimpleElementText(vecElem));
278       }
279 
280       vec.setElements(vecElemList);
281       vectorMap.put(vec.getName(), vec);
282     }
283 
284     return vectorMap;
285   }

331   public static Document getDocumentRoot(String xmlFile) {
332     // open up the XML file
333     DocumentBuilderFactory factory;
334     DocumentBuilder parser;
335     Document document;
336     InputSource inputSource;
337 
338     InputStream xmlInputStream;
339 
340     try {
341       xmlInputStream = new File(xmlFile).toURL().openStream();
342     } catch (IOException e) {
343       LOG.log(Level.WARNING, "IOException when getting input stream from ["
344           + xmlFile + "]: returning null document root");
345       return null;
346     }
347 
348     inputSource = new InputSource(xmlInputStream);
349 
350     try {
351       factory = DocumentBuilderFactory.newInstance();
352       parser = factory.newDocumentBuilder();
353       document = parser.parse(inputSource);
354     } catch (Exception e) {
355       LOG.warning("Unable to parse xml file [" + xmlFile + "]." + "Reason is ["
356           + e + "]");
357       return null;
358     }
359 
360     return document;
361   }
