47  /**
48   * 
49   * Extends the existing {@link ProductBrowser} and adds a Web form and
50   * associated pedigree tree container to expose the {@link Trace} component.
51   * 
52   * @author mattmann
53   * @version $Revision$
54   * 
55   */
56  public class TraceableProductBrowser extends ProductBrowser {
57  
58    private static final long serialVersionUID = 5512878676145737818L;
59  
60    private static final Logger LOG = Logger
61        .getLogger(TraceableProductBrowser.class.getName());
62  
63    /**
64     * @param componentId
65     * @param fmUrlStr
66     * @param productId
67     */
68    public TraceableProductBrowser(String componentId, final String fmUrlStr,
69        final String productId, final boolean enableNotCat, final List<String> excludes) {
70      super(componentId, fmUrlStr, productId);
71      final FileManagerConn fm = new FileManagerConn(fmUrlStr);
72      final Product product = fm.safeGetProductById(productId);
73      final OpsuiApp app = (OpsuiApp)getApplication();
74  
75      Form traceForm = new Form("trace_form");
76      traceForm.add(new Button("trace_button") {
77        /*
78         * (non-Javadoc)
79         * 
80         * @see org.apache.wicket.markup.html.form.Button#onSubmit()
81         */
82        @Override
83        public void onSubmit() {
84          Trace tracer = new Trace("pedigree", fmUrlStr, enableNotCat, excludes,
85              product);
86          tracer.setVisible(true);
87          getParent().getParent().replace(tracer);
88          setVisible(false);
89        }
90  
91      });
92      add(traceForm);
93      add(new WebMarkupContainer("pedigree").setVisible(false));
94  
95      List<Reference> refs = fm.getProductReferences(product);
96      if (refs != null && refs.size() > 0) {
97        replace(new Label("no_prod_ref_display") {
98          /*
99           * (non-Javadoc)
100          * 
101          * @see org.apache.wicket.Component#isVisible()
102          */
103         @Override
104         public boolean isVisible() {
105           return false;
106         }
107       });
108 
109       replace(new ListView<Reference>("ref_list", new ListModel<Reference>(refs)) {
110         /*
111          * (non-Javadoc)
112          * 
113          * @see
114          * org.apache.wicket.markup.html.list.ListView#populateItem(org.apache
115          * .wicket.markup.html.list.ListItem)
116          */
117         @Override
118         protected void populateItem(ListItem<Reference> refItem) {
119           Reference r = refItem.getModelObject();
120           String filePath = null;
121           try {
122             filePath = new File(new URI(r.getDataStoreReference()))
123                 .getAbsolutePath();
124           } catch (Exception ignore) {
125           }
126           
127           ExternalLink refLink = new ExternalLink("ref_file_path_link", 
128               "/" + app.getRootContext() + "/data?productID="+productId);
129           refLink.add(new Label("ref_file_path", filePath));
130           refItem.add(refLink);
131           refItem.add(new Label("ref_file_size",
132               String.valueOf(r.getFileSize())));
133           try {
134             refItem.add(new Label("ref_pct_transferred", NumberFormat
135                 .getPercentInstance()
136                 .format(fm.getFm().getRefPctTransferred(r))));
137           } catch (DataTransferException e) {
138             LOG.log(
139                 Level.WARNING,
140                 "Unable to determine product reference size: Reason: "
141                     + e.getMessage());
142             refItem.add(new Label("ref_pct_transferred", "N/A"));
143           }
144 
145         }
146       });
147     } else {
148       replace(new Label("no_prod_ref_display", "No Product References!"));
149     }
150 
151   }
