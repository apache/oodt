64    private static Logger LOG = Logger.getLogger(OFSNUtils.class.getName());
65    public static String extractFieldFromQuery(XMLQuery query, String name) {
66      for (Iterator<QueryElement> i = query.getWhereElementSet().iterator(); i
67          .hasNext();) {
68        QueryElement element = i.next();
69        if (element.getRole().equals(XMLQUERY_QUERYELEM_ROLE_ELEM)
70            && element.getValue().equalsIgnoreCase(name)) {
71          // get the next element and ensure that it is a LITERAL, and
72          // return that
73          QueryElement litElement = i.next();
74          return litElement.getValue();
75        }
76      }
77  
78      return null;
79    }

81    public static Document getOFSNDoc(List<File> fileList, OFSNHandlerConfig cfg,
82        String productRoot, boolean showDirSize, boolean showFileSize) {
83      DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
84      factory.setNamespaceAware(true);
85      Document document;
86  
87      try {
88        DocumentBuilder builder = factory.newDocumentBuilder();
89        document = builder.newDocument();
90  
91        Element root = (Element) document.createElement(DIR_RESULT_TAG);
92        XMLUtils.addAttribute(document, root, "xmlns", DIR_LISTING_NS);
93        document.appendChild(root);
94  
95        for (File file : fileList) {
96          Element dirEntryElem = XMLUtils.addNode(document, root, DIR_ENTRY_TAG);
97          String ofsn = toOFSN(file.getAbsolutePath(), productRoot);
98          //This ensures that we get ofsn names with unix style separators.
99          //On a Windows machine, the product server would return '\'
100         //separators.
101         String unixStyleOFSN = FilenameUtils.separatorsToUnix(ofsn);
102         if (cfg.getType().equals(LISTING_CMD)) {
103           if (!Boolean.valueOf(cfg.getHandlerConf().getProperty("isSizeCmd"))) {
104             XMLUtils.addNode(document, dirEntryElem, OFSN_TAG, unixStyleOFSN);
105           }
106         }
107 
108         long size = Long.MIN_VALUE;
109 
110         if (file.isDirectory()) {
111           if (showDirSize) {
112             size = FileUtils.sizeOfDirectory(file);
113           }
114         } else {
115           if (showFileSize) {
116             size = file.length();
117           }
118         }
119 
120         if (size != Long.MIN_VALUE) {
121           XMLUtils.addNode(document, dirEntryElem, FILE_SIZE_TAG, String
122               .valueOf(size));
123         }
124       }
125 
126       return document;
127     } catch (ParserConfigurationException e) {
128       LOG.log(Level.SEVERE, e.getMessage());
129       return null;
130     }
131 
132   }

138   public static File buildZipFile(String zipFileFullPath, File[] files) {
139     // Create a buffer for reading the files
140     byte[] buf = new byte[INT];
141     ZipOutputStream out = null;
142 
143     try {
144       // Create the ZIP file
145       out = new ZipOutputStream(new FileOutputStream(zipFileFullPath));
146 
147       for (File file : files) {
148         FileInputStream in = new FileInputStream(file);
149 
150         // Add ZIP entry to output stream.
151         out.putNextEntry(new ZipEntry(file.getName()));
152 
153         // Transfer bytes from the file to the ZIP file
154         int len;
155         while ((len = in.read(buf)) > 0) {
156           out.write(buf, 0, len);
157         }
158 
159         // Complete the entry
160         out.closeEntry();
161         in.close();
162       }
163     } catch (IOException e) {
164       LOG.log(Level.SEVERE, e.getMessage());
165     } finally {
166       if (out != null) {
167         try {
168           out.close();
169         } catch (Exception ignore) {
170         }
171 
172       }
173     }
174 
175     return new File(zipFileFullPath);
176 
177   }

179   public static boolean validateOFSN(String ofsn) {
180       if (ofsn == null) {
181           return false;
182       } else {
183           return !ofsn.equals("") && !ofsn.matches(".*\\.\\..*");
184       }
185   }

187   private static String toOFSN(String absolutePath, String productRootPath) {
188     if (absolutePath.startsWith(productRootPath)) {
189       return absolutePath.substring(productRootPath.length());
190     } else {
191       // must have been a *.zip file, generated in some cache dir
192       // just return the file name
193       return new File(absolutePath).getName();
194     }
195   }
