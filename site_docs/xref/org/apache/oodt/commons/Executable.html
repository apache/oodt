18  package org.apache.oodt.commons;
19  
20  import java.io.*;
21  
22  /** An executable object.
23   *
24   * Objects of this class are programs that can be executed asynchronously.  Their standard
25   * output and standard error streams are shared with the parent process' standard output
26   * and error streams.
27   *
28   * @author Kelly
29   */
30  public abstract class Executable {
31  
32    public static final int INT = 1024;
33  
34    /** Construct an executable.
35  	 */
36  	public Executable() {}
37  
38  	/** Get the command-line.
39  	 *
40  	 * @return Command-line.
41  	 */
42  	protected abstract String[] getCommandLine();
43  
44  	/** Start executing this program.
45  	 *
46  	 * If any error occurs while launching the program, this method notes a message to
47  	 * stderr, but otherwise takes no further action.
48  	 */
49  	public void execute() {
50  		// We force program execution to be asynchronous by cutting a new thread.
51  		new Thread() {
52  			public void run() {
53  				try {
54  					process = Runtime.getRuntime().exec(getCommandLine());
55  					
56  					// Read all output (standard and error output)
57  					// from the process and pass it to the user.
58  					redirect(process.getErrorStream(), System.err);
59  					redirect(process.getInputStream(), System.out);
60  				} catch (IOException ex) {
61  					System.err.println("Can't execute command \"" + getCommandLine()[0] + "\": "
62  						+ ex.getMessage());
63  				}
64  			}
65  
66  			/** Redirect the given input onto the given output.
67  			 *
68  			 * @param in The stream to read.
69  			 * @param out Where to write it.
70  			 */
71  			private void redirect(final InputStream in, final OutputStream out) {
72  				// Do this in the background, too.
73  				new Thread() {
74  					public void run() {
75  						try {
76  							byte[] buf = new byte[INT];
77  							for (;;) {
78  								int numRead = in.read(buf);
79  								if (numRead == -1) {
80  									in.close();
81  									break;
82  								}
83  								out.write(buf, 0, numRead);
84  							}
85  						} catch (IOException ex) {
86  							try {
87  								in.close();
88  							} catch (IOException ignore) {}
89  						}
90  					}
91  				}.start();
92  			}
93  		}.start();
94  
95  		// Spin until the process field is set.
96  		while (process == null) {
97  		  Thread.yield();
98  		}
99  	}
100 
101 	/** Wait for the process to terminate.
102 	 *
103 	 * @throws InterruptedException If the thread waiting for the termination is interrupted.
104 	 * @return The exit value of the process.
105 	 */
106 	public int waitFor() throws InterruptedException {
107 		return process.waitFor();
108 	}
109 
110 	/** Terminate this process.
111 	 */
112 	public void terminate() {
113 		process.destroy();
114 	}
115 
116 	/** The process backing this executable. */
117 	protected Process process;
118 
119 }
