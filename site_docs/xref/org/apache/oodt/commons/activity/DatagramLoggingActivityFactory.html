18  package org.apache.oodt.commons.activity;
19  
20  import java.io.ByteArrayOutputStream;
21  import java.io.IOException;
22  import java.io.ObjectOutputStream;
23  import java.net.DatagramPacket;
24  import java.net.DatagramSocket;
25  import java.net.InetAddress;
26  import java.net.SocketException;
27  import java.net.UnknownHostException;
28  
29  /**
30   * This activity factory produces activities that log their incidents using datagrams.  It
31   * examines the system properties to determine where to send the incidents.  The property
32   * <code>org.apache.oodt.commons.activity.DatagramLoggingActivityFactory.host</code> (or
33   * <code>activity.host</code> if not defined) teslls to what host to send the datagrams.
34   * There's no default value, and it's an runtime exception if neither property's defined.
35   *
36   * <p>The property <code>org.apache.oodt.commons.activity.DatagramLoggingActivityFactory.port</code> (or
37   * <code>activity.port</code> if not defined) tells to what port to send the datagrams.
38   * It defaults to 4556.
39   *
40   * <p>Once created, it records incidents by serializing them into a byte buffer and
41   * sending that buffer as a datagram to the activity host/port.  This does limit the size
42   * of incidents that can go over, but most incidents are pretty small.
43   *
44   * @author Kelly
45   * @version $Revision: 1.1 $
46   */
47  public class DatagramLoggingActivityFactory implements ActivityFactory {
48  
49    public static final int VAL = 4556;
50  
51    /**
52  	 * Creates a new {@link DatagramLoggingActivityFactory} instance.
53  	 */
54  	public DatagramLoggingActivityFactory() {
55  		String hostname = System.getProperty("org.apache.oodt.commons.activity.DatagramLoggingActivityFactory.host",
56  			System.getProperty("activity.host", ""));
57  		port = Integer.getInteger("org.apache.oodt.commons.activity.DatagramLoggingActivityFactory.port",
58  			Integer.getInteger("activity.port", VAL));
59  		if (hostname.length() == 0) {
60  		  throw new IllegalStateException(
61  			  "System property `org.apache.oodt.commons.activity.DatagramLoggingActivityFactory.host'"
62  			  + " (or simply `activity.host') not defined or is empty");
63  		}
64  		try {
65  			host = InetAddress.getByName(hostname);
66  			socket = new DatagramSocket();
67  		} catch (UnknownHostException ex) {
68  			throw new IllegalStateException("Activity host `" + host + "' unknown");
69  		} catch (SocketException ex) {
70  			throw new IllegalStateException("Cannot create anonymous datagram socket");
71  		}
72  	}
73  
74  	/**
75  	 * Create an activity that records incidents in a UDP datagram.
76  	 *
77  	 * @return an {@link Activity} value.
78  	 */
79  	public Activity createActivity() {
80  		return new Activity() {
81  			public void recordIncident(Incident incident) {
82  				try {
83  					ByteArrayOutputStream baos = new ByteArrayOutputStream();
84  					ObjectOutputStream oos = new ObjectOutputStream(baos);
85  					oos.writeObject(incident);
86  					oos.close();
87  					baos.close();
88  					byte[] buf = baos.toByteArray();
89  					DatagramPacket packet = new DatagramPacket(buf, buf.length, host, port);
90  					socket.send(packet);
91  				} catch (IOException ignore) {}
92  			}
93  		};
94  	}
95  
96  	/** To what host to send datagrams. */
97  	InetAddress host;
98  
99  	/** To what port on {@link #host}. */
100 	int port;
101 
102 	/** Transmission socket for datagrams. */
103 	DatagramSocket socket;
104 }
