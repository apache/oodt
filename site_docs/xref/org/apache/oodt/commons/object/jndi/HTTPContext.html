223 	/**
224 	 * Ensure the name is an HTTP product context name.
225 	 *
226 	 * @param name Name to check.
227 	 * @throws InvalidNameException If <var>name</var>'s not an RMI object context name.
228 	 */
229 	protected void checkName(String name) throws InvalidNameException {
230 		if (name == null) {
231 		  throw new IllegalArgumentException("Can't check a null name");
232 		}
233 		if (name.length() == 0) {
234 		  throw new InvalidNameException("Name's length is zero");
235 		}
236 		if (name.startsWith("http:") || name.startsWith("https:")) {
237 		  return;
238 		}
239 		throw new InvalidNameException("Not an HTTP name; try http://some.host/some-context/...");
240 	}
