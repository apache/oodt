23  /** The log.
24   *
25   * This class represents the application- or applet-wide logging facility.  If your
26   * application needs to log messages, here's your class.
27   *
28   * <p>To log a message, you call one of the <code>get</code> methods of this class to
29   * yield a {@link LogWriter} object.  You can then call methods like {@link
30   * LogWriter#println(String)} to log a line of text.  A typical invocation is
31   *
32   * <pre>Log.get().println("Buffer of length " + length + " allocated.");</pre>
33   *
34   * <p>This logs the given text using the default source string, and under the default
35   * category, and timestamped with the current time.  You can reuse the
36   * <code>LogWriter</code>, but the timestamp, source, and category won't change.  You
37   * should get a fresh <code>LogWriter</code>.
38   *
39   * <p><strong>Sources</strong> identify independent origins of log messages, such as
40   * independent threads in a program, or independent programs.  Sources are just strings.
41   * If you don't specify a source, you get a default source.  You can set the default
42   * source with {@link #setDefaultSource}.  You <em>always</em> get a source with every
43   * message, even if it's a default source.  If you don't otherwise set a default source
44   * string, the source is "app".
45   *
46   * <p><strong>Categories</strong> identify different classes or priorites of messages.
47   * Categories can be simple strings like "warning" or "debug", or they can be complex
48   * objects.  You get to define your categories.  Your group ought to agree on categories,
49   * though.  If you don't specify a category, you get a default category.  You can set the
50   * default category with {@link #setDefaultCategory}.  If you don't call that method, the
51   * default category is the String object "message".
52   *
53   * <p><strong>Streams</strong> identify independent activities within a program, which
54   * often have transient lifespans.  They're <em>not</em> separate output streams, but
55   * instead are separate, named entities representing separate activities (although a
56   * {@link LogListener} may put messages into separate output streams identified by each
57   * stream).  Activity streams are identified by strings.  To indicate the start and stop
58   * of streams, call {@link #startStream} and {@link #stopStream}.  These send
59   * stream-started and stream-stopped events to the log listeners, who may choose to pay
60   * attention to them or ignore them.  You can use streams to indicate to the log the start
61   * and the stop of activities such as an analyst examining the system, or a server
62   * handling a particular client.
63   *
64   * <p>All messages logged with this class go to one or more {@link LogListener} objects.
65   * A LogListener accepts logging events, such as a message being logged, and does
66   * something with it, such as writing the message to a file.  Call {@link #addLogListener}
67   * to add a log listener.  Logging of a message, starting a stream, and stopping a stream
68   * all get turned into {@link LogEvent}s and are multicasted to every registered listener.
69   *
70   * <p>The logging facility bootstraps itself with one or more log listeners specified by
71   * the system property <code>org.apache.oodt.commons.io.Log.loggers</code>.  This property must be a
72   * space-separated list of complete class names.  The logging facility will create an
73   * object of each class listed and add it as if you had called
74   * <code>addLogListener</code>.  (You can specify system properties on the command line or
75   * in the applet tag.)
76   *
77   * @see LogListener
78   * @see LogWriter
79   * @author Kelly
80   */
81  public class Log {
82  	/** Currently registered LogListeners.
83  	 */
84  	private static Vector listeners;
85  
86  	static {
87  		listeners = new Vector();
88  		String loggers = System.getProperty("org.apache.oodt.commons.io.Log.loggers", "");
89  		StringTokenizer tokenizer = new StringTokenizer(loggers);
90  		while (tokenizer.hasMoreTokens()) {
91  			String className = tokenizer.nextToken();
92  			try {
93  				Class clazz = Class.forName(className);
94  				LogListener listener = (LogListener) clazz.newInstance();
95  				addLogListener(listener);
96  			} catch (Exception e) {
97  				System.err.println("Can't create log listener object from class " + className + ": " + e);
98  				System.exit(1);
99  			}
100 		}
101 	}

103 	/** Get a writer to log messages.
104 	 *
105 	 * The writer will log messages with the current time, default category, and
106 	 * default source.  Messages will go into the general log.
107 	 *
108 	 * @return A writer with which you can log messages.
109 	 */
110 	public static LogWriter get() {
111 		if (lastWriter != null && !lastWriter.isFlushed()) {
112 		  return lastWriter;
113 		} else {
114 		  return get(new Date(), getDefaultSource(), getDefaultCategory());
115 		}
116 	}

130 	/** Get a writer to log messages.
131 	 *
132 	 * The writer will log messages with the specified time, specified category, and
133 	 * specified source.
134 	 *
135 	 * @param timestamp The time for messages logged with the returned writer.
136 	 * @param source The source of the log message.
137 	 * @param category The messages' category.
138 	 * @return A writer with which you can log messages.
139 	 */
140 	public static synchronized LogWriter get(Date timestamp, String source, Object category) {
141 		lastWriter = new LogWriter(timestamp, source, category);
142 		return lastWriter;
143 	}

145 	/** Start a new log stream.
146 	 *
147 	 * This method notifies the {@link LogListener}s that a new logging stream has
148 	 * started.
149 	 *
150 	 * @param stream The name of the stream.
151 	 * @param timestamp The time the stream started.  To use the current time, pass a new {@link Date} object.
152 	 * @param source A string identifying who or what started the stream.
153 	 */
154 	public static void startStream(String stream, Date timestamp, String source) {
155 		LogEvent event = null;
156 		for (Enumeration e = listeners.elements(); e.hasMoreElements();) {
157 			// Lazily create the event.
158 			if (event == null) {
159 			  event = new LogEvent(stream, timestamp, source);
160 			}
161 			((LogListener) e.nextElement()).streamStarted(event);
162 		}
163 	}
164 
165 	/** Stop a stream.
166 	 *	    
167 	 * This method notifies the {@link LogListener}s that a logging stream has stopped.
168 	 *
169 	 * @param stream The name of the stream that stopped.
170 	 */
171 	public static void stopStream(String stream) {
172 		LogEvent event = null;
173 		for (Enumeration e = listeners.elements(); e.hasMoreElements();) {
174 			// Lazily create the event.
175 			if (event == null) {
176 			  event = new LogEvent(stream);
177 			}
178 			((LogListener) e.nextElement()).streamStopped(event);
179 		}
180 	}

182 	/** Log a message.
183 	 *
184 	 * The {@link LogWriter}s call this when they've built up a complete message and
185 	 * want it multicasted to the {@link LogListener}s.
186 	 *
187 	 * @param timestamp The message's timestamp.
188 	 * @param source The source label of the message.
189 	 * @param category The message's category.
190 	 * @param message The message.
191 	 */
192 	static void logMessage(Date timestamp, String source, Object category, String message) {
193 		LogEvent event = null;
194 		for (Enumeration e = listeners.elements(); e.hasMoreElements();) {
195 			// Lazily create the event.
196 			if (event == null) {
197 			  event = new LogEvent(timestamp, source, category, message);
198 			}
199 			((LogListener) e.nextElement()).messageLogged(event);
200 		}
201 	}

245 	/** Add a log listener.
246 	 *
247 	 * The listener will be notified whenever a message is logged, a stream started,
248 	 * or a stream stopped.
249 	 *
250 	 * @param listener The listener to add.
251 	 */
252 	public static void addLogListener(LogListener listener) {
253 		if (listener == null) {
254 		  throw new IllegalArgumentException("Can't add a null log listener");
255 		}
256 		listeners.addElement(listener);

259 	/** Remove a log listener.
260 	 *

262 	 *
263 	 * @param listener The listener to remove.
264 	 */
265 	public static void removeLogListener(LogListener listener) {
266 		if (listener == null) {

269 		listeners.removeElement(listener);
