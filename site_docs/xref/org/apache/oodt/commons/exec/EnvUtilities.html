81      /**
82       * This method grabs the current environment using the Linux shell program
83       * called <code>env</code>. Unfortunately, this method also uses
84       * {@link Process#waitFor()}, which seems to hang when folks have weird
85       * environment variables and such.
86       * 
87       * A more appropriate method to call is {@link #getEnv()} because it uses
88       * the {@link ExecHelper} class, which seems to handle process termination a
89       * bit more elegantly.
90       * 
91       * @deprecated
92       * @return A {@link Properites} object containing the user's current
93       *         environment variables.
94       */
95      public static Properties getEnvUsingWaitFor() {
96          String commandLine = "env";
97  
98          Process p = null;
99          Properties envProps = null;
100 
101         try {
102             p = Runtime.getRuntime().exec(commandLine);
103             int retVal = p.waitFor();
104             envProps = new Properties();
105             envProps.load(preProcessInputStream(p.getInputStream()));
106         } catch (Exception e) {
107             LOG.log(Level.SEVERE, e.getMessage());
108             LOG.log(Level.WARNING, "Error executing env command: Message: "
109                     + e.getMessage());
110         } finally {
111             try {
112                 if (p.getErrorStream() != null) {
113                     p.getErrorStream().close();
114                 }
115             } catch (Exception ignored) {
116             }
117             try {
118                 if (p.getInputStream() != null) {
119                     p.getInputStream().close();
120                 }
121             } catch (Exception ignored) {
122             }
123             try {
124                 if (p.getOutputStream() != null) {
125                     p.getOutputStream().close();
126                 }
127             } catch (Exception ignored) {
128             }
129         }
130 
131         return envProps;
132     }

134     /**
135      * This method turns an {@link InputStream} into a String. Method taken
136      * from:<br>
137      * 
138      * <a
139      * href="http://snippets.dzone.com/posts/show/555">http://snippets.dzone.com/posts/show/555</a>.
140      * 
141      * @param in
142      *            The {@link InputStream} to Stringify.
143      * @return A String constructed from the given {@link InputStream}.
144      * @throws IOException
145      *             If any error occurs.
146      */
147     public static String slurp(InputStream in) throws IOException {
148         StringBuilder out = new StringBuilder();
149         byte[] b = new byte[INT];
150         for (int n; (n = in.read(b)) != -1;) {
151             out.append(new String(b, 0, n));
152         }
153         return out.toString();
154     }

156     protected static InputStream preProcessInputStream(InputStream is)
157         throws IOException {
158         // basically read this sucker into a BufferedReader
159         // line by line, and replaceAll on \ with \\
160         // so \\\\ with \\\\\\\\
161         BufferedReader reader = new BufferedReader(new InputStreamReader(is));
162         String line;
163         StringBuilder buf = new StringBuilder();
164 
165         while ((line = reader.readLine()) != null) {
166             // fix the line
167             line = line.replaceAll("\\\\", "\\\\\\\\");
168             buf.append(line).append("\n");
169         }
170 
171         try {
172             reader.close();
173         } catch (Exception ignore) {
174         }
175 
176         return new ByteArrayInputStream(buf.toString().getBytes());
177     }
