274     /**
275      * Executes the specified command using a shell. On windows uses cmd.exe or
276      * command.exe. On other platforms it uses /bin/sh.
277      * <p>
278      * A shell should be used to execute commands when features such as file
279      * redirection, pipes, argument parsing are desired.
280      * <p>
281      * No input is passed to the process on STDIN.
282      * 
283      * @param command
284      *            String containing a command to be parsed by the shell and
285      *            executed.
286      * @param charset
287      *            Output from the executed command is expected to be in this
288      *            character set.
289      * @return The results of the execution in an ExecHelper object.
290      * @throws SecurityException
291      *             if a security manager exists and its checkExec method doesn't
292      *             allow creation of a subprocess.
293      * @throws IOException -
294      *             if an I/O error occurs
295      * @throws NullPointerException -
296      *             if command is null
297      * 
298      * 
299      */
300     public static ExecHelper execUsingShell(String command, String charset)
301             throws IOException {
302         if (command == null) {
303           throw new NullPointerException();
304         }
305         String[] cmdarray;
306         String os = System.getProperty("os.name");
307         if (os.equals("Windows 95") || os.equals("Windows 98")
308                 || os.equals("Windows ME")) {
309             cmdarray = new String[] { "command.exe", "/C", command };
310         } else if (os.startsWith("Windows")) {
311             cmdarray = new String[] { "cmd.exe", "/C", command };
312         } else {
313             cmdarray = new String[] { "/bin/sh", "-c", command };
314         }
315         return new ExecHelper(Runtime.getRuntime().exec(cmdarray), charset);
316     }

318     /**
319      * Take a process, record its standard error and standard out streams, wait
320      * for it to finish
321      * 
322      * @param process
323      *            process to watch
324      * @throws SecurityException
325      *             if a security manager exists and its checkExec method doesn't
326      *             allow creation of a subprocess.
327      * @throws IOException -
328      *             if an I/O error occurs
329      * @throws NullPointerException -
330      *             if cmdarray is null
331      * @throws IndexOutOfBoundsException -
332      *             if cmdarray is an empty array (has length 0).
333      * 
334      * 
335      */
336     private ExecHelper(Process process, String charset) throws IOException {
337     	try {
338 	        StringBuilder output = new StringBuilder();
339 	        StringBuilder error = new StringBuilder();
340 	
341 	        Reader stdout;
342 	        Reader stderr;
343 	
344 	        if (charset == null) {
345 	            // This is one time that the system charset is appropriate,
346 	            // don't specify a character set.
347 	            stdout = new InputStreamReader(process.getInputStream());
348 	            stderr = new InputStreamReader(process.getErrorStream());
349 	        } else {
350 	            stdout = new InputStreamReader(process.getInputStream(), charset);
351 	            stderr = new InputStreamReader(process.getErrorStream(), charset);
352 	        }
353 	        char[] buffer = new char[INT];
354 	
355 	        boolean done = false;
356 	        boolean stdoutclosed = false;
357 	        boolean stderrclosed = false;
358 	        while (!done) {
359 	            boolean readSomething = false;
360 	            // read from the process's standard output
361 	            if (!stdoutclosed && stdout.ready()) {
362 	                readSomething = true;
363 	                int read = stdout.read(buffer, 0, buffer.length);
364 	                if (read < 0) {
365 	                    readSomething = true;
366 	                    stdoutclosed = true;
367 	                } else if (read > 0) {
368 	                    readSomething = true;
369 		                    output.append(buffer, 0, read);
370 	                }
371 	            }
372 	            // read from the process's standard error
373 	            if (!stderrclosed && stderr.ready()) {
374 	                int read = stderr.read(buffer, 0, buffer.length);
375 	                if (read < 0) {
376 	                    readSomething = true;
377 	                    stderrclosed = true;
378 	                } else if (read > 0) {
379 	                    readSomething = true;
380 	                    error.append(buffer, 0, read);
381 	                }
382 	            }
383 	            // Check the exit status only we haven't read anything,
384 	            // if something has been read, the process is obviously not dead
385 	            // yet.
386 	            if (!readSomething) {
387 	                try {
388 	                    this.status = process.exitValue();
389 	                    done = true;
390 	                } catch (IllegalThreadStateException itx) {
391 	                    // Exit status not ready yet.
392 	                    // Give the process a little breathing room.
393 	                    try {
394 	                        Thread.sleep(100);
395 	                    } catch (InterruptedException ix) {
396 	                        process.destroy();
397 	                        throw new IOException("Interrupted - processes killed");
398 	                    }
399 	                }
400 	            }
401 	        }
402 	
403 	        this.output = output.toString();
404 	        this.error = error.toString();
405     	}catch (Exception e) {
406     		LOG.log(Level.SEVERE, e.getMessage());
407     		throw new IOException("Process exec failed : " + e.getMessage());
408     	}finally {
409             try {
410             	process.getErrorStream().close();
411             } catch (Exception ignored) {}
412             try {
413                	process.getInputStream().close();
414             } catch (Exception ignored) {}
415             try {
416             	process.getOutputStream().close();
417             } catch (Exception ignored) {}
418     	}
419     }
