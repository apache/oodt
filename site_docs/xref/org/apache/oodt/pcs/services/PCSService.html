35  /**
36   *
37   * Base class for PCS JAX-RS services. Loads up 
38   * the {@link PCSServiceConfig} and makes it available
39   * statically to downstream descendants.
40   *
41   * @author mattmann
42   * @version $Revision$
43   *
44   */
45  public class PCSService extends HttpServlet {
46  
47    protected static final long serialVersionUID = 6256089700429938496L;
48    
49    protected static PCSServiceConfig conf;
50    
51  
52    /* (non-Javadoc)
53     * @see javax.servlet.GenericServlet#init(javax.servlet.ServletConfig)
54     */
55    @Override
56    public void init(ServletConfig config) throws ServletException {
57      super.init(config);
58      conf = new PCSServiceConfig(config);
59    }
60    
61  
62    protected class ResourceNotFoundException extends WebApplicationException {
63  
64      public ResourceNotFoundException() {
65        super(Response.status(Status.NOT_FOUND).type(MediaType.TEXT_PLAIN)
66            .build());
67      }
68  
69      public ResourceNotFoundException(String msg) {
70        super(Response.status(Status.NOT_FOUND).entity(msg).type(
71            MediaType.TEXT_PLAIN).build());
72      }
73    }
74  
75  }
