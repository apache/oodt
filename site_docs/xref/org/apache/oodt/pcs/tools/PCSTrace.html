86    /**
87     * Outputs a trace report in the following format:
88     * 
89     * Product Name: <i>oco_L1aRad09088_0939838838.hdf</i>
90     * -------------------------------------------------- Location:
91     * /preflt/jpl/Rad/9088/l1a/oco_L1aRad09088_0939838838.hdf
92     * 
93     * Metadata:
94     * 
95     * &lt;Met Field Name&gt; =&gt; &lt;Met Field Value&gt; &lt;Met Field
96     * Name2&gt; =&gt; &lt;Met Field Value2&gt;
97     * 
98     * --------------------------------------------------- Generated from
99     * workflows:
100    * 
101    * CPT Pipeline: ID: [99299-add99-92983-9d9d9-0099d] State: FINISHED
102    * Associated products:
103    * 
104    * oco_RawL017_93983993938.pkt oco_RawL032_93939388383.pkt
105    * oco_RawL033_93838383838.pkt oco_RawL034_93938378585.pkt
106    * 
107    * --------------------------------------------------- Full lineage:
108    * 
109    * Upstream: CPTOverview_98383388d.txt=&gt;oco_RawL017_93983993938.pkt=&gt;
110    * oco_RawL032_93939388383
111    * .pkt=&gt;oco_RawL033_93838383838.pkt=&gt;oco_RawL034_93938378585.pkt
112    * 
113    * Downstream: oco_L1BRad_19889_983838383.hdf
114    * ---------------------------------------------------
115    * 
116    * 
117    * @param productName
118    *          The name of the product to trace lineage for.
119    */
120   public void doTrace(String productName) {
121 
122     Product prod = fm.safeGetProductByName(productName);
123 
124     Pedigree pedigree = new Pedigree(fm, this.listNotCataloged,
125         this.excludeTypeList);
126 
127     System.out.println("");
128     System.out.println(REPORT_LINE_SEPARATOR);
129     System.out.println("Product: " + prod.getProductName());
130     System.out.println(REPORT_LINE_SEPARATOR);
131 
132     prod.setProductReferences(fm.safeGetProductReferences(prod));
133     System.out.println("Location: " + fm.getFilePath(prod));
134     System.out.println(REPORT_LINE_SEPARATOR);
135     System.out.println("Metadata: ");
136     Metadata met = fm.safeGetMetadata(prod);
137     if (met != null && met.getMap() != null
138         && met.getMap().keySet().size() > 0) {
139       for (String key : met.getMap().keySet()) {
140         List vals = met.getAllMetadata(key);
141         System.out.println(key + "=>" + vals);
142       }
143     }
144 
145     System.out.println(REPORT_LINE_SEPARATOR);
146     System.out.println("Generated from workflow:");
147     System.out.println("");
148     WorkflowInstance inst;
149 
150     try {
151       inst = getWorkflowInstanceById(wm.safeGetWorkflowInstances(), met
152           .getMetadata(JOB_ID));
153 
154       System.out.println(inst.getWorkflow().getName() + ": ID: ["
155           + inst.getId() + "]");
156       System.out.println("Status: " + inst.getStatus());
157       List wInstProds = pedigree.getWorkflowInstProds(inst.getId());
158       if (wInstProds != null && wInstProds.size() > 0) {
159         System.out.println("Associated products:");
160         System.out.println("");
161         for (Object wInstProd1 : wInstProds) {
162           Product wInstProd = (Product) wInstProd1;
163           System.out.println(wInstProd.getProductName());
164         }
165       }
166     } catch (Exception e) {
167       System.out.println("Unable to obtain workflow instance for product!");
168     }
169 
170     System.out.println(REPORT_LINE_SEPARATOR);
171     System.out.println("Full lineage:");
172     System.out.println("");
173     System.out.println("Downstream:");
174 
175     PedigreeTree downstreamLineageTree = pedigree.doPedigree(prod, false);
176     downstreamLineageTree.printPedigreeTree();
177 
178     System.out.println("");
179     System.out.println("Upstream:");
180     PedigreeTree upstreamLineageTree = pedigree.doPedigree(prod, true);
181     upstreamLineageTree.printPedigreeTree();
182 
183   }

189   /**
190    * 
191    * @param args
192    * @throws Exception
193    */
194   public static void main(String[] args) {
195     String productName = null;
196     String workflowUrlStr = null, filemgrUrlStr = null;
197     boolean enableNotCat = false;
198     List exList = null;
199 
200     String usage = "PCSTrace --fm <url> --wm <url> --product <name> "
201         + " [--enableNonCat] [--exclude <type name 1>,<type name 2>...,<type name n>]\n";
202 
203     for (int i = 0; i < args.length; i++) {
204       if (args[i].equals("--fm")) {
205         filemgrUrlStr = args[++i];
206       } else if (args[i].equals("--wm")) {
207         workflowUrlStr = args[++i];
208       } else if (args[i].equals("--product")) {
209         productName = args[++i];
210       } else if (args[i].equals("--enableNonCat")) {
211         enableNotCat = true;
212       } else if (args[i].equals("--exclude")) {
213         String[] excludeTypes = args[++i].split(",");
214         exList = Arrays.asList(excludeTypes);
215       }
216 
217     }
218 
219     if (productName == null || workflowUrlStr == null || filemgrUrlStr == null) {
220       System.err.println(usage);
221       System.exit(1);
222     }
223 
224     PCSTrace tracer = new PCSTrace(workflowUrlStr, filemgrUrlStr);
225     if (enableNotCat) {
226       tracer.enableNonCatalogProductsInPed();
227     }
228     if (exList != null) {
229       tracer.setExcludeTypeList(exList);
230     }
231     tracer.doTrace(productName);
232 
233   }

235   private WorkflowInstance getWorkflowInstanceById(List insts, String id) {
236     if (insts == null) {
237       return null;
238     }
239 
240     for (Object inst1 : insts) {
241       WorkflowInstance inst = (WorkflowInstance) inst1;
242       if (inst.getId().equals(id)) {
243         return inst;
244       }
245     }
246 
247     return null;
248   }
