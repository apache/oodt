16  package org.apache.oodt.pcs.input;
17  
18  //JDK imports
19  import java.util.List;
20  import java.util.Vector;
21  
22  /**
23   * 
24   * <p>
25   * A PGEMatrix is a set of rows and columns with values in each cell defined by
26   * a row number and column number.
27   * </p>
28   * 
29   * @author mattmann
30   * @version $Revision$
31   * 
32   */
33  public class PGEMatrix {
34  
35    private List<List<Object>> rows;
36  
37    private int numCols;
38  
39    private String name;
40  
41    /**
42     * <p>
43     * Constructs a new PGEMatrix with no rows or columns.
44     * </p>
45     */
46    public PGEMatrix() {
47      this(null, 0, 0);
48    }
49  
50    /**
51     * <p>
52     * Constructs a new PGEMatrix with the specified <code>numrows</code> and
53     * <code>numcols</code>.
54     * 
55     * @param numrows
56     *          The number of rows in the matrix.
57     * @param numcols
58     *          The number of columns for each row in the matrix.
59     */
60    public PGEMatrix(String name, int numrows, int numcols) {
61      super();
62      this.rows = new Vector<List<Object>>(numrows);
63      this.numCols = numcols;
64      this.name = name;
65  
66      for (int i = 0; i < numrows; i++) {
67        List<Object> colVector = new Vector<Object>(numcols);
68        this.rows.add(i, colVector);
69      }
70    }
71  
72    public void addValue(Object value, int row, int col) {
73      if ((row > this.rows.size() || row < 0) || (col > this.numCols || col < 0)) {
74        return;
75      }
76  
77      ((List<Object>) this.rows.get(row)).add(col, value);
78    }
79  
80    public Object getValue(int row, int col) {
81      if ((row > this.rows.size() || row < 0) || (col > this.numCols || col < 0)) {
82        return null;
83      }
84  
85      return ((List<Object>) this.rows.get(row)).get(col);
86    }
87  
88    /**
89     * @return Returns the rows.
90     */
91    public List<List<Object>> getRows() {
92      return this.rows;
93    }
94  
95    /**
96     * @param rows
97     *          The rows to set.
98     */
99    public void setRows(List<List<Object>> rows) {
100     this.rows = rows;
101   }
102 
103   /**
104    * @return Returns the name.
105    */
106   public String getName() {
107     return this.name;
108   }
109 
110   /**
111    * @param name
112    *          The name to set.
113    */
114   public void setName(String name) {
115     this.name = name;
116   }
117 
118   /**
119    * @return Returns the numCols.
120    */
121   public int getNumCols() {
122     return this.numCols;
123   }
124 
125   /**
126    * @param numCols
127    *          The numCols to set.
128    */
129   public void setNumCols(int numCols) {
130     this.numCols = numCols;
131   }
