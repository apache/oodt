22  /**

24   * <p>
25   * A PGEVector is a dynamic set of information, coupled with a name. The
26   * information may include number, or string data.
27   * </p>
28   * 
29   * @author mattmann
30   * @version $Revision$
31   * 
32   */
33  public class PGEVector {
34  
35    private String name;
36  
37    private List<Object> elements;
38  
39    /**
40     * @param name
41     * @param elements
42     */
43    public PGEVector(String name, List<Object> elements) {
44      super();
45      this.name = name;
46      this.elements = elements;
47    }
48  
49    /**
50     * 
51     */
52    public PGEVector() {
53      this(null, new Vector<Object>());
54    }
55  
56    /**
57     * @return the name
58     */
59    public String getName() {
60      return name;
61    }
62  
63    /**
64     * @param name
65     *          the name to set
66     */
67    public void setName(String name) {
68      this.name = name;
69    }
70  
71    /**
72     * @return the elements
73     */
74    public List<Object> getElements() {
75      return elements;
76    }
77  
78    /**
79     * @param elements
80     *          the elements to set
81     */
82    public void setElements(List<Object> elements) {
83      this.elements = elements;
84    }

86  }
