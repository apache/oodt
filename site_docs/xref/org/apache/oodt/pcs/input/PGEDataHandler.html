16  package org.apache.oodt.pcs.input;
17  
18  //JDK imports
19  import org.xml.sax.Attributes;
20  import org.xml.sax.SAXException;
21  import org.xml.sax.helpers.DefaultHandler;
22  
23  import java.util.Map;
24  import java.util.concurrent.ConcurrentHashMap;
25  import java.util.logging.Logger;
26  
27  /**
28   * 
29   * <p>
30   * A SAX-based event handler to parse PGE data out of an underlying XML file.
31   * </p>
32   * 
33   * 
34   * @author mattmann
35   * @version $Revision$
36   */
37  public class PGEDataHandler extends DefaultHandler implements PGEDataParseKeys {
38  
39    /* our log stream */
40    private static final Logger LOG = Logger.getLogger(PGEDataHandler.class
41        .getName());
42  
43    /* scalars to be set as tags are encountered */
44    private Map scalars = new ConcurrentHashMap();
45  
46    /* vectors to be set as tags are encountered */
47    private Map vectors = new ConcurrentHashMap();
48  
49    /* matrices to be set as tags are encountered */
50    private Map matrices = new ConcurrentHashMap();
51  
52    /* the status of the parse handler */
53    private int parseStatus = UNSET;
54  
55    private boolean vecElement;
56  
57    private boolean matrixElement;
58  
59    private PGEScalar currentScalar = null;
60  
61    private PGEVector currentVector = null;
62  
63    private PGEMatrix currentMatrix = null;
64  
65    private StringBuffer charBuf;
66  
67    // needed for parsing matrix
68    private int currentRow = 0;
69  
70    private int currentCol = 0;
71  
72    public PGEDataHandler() {
73      vecElement = false;
74      matrixElement = false;
75    }
76  
77    /*
78     * (non-Javadoc)
79     * 
80     * @see org.xml.sax.helpers.DefaultHandler#startElement(java.lang.String,
81     * java.lang.String, java.lang.String, org.xml.sax.Attributes)
82     */
83    public void startElement(String uri, String localName, String qName,
84        Attributes attributes) throws SAXException {
85      if (notParsing()) {
86        // only care about if it's a scalar/vector or matrix
87        if (qName.equals(SCALAR_TAG_NAME)) {
88          parseStatus = PARSING_SCALAR;
89          currentScalar = new PGEScalar();
90          currentScalar.setName(attributes.getValue(NAME_ATTR));
91          charBuf = new StringBuffer();
92        } else if (qName.equals(VECTOR_TAG_NAME)) {
93          parseStatus = PARSING_VEC;
94          currentVector = new PGEVector();
95          currentVector.setName(attributes.getValue(NAME_ATTR));
96        } else if (qName.equals(MATRIX_TAG_NAME)) {
97          parseStatus = PARSING_MATRIX;
98          currentMatrix = new PGEMatrix(attributes.getValue(NAME_ATTR), Integer
99              .parseInt(attributes.getValue(ROWS_ATTR)), Integer
100             .parseInt(attributes.getValue(COLS_ATTR)));
101       }
102     } else if (isParsingScalar()) {
103       // shouldn't encountere another tag here
104       throw new SAXException("Parsing scalar: [" + this.currentScalar.getName()
105           + "]: Should not encounter another tag within");
106     } else if (isParsingVector()) {
107       if (qName.equals(VECTOR_ELEMENT_TAG)) {
108         vecElement = true;
109         charBuf = new StringBuffer();
110       }
111 
112     } else if (isParsingMatrix()) {
113       if (qName.equals(MATRIX_COL_TAG)) {
114         matrixElement = true;
115         charBuf = new StringBuffer();
116       }
117     }
118 
119   }
120 
121   /*
122    * (non-Javadoc)
123    * 
124    * @see org.xml.sax.helpers.DefaultHandler#endElement(java.lang.String,
125    * java.lang.String, java.lang.String)
126    */
127   public void endElement(String uri, String localName, String qName)
128       throws SAXException {
129     if (qName.equals(SCALAR_TAG_NAME)) {
130       this.currentScalar.setValue(this.charBuf.toString());
131       this.scalars.put(this.currentScalar.getName(), this.currentScalar);
132       clearCharBuf();
133       parseStatus = UNSET;
134     } else if (qName.equals(VECTOR_ELEMENT_TAG)) {
135       this.currentVector.getElements().add(this.charBuf.toString());
136       clearCharBuf();
137       this.vecElement = false;
138     } else if (qName.equals(VECTOR_TAG_NAME)) {
139       // add the vector
140       if (this.currentVector != null) {
141         this.vectors.put(this.currentVector.getName(), this.currentVector);
142         parseStatus = UNSET;
143       }
144     } else if (qName.equals(MATRIX_TAG_NAME)) {
145       if (this.currentMatrix != null) {
146         this.matrices.put(this.currentMatrix.getName(), this.currentMatrix);
147 
148         parseStatus = UNSET;
149         this.currentCol = 0;
150         this.currentRow = 0;
151       }
152     } else if (qName.equals(MATRIX_ROW_TAG)) {
153       this.currentRow++;
154       this.currentCol = 0;
155     } else if (qName.equals(MATRIX_COL_TAG)) {
156       this.currentMatrix.addValue(this.charBuf.toString(), this.currentRow,
157           this.currentCol);
158       this.currentCol++;
159       clearCharBuf();
160       this.matrixElement = false;
161     }
162   }
163 
164   /*
165    * (non-Javadoc)
166    * 
167    * @see org.xml.sax.helpers.DefaultHandler#characters(char[], int, int)
168    */
169   public void characters(char[] ch, int start, int length) throws SAXException {
170     if (isParsingScalar() || isParsingVectorElement()
171         || isParsingMatrixElement()) {
172       this.charBuf.append(ch, start, length);
173     }
174   }

221   public boolean isParsingScalar() {
222     return parseStatus == PARSING_SCALAR;
223   }
224 
225   public boolean isParsingVector() {
226     return parseStatus == PARSING_VEC;
227   }
228 
229   public boolean isParsingMatrix() {
230     return parseStatus == PARSING_MATRIX;
231   }
232 
233   public boolean isParsingMatrixElement() {
234     return matrixElement;
235   }
236 
237   public boolean isParsingVectorElement() {
238     return vecElement;
239   }
240 
241   public boolean notParsing() {
242     return parseStatus == UNSET;
243   }

245   private void clearCharBuf() {
246     this.charBuf.setLength(0);
247     this.charBuf = null;
248   }
