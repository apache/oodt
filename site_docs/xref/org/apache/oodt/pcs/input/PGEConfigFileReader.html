32  /**
33   * 
34   * <p>
35   * A Reader for reading the constructs defined in a {@link PGEConfigurationFile}.
36   * The constructs are read and a new {@link PGEConfigurationFile} object is
37   * constructed and returned.
38   * </p>
39   * 
40   * @author mattmann
41   * @version $Revision$
42   * 
43   */
44  public class PGEConfigFileReader {

47     * <p>
48     * Default Constructor
49     * </p>
50     */
51    public PGEConfigFileReader() {

54    /**
55     * 
56     * <p>
57     * Reads the PGE Configuration file from the given <code>url</code>.
58     * </p>
59     * 
60     * @param url
61     *          The {@link URL} pointer to the pge configuration file.
62     * @return A new {@link PGEConfigurationFile} object, created from the
63     *         specified URL.
64     * @throws PGEConfigFileException
65     *           If there is an error reading the url.
66     */
67    public PGEConfigurationFile read(URL url) throws PGEConfigFileException {
68      PGEConfigurationFile configFile;
69  
70      try {
71        configFile = read(url.openStream());
72      } catch (IOException e) {
73        throw new PGEConfigFileException("Unable to read PGE "
74            + "configuration file from url: " + url + ": Message: "
75            + e.getMessage());
76      }
77  
78      return configFile;
79    }

81    /**
82     * 
83     * <p>
84     * Reads a PGEConfigurationFile from the specified <code>is</code>
85     * InputStream.
86     * </p>
87     * 
88     * @param is
89     *          The InputStream to read the PGEConfigurationFile from.
90     * @return A new {@link PGEConfigurationFile}, created from the specified
91     *         InputStream.
92     * @throws PGEConfigFileException
93     *           If any error occurs.
94     */
95    public PGEConfigurationFile read(InputStream is)
96        throws PGEConfigFileException {
97      PGEConfigurationFile configFile;
98  
99      DocumentBuilderFactory factory;
100     DocumentBuilder parser;
101     Document document;
102 
103     if (is == null) {
104       return null;
105     }
106 
107     InputSource inputSource = new InputSource(is);
108 
109     try {
110       factory = DocumentBuilderFactory.newInstance();
111       parser = factory.newDocumentBuilder();
112       document = parser.parse(inputSource);
113     } catch (Exception parseException) {
114       parseException.printStackTrace();
115       return null;
116     }
117 
118     // okay, construct the PGEConfigurationFile now
119     configFile = new PGEConfigurationFile();
120 
121     Element pgeConf = document.getDocumentElement();
122 
123     NodeList pgeGroups = pgeConf.getElementsByTagName("group");
124 
125     for (int i = 0; i < pgeGroups.getLength(); i++) {
126       // get the name of the group
127       Element group = (Element) pgeGroups.item(i);
128       String groupName = group.getAttribute("name");
129 
130       if (groupName.equals("PGENameGroup")) {
131         addPGEName(configFile, group);
132       } else if (groupName.equals("InputProductFiles")) {
133         addInputProductFiles(configFile, group);
134       } else if (groupName.equals("StaticFileIdentificationFiles")) {
135         addSFIFFiles(configFile, group);
136       } else if (groupName.equals("DynamicAuxiliaryInputFiles")) {
137         addDynamicAuxInputFiles(configFile, group);
138       } else if (groupName.equals("RecordedAuxiliaryInputFiles")) {
139         addRecAuxInputFiles(configFile, group);
140       } else if (groupName.equals("ProductPathGroup")) {
141         addProductPath(configFile, group);
142       } else if (groupName.equals("MonitorGroup")) {
143         addMonitorGroup(configFile, group);
144       } else if (groupName.equals("MonitorLevel")) {
145         addMonitorLevels(configFile, group);
146       } else {
147         // pge specific groups, just add generic groups
148         addPGESpecificGroup(configFile, group);
149       }
150     }
151 
152     return configFile;
153   }

155   private void addPGESpecificGroup(PGEConfigurationFile configFile,
156       Element group) throws PGEConfigFileException {
157 
158     List scalars = PGEXMLFileUtils.getScalars(group);
159     List vectors = PGEXMLFileUtils.getVectors(group);
160     List matrixs = PGEXMLFileUtils.getMatrixs(group);
161 
162     PGEGroup pgeGroup = new PGEGroup(group.getAttribute("name"));
163 
164     for (Object scalar : scalars) {
165       PGEScalar s = (PGEScalar) scalar;
166       pgeGroup.addScalar(s);
167     }
168 
169     for (Object vector : vectors) {
170       PGEVector v = (PGEVector) vector;
171       pgeGroup.addVector(v);
172     }
173 
174     for (Object matrix : matrixs) {
175       PGEMatrix m = (PGEMatrix) matrix;
176       pgeGroup.addMatrix(m);
177     }
178 
179     configFile.getPgeSpecificGroups().put(pgeGroup.getName(), pgeGroup);
180 
181   }

196   private void addMonitorGroup(PGEConfigurationFile configFile, Element group)
197       throws PGEConfigFileException {
198 
199     List scalars = PGEXMLFileUtils.getScalars(group);
200 
201     // the list should be not be null
202     if (scalars == null) {
203       throw new PGEConfigFileException(
204           "There is no monitor path or monitor filename format defined in the MonitorGroup!");
205     }
206 
207     PGEScalar monPath = null, monFilenameFormat = null;
208 
209     for (Object scalar1 : scalars) {
210       PGEScalar scalar = (PGEScalar) scalar1;
211 
212       if (scalar.getName().equals("MonitorPath")) {
213         monPath = scalar;
214       } else if (scalar.getName().equals("MonitorFilenameFormat")) {
215         monFilenameFormat = scalar;
216       }
217     }
218 
219     configFile.setMonitorPath(monPath);
220     configFile.setMonitorFilenameFormat(monFilenameFormat);
221 
222   }

224   private void addProductPath(PGEConfigurationFile configFile, Element group)
225       throws PGEConfigFileException {
226     List scalars = PGEXMLFileUtils.getScalars(group);
227 
228     // the list should be size 1
229     if (scalars == null || (scalars.size() != 1)) {
230       throw new PGEConfigFileException(
231           "There is no product path defined in the configuration file, or there is more than one scalar listed in the ProductPathGroup!");
232     }
233 
234     PGEScalar scalar = (PGEScalar) scalars.get(0);
235 
236     // the name of the product path should be ProductPath
237 
238     if (!scalar.getName().equals("ProductPath")) {
239       throw new PGEConfigFileException(
240           "The product path should be defined as a scalar with the name \"ProductPath\"!");
241     }
242 
243     configFile.setProductPath(scalar);
244   }

246   private void addPGEName(PGEConfigurationFile configFile, Element group)
247       throws PGEConfigFileException {
248 
249     // get the scalars, there should be only one
250     List scalars = PGEXMLFileUtils.getScalars(group);
251 
252     // the list should be size 1
253     if (scalars == null || (scalars.size() != 1)) {
254       throw new PGEConfigFileException(
255           "There is no PGEName defined in the configuration file, or there is more than one scalar listed in the PGENameGroup");
256     }
257 
258     // the name of the scalar should be PGEName
259     PGEScalar scalar = (PGEScalar) scalars.get(0);
260 
261     if (!scalar.getName().equals("PGEName")) {
262       throw new PGEConfigFileException(
263           "The name of the PGE should be defined as a scalar with the name \"PGEName\"!");
264     }
265 
266     // okay, we're set, set the PGE Name
267     configFile.setPgeName(scalar);
268 
269   }

292   private void addScalarFilesToGroup(Element group, PGEGroup pgeGroup) {
293     // get the scalars, and add them to the group
294     List scalars = PGEXMLFileUtils.getScalars(group);
295 
296     if (scalars != null && scalars.size() > 0) {
297       for (Object scalar1 : scalars) {
298         PGEScalar scalar = (PGEScalar) scalar1;
299         pgeGroup.addScalar(scalar);
300       }
301     }
302   }

309     if (vectors != null && vectors.size() > 0) {
