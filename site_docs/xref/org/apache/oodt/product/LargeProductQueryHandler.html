19  package org.apache.oodt.product;
20  
21  /**
22   * Handle requests for products that are too large to fit in an {@link
23   * org.apache.oodt.xmlquery.XMLQuery} object.
24   *
25   * @author Kelly
26   * @version $Revision: 1.1.1.1 $
27   */
28  public interface LargeProductQueryHandler extends QueryHandler {
29  	/**
30  	 * Retrieve a chunk of a product.
31  	 *
32  	 * The product is identified by a string ID.  The query handler should return a
33  	 * binary chunk of the product using the given offset and length.  If the ID isn't
34  	 * recognized, it should return null.  It should throw an exception if retrieval
35  	 * fails for some reason.
36  	 *
37  	 * @param id Product ID.
38  	 * @param offset Where in the product to get a chunk of it.
39  	 * @param length How much of the product to get.
40  	 * @return A chunk, or null if the <var>id</var> is unknown.
41  	 * @throws ProductException if an error occurs.
42  	 */
43  	byte[] retrieveChunk(String id, long offset, int length) throws ProductException;
44  
45  	/**
46  	 * Close off a product.
47  	 *
48  	 * This method indicates that the product is no longer required and its resources
49  	 * can be freed by the query handler.  If the ID is unknown, no untoward action is
50  	 * required.  It should throw an exception if there is an error during the
51  	 * resource release (such as an {@link java.io.IOException} when closing a file.
52  	 *
53  	 * @param id Product ID.
54  	 * @throws ProductException if an error occurs.
55  	 */
56  	void close(String id);
57  }
