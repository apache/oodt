19  package org.apache.oodt.security.sso;
20  
21  //JDK imports
22  import java.util.List;
23  
24  /**
25   * 
26   * The CAS java-based single sign on API.
27   * 
28   * @author mattmann
29   * @version $Revision$
30   * 
31   */
32  public interface SingleSignOn {
33  
34    /**
35     * Should return the current logged in Single Sign On username returned from
36     * the implementation-specific authentication API.
37     * 
38     * @return A string representation of the current SSO username.
39     */
40    String getCurrentUsername();
41  
42    /**
43     * Returns <code>true</code> when the user is logged in, or false otherwise.
44     * 
45     * @return True if the user is logged in, false otherwise.
46     */
47    boolean isLoggedIn();
48  
49    /**
50     * Logs the user with the provided <code>username</code> and
51     * <code>password</code> in to the SSO authentication mechanism.
52     * 
53     * @param username
54     *          The username credentials.
55     * @param password
56     *          The password credentials.
57     * @return True if the login was successful, false otherwise.
58     */
59    boolean login(String username, String password);
60  
61    /**
62     * Logs the current SSO user out of her session.
63     */
64    void logout();
65  
66    /**
67     * Should provide information (true or false) as to whether the last
68     * connection to the SSO authentication service was successful.
69     * 
70     * @return True if the last authentication was successful, false otherwise.
71     */
72    boolean getLastConnectionStatus();
73  
74    /**
75     * Obtains a user's groups from the security principal that this SSO object
76     * talks to.
77     * 
78     * @param username
79     *          The username to obtain the groups for.
80     * @return A {@link List} of string group names obtained from the security
81     *         principal.
82     **/
83    List<String> retrieveGroupsForUser(String username);
84  
85  }
