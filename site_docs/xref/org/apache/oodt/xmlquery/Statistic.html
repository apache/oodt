33  public class Statistic implements java.io.Serializable, Cloneable

35          /** Serial version unique ID. */
36          static final long serialVersionUID = 8611279755682736062L;

38          /** URL. */
39  	private String url;
40  
41  	/** Time. */
42  	private long time;
43  
44  	public Statistic()
45          {
46  		url = "UNKNOWN";
47  		time = 0;
48  	}
49  
50  	/** Constructor.
51  	 *
52  	 * @param url  The server's URL. 
53  	 * @param time The search time used by the server indicted by url.
54           */
55  	public Statistic(String url, long time) 
56  	{
57  		this.url=url;
58  		this.time= time;
59  	}
60  
61           /**
62           * Instantiates an Statistic instance from an Statistic structure in DOM node format.
63           *
64           * @param root  The DOM node.
65           */
66  	public Statistic(Node root)
67  	{
68  		Node node;
69  	        String nodeName;
70  
71  	        for (node = root.getFirstChild();
72              		node != null; node = node.getNextSibling())
73          	{
74                  	if (node instanceof Element)
75                  	{
76                      		nodeName = node.getNodeName();
77                      		if (nodeName.compareTo("url") == 0) {
78  							  url = XML.unwrappedText(node);
79  							} else
80                      		if (nodeName.compareTo("time") == 0) {
81  							  time = Long.parseLong(XML.unwrappedText(node));
82  							}
83                   	}
84           	}
85     	}
86  
87          /**
88           *  doc The org.w3c.dom.Document object.
89           */
90  	public synchronized Node toXML(Document doc) throws DOMException
91    	{
92                  Element root = doc.createElement("statistic");
93                  XML.add(root, "url", url);
94                  XML.add(root, "time", ""+time);
95                  return root;
96          }
