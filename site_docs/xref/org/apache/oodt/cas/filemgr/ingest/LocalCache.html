108     /*
109      * (non-Javadoc)
110      * 
111      * @see org.apache.oodt.cas.filemgr.ingest.Cache#sync(java.lang.String,
112      *      java.util.List)
113      */
114     public void sync(String uniqueElementName,
115             List<String> uniqueElementProductTypeNames) throws CacheException {
116         try {
117             this.uniqueElementName = uniqueElementName;            
118             this.uniqueElementProductTypeNames = uniqueElementProductTypeNames;
119             List<Product> products = new Vector<Product>();
120             for (String productType : this.uniqueElementProductTypeNames) {
121                 products.addAll(getProductsOverDateRange(
122                     this.rangeQueryElementName, productType, this.startOfQuery,
123                     this.endOfQuery));
124             }
125             clear();
126             for (Product product : products) {
127                 String value = getValueForMetadata(product, uniqueElementName);
128                 this.uniqueElements.add(value);
129             }
130         } catch (Exception e) {
131             throw new CacheException("Failed to sync with database : "
132                     + e.getMessage(), e);
133         }
134     }

202     /**
203      * Performs a {@link XmlRpcFileManagerClient#hasProduct(String)} check
204      * against a live File Manager, bypassing the {@link Cache}.
205      * 
206      * @param uniqueElementName
207      *            The product identifier element to identify whether the product
208      *            was received yet.
209      * @param uniqueElementValue
210      *            The value of the product identifier element.
211      * @param productTypeName
212      *            The {@link ProductType} of the Product that you would like to
213      *            check existence for.
214      * @return True if the given Product (identified by its
215      *         <code>uniqueElement</code>) exists, False otherwise.
216      * @throws CacheException
217      *             If any error occurs.
218      */
219     public boolean liveHasProduct(String uniqueElementName,
220             String uniqueElementValue, String productTypeName)
221             throws CacheException {
222         Query query = new Query();
223         query.addCriterion(new TermQueryCriteria(uniqueElementName,
224                 uniqueElementValue));
225         try {
226             return !(fm.query(query, fm.getProductTypeByName(productTypeName))
227                     .isEmpty());
228         } catch (Exception e) {
229             throw new CacheException(
230                     "Unable to check for product reception from file manager: ["
231                             + fm.getFileManagerUrl() + "]: Message: "
232                             + e.getMessage(), e);
233         }
234     }

267     private List<Product> getProductsOverDateRange(String elementName, String productType,
268             String startOfQuery, String endOfQuery) throws CacheException {
269         List<Product> products = new Vector<Product>();
270         try {        
271             Query query = new Query();
272             query.addCriterion(new RangeQueryCriteria(elementName,
273                     startOfQuery, endOfQuery));
274             if(this.uniqueElementProductTypeNames != null && 
275                     this.uniqueElementProductTypeNames.size() > 0){
276                 for (String productTypeName : this.uniqueElementProductTypeNames) {
277                     products.addAll(getProducts(query, productTypeName));
278                 }
279                 
280             }
281         } catch (Exception e) {
282             throw new CacheException("Failed to query for product via element "
283                     + elementName + " and range " + startOfQuery + " to "
284                     + endOfQuery + " : " + e.getMessage(), e);
285         }
286         
287         return products;
288     }
289 
290     private List<Product> getProducts(Query query, String productType)
291             throws CacheException {
292         try {
293             return fm.query(query, fm.getProductTypeByName(productType));
294         } catch (Exception e) {
295             throw new CacheException("Failed to get product list for query "
296                     + query + " : " + e.getMessage(), e);
297         }
298     }

300     private String getValueForMetadata(Product product,
301             String metadataElementName) throws CacheException {
302         try {
303             return fm.getMetadata(product).getMetadata(metadataElementName);
304         } catch (Exception e) {
305             throw new CacheException("Failed to get metadata value for "
306                     + metadataElementName + " : " + e.getMessage(), e);
307         }
308     }
