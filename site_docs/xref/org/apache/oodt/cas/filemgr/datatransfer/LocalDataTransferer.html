92     /*
93      * (non-Javadoc)
94      * 
95      * @see
96      * org.apache.oodt.cas.datatransfer.DataTransfer#transferProduct(org.apache
97      * .oodt.cas.data.structs.Product)
98      */
99     public void transferProduct(Product product) throws DataTransferException,
100          IOException {
101       // check whether or not it's a set of files, or it's actually a dir
102       // structure
103       if (product.getProductStructure().equals(Product.STRUCTURE_HIERARCHICAL)) {
104          try {
105             moveDirToProductRepo(product);
106          } catch (URISyntaxException e) {
107             LOG.log(
108                   Level.WARNING,
109                   "URI Syntax Exception when moving dir "
110                         + product.getProductReferences().get(0)
111                               .getOrigReference() + ": Message: "
112                         + e.getMessage());
113             throw new DataTransferException(e);
114          }
115       } else if (product.getProductStructure().equals(Product.STRUCTURE_FLAT)) {
116          try {
117             moveFilesToProductRepo(product);
118          } catch (URISyntaxException e) {
119             LOG.log(
120                   Level.WARNING,
121                   "URI Syntax Exception when moving files: Message: "
122                         + e.getMessage());
123             throw new DataTransferException(e);
124          }
125       } else if (product.getProductStructure().equals(Product.STRUCTURE_STREAM)) {
126             LOG.log(Level.INFO,"Streaming products are not moved.");
127       } else {
128          throw new DataTransferException(
129                "Cannot transfer product on unknown ProductStructure: "
130                      + product.getProductStructure());
131       }
132    }

141    public void retrieveProduct(Product product, File directory)
142          throws DataTransferException, IOException {
143       // check whether or not it's a set of files, or it's actually a dir
144       // structure
145       if (product.getProductStructure().equals(Product.STRUCTURE_HIERARCHICAL)) {
146          try {
147             copyDirToDir(product, directory);
148          } catch (URISyntaxException e) {
149             LOG.log(
150                   Level.WARNING,
151                   "URI Syntax Exception when moving dir "
152                         + product.getProductReferences().get(0)
153                               .getDataStoreReference() + ": Message: "
154                         + e.getMessage());
155             throw new DataTransferException(e);
156          }
157       } else if (product.getProductStructure().equals(Product.STRUCTURE_FLAT)) {
158          try {
159             copyFilesToDir(product, directory);
160          } catch (URISyntaxException e) {
161             LOG.log(
162                   Level.WARNING,
163                   "URI Syntax Exception when moving files: Message: "
164                         + e.getMessage());
165             throw new DataTransferException(e);
166          }
167       } else if (product.getProductStructure().equals(Product.STRUCTURE_STREAM)) {
168          LOG.log(Level.INFO,"Streaming products are not transfered.");
169       } else {
170          throw new DataTransferException(
171                "Cannot transfer product on unknown ProductStructure: "
172                      + product.getProductStructure());
173       }
174    }

176    @Override
177    public void deleteProduct(Product product) throws DataTransferException, IOException {
178      for (Reference ref : product.getProductReferences()) {
179        String u;
180        try {
181           u = URI.create(ref.getDataStoreReference()).toURL().getPath();
182        }
183        catch (IllegalArgumentException e) {
184          u = URI.create("file://"+ref.getDataStoreReference()).toURL().getPath();
185        }
186        File dataFile = new File(u);
187        if (!dataFile.delete()) {
188         throw new IOException(String.format("Failed to delete file %s - delete returned false",
189             dataFile));
190        }
191      }
192    }

197    public static void main(String[] args) throws DataTransferException,
198          IOException, URISyntaxException {
199       String usage = "LocalFileTransfer --productName <name> --productRepo <repo> [--dir <dirRef>] [--files <origRef 1>...<origRef N>]\n";
200 
201       MimeTypes mimeTypeRepo;
202       try {
203          mimeTypeRepo = MimeTypesFactory
204                .create(System
205                      .getProperty("org.apache.oodt.cas.filemgr.mime.type.repository"));
206       } catch (MimeTypeException e) {
207          LOG.log(Level.SEVERE, e.getMessage());
208          throw new IOException(e.getMessage());
209       }
210 
211       String productName = null;
212       String productRepo = null;
213       String transferType = null;
214       Reference dirReference = null;
215 
216       List<Reference> fileReferences = null;
217 
218       for (int i = 0; i < args.length; i++) {
219          if (args[i].equals("--dir")) {
220             transferType = "dir";
221             dirReference = new Reference();
222             dirReference.setOrigReference(new File(new URI(args[++i])).toURI()
223                   .toString());
224             LOG.log(Level.FINER,
225                   "LocalFileTransfer.main: Generated orig reference: "
226                         + dirReference.getOrigReference());
227          } else if (args[i].equals("--files")) {
228             transferType = "files";
229             fileReferences = new Vector<Reference>();
230             for (int j = i + 1; j < args.length; j++) {
231                LOG.log(Level.FINER, "LocalFileTransfer.main: Adding file ref: "
232                      + args[j]);
233                fileReferences.add(new Reference(args[j], null,
234                      new File(args[j]).length(), mimeTypeRepo
235                            .getMimeType(args[j])));
236             }
237          } else if (args[i].equals("--productName")) {
238             productName = args[++i];
239          } else if (args[i].equals("--productRepo")) {
240             productRepo = args[++i];
241          }
242       }
243 
244       if (transferType == null || (((transferType.equals("dir") && dirReference == null) || (
245           transferType.equals("files") && fileReferences == null) || (!(
246           transferType.equals("dir") || transferType
247               .equals("files"))) || productName == null || productRepo == null))) {
248          System.err.println(usage);
249          System.exit(1);
250       }
251 
252       // construct a new Product
253       Product p = new Product();
254       p.setProductName(productName);
255 
256       if (transferType.equals("dir")) {
257          p.setProductStructure(Product.STRUCTURE_HIERARCHICAL);
258          dirReference.setDataStoreReference(new File(new URI(productRepo))
259                .toURI().toURL().toExternalForm()
260                + URLEncoder.encode(p.getProductName(), "UTF-8") + "/");
261          p.getProductReferences().add(dirReference);
262          /* we'll do a simple versioning scheme ourselves: no versioning! */
263          p.getProductReferences().addAll(
264                VersioningUtils.getReferencesFromDir(new File(new URI(
265                      dirReference.getOrigReference()))));
266          VersioningUtils.createBasicDataStoreRefsHierarchical(p
267                .getProductReferences());
268       } else if (transferType.equals("files")) {
269          p.setProductStructure("Flat");
270          p.getProductReferences().addAll(fileReferences);
271          VersioningUtils.createBasicDataStoreRefsFlat(productName, productRepo,
272                p.getProductReferences());
273       }
274 
275       DataTransfer transfer = new LocalDataTransferer();
276       transfer.transferProduct(p);
277 
278    }

280    private void copyDirToDir(Product product, File directory)
281          throws IOException, URISyntaxException {
282       Reference dirRef = product.getProductReferences().get(0);
283       LOG.log(
284             Level.INFO,
285             "LocalDataTransferer: Staging Directory: "
286                   + dirRef.getDataStoreReference() + " into directory "
287                   + directory.getAbsolutePath());
288 
289      for (Reference r : product.getProductReferences()) {
290        File fileRef = new File(new URI(r.getDataStoreReference()));
291 
292        if (fileRef.isFile()) {
293          copyFile(r, directory);
294        } else if (fileRef.isDirectory()
295                   && (fileRef.list() != null && fileRef.list().length == 0)) {
296          // if it's a directory and it doesn't exist yet, we should
297          // create it
298          // just in case there's no files in it
299          File dest = new File(directory, fileRef.getName());
300          if (!new File(new URI(dest.getAbsolutePath())).exists()) {
301            LOG.log(Level.FINER, "Directory: [" + dest.getAbsolutePath()
302                                 + "] doesn't exist: creating it");
303            try {
304              FileUtils.forceMkdir(new File(new URI(dest.getAbsolutePath())));
305            } catch (IOException e) {
306              LOG.log(
307                  Level.WARNING,
308                  "Unable to create directory: ["
309                  + dest.getAbsolutePath()
310                  + "] in local data transferer");
311 
312            }
313          }
314        }
315      }
316    }

318    private void moveDirToProductRepo(Product product) throws IOException,
319          URISyntaxException {
320       Reference dirRef = product.getProductReferences().get(0);
321       LOG.log(
322             Level.INFO,
323             "LocalDataTransferer: Moving Directory: "
324                   + dirRef.getOrigReference() + " to "
325                   + dirRef.getDataStoreReference());
326 
327       // notify the file manager that we started
328       quietNotifyTransferProduct(product);
329 
330      for (Reference r : product.getProductReferences()) {
331        File fileRef = new File(new URI(r.getOrigReference()));
332 
333        if (fileRef.isFile()) {
334          moveFile(r, false);
335        } else if (fileRef.isDirectory()
336                   && (fileRef.list() != null && fileRef.list().length == 0)) {
337          // if it's a directory and it doesn't exist yet, we should
338          // create it
339          // just in case there's no files in it
340          if (!new File(new URI(r.getDataStoreReference())).exists()) {
341            LOG.log(Level.FINER, "Directory: [" + r.getDataStoreReference()
342                                 + "] doesn't exist: creating it");
343            try {
344              FileUtils.forceMkdir(new File(new URI(r.getDataStoreReference())));
345            } catch (IOException e) {
346              LOG.log(
347                  Level.WARNING,
348                  "Unable to create directory: ["
349                  + r.getDataStoreReference()
350                  + "] in local data transferer");
351            }
352          }
353        }
354      }
355 
356       // notify the file manager that we're done
357       quietNotifyProductTransferComplete(product);
358 
359    }

361    private void moveFilesToProductRepo(Product product) throws IOException,
362          URISyntaxException {
363       List<Reference> refs = product.getProductReferences();
364 
365       // notify the file manager that we started
366       quietNotifyTransferProduct(product);
367 
368      for (Reference r : refs) {
369        moveFile(r, true);
370      }
371 
372       // notify the file manager that we're done
373       quietNotifyProductTransferComplete(product);
374    }

376    private void copyFilesToDir(Product product, File directory)
377          throws IOException, URISyntaxException {
378       List<Reference> refs = product.getProductReferences();
379      for (Reference r : refs) {
380        copyFile(r, directory);
381      }
382    }

384    private void moveFile(Reference r, boolean log) throws IOException,
385          URISyntaxException {
386       if (log) {
387          LOG.log(Level.INFO,
388                "LocalDataTransfer: Moving File: " + r.getOrigReference()
389                      + " to " + r.getDataStoreReference());
390       }
391       File srcFileRef = new File(new URI(r.getOrigReference()));
392       File destFileRef = new File(new URI(r.getDataStoreReference()));
393 
394       FileUtils.copyFile(srcFileRef, destFileRef);
395    }
396 
397    private void copyFile(Reference r, File directory) throws IOException,
398          URISyntaxException {
399       File srcFileRef = new File(new URI(r.getDataStoreReference()));
400       LOG.log(Level.INFO,
401             "LocalDataTransfer: Copying File: " + r.getDataStoreReference()
402                   + " to file:" + directory.getAbsolutePath() + "/"
403                   + srcFileRef.getName());
404       FileUtils.copyFile(srcFileRef, new File(directory, srcFileRef.getName()));
405    }

407    private void quietNotifyTransferProduct(Product p) {
408       if (client == null) {
409          LOG.log(Level.WARNING,
410                "File Manager service not defined: this transfer will not be tracked");
411          return;
412       }
413 
414       try {
415          client.transferringProduct(p);
416       } catch (DataTransferException e) {
417          LOG.log(Level.SEVERE, e.getMessage());
418          LOG.log(Level.WARNING,
419                "Error notifying file manager of product transfer initiation for product: ["
420                      + p.getProductId() + "]: Message: " + e.getMessage());
421       }
422    }
423 
424    private void quietNotifyProductTransferComplete(Product p) {
425       if (client == null) {
426          LOG.log(Level.WARNING,
427                "File Manager service not defined: this transfer will not be tracked");
428          return;
429       }
430 
431       try {
432          client.removeProductTransferStatus(p);
433       } catch (DataTransferException e) {
434          LOG.log(Level.SEVERE, e.getMessage());
435          LOG.log(Level.WARNING,
436                "Error notifying file manager of product transfer completion for product: ["
437                      + p.getProductId() + "]: Message: " + e.getMessage());
438       }
439    }
