78    public void addMetadata(Metadata m, Product p) throws CatalogException {
79  
80      try {
81        // First, understand which dataset this metadata belongs to
82        // by examining the dataset_id metadata field
83        int datasetId = Integer.parseInt(m.getMetadata("dataset_id"));
84  
85        // Then, create a record for the granule in the database,
86        // and store its id.
87        int granuleId = this.createGranule(datasetId,
88            m.getMetadata("granule_filename"));
89  
90        // Then, create a record for any parameters detected in file
91        // that do not already exist in the database
92        Metadata paramMetadata = getMetadataSubset(m, Pattern.compile("param_*"));
93  
94        for (Enumeration keys = paramMetadata.getHashTable().keys(); keys
95            .hasMoreElements();) {
96  
97          String key = (String) keys.nextElement();
98          String keyName = key.substring(6); // trim "param_"
99  
100         int paramId = this.createParameter(datasetId, keyName);
101         LOG.log(Level.INFO, "Currently extracting data for variable '"
102             + keyName + "' ");
103 
104         // Create a record for each dataPoint for each parameter
105         // detected in the file
106 
107         List<String> dataPoints = m.getAllMetadata("data_" + keyName);
108 
109         LOG.log(Level.INFO, "Will now extract " + dataPoints.size()
110             + " data points for variable '" + keyName + "'... ");
111 
112         StringBuffer queryBuffer = new StringBuffer(
113             "INSERT INTO `dataPoint` (`granule_id`,`dataset_id`,`parameter_id`,"
114                 + "`latitude`,`longitude`,`vertical`,`time`,`value`) VALUES ");
115         // grab a count of items in the dataPoints list
116 
117         LOG.log(Level.INFO, "there are " + dataPoints.size()
118             + " data points in the List");
119 
120         // CGOODALE int to count how many times this thing loops
121         int iterCount = 0;
122 
123         for (String dataPoint : dataPoints) {
124 
125           // Decompose each dataPoint into its component parts:
126           // lat,lon,vertical,time,value
127           String[] components = dataPoint.split(",");
128 
129           // Append the dataPoint to the StringBuffer
130           queryBuffer.append(" ( ");
131           queryBuffer.append(granuleId); // granuleId
132           queryBuffer.append(" , ");
133           queryBuffer.append(datasetId); // datasetId
134           queryBuffer.append(" , ");
135           queryBuffer.append(paramId); // parameterId
136           queryBuffer.append(" , ");
137           queryBuffer.append(components[0]); // Lat
138           queryBuffer.append(" , ");
139           queryBuffer.append(components[1]); // Lon
140           queryBuffer.append(" , ");
141           queryBuffer.append(components[2]); // vertical
142           queryBuffer.append(" , ");
143           queryBuffer.append('"'); // Conversion from ISO to MySQL Compliant
144           // DateTime Format
145           queryBuffer.append(components[3].substring(0, 4)); // YYYY
146           queryBuffer.append("-");
147           queryBuffer.append(components[3].substring(4, 6)); // MM
148           queryBuffer.append("-");
149           queryBuffer.append(components[3].substring(6, 8)); // DD
150           queryBuffer.append(" ");
151           queryBuffer.append(components[3].substring(9, 11)); // HH
152           queryBuffer.append(":");
153           queryBuffer.append(components[3].substring(11, 13)); // mm
154           // no datasets contain seconds at this time
155           // including this code would cause some datasets to
156           // parse Z, into seconds
157           queryBuffer.append('"'); // End time
158           queryBuffer.append(" , ");
159           queryBuffer.append(components[4]); // Value
160           queryBuffer.append(" ) ");
161 
162           queryBuffer.append(',');// Statement seperator
163 
164           // Periodically commit the query
165           if (iterCount > 100) {
166 
167             queryBuffer.deleteCharAt(queryBuffer.length() - 1);
168             queryBuffer.append(";");
169 
170             // Commit the query;
171             this.commitQuery(queryBuffer, null);
172 
173             // RESET query
174             queryBuffer = new StringBuffer(
175                 "INSERT INTO `dataPoint` (`granule_id`,`dataset_id`,`parameter_id`,`latitude`,"
176                     + "`longitude`,`vertical`,`time`,`value`) VALUES ");
177 
178             // RESET iterCount
179             iterCount = 0;
180           } else {
181             // Increment the counter
182             iterCount++;
183           }
184         }
185 
186         // Commit any remaining data points
187         queryBuffer.deleteCharAt(queryBuffer.length() - 1);
188         queryBuffer.append(";");
189         this.commitQuery(queryBuffer, null);
190 
191         LOG.log(Level.INFO, "Extracted " + dataPoints.size()
192             + " data points for variable '" + keyName + "' ");
193       }
194 
195     } catch (Exception e) {
196       LOG.log(Level.WARNING,
197           "Exception adding product metadata. Message: " + e.getMessage());
198       LOG.log(Level.SEVERE, e.getMessage());
199       throw new CatalogException(e.getMessage(), e);
200     }
201   }

203   public int createGranule(int datasetId, String filename)
204       throws CatalogException, SQLException {
205     // Detect duplicate granule
206     Connection conn = null;
207     Statement statement = null;
208     boolean alreadyExists = false;
209     int granuleId = 0;
210     String queryExists = "SELECT granule_id FROM `granule` WHERE `filename`='"
211         + filename + "' AND `dataset_id`='" + datasetId + "'; ";
212     try {
213       conn = this.dataSource.getConnection();
214       conn.setAutoCommit(true);
215       statement = conn.createStatement();
216       ResultSet rs = statement.executeQuery(queryExists);
217       while (rs.next()) {
218         granuleId = rs.getInt("granule_id");
219       }
220       alreadyExists = (granuleId > 0);
221     } catch (SQLException e) {
222       LOG.log(Level.WARNING,
223           "SQL Exception querying for granule existence. Last query was: "
224               + queryExists + " , Message: " + e.getMessage());
225       throw new SQLException(e.getMessage(), e);
226     } finally {
227       if (statement != null) {
228         try {
229           statement.close();
230         } catch (SQLException ignore) {
231         }
232 
233       }
234 
235       if (conn != null) {
236         try {
237           conn.close();
238         } catch (SQLException ignore) {
239         }
240 
241       }
242     }
243 
244     if (!alreadyExists) {
245       String query = "INSERT INTO `granule` (`dataset_id`,`filename`) VALUES ('"
246           + datasetId + "','" + filename + "');";
247       try {
248         String maxQuery = "SELECT MAX(granule_id) AS max_id FROM granule";
249         granuleId = this.commitQuery(new StringBuffer(query), maxQuery);
250       } catch (Exception e) {
251         throw new CatalogException(e.getMessage(), e);
252       }
253     }
254 
255     // Return the granule id
256     return granuleId;
257   }

259   public int createParameter(int datasetId, String longName)
260       throws CatalogException, SQLException {
261     // Detect duplicate parameter
262     Connection conn = null;
263     Statement statement = null;
264     boolean alreadyExists = false;
265     int parameterId = 0;
266     String queryExists = "SELECT parameter_id FROM `parameter` WHERE `longName`='"
267         + longName + "' AND `dataset_id`='" + datasetId + "'; ";
268     try {
269       conn = this.dataSource.getConnection();
270       conn.setAutoCommit(true);
271       int count = 0;
272       statement = conn.createStatement();
273       ResultSet rs = statement.executeQuery(queryExists);
274       while (rs.next()) {
275         parameterId = rs.getInt("parameter_id");
276       }
277       alreadyExists = (parameterId > 0);
278     } catch (SQLException e) {
279       LOG.log(Level.WARNING,
280           "SQL Exception querying for parameter existence. Last query was: "
281               + queryExists + " , Message: " + e.getMessage());
282       throw new SQLException(e.getMessage(), e);
283     } finally {
284       if (statement != null) {
285         try {
286           statement.close();
287         } catch (SQLException ignore) {
288         }
289 
290       }
291 
292       if (conn != null) {
293         try {
294           conn.close();
295         } catch (SQLException ignore) {
296         }
297 
298       }
299     }
300 
301     if (!alreadyExists) {
302       // Insert the parameter into the database
303       String query = "INSERT INTO `parameter` (`longName`,`dataset_id`,`description`) VALUES ('"
304           + longName + "','" + datasetId + "','') ";
305 
306       try {
307         String maxQuery = "SELECT MAX(parameter_id) AS max_id FROM parameter";
308         parameterId = this.commitQuery(new StringBuffer(query), maxQuery);
309       } catch (Exception e) {
310         throw new CatalogException(e.getMessage(), e);
311       }
312     }
313 
314     // Return the id of the parameter
315     return parameterId;
316   }

318   public int commitQuery(StringBuffer query, String maxQuery)
319       throws SQLException {
320     Connection conn = null;
321     Statement statement = null;
322     int returnId = 0;
323 
324     // Try to commit the query
325     try {
326       conn = this.dataSource.getConnection();
327       conn.setAutoCommit(false);
328       statement = conn.createStatement();
329       statement.execute(query.toString());
330 
331       if (maxQuery != null) {
332         ResultSet rs = statement.executeQuery(maxQuery);
333         while (rs.next()) {
334           returnId = rs.getInt("max_id");
335         }
336       }
337       conn.commit();
338       return returnId;
339 
340     } catch (SQLException e) {
341       // TODO Auto-generated catch block
342       LOG.log(Level.WARNING,
343           "SQL Exception adding product metadata. Last query was: " + query
344               + " , Message: " + e.getMessage());
345       try {
346         if (conn != null) {
347           conn.rollback();
348         }
349       } catch (SQLException e2) {
350         LOG.log(
351             Level.SEVERE,
352             "Unable to rollback addMetadata transaction. Message: "
353                 + e2.getMessage());
354       }
355       throw new SQLException(e.getMessage(), e);
356     } finally {
357       if (statement != null) {
358         try {
359           statement.close();
360         } catch (SQLException ignore) {
361         }
362 
363       }
364 
365       if (conn != null) {
366         try {
367           conn.close();
368         } catch (SQLException ignore) {
369         }
370 
371       }
372     }
373   }
