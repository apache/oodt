151     public synchronized void addMetadata(Metadata m, Product product)
152             throws CatalogException {
153         if(product.getProductId()!=null && CATALOG_CACHE.containsKey(product.getProductId())) {
154             CompleteProduct p = CATALOG_CACHE.get(product.getProductId());
155 
156                 p.setMetadata(m);
157                 if (hasMetadataAndRefs(p)) {
158                     LOG.log(Level.FINE,
159                         "metadata and references present for product: ["
160                         + product.getProductId() + "]");
161                     addCompleteProductToIndex(p);
162                     // now remove its entry from the cache
163                     CATALOG_CACHE.remove(product.getProductId());
164             }
165         }
166         else{
167                 // move product from index to cache
168                 // it will be moved back after metadata is added
169                 getCompleteProductById(product.getProductId(), true, true);
170                 LOG.log(Level.FINE, "Product not found in local cache, retrieved from index");
171                 removeProduct(product);
172 
173         }
174     }

176     /*
177      * (non-Javadoc)
178      * 
179      * @see org.apache.oodt.cas.filemgr.catalog.Catalog#removeMetadata(org.apache.oodt.cas.metadata.Metadata,
180      *      org.apache.oodt.cas.filemgr.structs.Product)
181      */
182     public synchronized void removeMetadata(Metadata m, Product product)
183             throws CatalogException {
184         CompleteProduct p;
185 
186         if(product.getProductId()!=null && CATALOG_CACHE.containsKey(product.getProductId())) {
187              p = CATALOG_CACHE.get(product.getProductId());
188         }
189         else{
190             String prodId = product.getProductId();
191             p = getCompleteProductById(prodId, true, true);
192             removeProductDocument(product);
193         }
194 
195 
196 
197 
198         Metadata currMet = p.getMetadata();
199         List<String> metadataTypes = new ArrayList<String>();
200 
201         if (valLayer!=null) {
202 	        try {
203 	        		// remove metadata elements specified by validation layer
204 	        		for (Element element : valLayer.getElements(product.getProductType())) {
205 	        			metadataTypes.add(element.getElementName());
206 	        		}
207 	        } catch (ValidationLayerException e) {
208 	            LOG.log(Level.SEVERE, e.getMessage());
209 	            throw new CatalogException(
210 	                    "ValidationLayerException when trying to obtain element list for product type: "
211 	                            + product.getProductType().getName()
212 	                            + ": Message: " + e.getMessage(), e);
213 	        }
214         } else {
215         	// remove all metadata
216         	metadataTypes = currMet.getAllKeys();
217         }
218 
219         for (String name : metadataTypes) {
220             currMet.removeMetadata(name);
221         }
222 
223         p.setMetadata(currMet);
224 
225         if (hasMetadataAndRefs(p)) {
226             LOG.log(Level.FINE,
227                     "metadata and references present for product: ["
228                             + product.getProductId() + "]");
229             addCompleteProductToIndex(p);
230             // now remove its entry from the cache
231             CATALOG_CACHE.remove(product.getProductId());
232         }
233     }

240     public synchronized void addProduct(Product product)
241             throws CatalogException {
242         if(product.getProductId()!=null && CATALOG_CACHE.containsKey(product.getProductId())) {
243             throw new CatalogException(
244                 "Attempt to add a product that already existed: product: ["
245                 + product.getProductName() + "]");
246 
247 
248 
249 
250 
251         } else {
252             // haven't cached this product yet, so let's cache it
253             CompleteProduct completeProduct = new CompleteProduct();
254 
255             // NOTE: reuse existing ID if possible
256             if (product.getProductId() == null) {
257                 synchronized (completeProduct) {
258                     // now generate a unique ID for the product
259                     UUID prodUUID = generator.generateTimeBasedUUID();
260                     product.setProductId(prodUUID.toString());
261                 }
262             }
263 
264             completeProduct.setProduct(product);
265             CATALOG_CACHE.put(product.getProductId(), completeProduct);
266 
267         }
268 
269     }

276     public synchronized void modifyProduct(Product product)
277             throws CatalogException {
278         if (product.getProductId()!=null && CATALOG_CACHE.containsKey(product.getProductId())) {
279             LOG.log(Level.FINE, "Modifying product: [" + product.getProductId()
280                     + "]: found product in cache!");
281             CompleteProduct cp = CATALOG_CACHE.get(product
282                     .getProductId());
283             cp.setProduct(product);
284         } else {
285             // need to grab the metadata for the existing product, and make sure
286             // we don't lose it
287             Metadata metadata = getMetadata(product);
288 
289             CompleteProduct completeProduct = new CompleteProduct();
290             completeProduct.setMetadata(metadata);
291 
292             // now remove the product's document from the catalog
293             removeProductDocument(product);
294 
295             // now add it back
296             completeProduct.setProduct(product);
297             addCompleteProductToIndex(completeProduct);
298         }
299     }

382     private CompleteProduct getCompleteProductById(String productId,
383             boolean getRefs, boolean getMet) throws CatalogException {
384         IndexSearcher searcher = null;
385         try {
386             try {
387                 reader = DirectoryReader.open(indexDir);
388             } catch (IOException e) {
389                 e.printStackTrace();
390             }
391 
392             searcher = new IndexSearcher(reader);
393             Term productIdTerm = new Term("product_id", productId);
394             org.apache.lucene.search.Query query = new TermQuery(productIdTerm);
395             TopDocs topDocs = searcher.search(query,1);
396 
397             ScoreDoc[] hits = topDocs.scoreDocs;
398 
399             // should be exactly 1 hit
400             if (topDocs.totalHits == 0) {
401             	throw new CatalogException("Product: [" + productId + "] NOT found in the catalog!");
402             }
403             if (topDocs.totalHits > 1) {
404                 throw new CatalogException("Product: [" + productId+ "] is not unique in the catalog!");
405             }
406 
407             Document productDoc = searcher.doc(hits[0].doc);
408             return toCompleteProduct(productDoc, getRefs,
409                     getMet);
410         } catch (IOException e) {
411             LOG.log(Level.WARNING,
412                     "IOException when opening index directory: ["
413                             + indexFilePath + "] for search: Message: "
414                             + e.getMessage());
415             throw new CatalogException(e.getMessage(), e);
416         } finally {
417             if (searcher != null) {
418                 try {
419                     //TODO shutdown  reader
420                 } catch (Exception ignore) {
421                 }
422             }
423         }
424     }

512     private List<Product> getProducts(boolean getRefs) throws CatalogException {
513         IndexSearcher searcher = null;
514         List<Product> products = null;
515 
516         try {
517             try {
518                 reader = DirectoryReader.open(indexDir);
519             } catch (IOException e) {
520                 e.printStackTrace();
521             }
522             searcher = new IndexSearcher(reader);
523             Term productIdTerm = new Term("myfield", "myvalue");
524             org.apache.lucene.search.Query query = new TermQuery(productIdTerm);
525             Sort sort = new Sort(new SortField("CAS.ProductReceivedTime",
526                     SortField.Type.STRING, true));
527             //TODO FIX NUMBER OF RECORDS
528             TopDocs check = searcher.search(query, 1, sort);
529             TopDocs topDocs = searcher.search(query, check.totalHits, sort);
530 
531             ScoreDoc[] hits = topDocs.scoreDocs;
532 
533             // should be > 0 hits
534             if (hits.length > 0) {
535                 products = new Vector<Product>(hits.length);
536                 for (ScoreDoc hit : hits) {
537                     Document productDoc = searcher.doc(hit.doc);
538                     CompleteProduct prod = toCompleteProduct(productDoc,

541                 }

544                         "Request for products returned no results");

546             }

549             LOG.log(Level.WARNING,
550                     "IOException when opening index directory: ["
551                             + indexFilePath + "] for search: Message: "

556                 try {
557                     //TODO close searcher
558                 } catch (Exception ignore) {
559                 }

678     public Metadata getReducedMetadata(Product product, List<String> elements) throws CatalogException {
679         Metadata fullMetadata = getMetadata(product);
680         Metadata reducedMetadata = new Metadata();
681         for (String element : elements) {
682             if (fullMetadata.containsKey(element)) {
683                 reducedMetadata.replaceMetadata(element, fullMetadata.getAllMetadata(element));
684             }
685         }
686         return reducedMetadata;
687     }

691      * 
692      * @see org.apache.oodt.cas.filemgr.catalog.Catalog#query(org.apache.oodt.cas.filemgr.structs.Query,
693      *      org.apache.oodt.cas.filemgr.structs.ProductType)
694      */
695     public List<String> query(Query query, ProductType type) throws CatalogException {
696         // paginate products returns full products, but the query method
697         // is expected to return product ids
698         List<Product> fullProducts = paginateQuery(query, type, -1, null);
699         List<String> productIds = null;
700 
701         if (fullProducts != null && fullProducts.size() > 0) {
702             productIds = new Vector<String>(fullProducts.size());

1090     private CompleteProduct toCompleteProduct(Document doc, boolean getRefs,
1091             boolean getMetadata) {
1092         Product product = new Product();
1093         Metadata metadata = new Metadata();
1094         CompleteProduct completeProduct = new CompleteProduct();
1095 
1096         product.setProductId(doc.get("product_id"));
1097         product.setProductName(doc.get("product_name"));
1098         product.setProductStructure(doc.get("product_structure"));
1099         product.setTransferStatus(doc.get("product_transfer_status"));
1100 
1101         ProductType type = new ProductType();
1102         type.setDescription(doc.get("product_type_desc"));
1103         type.setProductTypeId(doc.get("product_type_id"));
1104         type.setName(doc.get("product_type_name"));
1105         type.setProductRepositoryPath(doc.get("product_type_repoPath"));
1106         type.setVersioner(doc.get("product_type_versioner"));
1107         product.setProductType(type);
1108 
1109         if (getMetadata) {
1110             List<String> names = new ArrayList<String>();
1111 
1112             if (valLayer!=null) {
1113             	// only add metadata elements specified by validation layer
1114 	            try {
1115 	                for (Element element : valLayer.getElements(type)) {
1116 	                	names.add(element.getElementName());
1117 	                }
1118 	            } catch (ValidationLayerException e) {
1119 	                LOG.log(Level.WARNING,
1120 	                        "Unable to obtain metadata for product: ["
1121 	                                + product.getProductName() + "]: Message: "
1122 	                                + e.getMessage());
1123 	            }
1124             } else {
1125             	// add all metadata elements found in document
1126             	List<IndexableField> fields = doc.getFields();
1127                 for(IndexableField field: fields){
1128                     if (!names.contains(field.name())) {
1129                         names.add(field.name());
1130                     }
1131                 }
1132             	
1133             }
1134 
1135             // loop over field names to add to metadata
1136             for (String name : names) {
1137             		if (metadata.getAllMetadata(name)==null || metadata.getAllMetadata(name).size()==0) {
1138 	                String[] elemValues = doc.getValues(name);
1139 	                	
1140 	                if (elemValues != null && elemValues.length > 0) {
1141                         for (String elemValue : elemValues) {
1142                             metadata.addMetadata(name, elemValue);
1143                         }
1144 	                }
1145             		}
1146             }
1147 
1148             completeProduct.setMetadata(metadata);
1149         }
1150 
1151         if (getRefs) {
1152             // now add the references
1153             String[] origRefs = doc.getValues("reference_orig");
1154             String[] dataStoreRefs = doc.getValues("reference_data_store");
1155             String[] refLengths = doc.getValues("reference_fileSize");
1156             String[] refMimeTypes = doc.getValues("reference_mimeType");
1157 
1158             if ((origRefs.length == dataStoreRefs.length)
1159                     && (origRefs.length == refLengths.length)) {
1160                 List<Reference> references = new Vector<Reference>();
1161                 for (int i = 0; i < origRefs.length; i++) {
1162                     Reference r = new Reference();
1163                     r.setOrigReference(origRefs[i]);
1164                     r.setDataStoreReference(dataStoreRefs[i]);
1165                     r.setFileSize((Long.parseLong(refLengths[i])));
1166                     if (refMimeTypes != null) {
1167                         r.setMimeType(refMimeTypes[i]);
1168                     }
1169                     references.add(r);
1170                 }
1171 
1172                 product.setProductReferences(references);
1173             } else {
1174                 LOG.log(Level.WARNING, "Number of original refs: ["
1175                         + origRefs.length + "] for product: ["
1176                         + product.getProductName()
1177                         + "] not equivalent to number of data store refs: ["
1178                         + dataStoreRefs.length
1179                         + "]: Skipping product references");
1180             }
1181         }
1182 
1183         completeProduct.setProduct(product);
1184         return completeProduct;
1185     }

1198                 .add(new Field("product_transfer_status", product

1202         doc
1203                 .add(new Field("product_type_id", product.getProductType()
1204                         .getProductTypeId(), StringField.TYPE_STORED));
1205         doc.add(new Field("product_type_name", product.getProductType()
1206                 .getName(), StringField.TYPE_STORED));
1207         doc.add(new Field("product_type_desc", product.getProductType()
1208                 .getDescription() != null ? product.getProductType()
1209                 .getDescription() : "", StringField.TYPE_STORED));
1210         doc.add(new Field("product_type_repoPath", product.getProductType()
1211                 .getProductRepositoryPath() != null ? product.getProductType()
1212                 .getProductRepositoryPath() : "", StringField.TYPE_STORED));
1213         doc.add(new Field("product_type_versioner", product.getProductType()
1214                 .getVersioner() != null ? product.getProductType()
1215                 .getVersioner() : "", StringField.TYPE_STORED));
1216         
1217         // write metadata fields to the Lucene document
1218         List<String> keys = new ArrayList<String>();
1219         // validation layer: add only specifically configured keys
1220         if (valLayer!=null) {
1221         	List<Element> elements = quietGetElements(product.getProductType());
1222             for (Element element : elements) {
1223                 String key = element.getElementName();
1224                 keys.add(key);
1225             }
1226         // no validation layer: add all keys that are NOT already in doc
1227         // (otherwise some keys such as the product_* keys are duplicated)
1228         } else {
1229         	for (String key : metadata.getAllKeys()) {
1230         		if (doc.getField(key)==null) {
1231         				keys.add(key);
1232         		}
1233         	}
1234         }
1235 
1236 
1237         for (String key : keys) {
1238           List<String> values = metadata.getAllMetadata(key);
1239 
1240             if (values == null) {
1241                 LOG
1242                         .log(
1243                                 Level.WARNING,
1244                                 "No Metadata specified for product ["
1245                                         + product.getProductName()
1246                                         + "] for required field ["
1247                                         + key
1248                                         + "]: Attempting to continue processing metadata");
1249                 continue;
1250             }
1251 
1252             for (String val : values) {
1253                 doc.add(new Field(key, val, StringField.TYPE_STORED));
1254                 if(values.size()==1) {
1255                     doc.add(new SortedDocValuesField(key, new BytesRef(val)));
1256                 }
1257             }
1258         }
1259 
1260         // add the product references
1261         for (Reference r : product.getProductReferences()) {
1262             doc.add(new Field("reference_orig", r.getOrigReference(),
1263                     StringField.TYPE_STORED));
1264             doc
1265                 .add(new Field("reference_data_store", r
1266                     .getDataStoreReference(), StringField.TYPE_STORED));
1267             doc.add(new Field("reference_fileSize", String.valueOf(r
1268                 .getFileSize()), StringField.TYPE_STORED));
1269             doc.add(new Field("reference_mimeType", r.getMimeType() != null ? r
1270                 .getMimeType().getName() : "", StringField.TYPE_STORED));
1271         }
1272 
1273         // add special field for all products
1274         // then can use that field to retrieve back all products
1275         doc.add(new Field("myfield", "myvalue", StringField.TYPE_STORED));
1276 
1277         return doc;
1278     }

1451     private org.apache.lucene.search.Query getQuery(QueryCriteria queryCriteria) throws CatalogException {
1452         if (queryCriteria instanceof BooleanQueryCriteria) {
1453             BooleanQuery.Builder booleanQuery = new BooleanQuery.Builder();
1454             BooleanClause.Occur occur;
1455             switch (((BooleanQueryCriteria) queryCriteria).getOperator()) {
1456             case BooleanQueryCriteria.AND:
1457                 occur = BooleanClause.Occur.MUST;
1458                 break;
1459             case BooleanQueryCriteria.OR:
1460                 occur = BooleanClause.Occur.SHOULD;
1461                 break;
1462             case BooleanQueryCriteria.NOT:
1463                 occur = BooleanClause.Occur.MUST_NOT;
1464                 booleanQuery.add(new WildcardQuery(new Term(((BooleanQueryCriteria) queryCriteria)
1465                         .getTerms().get(0).getElementName(), "*")), BooleanClause.Occur.SHOULD);
1466                 break;
1467             default:
1468                 throw new CatalogException("Invalid BooleanQueryCriteria opertor [" 
1469                         + ((BooleanQueryCriteria) queryCriteria).getOperator() + "]");
1470             }
1471             for (QueryCriteria qc : ((BooleanQueryCriteria) queryCriteria).getTerms()) {
1472                 booleanQuery.add(this.getQuery(qc), occur);
1473             }
1474 
1475             return booleanQuery.build();
1476         } else if (queryCriteria instanceof TermQueryCriteria) {
1477             String val = ((TermQueryCriteria) queryCriteria).getValue();
1478             return new TermQuery(new Term(queryCriteria.getElementName(), val));
1479         } else if (queryCriteria instanceof RangeQueryCriteria) {
1480             String startVal = ((RangeQueryCriteria) queryCriteria).getStartValue();
1481             String endVal = ((RangeQueryCriteria) queryCriteria).getEndValue();
1482             boolean inclusive = ((RangeQueryCriteria) queryCriteria).getInclusive();
1483             Term startTerm = null;
1484             if (!startVal.equals("")) {
1485                 startTerm = new Term(queryCriteria.getElementName(), startVal);
1486             }
1487 
1488             return TermRangeQuery.newStringRange(startTerm.field(), startVal, endVal, inclusive,inclusive);
1489         }else {
1490             throw new CatalogException("Invalid QueryCriteria ["
1491                     + queryCriteria.getClass().getCanonicalName() + "]");
1492         }
