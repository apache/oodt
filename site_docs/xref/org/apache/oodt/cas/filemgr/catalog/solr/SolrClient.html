63  	/**
64  	 * Method to send one or more documents to be indexed to Solr.
65  	 * 
66  	 * @param docs
67  	 * @param commit
68  	 * @param mimeType : the mime-type format of the documents
69  	 * @return
70  	 * @throws CatalogException
71  	 */
72  	public String index(List<String> docs, boolean commit, String mimeType) throws CatalogException {
73  		
74  		try {
75  			
76  			final String url = this.buildUpdateUrl();
77  			
78  			// build message
79  			StringBuilder message = new StringBuilder("<add>");
80  			for (String doc : docs) {
81  				message.append(doc);
82  			}
83  			message.append("</add>");	
84  	    
85  	    // send POST request
86  	    LOG.info("Posting message:"+message+" to URL:"+url);
87  	    String response = doPost(url, message.toString(), mimeType);
88  	    LOG.info(response);
89  	    
90  	    // commit changes ?
91  	    if (commit) {
92  		  this.commit();
93  		}
94  	    
95  	    LOG.info(response);
96  	    return response;
97      
98  		} catch(Exception e) {
99  			LOG.log(Level.SEVERE, e.getMessage());
100 			throw new CatalogException(e.getMessage());
101 		}
102     
103 	}

230 	/**
231 	 * Method to execute a GET request to the given URL with the given parameters.
232 	 * @param url
233 	 * @param parameters
234 	 * @return
235 	 */
236 	private String doGet(String url, Map<String, String[]> parameters, String mimeType)
237 		throws IOException, CatalogException {
238 				    
239 		// build HTTP/GET request
240     GetMethod method = new GetMethod(url);
241     List<NameValuePair> nvps = new ArrayList<NameValuePair>();
242     for (Map.Entry<String, String[]> key : parameters.entrySet()) {
243     	for (String value : key.getValue()) {
244     		nvps.add(new NameValuePair(key.getKey(), value));
245     	}
246     }
247     // request results in JSON format
248     if (mimeType.equals(Parameters.MIME_TYPE_JSON)) {
249 	  nvps.add(new NameValuePair("wt", "json"));
250 	}
251     method.setQueryString( nvps.toArray( new NameValuePair[nvps.size()] ) );
252     LOG.info("GET url: "+url+" query string: "+method.getQueryString());
253     
254     // send HTTP/GET request, return response
255     return doHttp(method);
256     
257   }

259 	/**
260 	 * Method to execute a POST request to the given URL.
261 	 * @param url
262 	 * @param document
263 	 * @return
264 	 */
265 	private String doPost(String url, String document, String mimeType) throws IOException, CatalogException {
266     
267 		// build HTTP/POST request
268     PostMethod method = new PostMethod(url);
269     StringRequestEntity requestEntity = new StringRequestEntity(document, mimeType, "UTF-8");
270     method.setRequestEntity(requestEntity);
271     
272     // send HTTP/POST request, return response
273     return doHttp(method);
274 		
275 	}

277 	/**
278 	 * Common functionality for HTTP GET and POST requests.
279 	 * @param method
280 	 * @return
281 	 * @throws Exception
282 	 */
283 	private String doHttp(HttpMethod method) throws IOException, CatalogException {
284 		
285 		StringBuilder response = new StringBuilder();
286 		BufferedReader br = null;
287 		try {
288 			
289 	    // send request
290 	    HttpClient httpClient = new HttpClient();
291             // OODT-719 Prevent httpclient from spawning closewait tcp connections
292             method.setRequestHeader("Connection", "close");
293 
294 	    int statusCode = httpClient.executeMethod(method);	    
295 	    
296 	    // read response
297 	    if (statusCode != HttpStatus.SC_OK) {
298 	    	
299 	    	// still consume the response
300 	    	method.getResponseBodyAsString();
301 	      throw new CatalogException("HTTP method failed: " + method.getStatusLine());
302 	      
303 	    } else {
304 	
305 		    // read the response body.
306 		    br = new BufferedReader(new InputStreamReader(method.getResponseBodyAsStream()));
307         String readLine;
308         while(((readLine = br.readLine()) != null)) {
309           response.append(readLine);
310         }
311 	    
312 	    }
313 	  	
314 	  } finally {
315 	    // must release the connection even if an exception occurred
316 	    method.releaseConnection();
317 	    if (br!=null) {
318 		  try {
319 			br.close();
320 		  } catch (Exception ignored) {
321 		  }
322 		}
323 	  }  
324   
325 	  return response.toString();
326   
327 	}
