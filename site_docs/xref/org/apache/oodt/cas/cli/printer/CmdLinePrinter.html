17  package org.apache.oodt.cas.cli.printer;
18  
19  //JDK imports
20  import java.util.List;
21  import java.util.Set;
22  
23  //OODT imports
24  import org.apache.oodt.cas.cli.action.CmdLineAction;
25  import org.apache.oodt.cas.cli.option.CmdLineOption;
26  import org.apache.oodt.cas.cli.option.validator.CmdLineOptionValidator.Result;
27  
28  /**
29   * Printer which is responsible for generating a {@link String} representation
30   * of help, validation errors, and missing required options errors.
31   *
32   * @author bfoster (Brian Foster)
33   */
34  public interface CmdLinePrinter {
35  
36     /**
37      * Should generate help message for action specified by
38      * .
39      * 
40      * @param action
41      *           {@link CmdLineAction} for which help will be generate
42      * @param options
43      *           Supported {@link CmdLineOption}s
44      * @return Help message for specified action
45      */
46     String printActionHelp(CmdLineAction action, Set<CmdLineOption> options);
47  
48     /**
49      * Generates help messages given {@link CmdLineAction}s.
50      * 
51      * @param actions
52      *           {@link CmdLineAction}s to print help for
53      * @return Help message for given {@link CmdLineAction}s
54      */
55     String printActionsHelp(Set<CmdLineAction> actions);
56  
57     /**
58      * Generates help message for given {@link CmdLineOption}s.
59      * 
60      * @param options
61      *           {@link CmdLineOption}s for which help message will be generated
62      * @return Help message for given {@link CmdLineOption}s
63      */
64     String printOptionsHelp(Set<CmdLineOption> options);
65  
66     /**
67      * Generate validation error message for results of failed validations.
68      * 
69      * @param results
70      *           {@link List} of FAILed validations
71      * @return Generated validation error message
72      */
73     String printOptionValidationErrors(List<Result> results);
74  
75     /**
76      * Generates missing required options error message from given
77      * {@link CmdLineOption}s which where required and not specified.
78      * 
79      * @param missingOptions
80      *           {@link Set} of {@link CmdLineOption}s which where required and
81      *           not set.
82      * @return Generated missing required options error message.
83      */
84     String printRequiredOptionsMissingError(
85         Set<CmdLineOption> missingOptions);
86  
87     /**
88      * Generates {@link CmdLineAction} message from list of messages.
89      *
90      * @param messages The messages from a {@link CmdLineAction}
91      * @return printed message from list of messages.
92      */
93     String printActionMessages(List<String> messages);
94  }
