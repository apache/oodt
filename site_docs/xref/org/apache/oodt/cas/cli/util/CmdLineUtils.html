63      * Determines which of the given {@link CmdLineOption}s are either always
64      * required and are required because the given {@link CmdLineAction} was
65      * specified.
66      * 
67      * @param action
68      *           The {@link CmdLineAction} which was specified.
69      * @param options
70      *           The {@link CmdLineOption}s in question of being required or not.
71      * @return The {@link Set} of {@link CmdLineOption}s which are either always
72      *         required or are required because the given {@link CmdLineAction}
73      *         was specified.

75     public static Set<CmdLineOption> determineRequired(CmdLineAction action,
76           Set<CmdLineOption> options) {
77        return determineRequired(action, options, false);
78     }

80     /**
81      * Same as method above but optionally allows action options to be ignored.
82      */
83     public static Set<CmdLineOption> determineRequired(CmdLineAction action,
84           Set<CmdLineOption> options, boolean ignoreActionOption) {
85        Validate.notNull(action);
86        Validate.notNull(options);
87  
88        Set<CmdLineOption> requiredOptions = Sets.newHashSet();
89        for (CmdLineOption option : options) {
90           if (!(ignoreActionOption && isActionOption(option))
91                 && isRequired(action, option)) {
92              requiredOptions.add(option);
93           }
94        }
95        return requiredOptions;
96     }

98     /**
99      * Determines the given {@link GroupCmdLineOption}'s sub-options which
100     * affect the given {@link CmdLineAction}.
101     * 
102     * @param action
103     *           The {@link CmdLineAction} for which given
104     *           {@link GroupCmdLineOption}'s sub-options will be determined
105     *           relevant or not
106     * @param option
107     *           The {@link GroupCmdLineOption} whose sub-options are in question
108     *           of being relevant to the given {@link CmdLineAction}
109     * @return The {@link GroupCmdLineOption}'s sub-options who are relevant to
110     *         given {@link CmdLineAction}
111     */
112    public static Set<CmdLineOption> determineRelevantSubOptions(
113          CmdLineAction action, GroupCmdLineOption option) {
114       Set<CmdLineOption> relevantOptions = Sets.newHashSet();
115       for (GroupSubOption subOption : option.getSubOptions()) {
116          if (subOption.getOption().getRequirementRules().isEmpty()
117                || isRequired(action, subOption.getOption())
118                || isStrictlyOptional(action, subOption.getOption())) {
119             relevantOptions.add(subOption.getOption());
120          }
121       }
122       return relevantOptions;
123    }

174    /**
175     * Determines if the given {@link CmdLineOption} is required because the
176     * given {@link CmdLineAction} was specified or because it is always required.
177     * 
178     * @param action
179     *           The {@link CmdLineAction} which was specified.
180     * @param option
181     *           The {@link CmdLineOption} in question of being required or not.
182     * @return True is option is required, false otherwise.
183     */
184    public static boolean isRequired(CmdLineAction action, CmdLineOption option) {
185       Validate.notNull(option);
186       Validate.notNull(action);
187 
188       for (RequirementRule requirementRule : option.getRequirementRules()) {
189          if (requirementRule.getRelation(action) == Relation.REQUIRED) {
190             return true;
191          }
192       }
193       return option.getRequirementRules().isEmpty() && option.isRequired();
194    }

221    /**
222     * Determines if the given {@link CmdLineOption} is optional either because
223     * it is always optional or because the given {@link CmdLineAction} was
224     * specified.
225     * 
226     * @param action
227     *           The {@link CmdLineAction} which was specified.
228     * @param option
229     *           The {@link CmdLineOption} in question of being optional or not.
230     * @return True is option is optional, false otherwise.
231     */
232    public static boolean isOptional(CmdLineAction action, CmdLineOption option) {
233       Validate.notNull(action);
234       Validate.notNull(option);
235 
236       if (isPerformAndQuitOption(option)) {
237          return false;
238       }
239 
240       for (RequirementRule requirementRule : option.getRequirementRules()) {
241          if (requirementRule.getRelation(action) == Relation.OPTIONAL) {
242             return true;
243          }
244       }
245       return option.getRequirementRules().isEmpty() && !option.isRequired();
246    }

319    /**
320     * Sorts {@link CmdLineOption}s by requirement levels. {@link CmdLineOption}s
321     * which are always required have highest sort score, followed by
322     * {@link CmdLineOption}s which have {@link RequirementRule}s, followed by
323     * all others.
324     * 
325     * @param options
326     *           The {@link Set} of {@link CmdLineOption}s to sort.
327     * @return The {@link CmdLineOption}s sorted by requirement.
328     */
329    public static List<CmdLineOption> sortOptionsByRequiredStatus(
330          Set<CmdLineOption> options) {
331       Validate.notNull(options);
332 
333       List<CmdLineOption> optionsList = Lists.newArrayList(options);
334       Collections.sort(optionsList, new Comparator<CmdLineOption>() {
335          public int compare(CmdLineOption option1, CmdLineOption option2) {
336             int thisScore = (option1.isRequired() ? 2 : 0)
337                   + (!option1.getRequirementRules().isEmpty() ? 1 : 0);
338             int compareScore = (option2.isRequired() ? 2 : 0)
339                   + (!option2.getRequirementRules().isEmpty() ? 1 : 0);
340             if (thisScore == compareScore) {
341                return option2.getLongOption()
342                      .compareTo(option1.getLongOption());
343             } else {
344                return Integer.valueOf(thisScore).compareTo(compareScore);
345             }
346          }
347       });
348       Collections.reverse(optionsList);
349       return optionsList;
350    }

386    /**
387     * Finds {@link CmdLineOption} whose short name or long name equals given
388     * option name.
389     * 
390     * @param optionName
391     *           The short or long name of the {@link CmdLineOption} to find
392     * @param options
393     *           The {@link CmdLineOption}s to search in
394     * @return The {@link CmdLineOption} found or null if not found.
395     */
396    public static CmdLineOption getOptionByName(String optionName,
397          Set<CmdLineOption> options) {
398       Validate.notNull(optionName);
399       Validate.notNull(options);
400 
401       for (CmdLineOption option : options) {
402          if (option.getLongOption().equals(optionName)
403                || option.getShortOption().equals(optionName)) {
404             return option;
405          } else if (isGroupOption(option)) {
406             Set<CmdLineOption> subOptions = Sets.newHashSet(); 
407             for (GroupSubOption subOption : asGroupOption(option).getSubOptions()) {
408                subOptions.add(subOption.getOption());
409             }
410             CmdLineOption foundOption = getOptionByName(optionName, subOptions);
411             if (foundOption != null) {
412                return foundOption;
413             }
414          }
415       }
416       return null;
417    }

446     * {@link CmdLineOption}.
447     * 
448     * @param option
449     *           {@link CmdLineOption} which is being checked if the other given

452     *           The {@link CmdLineOption} which is being check if it is a
453     *           sub-option of the other given {@link CmdLineOption}
454     * @return True if subOption is a sub-option of option

456    public static boolean isSubOption(CmdLineOption option,
457          CmdLineOption subOption) {
458       Validate.notNull(option);
459       Validate.notNull(subOption);
460 
461       if (isGroupOption(option)) {
462          for (GroupSubOption curSubOption : asGroupOption(option)
463                .getSubOptions()) {
464             if (curSubOption.getOption().equals(subOption)) {
465                return true;
466             }
467          }
468       }
469       return false;
470    }

1054       String[] splitStrings = StringUtils.split(string, " ");
