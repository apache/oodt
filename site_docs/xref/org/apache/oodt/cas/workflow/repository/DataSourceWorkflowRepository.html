74    public Workflow getWorkflowByName(String workflowName, boolean getTasks,
75        boolean getConditions) throws RepositoryException {
76      Connection conn = null;
77      Statement statement = null;
78      ResultSet rs = null;
79  
80      Workflow workflow = null;
81  
82      try {
83        if(dataSource!=null) {
84          conn = dataSource.getConnection();
85        }
86        else{
87          throw new RepositoryException("Null datasource");
88        }
89        statement = conn.createStatement();
90  
91        String getWorkflowSql = "SELECT * from workflows WHERE workflow_name = '"
92            + workflowName + "'";
93  
94        LOG.log(Level.FINE, "getWorkflowByName: Executing: " + getWorkflowSql);
95        rs = statement.executeQuery(getWorkflowSql);
96  
97        while (rs.next()) {
98          workflow = DbStructFactory.getWorkflow(rs);
99  
100         if (getTasks) {
101           workflow.setTasks(getTasksByWorkflowId(workflow.getId()));
102         }
103 
104         if (getConditions) {
105           workflow.setConditions(getConditionsByWorkflowId(workflow.getId()));
106           handleGlobalWorkflowConditions(workflow);
107         }
108       }
109 
110     } catch (Exception e) {
111       LOG.log(Level.SEVERE, e.getMessage());
112       LOG.log(Level.WARNING,
113           "Exception getting workflow. Message: " + e.getMessage());
114       try {
115         if (conn != null) {
116           conn.rollback();
117         }
118       } catch (SQLException e2) {
119         LOG.log(
120             Level.SEVERE,
121             "Unable to rollback getWorkflowByName transaction. Message: "
122                 + e2.getMessage());
123       }
124       throw new RepositoryException(e.getMessage());
125     } finally {
126 
127       if (rs != null) {
128         try {
129           rs.close();
130         } catch (SQLException ignore) {
131         }
132 
133       }
134 
135       if (statement != null) {
136         try {
137           statement.close();
138         } catch (SQLException ignore) {
139         }
140 
141       }
142 
143       if (conn != null) {
144         try {
145           conn.close();
146 
147         } catch (SQLException ignore) {
148         }
149 
150       }
151     }
152 
153     return workflow;
154   }

1208             + task.getTaskId() + "]");
1209       }
1210 
1211       // check its conditions
1212       if (task.getConditions() != null && task.getConditions().size() > 0) {
1213         List<WorkflowCondition> conditions = this.getConditionsByTaskId(task
1214             .getTaskId());
1215         for (WorkflowCondition cond : (List<WorkflowCondition>) task

1218             throw new RepositoryException("Reference in new workflow: ["
