272   /* (non-Javadoc)
273    * @see org.apache.oodt.cas.workflow.repository.WorkflowRepository#addTask(org.apache.oodt.cas.workflow.structs.WorkflowTask)
274    */
275   @Override
276   public String addTask(WorkflowTask task) throws RepositoryException {
277     // check its conditions
278     if(task.getPreConditions() != null && task.getPreConditions().size() > 0){
279       for(WorkflowCondition cond: task.getPreConditions()){
280         if(!this.conditions.containsKey(cond.getConditionId())){
281           throw new RepositoryException("Reference in new task: ["+task.getTaskName()+"] to undefined pre condition ith id: ["+cond.getConditionId()+"]");            
282         }          
283       }
284       
285       for(WorkflowCondition cond: task.getPostConditions()){
286         if(!this.conditions.containsKey(cond.getConditionId())){
287           throw new RepositoryException("Reference in new task: ["+task.getTaskName()+"] to undefined post condition ith id: ["+cond.getConditionId()+"]");            
288         }              
289       }
290     }
291     
292       String taskId = task.getTaskId() != null ? 
293         task.getTaskId():UUID.randomUUID().toString();
294       this.tasks.put(taskId, task);
295       return taskId;
296   }

298   /*
299    * (non-Javadoc)
300    * 
301    * @see
302    * org.apache.oodt.cas.workflow.repository.WorkflowRepository#addWorkflow(
303    * org.apache.oodt.cas.workflow.structs.Workflow)
304    */
305   @Override
306   public String addWorkflow(Workflow workflow) throws RepositoryException {
307     // first check to see that its tasks are all present
308     if (workflow.getTasks() == null || (workflow.getTasks().size() == 0)) {
309       throw new RepositoryException("Attempt to define a new worklfow: ["
310           + workflow.getName() + "] with no tasks.");
311     }
312 
313     for (WorkflowTask task : (List<WorkflowTask>) workflow.getTasks()) {
314       if (!this.tasks.containsKey(task.getTaskId())) {
315         throw new RepositoryException("Reference in new workflow: ["
316             + workflow.getName() + "] to undefined task with id: ["
317             + task.getTaskId() + "]");
318       }
319 
320       // check its conditions
321       if (task.getConditions() != null && task.getConditions().size() > 0) {
322         for (WorkflowCondition cond : (List<WorkflowCondition>) task
323             .getConditions()) {
324           if (!this.conditions.containsKey(cond.getConditionId())) {
325             throw new RepositoryException("Reference in new workflow: ["
326                 + workflow.getName() + "] to undefined condition ith id: ["
327                 + cond.getConditionId() + "]");
328           }
329         }
330       }
331     }
332 
333     // recast it as a parent/child workflow
334     String workflowId = workflow.getId();
335 	if (workflowId == null || (workflowId.equals(""))) {
336 		// generate its ID
337 		workflowId = UUID.randomUUID().toString();
338 		workflow.setId(workflowId);
339 	}
340       
341     ParentChildWorkflow pcw;
342     if(workflow instanceof ParentChildWorkflow) {
343         pcw = (ParentChildWorkflow) workflow;
344     }
345     else {
346         Graph graph = new Graph();
347         graph.setExecutionType("sequential");
348         pcw = new ParentChildWorkflow(graph);
349         pcw.setName(workflow.getName());
350         pcw.setTasks(workflow.getTasks());
351         pcw.setId(workflow.getId());
352     }
353     this.workflows.put(pcw.getId(), pcw);
354     this.eventWorkflowMap.put(workflowId, Collections.singletonList(pcw));
355 
356     // generate its ID
357     return workflowId;
358   }

387   private void init() throws RepositoryException {
388     this.workflows = new ConcurrentHashMap<String, ParentChildWorkflow>();
389     this.tasks = new ConcurrentHashMap<String, WorkflowTask>();
390     this.conditions = new ConcurrentHashMap<String, WorkflowCondition>();
391     this.eventWorkflowMap = new ConcurrentHashMap<String, List<ParentChildWorkflow>>();
392     this.globalConfGroups = new ConcurrentHashMap<String, Metadata>();
393     DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
394     DocumentBuilder parser;
395 
396     try {
397       parser = factory.newDocumentBuilder();
398       List<Element> rootElements = new Vector<Element>();
399       for (File file : files) {
400         rootElements.add(parser.parse(file).getDocumentElement());
401       }
402       for (Element root : rootElements) {
403         Metadata staticMetadata = new Metadata();
404         loadConfiguration(rootElements, root, staticMetadata);
405         loadTaskAndConditionDefinitions(rootElements, root, staticMetadata);
406         loadGraphs(rootElements, root, new Graph(), staticMetadata);
407         computeEvents();
408         computeWorkflowConditions();
409       }
410     } catch (Exception e) {
411       LOG.log(Level.SEVERE, e.getMessage());
412       throw new RepositoryException(e.getMessage());
413     }
414   }

416   private void computeWorkflowConditions() {
417     if (this.workflows != null && this.workflows.values().size() > 0) {
418       for (ParentChildWorkflow w : this.workflows.values()) {
419         if (w.getConditions() != null && w.getConditions().size() > 0) {
420           w.getTasks().add(0,
421               getGlobalWorkflowConditionsTask(w.getName(), w.getId(), w.getConditions()));
422         }
423       }
424     }
425   }

427   private void computeEvents() throws WorkflowException {
428     List<ParentChildWorkflow> workflows = new Vector<ParentChildWorkflow>();
429     for (ParentChildWorkflow w : this.workflows.values()) {
430       workflows.add(w);
431 
432     }
433     for (ParentChildWorkflow workflow : workflows) {
434 
435       // event for each workflow id
436       List<ParentChildWorkflow> wList = new Vector<ParentChildWorkflow>();
437       wList.add(workflow);
438       this.eventWorkflowMap.put(workflow.getId(), wList);
439 
440       // clear its tasks, we are going to re-add them back
441       workflow.getTasks().clear();
442       List<Graph> children = workflow.getGraph().getChildren();
443       if (workflow.getGraph().getExecutionType().equals("sequential")) {
444         for (Graph child : children) {
445           if (child.getWorkflow() != null) {
446             workflow.getTasks().add(
447                 generateRedirector(child.getWorkflow().getId()));
448           } else if (child.getTask() != null) {
449             workflow.getTasks().add(child.getTask());
450           }
451         }
452       } else if (workflow.getGraph().getExecutionType().equals("parallel")) {
453         // clear it as a workflow from the list
454         // to begin with
455         this.eventWorkflowMap.get(workflow.getId()).clear();
456         this.workflows.remove(workflow.getId());
457         for (Graph child : children) {
458           if (child.getWorkflow() != null) {
459             // add child workflow to the event kickoff for this id
460             this.eventWorkflowMap.get(workflow.getId())
461                 .add(child.getWorkflow());
462           } else if (child.getTask() != null) {
463             // add a new dynamic workflow
464             // with just this task
465             ParentChildWorkflow w = getDynamicWorkflow(child.getTask());
466             this.eventWorkflowMap.get(workflow.getId()).add(w);
467           }
468         }
469       } else {
470         throw new WorkflowException("Unsupported execution type: ["
471                                     + workflow.getGraph().getExecutionType() + "]");
472       }
473     }
474   }

476   private void loadTaskAndConditionDefinitions(List<Element> rootElements,
477       Element rootElem, Metadata staticMetadata)
478       throws CommonsException, CasMetadataException, WorkflowException, ParseException {
479 
480     List<Element> conditionBlocks = this.getChildrenByTagName(rootElem,
481         "condition");
482     List<Element> taskBlocks = this.getChildrenByTagName(rootElem, "task");
483 
484     if (conditionBlocks != null && conditionBlocks.size() > 0) {
485       LOG.log(Level.FINER, "Loading: [" + conditionBlocks.size()
486           + "] conditions from: ["
487           + rootElem.getOwnerDocument().getDocumentURI() + "]");
488 
489       for (Element condElem : conditionBlocks) {
490         loadGraphs(rootElements, condElem, new Graph(), staticMetadata);
491       }
492 
493     }
494 
495     if (taskBlocks != null && taskBlocks.size() > 0) {
496       LOG.log(Level.FINER, "Loading: [" + taskBlocks.size() + "] tasks from: ["
497           + rootElem.getOwnerDocument().getDocumentURI() + "]");
498       for (Element taskElem : taskBlocks) {
499         loadGraphs(rootElements, taskElem, new Graph(), staticMetadata);
500       }
501     }
502   }

504   private void loadGraphs(List<Element> rootElements, Element graphElem,
505       Graph parent, Metadata staticMetadata)
506       throws CommonsException, CasMetadataException, WorkflowException, ParseException {
507 
508     LOG.log(Level.FINEST, "Visiting node: [" + graphElem.getNodeName() + "]");
509     loadConfiguration(rootElements, graphElem, staticMetadata);
510     Graph graph = !graphElem.getNodeName().equals("cas:workflows") ? new Graph(
511         graphElem, staticMetadata) : new Graph();
512     parent.getChildren().add(graph);
513     graph.setParent(parent);
514     if (!graphElem.getNodeName().equals("cas:workflows")) {
515       expandWorkflowTasksAndConditions(graph, staticMetadata);
516     }
517 
518     for (String processorType : Graph.processorIds) {
519       LOG.log(Level.FINE, "Scanning for: [" + processorType + "] nodes");
520       List<Element> procTypeBlocks = this.getChildrenByTagName(graphElem,
521           processorType);
522       if (procTypeBlocks != null && procTypeBlocks.size() > 0) {
523         LOG.log(Level.FINE, "Found: [" + procTypeBlocks.size() + "] ["
524             + processorType + "] processor types");
525         for (Element procTypeBlock : procTypeBlocks) {
526           loadGraphs(rootElements, procTypeBlock, graph, staticMetadata);
527         }
528       } else {
529         if (processorType.equals("condition")) {
530           Element conditionsElem = XMLUtils.getFirstElement("conditions",
531               graphElem);
532           if (conditionsElem != null) {
533             List<Element> procTypeBlockNodes = this.getChildrenByTagName(
534                 conditionsElem, "condition");
535             if (procTypeBlockNodes != null && procTypeBlockNodes.size() > 0) {
536               LOG.log(Level.FINE, "Found: [" + procTypeBlockNodes.size()
537                   + "] linked condition definitions");
538               for (Element procTypeBlockNode : procTypeBlockNodes) {
539                 loadGraphs(rootElements, procTypeBlockNode, graph,
540                     staticMetadata);
541               }
542             }
543           }
544         }
545       }
546     }

551   private void loadConfiguration(List<Element> rootElements, Node workflowNode,
552       Metadata staticMetadata) throws ParseException, CommonsException, CasMetadataException, WorkflowException {
553     NodeList children = workflowNode.getChildNodes();
554     for (int i = 0; i < children.getLength(); i++) {
555       Node curChild = children.item(i);
556 
557       if (curChild.getNodeName().equals("configuration")) {
558         Metadata curMetadata = new Metadata();
559         if (!((Element) curChild).getAttribute("extends").equals("")) {
560           for (String extension : ((Element) curChild).getAttribute("extends")
561                                                       .split(",")) {
562             curMetadata
563                 .replaceMetadata(globalConfGroups.containsKey(extension) ? globalConfGroups
564                     .get(extension) : this.loadConfGroup(rootElements,
565                     extension, globalConfGroups));
566           }
567         }
568         curMetadata.replaceMetadata(XmlStructFactory
569             .getConfigurationAsMetadata(curChild));
570         NamedNodeMap attrMap = curChild.getAttributes();
571         String configName = null;
572         for (int j = 0; j < attrMap.getLength(); j++) {
573           Attr attr = (Attr) attrMap.item(j);
574           if (attr.getName().equals("name")) {
575             configName = attr.getValue();
576           }
577         }
578 
579         if (configName == null || (configName.equals(""))) {
580           NamedNodeMap workflowNodeAttrs = workflowNode.getAttributes();
581           for (int j = 0; j < workflowNodeAttrs.getLength(); j++) {
582             Attr attr = (Attr) workflowNodeAttrs.item(j);
583             if (attr.getName().equals("id")) {
584               configName = attr.getValue();
585             }
586           }
587         }
588 
589         this.globalConfGroups.put(configName, curMetadata);
590         staticMetadata.replaceMetadata(curMetadata);
591       }
592     }
593   }

595   private Metadata loadConfGroup(List<Element> rootElements, String group,
596       Map<String, Metadata> globalConfGroups)
597       throws ParseException, CommonsException, CasMetadataException, WorkflowException {
598     for (final Element rootElement : rootElements) {
599       NodeList nodes = rootElement.getElementsByTagName("configuration");
600       for (int i = 0; i < nodes.getLength(); i++) {
601         Node node = nodes.item(i);
602         String name = ((Element) node).getAttribute("name");
603         if (name.equals(group)) {
604           return XmlStructFactory.getConfigurationAsMetadata(node);
605         }
606       }
607     }
608     throw new WorkflowException("Configuration group '" + group + "' not defined!");
609   }

611   private void expandWorkflowTasksAndConditions(Graph graph,
612       Metadata staticMetadata) {
613     if (graph.getExecutionType().equals("workflow")
614         || graph.getExecutionType().equals("sequential")
615         || graph.getExecutionType().equals("parallel")) {
616       ParentChildWorkflow workflow = new ParentChildWorkflow(graph);
617       workflow.setId(graph.getModelId());
618       workflow.setName(graph.getModelName());
619       graph.setWorkflow(workflow);
620       if (graph.getParent() == null || (graph.getParent().getWorkflow() == null)) {
621         LOG.log(Level.FINEST, "Workflow: [" + graph.getModelId()
622             + "] has no parent: it's a top-level workflow");
623       }
624 
625       if (workflow.getName() == null || (workflow.getName().equals(""))) {
626         workflow.setName(graph.getExecutionType() + "-" + workflow.getId());
627       }
628       this.workflows.put(graph.getModelId(), workflow);
629     } else if (graph.getExecutionType().equals("condition")) {
630       WorkflowCondition cond;
631 
632       if (graph.getModelIdRef() != null && !graph.getModelIdRef().equals("")) {
633         cond = this.conditions.get(graph.getModelIdRef());
634       } else {
635         cond = new WorkflowCondition();
636         cond.setConditionId(graph.getModelId());
637         cond.setConditionName(graph.getModelName());
638         cond.setConditionInstanceClassName(graph.getClazz());
639         cond.setTimeoutSeconds(graph.getTimeout());
640         cond.setOptional(graph.isOptional());
641         cond.setCondConfig(convertToConditionConfiguration(staticMetadata));
642 
643         if (cond.getConditionName() == null || (cond.getConditionName()
644                                                     .equals(""))) {
645           cond.setConditionName(cond.getConditionId());
646         }
647         this.conditions.put(graph.getModelId(), cond);
648 
649       }
650 
651       graph.setCond(cond);
652       if (graph.getParent() != null) {
653         if (graph.getParent().getWorkflow() != null) {
654           LOG.log(Level.FINEST, "Adding condition: [" + cond.getConditionName()
655               + "] to parent workflow: ["
656               + graph.getParent().getWorkflow().getName() + "]");
657           graph.getParent().getWorkflow().getConditions().add(cond);
658         } else if (graph.getParent().getTask() != null) {
659           graph.getParent().getTask().getConditions().add(cond);
660         } else {
661           LOG.log(Level.FINEST, "Condition: [" + graph.getModelId()
662               + "] has not parent: it's a condition definition");
663         }
664       } else {
665         LOG.log(Level.FINEST, "Condition: [" + graph.getModelId()
666             + "]: parent is null");
667       }
668       // if parent doesn't have task or workflow set, then its parent
669       // is null and it's a condition definition, just add it
670 
671     } else if (graph.getExecutionType().equals("task")) {
672       WorkflowTask task;
673       if (graph.getModelIdRef() != null && !graph.getModelIdRef().equals("")) {
674         LOG.log(Level.FINER, "Model ID-Ref to: [" + graph.getModelIdRef() + "]");
675         task = this.tasks.get(graph.getModelIdRef());
676       } else {
677         task = new WorkflowTask();
678         task.setTaskId(graph.getModelId());
679         task.setTaskName(graph.getModelName());
680         task.setTaskConfig(convertToTaskConfiguration(staticMetadata));
681         task.setTaskInstanceClassName(graph.getClazz());
682 
683         if (task.getTaskName() == null || (task.getTaskName().equals(""))) {
684           task.setTaskName(task.getTaskId());
685         }
686         this.tasks.put(graph.getModelId(), task);
687       }
688 
689       graph.setTask(task);
690       if (graph.getParent() != null) {
691         if (graph.getParent().getWorkflow() != null) {
692           graph.getParent().getWorkflow().getTasks().add(task);
693         } else {
694           LOG.log(Level.FINEST, "Task: [" + graph.getModelId()
695               + "] has no parent: it's a task definition");
696         }
697       } else {
698         LOG.log(Level.FINEST, "Task: [" + graph.getModelId()
699             + "]: parent is null");
700       }
701     }
702 
703   }
