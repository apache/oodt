188   /**
189    * 
190    * @param id
191    *          Relative path from staging root to product. The met extension will
192    *          be added to this id to look up and see if a met file exists with
193    *          in the met area.
194    * @param configId
195    *          Reference to the extractor config. {@link ExtractorConfigReader}
196    *          will load the configuration
197    * @param overwrite
198    *          Flag to indicate whether or not to overwrite a met file if present
199    *          in the staging area.
200    * @return The {@link Metadata} retrieved from the met area path if present or
201    *         extracted using the met extractor config.
202    * @throws FileNotFoundException
203    * @throws InstantiationException
204    * @throws IOException
205    * @throws MetExtractionException
206    */
207   protected Metadata getStagingMetadata(String id, String configId,
208       Boolean overwrite) throws InstantiationException,
209       IOException, MetExtractionException {
210     if (configId == null || configId.trim().length() == 0) {
211       return this.readMetFile(id + CurationService.config.getMetExtension());
212     } else {
213       String relMetPath = id.startsWith("/") ? id : "/" + id;
214       String pathToMetFile = CurationService.config.getMetAreaPath()
215           + relMetPath + CurationService.config.getMetExtension();
216       if (!overwrite && new File(pathToMetFile).exists()) {
217         return this.readMetFile(id + CurationService.config.getMetExtension());
218       } else {
219         // Make sure the parent directory exists
220         new File(pathToMetFile).getParentFile().mkdirs();
221         Metadata metadata = this.runMetExtractor(id, ExtractorConfigReader
222             .readFromDirectory(
223                 new File(CurationService.config
224                 .getMetExtrConfUploadPath()), configId));
225         this.writeMetFile(id, metadata);
226         return metadata;
227       }
228     }
229   }

370   private Metadata getMetadataFromJSON(String metadataJSON) {
371     JSONObject json = (JSONObject) JSONSerializer.toJSON(metadataJSON);
372     Metadata metadata = new Metadata();
373 
374     Set<String> keys = json.keySet();
375     for (String key : keys) {
376       List values = (List) JSONSerializer.toJava((JSONArray) json.get(key));
377       metadata.addMetadata(key, values);
378     }
379 
380     return metadata;
381   }

383   private Metadata getMetadataFromMap(MultivaluedMap<String, String> formParams) {
384     Metadata metadata = new Metadata();
385     
386     for (Map.Entry<String, List<String>> entry : formParams.entrySet()) {
387       if (entry.getKey().startsWith("metadata.")) {
388         String newKey = entry.getKey().substring(entry.getKey().indexOf('.') + 1);
389         for (String value : entry.getValue()) {
390           metadata.addMetadata(newKey, value);
391         }
392       }
393     }
394     
395     return metadata;
396   }

398   protected String getMetadataAsHTML(Metadata metadata) {
399     if (metadata == null) {
400       return "<table></table>";
401     }
402 
403     StringBuilder html = new StringBuilder();
404 
405     html.append("<table>\r\n");
406     for (String key : (Set<String>) metadata.getMap().keySet()) {
407       html.append(" <tr>\r\n");
408       html.append("  <th>").append(key).append("</th>\r\n");
409       html.append("  <td class=\"").append(key).append("\">");
410       List<String> values = metadata.getAllMetadata(key);
411       for (Iterator<String> i = values.iterator(); i.hasNext();) {
412         html.append("<span>").append(i.next()).append("</span>");
413         if (i.hasNext()) {
414           html.append(", ");
415         }
416       }
417       for (String value : (List<String>) metadata.getAllMetadata(key)) {
418       }
419       html.append("</td>\r\n");
420       html.append(" </tr>\r\n");
421     }
422     html.append("</table>\r\n");
423 
424     return html.toString();
425   }

492   /**
493    * Method to update the catalog metadata for a given product. 
494    * All current metadata fields will be preserved, 
495    * except those specified in the HTTP POST request as 'metadata.<field_name>=<field_value>'.
496    * 
497    * @param id
498    * 	identifier of CAS product - either 'id' or 'name' must be specified
499    * @param name
500    * 	name of CAS product - either 'id' or 'name' must be specified
501    * @param formParams
502    * 	HTTP (name, value) form parameters. The parameter names MUST start with "metadata."
503    * @param replace
504    * 	optional flag set to false to add the new metadata values to the existing values, for the given flags
505    */
506   @POST
507   @Path(UPDATE)
508   @Consumes("application/x-www-form-urlencoded")
509   @Produces("text/plain")
510   public String updateMetadata(MultivaluedMap<String, String> formParams, 
511 		  @FormParam("id") String id, 
512 		  @FormParam("name") String name, 
513 		  @DefaultValue("true") @FormParam("replace") boolean replace,
514 		  @DefaultValue("false") @FormParam("remove") boolean remove) {
515   		      
516   	// new metadata from HTTP POST request
517     Metadata newMetadata = this.getMetadataFromMap(formParams);
518     
519     // client for interacting with remote File Manager
520     XmlRpcFileManagerClient fmClient = CurationService.config.getFileManagerClient();
521  
522     // empty metadata
523     Metadata metadata;
524     
525     try {
526     
527       // retrieve product from catalog
528       Product product;
529       if (StringUtils.hasText(id)) {
530     	  id = id.substring(id.lastIndexOf("/") + 1);
531     	  product = fmClient.getProductById(id);
532       } else if (StringUtils.hasText(name)) {
533     	  product = fmClient.getProductByName(name);
534       } else {
535     	  throw new Exception("Either the HTTP parameter 'id' or the HTTP parameter 'name' must be specified");
536       }
537             
538       // retrieve existing metadata
539       metadata = fmClient.getMetadata(product);
540 
541       // remove product references (as they will be added later)
542       metadata.removeMetadata("reference_orig");
543       metadata.removeMetadata("reference_data_store");
544       metadata.removeMetadata("reference_fileSize");
545       metadata.removeMetadata("reference_mimeType");
546       
547       // merge new and existing metadata
548       metadata.addMetadata(newMetadata);
549       
550       // replace metadata values for keys specified in HTTP request (not others)
551       if (replace) {
552     	  for (String key : newMetadata.getAllKeys()) {
553     		  metadata.replaceMetadata(key, newMetadata.getAllMetadata(key));
554     	  }
555       }
556       
557       // remove metadata tags
558       if (remove) {
559 	      for (String key : newMetadata.getAllKeys()) {
560 	      	metadata.removeMetadata(key);
561 	      }
562       }
563       
564       // insert old and new metadata
565       fmClient.updateMetadata(product, metadata);
566       
567       // return product id to downstream processors
568       return "id="+product.getProductId();
569           
570     } catch (Exception e) {
571     	
572       LOG.log(Level.SEVERE, e.getMessage());
573       // return error message
574       throw new WebApplicationException(e, Response.Status.INTERNAL_SERVER_ERROR);
575 
576     }
577 
578   }

580   /**
581    * Updates the cataloged {@link Metadata} for a {@link Product} in the CAS
582    * File Manager.
583    * 
584    * @param product
585    *          The {@link Product} to update {@link Metadata} for.
586    * @throws CatalogException
587    *           If any error occurs during the update.
588    * @throws IOException
589    * @throws FileNotFoundException
590    */
591   public void updateCatalogMetadata(Product product, Metadata newMetadata)
592       throws CatalogException, IOException {
593     InputStream is = new FileInputStream(CurationService.config.getFileMgrProps());
594     try {
595       System.getProperties().load(is);
596     }
597     finally{
598       is.close();
599     }
600     Catalog catalog = this.getCatalog();
601     
602     Metadata oldMetadata = catalog.getMetadata(product);
603     List<Reference> references = catalog.getProductReferences(product);
604     Product newProduct = new Product(product.getProductName(), product
605         .getProductType(), product.getProductStructure(), product
606         .getTransferStatus(), product.getProductReferences());
607     // Constructor is bugged and doesn't set transfer status
608     newProduct.setTransferStatus(product.getTransferStatus());
609     catalog.removeMetadata(oldMetadata, product);
610     catalog.removeProduct(product);
611     newProduct.setProductId(product.getProductId());
612     catalog.addProduct(newProduct);
613     newProduct.setProductReferences(references);
614     catalog.addProductReferences(newProduct);
615     catalog.addMetadata(newMetadata, newProduct);
616   }

677   private Metadata getProductTypeMetadataForPolicy(String policy,
678       String productTypeName) throws MalformedURLException,
679       InstantiationException, RepositoryManagerException {
680     String rootPolicyPath = this.cleanse(CurationService.config
681         .getPolicyUploadPath());
682     String policyPath = new File(rootPolicyPath + policy).toURL()
683         .toExternalForm();
684     String[] policies = { policyPath };

927   private void removeUnusedElements(List<Element> elements, 
928 		  XMLRepositoryManager xmlRepo, XMLValidationLayer vLayer) 
929 				  throws ValidationLayerException, RepositoryManagerException {
930       // Remove Elements that aren't used in any product type
931       List<ProductType> ptypelist = xmlRepo.getProductTypes();
932       ConcurrentHashMap<String, Boolean> usedElementIds = new ConcurrentHashMap<String, Boolean>();
933       for(ProductType ptype: ptypelist) {
934           List<Element> ptypeElements = 
935               vLayer.getElements(ptype);
936           for(Element el: ptypeElements) {
937               usedElementIds.put(el.getElementId(), true);
938           }
939       }
940       for(Element el: elements) {
941           if(!usedElementIds.containsKey(el.getElementId())) {
942             vLayer.removeElement(el);
943           }
944       }
945   }
