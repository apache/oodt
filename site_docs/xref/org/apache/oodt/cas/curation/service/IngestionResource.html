113   @GET
114   @Path("create")
115   @Produces("text/plain")
116   public String createTask(@QueryParam("files") String fileList,
117       @QueryParam("numfiles") Integer numFiles,
118       @QueryParam("metExtCfgId") String metExtractorConfigId,
119       @QueryParam("policy") String policy,
120       @QueryParam("ptype") String productType) {
121 
122     IngestionTask newTask = new IngestionTask();
123     newTask.setCreateDate(new Date());
124     try {
125       newTask.setExtConf(ExtractorConfigReader.readFromDirectory(
126           new File(CurationService.config.getMetExtrConfUploadPath()),
127           metExtractorConfigId));
128     } catch (Exception e) {
129       LOG.log(Level.SEVERE, e.getMessage());
130       String errorMsg = "Unable to load extractor config from metExtCfgId: ["
131           + metExtractorConfigId + "]";
132       LOG.log(Level.WARNING, errorMsg);
133       return errorMsg;
134     }
135     newTask.setFileList(deducePaths(Arrays.asList(fileList.split(","))));
136     newTask.setPolicy(policy);
137     newTask.setProductType(productType);
138     newTask.setStatus(IngestionTask.NOT_STARTED);
139     String addedTaskID = this.taskList.addIngestionTask(newTask);
140     saveTaskListState();
141     return addedTaskID;
142   }

173   @GET
174   @Path("start")
175   @Produces("text/plain")
176   public String doIngest(@QueryParam("taskId") String ingestTaskId) {
177     IngestionTask task = this.taskList.getIngestionTaskById(ingestTaskId);
178     if (task == null) {
179       String errorMsg = "Task with ID [" + ingestTaskId
180           + "] is not being managed by this Ingestion Resource!";
181       LOG.log(Level.WARNING, errorMsg);
182       return this.encodeIngestResponseAsJSON(false, errorMsg);
183     }
184 
185     Ingester ingest = this.configureIngester();
186     MetadataResource metService = new MetadataResource();
187     for (String file : task.getFileList()) {
188       Metadata fileMet;
189       try {
190         String vFilePath = this
191             .getVirtualPath(CurationService.config.getStagingAreaPath(), file);
192         LOG.log(Level.FINE,
193             "IngestionResource: getting staging metadata for virtual path: ["
194                 + vFilePath + "]");
195         fileMet = metService.getStagingMetadata(vFilePath,
196             task.getExtConf().getIdentifier(), false);
197       } catch (Exception e) {
198         LOG.log(Level.SEVERE, e.getMessage());
199         return this.encodeIngestResponseAsHTML(false, e.getMessage());
200       }
201 
202       try {
203         ingest.ingest(safeGetUrl(CurationService.config.getFileMgrURL()),
204             new File(file), fileMet);
205       } catch (IngestException e) {
206         LOG.log(Level.SEVERE, e.getMessage());
207         return this.encodeIngestResponseAsHTML(false, e.getMessage());
208       }
209 
210       // set task status to success
211       task.setStatus(IngestionTask.FINISHED);
212     }
213 
214     String response = this.encodeIngestResponseAsHTML(true, null);
215     saveTaskListState();
216     return response;
217 
218   }

261   private String encodeTaskListAsJSON(List<IngestionTask> taskList) {
262     List<Map<String, String>> jsonFriendlyTaskList = new Vector<Map<String, String>>();
263     for (IngestionTask task : taskList) {
264       Map<String, String> taskPropMap = new ConcurrentHashMap<String, String>();
265       taskPropMap.put("id", task.getId());
266       taskPropMap.put("createDate",
267           DateUtils.getDateAsISO8601String(task.getCreateDate()));
268       taskPropMap.put("policy", task.getPolicy());
269       taskPropMap.put("productType", task.getProductType());
270       taskPropMap.put("status", task.getStatus());
271       taskPropMap.put("fileList", task.getFileList().toString());
272       taskPropMap.put("extractorClass", task.getExtConf().getClassName());
273       taskPropMap.put("extractorConfFiles",
274           task.getExtConf().getConfigFiles().toString());
275       jsonFriendlyTaskList.add(taskPropMap);
276     }
277 
278     JSONObject resObj = new JSONObject();
279     resObj.put("taskList", jsonFriendlyTaskList);

373   private void rebuildTaskList(Document document)
374       throws FileNotFoundException, IOException, ParseException {
375 
376     NodeList tasks = document.getElementsByTagName("Task");
377 
378     for (int i = 0; i < tasks.getLength(); i++) {
379       Element taskElement = (Element) tasks.item(i);
380 
381       String id = getNodeValue(taskElement, "ID");
382       String dateString = getNodeValue(taskElement, "CreateDate");
383       List<String> newFileList = getTaskFileList(taskElement);
384       String policy = getNodeValue(taskElement, "Policy");
385       String productType = getNodeValue(taskElement, "ProductType");
386       String status = getNodeValue(taskElement, "Status");
387       String[] extConfParams = getTaskExtractorConfigParams(taskElement);
388 
389       IngestionTask newTask = createTaskFromParameters(id, dateString,
390           newFileList, policy, productType, status, extConfParams);
391 
392       this.taskList.addIngestionTaskWithoutIdGen(newTask);
393     }
394   }

425

437     newTask.setStatus(status);

481     public List<IngestionTask> getTaskList() {
482       List<IngestionTask> taskList = Arrays.asList(
483           taskMap.values().toArray(new IngestionTask[taskMap.values().size()]));
484       Collections.sort(taskList, new Comparator<IngestionTask>() {
485 
486         public int compare(IngestionTask o1, IngestionTask o2) {
487           if (o1.getCreateDate().before(o2.getCreateDate())) {
488             return -1;
489           } else if (o1.getCreateDate().equals(o2.getCreateDate())) {
490             return 0;
491           } else {
492             return 1;
493           }
494         }
495       });
496       return taskList;
497     }

500     public void exportTaskListAsXMLToFile(String fileName) {
501       try {

530     private Document generateXMLDocument() throws ParserConfigurationException {
531       DocumentBuilderFactory documentFactory = DocumentBuilderFactory
532           .newInstance();
533       DocumentBuilder documentBuilder = documentFactory.newDocumentBuilder();
534 
535       Document xmlDocument = documentBuilder.newDocument();
536 
537       // Add root "Tasks" node
538       Element root = xmlDocument.createElement("Tasks");
539       xmlDocument.appendChild(root);
540 
541       // For each task, generate the following...
542       List<IngestionTask> taskList = getTaskList();
543       for (IngestionTask task : taskList) {
544         Element taskNode = xmlDocument.createElement("Task");
545         root.appendChild(taskNode);
546 
547         // Task ID Node
548         Element ID = xmlDocument.createElement("ID");
549         ID.appendChild(xmlDocument.createTextNode(task.getId()));
550         taskNode.appendChild(ID);
551 
552         // Task creation date Node
553         Element createDate = xmlDocument.createElement("CreateDate");
554         createDate.appendChild(
555             xmlDocument.createTextNode(task.getCreateDate().toString()));
556         taskNode.appendChild(createDate);
557 
558         // Task file list Node
559         Element fileList = xmlDocument.createElement("Files");
560         for (String file : task.getFileList()) {
561           // Individual file Node
562           Element fileNode = xmlDocument.createElement("File");
563           fileNode.appendChild(xmlDocument.createTextNode(file));
564           fileList.appendChild(fileNode);
565         }
566         taskNode.appendChild(fileList);
567 
568         // Task policy Node
569         Element policy = xmlDocument.createElement("Policy");
570         policy.appendChild(xmlDocument.createTextNode(task.getPolicy()));
571         taskNode.appendChild(policy);
572 
573         // Task product type Node
574         Element productType = xmlDocument.createElement("ProductType");
575         productType
576             .appendChild(xmlDocument.createTextNode(task.getProductType()));
577         taskNode.appendChild(productType);
578 
579         // Task status Node
580         Element status = xmlDocument.createElement("Status");
581         status.appendChild(xmlDocument.createTextNode(task.getStatus()));
582         taskNode.appendChild(status);
583 
584         // Task extractor configuration Node
585         Element extractorConf = xmlDocument.createElement("ExtractorConfig");
586         // Extractor configuration upload path Node
587         Element uploadPath = xmlDocument.createElement("UploadPath");
588         uploadPath.appendChild(xmlDocument
589             .createTextNode(CurationService.config.getMetExtrConfUploadPath()));
590         extractorConf.appendChild(uploadPath);
591         // Extractor configuration ID Node
592         Element extractID = xmlDocument.createElement("ID");
593         extractID.appendChild(
594             xmlDocument.createTextNode(task.getExtConf().getIdentifier()));
595         extractorConf.appendChild(extractID);
596         taskNode.appendChild(extractorConf);
597       }
598 
599       return xmlDocument;
600     }
