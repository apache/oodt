46  /**
47   * Delivers back entire {@link ProductType}s (or <code>Dataset</code>s) as
48   * zipped up packages.
49   * 
50   * @author mattmann
51   * @version $Revision$
52   */
53  public class DatasetDeliveryServlet extends HttpServlet implements
54      DataDeliveryKeys {
55  
56    /* our log stream */
57    private static final Logger LOG = Logger
58        .getLogger(DatasetDeliveryServlet.class.getName());
59  
60    /* serial version UID */
61    private static final long serialVersionUID = -6692665690674186105L;
62    public static final int INT = 512;
63  
64    /* our file manager client */
65    private XmlRpcFileManagerClient client;
66  
67    /* the working dir in which to create huge zip files */
68    private String workingDirPath;
69  
70    /* indicates the product has been zipped up already */
71    private final static String ALREADY_ZIPPED = "00ga";
72  
73    /*
74     * (non-Javadoc)
75     * 
76     * @see
77     * javax.servlet.http.HttpServlet#doGet(javax.servlet.http.HttpServletRequest,
78     * javax.servlet.http.HttpServletResponse)
79     */
80    protected void doGet(HttpServletRequest req, HttpServletResponse res)
81        throws ServletException, IOException {
82      doPut(req, res);
83    }
84  
85    /**
86     * 
87     * Method requires a single parameter, <code>typeID</code>, specifying a valid
88     * {@link ProductType} within the filemgr. All {@link Product}s associated
89     * with the given <code>typeID</code> are packaged up (as zip files), and then
90     * aggregated and zipped and delivered as an entire collection called a
91     * <code>Dataset</code>.
92     * 
93     * @param req
94     *          Servlet request.
95     * @param res
96     *          Servlet response
97     * @throws ServletException
98     *           If any error occurs.
99     * @throws IOException
100    *           If an I/O error occurs.
101    */
102   protected void doPut(HttpServletRequest req, HttpServletResponse res)
103       throws ServletException, IOException {
104 
105     String typeID = req.getParameter("typeID");
106     ProductType type;
107 
108     if (typeID == null) {
109       throw new IllegalArgumentException("No typeID parameter specified!");
110     }
111 
112     try {
113       type = client.getProductTypeById(typeID);
114     } catch (Exception e) {
115       throw new ServletException("Unable to deduce product type: [" + typeID
116           + "]: Message: " + e.getMessage());
117     }
118 
119     // create a temporary product dir: we'll use working dir + typeName
120     String productDirPath = workingDirPath + type.getName();
121     if (!new File(productDirPath).mkdirs()) {
122       LOG.log(Level.WARNING,
123           "mkdirs returned false for temporary dataset dir: [" + productDirPath
124               + "]: errors may follow");
125     }
126 
127     // use the pagination API to iterate over each product
128     // for each product, zip it up
129     // after you zip up all products then create the dataset zip
130 
131     ProductPage page;
132 
133     try {
134       page = client.getFirstPage(type);
135       if (page == null || (page.getPageProducts() == null) || (page.getPageProducts().size() == 0)) {
136         throw new ServletException("No products for dataset: ["
137             + type.getName() + "]");
138       }
139 
140       Map productHash = new ConcurrentHashMap();
141 
142       do {
143         for (Product product : page.getPageProducts()) {
144           if (alreadyZipped(product, productHash)) {
145             continue;
146           }
147 
148           Metadata metadata;
149           product.setProductReferences(client.getProductReferences(product));
150           metadata = client.getMetadata(product);
151           DataUtils.createProductZipFile(product, metadata, productDirPath);
152           productHash.put(product.getProductName(), ALREADY_ZIPPED);
153         }
154 
155         page = client.getNextPage(type, page);
156 
157       } while ((page != null && !page.isLastPage())
158           && (page.getPageProducts() != null && page.getPageProducts().size() > 0));
159     } catch (Exception e) {
160       LOG.log(Level.SEVERE, e.getMessage());
161       throw new ServletException(e.getMessage());
162     }
163 
164     // now that all product zips have been created, create the dataset
165     // zip
166     String datasetZipFilePath;
167     File datasetZipFile = null;
168     InputStream in = null;
169     OutputStream o2 = null;
170 
171     try {
172       datasetZipFilePath = DataUtils.createDatasetZipFile(type, productDirPath);
173 
174       datasetZipFile = new File(datasetZipFilePath);
175       String datasetZipFileName = datasetZipFile.getName();
176 
177       res.addHeader(CONTENT_DISPOSITION_HDR, "attachment; filename=\""
178           + datasetZipFileName + "\"");
179 
180       // add the content length for the zip file size
181       res.addHeader(CONTENT_LENGTH_HDR, String.valueOf(new File(
182           datasetZipFilePath).length()));
183 
184       // now we need to read the zip file in, and write it to the output
185       // stream
186       in = new FileInputStream(datasetZipFile);
187 
188       // Call java.io.File.delete() on the file, while still open.
189       if (!datasetZipFile.delete()) {
190         throw new RuntimeException("Unable to delete streaming file: ["
191             + datasetZipFile.getAbsolutePath() + "] while delivering!");
192       }
193 
194       // 3. Deliver the data.
195       o2 = res.getOutputStream();
196       byte[] buf = new byte[INT];
197       int n;
198       while ((n = in.read(buf)) != -1) {
199         o2.write(buf, 0, n);
200       }
201 
202     } catch (Exception e) {
203       LOG.log(Level.SEVERE, e.getMessage());
204       LOG.log(Level.WARNING, "Exception delivering dataset: Message: "
205           + e.getMessage());
206     } finally {
207       if (in != null) {
208         try {
209           in.close();
210         } catch (Exception ignore) {
211         }
212 
213       }
214 
215       if (o2 != null) {
216         try {
217           o2.close();
218         } catch (Exception ignore) {
219         }
220 
221       }
222 
223       // now try and remove the tmp working directory for the
224       // dataset zip
225       if (datasetZipFile != null) {
226         datasetZipFile.getParentFile().delete();
227       }
228     }
229 
230   }

232   /*
233    * (non-Javadoc)
234    * 
235    * @see javax.servlet.GenericServlet#init(javax.servlet.ServletConfig)
236    */
237   public void init(ServletConfig config) throws ServletException {
238     super.init(config);
239 
240     try {
241       String fileMgrURL;
242       try {
243         fileMgrURL = PathUtils.replaceEnvVariables(config.getServletContext().getInitParameter(
244             "filemgr.url") );
245       } catch (Exception e) {
246         throw new ServletException("Failed to get filemgr url : " + e.getMessage(), e);
247       }
248       client = new XmlRpcFileManagerClient(new URL(fileMgrURL));
249     } catch (MalformedURLException ex) {
250       throw new ServletException(ex);
251     } catch (ConnectionException ex) {
252       throw new ServletException(ex);
253     }
254 
255     workingDirPath = config.getServletContext().getInitParameter(
256         "filemgr.working.dir");
257 
258     // need the working dir to be specified, else throw exception
259     if (workingDirPath == null) {
260       throw new ServletException("no servlet working dir path specified!");
261     } else {
262       // clean it
263       this.workingDirPath += this.workingDirPath.endsWith("/") ? "" : "/";
264     }
265 
266   }
267 
268   private boolean alreadyZipped(Product p, Map hash) {
269     return hash.containsKey(p.getProductName());
270   }
