60    /** {@inheritDoc} */
61    public void init(ServletConfig config) throws ServletException {
62      super.init(config);
63      try {
64        String fileMgrURL;
65        try {
66          fileMgrURL = PathUtils.replaceEnvVariables(config.getServletContext().getInitParameter(
67              "filemgr.url") );
68        } catch (Exception e) {
69          throw new ServletException("Failed to get filemgr url : " + e.getMessage(), e);
70        }
71        client = new XmlRpcFileManagerClient(new URL(fileMgrURL));
72      } catch (MalformedURLException ex) {
73        throw new ServletException(ex);
74      } catch (ConnectionException ex) {
75        throw new ServletException(ex);
76      }
77  
78      workingDirPath = config.getServletContext().getInitParameter(
79          "filemgr.working.dir");
80  
81      // need the working dir to be specified, else throw exception
82      if (workingDirPath == null) {
83        throw new ServletException("no servlet working dir path specified!");
84      } else {
85        // clean it
86        this.workingDirPath += this.workingDirPath.endsWith("/") ? "" : "/";
87      }
88    }

96    /**
97     * Handle a request for product data. Request parameters include a mandatory
98     * <code>productID</code> identifying what product to retrieve, and an
99     * optional <code>refIndex</code> in order to select which reference file to
100    * send. By default it sends the zeroth reference.
101    * 
102    * In addition, an optional <code>format</code> option can be specified to
103    * indicate that the product data be zipped up and delivered back as a
104    * zip file.
105    * 
106    * @param req
107    *          Servlet request
108    * @param res
109    *          Servlet response
110    * @throws ServletException
111    *           If an error occurs
112    * @throws IOException
113    *           If an I/O error occurs
114    */
115   public void doPut(HttpServletRequest req, HttpServletResponse res)
116       throws ServletException, IOException {
117     try {
118       String productID = req.getParameter("productID");
119       if (productID == null) {
120         throw new IllegalArgumentException("productID is required");
121       }
122       String refIndex = req.getParameter("refIndex");
123       if (refIndex == null) {
124         refIndex = "0";
125       }
126       int index = Integer.parseInt(refIndex);
127       String format = req.getParameter("format");
128 
129       if (format == null) {
130         // then just deliver back the particular product file
131         deliverProductFile(req, res, index, productID);
132       } else {
133         // make sure it's application/x-zip
134         if (!format.equals(FORMAT_ZIP)) {
135           throw new IllegalArgumentException("unknown product return format: "
136               + format);
137         }
138 
139         try {
140           deliverProductAsZip(req, res, productID);
141         } catch (Exception e) {
142           throw new ServletException(e);
143         }
144       }
145 
146     } catch (IllegalArgumentException ex) {
147       throw new ServletException(ex);
148     } catch (CatalogException ex) {
149       throw new ServletException(ex);
150     }
151   }

153   private void deliverProductAsZip(HttpServletRequest req,
154       HttpServletResponse res, String productID) throws CatalogException {
155     Product product = client.getProductById(productID);
156     product.setProductReferences(client.getProductReferences(product));
157     Metadata metadata = client.getMetadata(product);
158 
159     // we'll be delivering a zip
160     res.addHeader(CONTENT_TYPE_HDR, FORMAT_ZIP);
161 
162     String productZipFilePath;
163     File productZipFile;
164     InputStream in = null;
165     OutputStream o2 = null;
166 
167     try {
168       productZipFilePath = DataUtils.createProductZipFile(product, metadata,
169           workingDirPath);
170       productZipFile = new File(productZipFilePath);
171       String productZipFileName = productZipFile.getName();
172 
173       res.addHeader(CONTENT_DISPOSITION_HDR, "attachment; filename=\""
174           + productZipFileName + "\"");
175 
176       // add the content length for the zip file size
177       res.addHeader(CONTENT_LENGTH_HDR, String.valueOf(new File(
178           productZipFilePath).length()));
179 
180       // now we need to read the zip file in, and write it to the output
181       // stream
182       in = new FileInputStream(productZipFile);
183 
184       // Call java.io.File.delete() on the file, while still open.
185       if (!productZipFile.delete()) {
186         throw new RuntimeException("Unable to delete streaming file: ["
187             + productZipFile.getAbsolutePath() + "] while delivering!");
188       }
189 
190       // 3. Deliver the data.
191       o2 = res.getOutputStream();
192       byte[] buf = new byte[INT];
193       int n;
194       while ((n = in.read(buf)) != -1) {
195         o2.write(buf, 0, n);
196       }
197     } catch (Exception e) {
198       LOG.log(Level.SEVERE, e.getMessage());
199       LOG.log(Level.WARNING, "Exception delivering data!: Message: "
200           + e.getMessage());
201     } finally {
202       if (in != null) {
203         try {
204           in.close();
205         } catch (Exception ignore) {
206         }
207 
208       }
209 
210       if (o2 != null) {
211         try {
212           o2.close();
213         } catch (Exception ignore) {
214         }
215 
216       }
217 
218     }
219 
220   }

222   private void deliverProductFile(HttpServletRequest req,
223       HttpServletResponse res, int index, String productID)
224       throws CatalogException, IOException {
225     Product product = null; 
226     List refs = null;
227     
228     try{
229       product = client.getProductById(productID);
230       refs = client.getProductReferences(product);      
231     }
232     catch(Exception e){
233       LOG.warning("Unable to deliver product: ID: ["+productID+"]: "
234           + "Message: "+e.getMessage()+" throwing 404.");
235       res.sendError(HttpServletResponse.SC_NOT_FOUND);
236       return;
237     }
238     
239     Reference ref = (Reference) refs.get(index);
240     res.addHeader(CONTENT_LENGTH_HDR, String.valueOf(ref.getFileSize()));
241     String contentType = (ref.getMimeType() != null
242         && ref.getMimeType().getName() != null && !ref.getMimeType().getName()
243         .equals("")) ? ref.getMimeType().getName() : DataUtils
244         .guessTypeFromName(ref.getDataStoreReference());
245     res.addHeader(CONTENT_TYPE_HDR, contentType);
246     try {
247       res.addHeader(CONTENT_DISPOSITION_HDR, "attachment; filename=\""
248           + new File(new URI(ref.getDataStoreReference())).getName() + "\"");
249     } catch (URISyntaxException e) {
250       LOG.log(Level.WARNING,
251           "Unable to sense filename from data store URI: Message: "
252               + e.getMessage());
253     }
254     URL url = new URL(ref.getDataStoreReference());
255     URLConnection c = url.openConnection();
256     InputStream in = c.getInputStream();
257     OutputStream out = res.getOutputStream();
258     byte[] buf = new byte[INT];
259     int n;
260     while ((n = in.read(buf)) != -1) {
261       out.write(buf, 0, n);
262     }
263     in.close();
264     out.close();
265   }
