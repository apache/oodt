52    public WorkflowMgrConn(String wmUrlStr) {
53      try {
54        this.wm = new XmlRpcWorkflowManagerClient(new URL(wmUrlStr));
55      } catch (MalformedURLException e) {
56        LOG.log(Level.SEVERE, "Unable to contact Workflow Manager at URL: ["
57            + wmUrlStr + "]: Message: " + e.getMessage());
58        this.wm = null;
59      }
60    }

62    public WorkflowTask safeGetTaskById(String taskId) {
63      try {
64        return new SerializableWorkflowTask(this.wm.getTaskById(taskId));
65      } catch (Exception e) {
66        LOG.log(Level.WARNING, "Unable to obtain workflow task with ID: ["
67            + taskId + "]: Message: " + e.getMessage());
68        return null;
69      }
70    }
