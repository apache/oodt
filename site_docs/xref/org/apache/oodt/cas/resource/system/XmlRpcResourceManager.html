72      public XmlRpcResourceManager(int port) throws IOException {
73          // load properties from workflow manager properties file, if specified
74          if (System.getProperty("org.apache.oodt.cas.resource.properties") != null) {
75              String configFile = System
76                      .getProperty("org.apache.oodt.cas.resource.properties");
77              LOG.log(Level.INFO,
78                      "Loading Resource Manager Configuration Properties from: ["
79                              + configFile + "]");
80              System.getProperties().load(
81                      new FileInputStream(new File(configFile)));
82          }
83  
84          String schedulerClassStr = System.getProperty(
85                  "resource.scheduler.factory",
86                  "org.apache.oodt.cas.resource.scheduler.LRUSchedulerFactory");
87  
88          scheduler = GenericResourceManagerObjectFactory
89                  .getSchedulerServiceFromFactory(schedulerClassStr);
90  
91          // start up the scheduler
92          new Thread(scheduler).start();
93  
94  
95  
96          // start up the web server
97          webServer = new WebServer(port);
98          webServer.addHandler("resourcemgr", this);
99          webServer.start();
100 
101         LOG.log(Level.INFO, "Resource Manager started by "
102                 + System.getProperty("user.name", "unknown"));
103 
104     }

251     public boolean killJob(String jobId) throws MonitorException {
252         String resNodeId = scheduler.getBatchmgr().getExecutionNode(jobId);
253         if (resNodeId == null) {
254             LOG.log(Level.WARNING, "Attempt to kill job: [" + jobId
255                     + "]: cannot find execution node"
256                     + " (has the job already finished?)");
257             return false;
258         }
259         ResourceNode node = scheduler.getMonitor().getNodeById(resNodeId);
260         return scheduler.getBatchmgr().killJob(jobId, node);

358     public String getNodeReport() throws MonitorException{
359     	StringBuilder report = new StringBuilder();
360     	
361     	try{
362     		
363     		// get a sorted list of nodes
364     		List nodes = scheduler.getMonitor().getNodes();
365     		Collections.sort(nodes, new ResourceNodeComparator());
366     		
367     		// formulate the report string
368             for (Object node1 : nodes) {
369                 ResourceNode node = (ResourceNode) node1;
370                 String nodeId = node.getNodeId();
371                 report.append(nodeId);
372                 report.append(" (").append(getNodeLoad(nodeId)).append("/").append(node.getCapacity()).append(")");
373                 List<String> nodeQueues = getQueuesWithNode(nodeId);
374                 if (nodeQueues != null && nodeQueues.size() > 0) {
375                     report.append(" -- ").append(nodeQueues.get(0));
376                     for (int j = 1; j < nodeQueues.size(); j++) {
377                         report.append(", ").append(nodeQueues.get(j));
378                     }
379                 }
380                 report.append("\n");
381             }
382     	
383     	}catch(Exception e){
384     		throw new MonitorException(e.getMessage(), e);
385     	}
386     	
387     	return report.toString();
388     }

390     public String getExecutionReport() throws JobRepositoryException{
391     	StringBuilder report = new StringBuilder();
392     	
393     	try{
394     	
395 	    	// get a sorted list of all nodes, since the report should be
396 	    	// alphabetically sorted by node
397 	    	List resNodes = scheduler.getMonitor().getNodes();
398 	    	if(resNodes.size() == 0){
399 	    		throw new MonitorException(
400 	    				"No jobs can be executing, as there are no nodes in the Monitor");
401 	    	}
402 	    	Vector<String> nodeIds = new Vector<String>();
403             for (Object resNode : resNodes) {
404                 nodeIds.add(((ResourceNode) resNode).getNodeId());
405             }
406 	    	Collections.sort(nodeIds);
407 	    	
408 	    	// generate the report string
409 	    	for(String nodeId: nodeIds){
410 	    		List execJobIds = this.scheduler.getBatchmgr().getJobsOnNode(nodeId);
411 	    		if(execJobIds != null && execJobIds.size() > 0){
412                     for (Object execJobId : execJobIds) {
413                         String jobId = (String) execJobId;
414                         Job job = scheduler.getJobQueue().getJobRepository()
415                                            .getJobById(jobId).getJob();
416                         report.append("job id=").append(jobId);
417                         report.append(", load=").append(job.getLoadValue());
418                         report.append(", node=").append(nodeId);
419                         report.append(", queue=").append(job.getQueueName()).append("\n");
420                     }
421 	    		}
422 	    	}
423     	
424     	}catch(Exception e){
425     		throw new JobRepositoryException(e.getMessage(), e);
426     	}
427     	
428     	return report.toString();
429     }

467     private String genericHandleJob(Hashtable jobHash, Object jobIn)
468         throws SchedulerException {
469         return this.genericHandleJobCore(jobHash, jobIn);
470     }
471         private String genericHandleJobCore(Map jobHash, Object jobIn)
472             throws SchedulerException {
473 
474         Job exec = XmlRpcStructFactory.getJobFromXmlRpc(jobHash);
475         JobInput in = GenericResourceManagerObjectFactory
476                 .getJobInputFromClassName(exec.getJobInputClassName());
477         if (in != null) {
478             in.read(jobIn);
479         }
480 
481         JobSpec spec = new JobSpec(in, exec);
482 
483         // queue the job up
484         String jobId;
485 
486         try {
487             jobId = scheduler.getJobQueue().addJob(spec);
488         } catch (JobQueueException e) {
489             LOG.log(Level.WARNING, "JobQueue exception adding job: Message: "
490                     + e.getMessage());
491             throw new SchedulerException(e.getMessage());
492         }
493         return jobId;
494     }

496     private boolean genericHandleJob(Map jobHash, Object jobIn,
497                                          String urlStr) throws JobExecutionException {
498         return this.genericHandleJobCore(jobHash,jobIn,urlStr);
499     }
500     private boolean genericHandleJobCore(Map jobHash, Object jobIn,
501             String urlStr) throws JobExecutionException {
502         Job exec = XmlRpcStructFactory.getJobFromXmlRpc(jobHash);
503         JobInput in = GenericResourceManagerObjectFactory
504             .getJobInputFromClassName(exec.getJobInputClassName());
505         if (in != null) {
506             in.read(jobIn);
507         }
508 
509         JobSpec spec = new JobSpec(in, exec);
510 
511         URL remoteUrl = safeGetUrlFromString(urlStr);
512         ResourceNode remoteNode = null;
513 
514         try {
515             remoteNode = scheduler.getMonitor().getNodeByURL(remoteUrl);
516         } catch (MonitorException ignored) {
517         }
518 
519         return remoteNode != null && scheduler.getBatchmgr().executeRemotely(spec, remoteNode);
520     }

522     private URL safeGetUrlFromString(String urlStr) {
523         URL url = null;
524 
525         try {
526             url = new URL(urlStr);
527         } catch (MalformedURLException e) {
528             LOG.log(Level.WARNING, "Error converting string: [" + urlStr
529                     + "] to URL object: Message: " + e.getMessage());
530         }
531 
532         return url;
533     }
