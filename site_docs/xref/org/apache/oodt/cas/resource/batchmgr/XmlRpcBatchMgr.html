50  public class XmlRpcBatchMgr implements Batchmgr {
51      /* our log stream */
52      private static final Logger LOG = Logger.getLogger(XmlRpcBatchMgr.class
53              .getName());
54  
55      private Monitor mon;
56  
57      private JobRepository repo;
58  
59      private Map nodeToJobMap;
60  
61      private Map specToProxyMap;
62  
63      public XmlRpcBatchMgr() {
64          nodeToJobMap = new ConcurrentHashMap();
65          specToProxyMap = new ConcurrentHashMap();
66      }
67  
68      /*
69       * (non-Javadoc)
70       * 
71       * @see org.apache.oodt.cas.resource.batchmgr.Batchmgr#executeRemotely(org.apache.oodt.cas.resource.structs.JobSpec,
72       *      java.net.URL)
73       */
74      public boolean executeRemotely(JobSpec jobSpec, ResourceNode resNode)
75              throws JobExecutionException {
76  
77          XmlRpcBatchMgrProxy proxy = new XmlRpcBatchMgrProxy(jobSpec, resNode,
78                  this);
79          if (!proxy.nodeAlive()) {
80              throw new JobExecutionException("Node: [" + resNode.getNodeId()
81                      + "] is down: Unable to execute job!");
82          }
83  
84          synchronized (this.specToProxyMap) {
85              specToProxyMap.put(jobSpec.getJob().getId(), proxy);
86          }
87  
88          synchronized (this.nodeToJobMap) {
89              this.nodeToJobMap
90                      .put(jobSpec.getJob().getId(), resNode.getNodeId());
91          }
92  
93          proxy.start();
94  
95          return true;
96      }

126     /*
127      * (non-Javadoc)
128      * 
129      * @see org.apache.oodt.cas.resource.batchmgr.Batchmgr#killJob(java.lang.String,
130      *      org.apache.oodt.cas.resource.structs.ResourceNode)

132     public boolean killJob(String jobId, ResourceNode node) {
133         JobSpec spec;
134         try {
135             spec = repo.getJobById(jobId);
136         } catch (Exception e) {
137             LOG.log(Level.WARNING, "Unable to get job by id: [" + jobId
138                     + "] to kill it: Message: " + e.getMessage());
139             return false;
140         }
141 
142         XmlRpcBatchMgrProxy proxy = new XmlRpcBatchMgrProxy(spec, node, this);
143         return proxy.killJob();
144     }

146     public List getJobsOnNode(String nodeId){
147     	Vector<String> jobIds = new Vector();
148     	
149     	if(this.nodeToJobMap.size() > 0){
150             for (Object o : this.nodeToJobMap.keySet()) {
151                 String jobId = (String) o;
152                 if (nodeId.equals(this.nodeToJobMap.get(jobId))) {
153                     jobIds.add(jobId);
154                 }
155             }
156     	}
157     	
158     	Collections.sort(jobIds); // sort the list to return as a courtesy to the user
159     	
160     	return jobIds;
161     }

163     protected void notifyMonitor(ResourceNode node, JobSpec jobSpec) {
164         Job job = jobSpec.getJob();
165         int reducedLoad = job.getLoadValue();
166         try {
167             mon.reduceLoad(node, reducedLoad);
168         } catch (MonitorException ignored) {
169         }
170     }

172     protected void jobSuccess(JobSpec spec) {
173         spec.getJob().setStatus(JobStatus.SUCCESS);
174         synchronized (this.nodeToJobMap) {
175             this.nodeToJobMap.remove(spec.getJob().getId());
176         }
177         synchronized (this.specToProxyMap) {
178             this.specToProxyMap
179                     .remove(spec.getJob().getId());
180         }
181 
182         try {
183             repo.updateJob(spec);
184         } catch (JobRepositoryException e) {
185             LOG.log(Level.WARNING, "Error set job completion status for job: ["
186                     + spec.getJob().getId() + "]: Message: " + e.getMessage());
187         }
188     }
189     
190     protected void jobFailure(JobSpec spec) {
191         spec.getJob().setStatus(JobStatus.FAILURE);
192         synchronized (this.nodeToJobMap) {
193             this.nodeToJobMap.remove(spec.getJob().getId());
194         }
195         synchronized (this.specToProxyMap) {
196             this.specToProxyMap
197                     .remove(spec.getJob().getId());
198 
199         }
200 
201         try {
202             repo.updateJob(spec);
203         } catch (JobRepositoryException e) {
204             LOG.log(Level.WARNING, "Error set job completion status for job: ["
205                     + spec.getJob().getId() + "]: Message: " + e.getMessage());
206         }
207     }
208 
209     protected void jobKilled(JobSpec spec) {
210         spec.getJob().setStatus(JobStatus.KILLED);
211         nodeToJobMap.remove(spec.getJob().getId());
212         try {
213             repo.updateJob(spec);
214         } catch (JobRepositoryException e) {
215             LOG.log(Level.WARNING, "Error setting job killed status for job: ["
216                     + spec.getJob().getId() + "]: Message: " + e.getMessage());
217         }
218     }
219 
220     protected void jobExecuting(JobSpec spec) {
221         spec.getJob().setStatus(JobStatus.EXECUTED);
222         try {
223             repo.updateJob(spec);
224         } catch (JobRepositoryException e) {
225             LOG.log(Level.WARNING,
226                     "Error setting job execution status for job: ["
227                             + spec.getJob().getId() + "]: Message: "
228                             + e.getMessage());
229         }
230     }
