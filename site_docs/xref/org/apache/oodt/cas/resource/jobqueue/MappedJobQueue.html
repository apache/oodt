18  package org.apache.oodt.cas.resource.jobqueue;
19  
20  import java.util.List;
21  
22  import org.apache.oodt.cas.resource.jobqueue.JobQueue;
23  import org.apache.oodt.cas.resource.structs.JobSpec;
24  import org.apache.oodt.cas.resource.structs.exceptions.JobQueueException;
25  import org.apache.oodt.cas.resource.structs.exceptions.JobRepositoryException;
26  
27  /**
28   * The interface for a {@link JobQueue} that is aware of the different queues in
29   * the ResourceManager and allows for the manipulation of jobs on a
30   * queue-by-queue basis.
31   * 
32   * @author resneck
33   *
34   */
35  public interface MappedJobQueue extends JobQueue {
36  
37    /**
38     * Returns a boolean value representing whether or not the job queue contains
39     * jobs in the given queue.
40     * 
41     * @param queueName
42     *          The name of the queue which you are checking.
43     * @return true, if the queue is empty, false otherwise.
44     * @throws JobQueueException
45     *           If the given queue name is null or if no queue with that name
46     *           exists.
47     */
48    public boolean isEmpty(String queueName) throws JobQueueException;
49  
50    /**
51     * Gets the next {@link JobSpec} in the queue with the given name.
52     * 
53     * @param queueName
54     *          The name of the queue from which the next {@link JobSpec} will be
55     *          returned.
56     * @return The next {@link JobSpec} from the jobqueue belonging to the queue
57     *         with the given name.
58     * @throws JobQueueException
59     *           If the given queue name is null or if no queue with that name
60     *           exists.
61     */
62    public JobSpec getNextJob(String queueName)
63        throws JobQueueException, JobRepositoryException;
64  
65    /**
66     * Gets the number of jobs in the queue with the given name.
67     * 
68     * @param queueName
69     *          The name of the queue whos size will be given.
70     * @return The number of {@link JobSpec}s in the queue with the given name.
71     * @throws JobQueueException
72     *           If the given queue name is null or if no queue with that name
73     *           exists.
74     */
75    public int getSize(String queueName) throws JobQueueException;
76  
77    /**
78     * Removes the {@link JobSpec} with the given ID from the queue.
79     * 
80     * @param id
81     *          The ID of the {@link JobSpec} that will be removed.
82     * @throws JobQueueException
83     *           If no {@link JobSpec} has the given ID.
84     */
85    public void removeJob(JobSpec spec) throws JobQueueException;
86  
87    /**
88     * Add a queue with the given name.
89     * 
90     * @param queueName
91     *          The name of the queue to be created.
92     * @throws JobQueueException
93     *           If the given queue name is null or if a queue with the given name
94     *           already exists.
95     */
96    public void addQueue(String queueName) throws JobQueueException;
97  
98    /**
99     * Remove the queue with the given name.
100    * 
101    * @param queueName
102    *          The name of the queue to be removed.
103    * @throws JobQueueException
104    *           If the given queue name is null or if no queue with the given
105    *           name exists.
106    */
107   public void removeQueue(String queueName) throws JobQueueException;
108 
109   /**
110    * Gets a list of all queued jobs that belong to the queue with the given
111    * name.
112    * 
113    * @param queueName
114    *          The name of the queue whose members will be given.
115    * @return A {@link List} of queued {@JobSpec}s from the given queue.
116    * @throws JobQueueException
117    *           If the given queue name is null or if no queue with the given
118    *           name exists.
119    */
120   public List<JobSpec> getQueuedJobs(String queueName)
121       throws JobQueueException, JobRepositoryException;
122 
123   public String addJob(JobSpec spec) throws JobQueueException;
124 
125   public String requeueJob(JobSpec spec) throws JobQueueException;
126 
127   public void promoteJob(JobSpec spec) throws JobQueueException;
128 
129   public void promoteKeyValPair(String key, String val)
130       throws JobQueueException, JobRepositoryException;
131 
132 }
