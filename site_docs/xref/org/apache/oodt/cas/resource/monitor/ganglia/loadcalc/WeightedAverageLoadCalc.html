21  import java.text.NumberFormat;
22  import java.util.Map;

25  import static org.apache.oodt.cas.resource.monitor.ganglia.GangliaMetKeys.MAXIMUM_FRACTION_DIGITS;
26  import static org.apache.oodt.cas.resource.monitor.ganglia.GangliaMetKeys.TN;
27  import static org.apache.oodt.cas.resource.monitor.ganglia.GangliaMetKeys.TMAX;
28  import static org.apache.oodt.cas.resource.monitor.ganglia.GangliaMetKeys.CPU_NUM;
29  import static org.apache.oodt.cas.resource.monitor.ganglia.GangliaMetKeys.LOAD_ONE;
30  import static org.apache.oodt.cas.resource.monitor.ganglia.GangliaMetKeys.LOAD_FIVE;
31  import static org.apache.oodt.cas.resource.monitor.ganglia.GangliaMetKeys.LOAD_FIFTEEN;

33  /**
34   * @author rajith
35   * @author mattmann
36   * @version $Revision$
37   */
38  public class WeightedAverageLoadCalc implements LoadCalculator {
39  
40      private double loadOneWeight;
41      private double loadFiveWeight;
42      private double loadFifteenWeight;
43  
44      /* to format the load value*/
45      private NumberFormat numberFormat;
46  
47      /**
48       * Make a new WeightedAverageLoadCalc {@link LoadCalculator}
49       * @param loadOneWeight weight for the load_one
50       * @param loadFiveWeight weight for the load_five
51       * @param loadFifteenWeight weight for the load_fifteen
52       */
53      public WeightedAverageLoadCalc (double loadOneWeight, double loadFiveWeight, double loadFifteenWeight){
54          this.loadOneWeight = loadOneWeight;
55          this.loadFiveWeight = loadFiveWeight;
56          this.loadFifteenWeight = loadFifteenWeight;
57  
58          numberFormat = NumberFormat.getNumberInstance();
59          numberFormat.setMaximumFractionDigits(MAXIMUM_FRACTION_DIGITS);
60      }
61      /**
62       * {@inheritDoc}
63       *
64       * load is calculated as follows
65       * weightedLoadOne = loadOneWeight * minimum of (numOfCPUs, ((loadOne/numOfCPUs) * numOfCPUs))
66       *
67       * load = (weightedLoadOne + weightedLoadFive + weightedLoadFifteen) /
68       *           (loadOneWeight + loadFiveWeight + loadFifteenWeight)
69       */
70      @Override
71      public double calculateLoad(Map<String, String> nodeMetrics) {
72          double tn = Double.valueOf(nodeMetrics.get(TN));
73          double tmax = Double.valueOf(nodeMetrics.get(TMAX));
74          double numCpus = Double.valueOf(nodeMetrics.get(CPU_NUM));
75  
76          if(tn > (4 * tmax)){
77             return numCpus; //if the node is offline assign the node's capacity as the load
78          }
79          else {
80              double weightedLoadOne = loadOneWeight * Math.min(numCpus,
81                      ((Double.valueOf(nodeMetrics.get(LOAD_ONE)) /
82                              Double.valueOf(nodeMetrics.get(CPU_NUM))) * numCpus));
83              double weightedLoadFive = loadFiveWeight * Math.min(numCpus,
84                      ((Double.valueOf(nodeMetrics.get(LOAD_FIVE)) /
85                              Double.valueOf(nodeMetrics.get(CPU_NUM)))* numCpus));
86              double weightedLoadFifteen = loadFifteenWeight * Math.min(numCpus,
87                      ((Double.valueOf(nodeMetrics.get(LOAD_FIFTEEN)) /
88                              Double.valueOf(nodeMetrics.get(CPU_NUM)))* numCpus));
89  
90              double weightedLoadAverage = (weightedLoadOne + weightedLoadFive + weightedLoadFifteen) /
91                      (loadOneWeight + loadFiveWeight + loadFifteenWeight);
92              
93              System.out.println("Weighted load one: ["+weightedLoadOne+"]: weighted load five: ["+weightedLoadFive+"] weighted load fifteen: ["+weightedLoadFifteen+"]");
94              return Double.valueOf(numberFormat.format(weightedLoadAverage));
95          }
96      }
97  }
