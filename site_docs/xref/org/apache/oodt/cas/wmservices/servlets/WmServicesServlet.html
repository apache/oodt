18  package org.apache.oodt.cas.wmservices.servlets;
19  
20  import java.io.File;
21  import java.net.MalformedURLException;
22  import java.net.URL;
23  import java.util.logging.Level;
24  import java.util.logging.Logger;
25  
26  import javax.servlet.ServletConfig;
27  import javax.servlet.ServletContext;
28  import javax.servlet.ServletException;
29  
30  import org.apache.cxf.jaxrs.servlet.CXFNonSpringJaxrsServlet;
31  import org.apache.oodt.cas.metadata.util.PathUtils;
32  import org.apache.oodt.cas.workflow.system.XmlRpcWorkflowManagerClient;
33  
34  /**
35   * Initialize workflow manager services servlet
36   * 
37   * Set the following Parameters in context.xml for the webapp:
38   * <ul>
39   *  <li>workflow.url
40   *  <li>packagedRepo.dir.path
41   * </ul>
42   * 
43   * @author vratnakar
44   */
45  public class WmServicesServlet extends CXFNonSpringJaxrsServlet {
46    private static final Logger LOGGER = Logger
47        .getLogger(WmServicesServlet.class.getName());
48    
49    private static final long serialVersionUID = -7830210280506307805L;
50  
51    // Default URL for the workflow manager
52    private static final String DEFAULT_WM_URL = "http://localhost:9001";
53    
54    // Servlet context parameter names.
55    private static final String PARAM_NAME_URL = "workflow.url";
56    private static final String PARAM_NAME_PKGREPO_DIR = "packagedRepo.dir.path"; // For Packaged Repo
57  
58    /**
59     * The name of the servlet context attribute that holds a client for the
60     * workflow manager, a {@link XmlRpcWorkflowManagerClient} object.
61     */
62    public static final String ATTR_NAME_CLIENT = "client";
63    
64    /**
65     * The name of the servlet context attribute that holds the workflow manager's
66     * packaged repository directory: a {@link File} object.
67     */
68    public static final String ATTR_NAME_PKG_REPO_DIR = "pkgRepoFilesDir";
69    
70    
71    @Override
72    public void init(ServletConfig configuration) throws ServletException {
73      super.init(configuration);
74      ServletContext context = configuration.getServletContext();
75  
76      // Initialize the workflow manager client.
77      try {
78        URL url = null;
79        String urlParameter = context.getInitParameter(PARAM_NAME_URL);
80        if (urlParameter != null) {
81          // Get the workflow manager URL from the context parameter.
82          url = new URL(PathUtils.replaceEnvVariables(urlParameter));
83        } else {
84          // Try the default URL for the workflow manager.
85          LOGGER.log(Level.WARNING,
86              "Unable to find the servlet context parameter" + " (\""
87                  + PARAM_NAME_URL + "\") for the workflow manager's URL.");
88          url = new URL(DEFAULT_WM_URL);
89        }
90        // Attempt to connect the client to the workflow manager and if successful
91        // store the client as a context attribute for other objects to access.
92        XmlRpcWorkflowManagerClient client = new XmlRpcWorkflowManagerClient(url);
93        context.setAttribute(ATTR_NAME_CLIENT, client);
94      } catch (MalformedURLException e) {
95        LOGGER.log(Level.SEVERE,
96            "Encountered a malformed URL for the workflow manager.", e);
97        throw new ServletException(e);
98      }
99  
100     // Initialize the repository directory
101     String pkgRepoDirPath = context.getInitParameter(PARAM_NAME_PKGREPO_DIR);
102     if (pkgRepoDirPath != null) {
103       pkgRepoDirPath = PathUtils.replaceEnvVariables(pkgRepoDirPath);
104       File workflowDir = new File(pkgRepoDirPath);
105       if (workflowDir.exists() && workflowDir.isDirectory()) {
106         context.setAttribute(ATTR_NAME_PKG_REPO_DIR, workflowDir);
107       } else {
108         LOGGER.log(Level.SEVERE,
109             "Unable to locate the Packaged repository directory ("
110                 + workflowDir.getAbsolutePath());
111       }
112     }
113   }
114 }
