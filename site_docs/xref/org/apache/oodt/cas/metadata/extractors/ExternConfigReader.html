37  /**
38   * @author mattmann
39   * @author bfoster
40   * @version $Revision$
41   * 
42   * <p>
43   * Reader that reads {@link ExternalMetExtractorConfig}s from an XML file.
44   * </p>.
45   */
46  public final class ExternConfigReader implements MetExtractorConfigReader,
47          ExternMetExtractorMetKeys, ExternConfigReaderMetKeys {
48  
49      /*
50       * (non-Javadoc)
51       * 
52       * @see org.apache.oodt.cas.metadata.MetExtractorConfigReader#parseConfigFile(java.io.File)
53       */
54      public MetExtractorConfig parseConfigFile(File file)
55              throws MetExtractorConfigReaderException {
56          try {
57              Document doc = XMLUtils.getDocumentRoot(new FileInputStream(file));
58  
59              ExternalMetExtractorConfig config = new ExternalMetExtractorConfig();
60              Element docElem = doc.getDocumentElement();
61              Element execElement = XMLUtils.getFirstElement(EXEC_TAG, docElem);
62              config.setWorkingDirPath(PathUtils.replaceEnvVariables(execElement
63                      .getAttribute(WORKING_DIR_ATTR)));
64              String metFileExt = PathUtils.replaceEnvVariables(execElement
65                      .getAttribute(MET_FILE_EXT_ATTR));
66              if (!metFileExt.equals("")) {
67                config.setMetFileExt(metFileExt);
68              }
69              Element binPathElem = XMLUtils.getFirstElement(
70                      EXTRACTOR_BIN_PATH_TAG, execElement);
71              String binPath = XMLUtils.getSimpleElementText(binPathElem);
72              if (Boolean.valueOf(binPathElem.getAttribute(ENV_REPLACE_ATTR))) {
73                  binPath = PathUtils.replaceEnvVariables(binPath);
74              }
75  
76              // make sure to do path replacement on binPath because it's always
77              // going
78              // to be
79              // a path
80              binPath = binPath.replaceAll("^\\s+", "").replaceAll("\\s+$", "").replaceAll("\\s", "\\\\ ");
81  
82              config.setExtractorBinPath(binPath);
83  
84              Element argsElem = XMLUtils.getFirstElement(ARGS_TAG, execElement);
85              if (argsElem != null) {
86                  NodeList argNodes = argsElem.getElementsByTagName(ARG_TAG);
87  
88                  if (argNodes != null && argNodes.getLength() > 0) {
89                      Vector argVector = new Vector();
90                      for (int i = 0; i < argNodes.getLength(); i++) {
91                          Element argElem = (Element) argNodes.item(i);
92                          String argStr;
93                          if (Boolean.valueOf(
94                              argElem.getAttribute(IS_DATA_FILE_ATTR)
95                                     .toLowerCase())) {
96                            argStr = DATA_FILE_PLACE_HOLDER;
97                          } else if (Boolean.valueOf(
98                              argElem.getAttribute(IS_MET_FILE_ATTR)
99                                     .toLowerCase())) {
100                           argStr = MET_FILE_PLACE_HOLDER;
101                         } else {
102                           argStr = XMLUtils.getSimpleElementText(argElem);
103                         }
104 
105                         String appendExt;
106                         if (!(appendExt = argElem.getAttribute(APPEND_EXT_ATTR))
107                                 .equals("")) {
108                           argStr += "." + appendExt;
109                         }
110 
111                         if (Boolean.valueOf(
112                             argElem.getAttribute(ENV_REPLACE_ATTR))) {
113                             argStr = PathUtils.replaceEnvVariables(argStr);
114                         }
115 
116                         if (Boolean.valueOf(argElem.getAttribute(IS_PATH_ATTR))) {
117                             argStr = argStr.replaceAll("\\s", "\\\\ ");
118                         }
119 
120                         argVector.add(argStr);
121                     }
122 
123                     config.setArgList((String[]) argVector
124                         .toArray(new String[argVector.size()]));
125                 }
126             }
127 
128             return config;
129         } catch (Exception e) {
130             throw new MetExtractorConfigReaderException("Failed to parser '"
131                     + file + "' : " + e.getMessage());
132         }
133 
134     }
