40  /**
41   * @author mattmann
42   * @author bfoster
43   * @version $Revision$
44   * 
45   * <p>
46   * Runs an external {@link MetExtractor} that is a command line program to
47   * extract {@link Metadata} from {@link Product} files.
48   * </p>.
49   */
50  public class ExternMetExtractor extends CmdLineMetExtractor implements
51          ExternMetExtractorMetKeys {
52  
53      private static ExternConfigReader reader = new ExternConfigReader();
54  
55      public ExternMetExtractor() {
56          super(reader);
57      }
58  
59      /*
60       * (non-Javadoc)
61       * 
62       * @see org.apache.oodt.cas.metadata.AbstractMetExtractor#extractMetadata(java.io.File)
63       */
64      public Metadata extrMetadata(File file) throws MetExtractionException {
65  
66          // determine working directory
67          String workingDirPath = ((ExternalMetExtractorConfig) this.config)
68                  .getWorkingDirPath();
69          if (workingDirPath == null || workingDirPath.equals("")) {
70            workingDirPath = file.getParentFile().getAbsolutePath();
71          }
72          File workingDir = new File(workingDirPath);
73  
74          // determine met file path
75          String metFileName = file.getName() + "."
76                  + ((ExternalMetExtractorConfig) this.config).getMetFileExt();
77          String metFilePath = workingDir.getAbsolutePath() + "/" + metFileName;
78          File metFile = new File(metFilePath);
79  
80          // get exe args
81          List commandLineList = new Vector();
82          commandLineList.add(((ExternalMetExtractorConfig) this.config)
83                  .getExtractorBinPath());
84          if (((ExternalMetExtractorConfig) this.config).getArgList() != null
85                  && ((ExternalMetExtractorConfig) this.config).getArgList().length > 0) {
86            commandLineList.addAll(Arrays
87                .asList(((ExternalMetExtractorConfig) this.config)
88                    .getArgList()));
89          }
90          String[] commandLineArgs = new String[commandLineList.size()];
91          for (int i = 0; i < commandLineList.size(); i++) {
92            commandLineArgs[i] = StringUtils.replace(StringUtils.replace(
93                (String) commandLineList.get(i), MET_FILE_PLACE_HOLDER,
94                metFilePath), DATA_FILE_PLACE_HOLDER, file
95                .getAbsolutePath());
96          }
97  
98          // generate metadata file
99          LOG.log(Level.INFO, "Generating met file for product file: ["
100                 + file.getAbsolutePath() + "]");
101         int status;
102         try {
103             LOG.log(Level.INFO, "Executing command line: ["
104                     + ExecUtils.printCommandLine(commandLineArgs)
105                     + "] with workingDir: [" + workingDir
106                     + "] to extract metadata");
107             status = ExecUtils.callProgram(commandLineArgs, workingDir);
108         } catch (IOException e) {
109             LOG.log(Level.WARNING,
110                     "IOException running met extraction: commandLine: ["
111                             + ExecUtils.printCommandLine(commandLineArgs)
112                             + "]: Message: " + e.getMessage());
113             return null;
114         }
115 
116         if (status != 0) {
117             throw new MetExtractionException(
118                     "Exit status for met extraction not 0");
119         } else if (!metFile.exists()) {
120             throw new MetExtractionException(
121                     "Met extractor failed to create metadata file");
122         } else {
123             LOG.log(Level.INFO, "Met extraction successful for product file: ["
124                     + file.getAbsolutePath() + "]");
125             try {
126                 SerializableMetadata sm = new SerializableMetadata("UTF-8",
127                         false);
128                 sm.loadMetadataFromXmlStream(new FileInputStream(metFile));
129                 return sm;
130             } catch (Exception e) {
131                 LOG.log(Level.SEVERE, e.getMessage());
132                 throw new MetExtractionException(e.getMessage());
133             }
134         }
135     }
136 
137     public static void main(String[] args) throws Exception {
138         processMain(args, new ExternMetExtractor());
139     }
